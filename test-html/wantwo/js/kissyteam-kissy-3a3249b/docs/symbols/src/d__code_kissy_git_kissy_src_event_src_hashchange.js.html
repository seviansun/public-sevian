1 /**
  2  * @fileOverview   event-hashchange
  3  * @author  yiminghe@gmail.com , xiaomacji@gmail.com
  4  */
  5 KISSY.add('event/hashchange', function (S, Event, DOM, UA, special) {
  6 
  7     var win=S.Env.host,
  8         doc = win.document,
  9         docMode = doc['documentMode'],
 10         ie = docMode || UA['ie'],
 11         HASH_CHANGE = 'hashchange';
 12 
 13     // ie8 支持 hashchange
 14     // 但IE8以上切换浏览器模式到IE7（兼容模式），会导致 'onhashchange' in window === true，但是不触发事件
 15 
 16     // 1. 不支持 hashchange 事件，支持 hash 导航(opera??)：定时器监控
 17     // 2. 不支持 hashchange 事件，不支持 hash 导航(ie67) : iframe + 定时器
 18     if ((!( 'on' + HASH_CHANGE in win)) || ie && ie < 8) {
 19 
 20 
 21         function getIframeDoc(iframe) {
 22             return iframe.contentWindow.document;
 23         }
 24 
 25         var POLL_INTERVAL = 50,
 26             IFRAME_TEMPLATE = "<html><head><title>" + (doc.title || "") +
 27                 " - {hash}</title>{head}</head><body>{hash}</body></html>",
 28 
 29             getHash = function () {
 30                 // 不能 location.hash
 31                 // http://xx.com/#yy?z=1
 32                 // ie6 => location.hash = #yy
 33                 // 其他浏览器 => location.hash = #yy?z=1
 34                 var url = location.href;
 35                 return '#' + url.replace(/^[^#]*#?(.*)$/, '$1');
 36             },
 37 
 38             timer,
 39 
 40             // 用于定时器检测，上次定时器记录的 hash 值
 41             lastHash,
 42 
 43             poll = function () {
 44                 var hash = getHash();
 45                 if (hash !== lastHash) {
 46                     // S.log("poll success :" + hash + " :" + lastHash);
 47                     // 通知完调用者 hashchange 事件前设置 lastHash
 48                     lastHash = hash;
 49                     // ie<8 同步 : hashChange -> onIframeLoad
 50                     hashChange(hash);
 51                 }
 52                 timer = setTimeout(poll, POLL_INTERVAL);
 53             },
 54 
 55             hashChange = ie && ie < 8 ? function (hash) {
 56                 // S.log("set iframe html :" + hash);
 57 
 58                 var html = S.substitute(IFRAME_TEMPLATE, {
 59                     hash:hash,
 60                     // 一定要加哦
 61                     head:DOM._isCustomDomain() ? "<script>document.domain = '" +
 62                         doc.domain
 63                         + "';</script>" : ""
 64                 }),
 65                     iframeDoc = getIframeDoc(iframe);
 66                 try {
 67                     // 写入历史 hash
 68                     iframeDoc.open();
 69                     // 取时要用 innerText !!
 70                     // 否则取 innerHtml 会因为 escapeHtml 导置 body.innerHTMl != hash
 71                     iframeDoc.write(html);
 72                     iframeDoc.close();
 73                     // 立刻同步调用 onIframeLoad !!!!
 74                 } catch (e) {
 75                     // S.log('doc write error : ', 'error');
 76                     // S.log(e, 'error');
 77                 }
 78             } : function () {
 79                 notifyHashChange();
 80             },
 81 
 82             notifyHashChange = function () {
 83                 // S.log("hash changed : " + getHash());
 84                 Event.fire(win, HASH_CHANGE);
 85             },
 86             setup = function () {
 87                 if (!timer) {
 88                     poll();
 89                 }
 90             },
 91             tearDown = function () {
 92                 timer && clearTimeout(timer);
 93                 timer = 0;
 94             },
 95             iframe;
 96 
 97         // ie6, 7, 覆盖一些function
 98         if (ie < 8) {
 99 
100             /*
101              前进后退 : start -> notifyHashChange
102              直接输入 : poll -> hashChange -> start
103              iframe 内容和 url 同步
104              */
105             setup = function () {
106                 if (!iframe) {
107                     var iframeSrc = DOM._genEmptyIframeSrc();
108                     //http://www.paciellogroup.com/blog/?p=604
109                     iframe = DOM.create('<iframe ' +
110                         (iframeSrc ? 'src="' + iframeSrc + '"' : '') +
111                         ' style="display: none" ' +
112                         'height="0" ' +
113                         'width="0" ' +
114                         'tabindex="-1" ' +
115                         'title="empty"/>');
116                     // Append the iframe to the documentElement rather than the body.
117                     // Keeping it outside the body prevents scrolling on the initial
118                     // page load
119                     DOM.prepend(iframe, doc.documentElement);
120 
121                     // init，第一次触发，以后都是 onIframeLoad
122                     Event.add(iframe, "load", function () {
123                         Event.remove(iframe, "load");
124                         // Update the iframe with the initial location hash, if any. This
125                         // will create an initial history entry that the user can return to
126                         // after the state has changed.
127                         hashChange(getHash());
128                         Event.add(iframe, "load", onIframeLoad);
129                         poll();
130                     });
131 
132                     // Whenever `document.title` changes, update the Iframe's title to
133                     // prettify the back/next history menu entries. Since IE sometimes
134                     // errors with "Unspecified error" the very first time this is set
135                     // (yes, very useful) wrap this with a try/catch block.
136                     doc.onpropertychange = function () {
137                         try {
138                             if (event.propertyName === 'title') {
139                                 getIframeDoc(iframe).title =
140                                     doc.title + " - " + getHash();
141                             }
142                         } catch (e) {
143                         }
144                     };
145 
146                     /*
147                      前进后退 ： onIframeLoad -> 触发
148                      直接输入 : timer -> hashChange -> onIframeLoad -> 触发
149                      触发统一在 start(load)
150                      iframe 内容和 url 同步
151                      */
152                     function onIframeLoad() {
153                         // S.log('iframe start load..');
154 
155                         // 2011.11.02 note: 不能用 innerHtml 会自动转义！！
156                         // #/x?z=1&y=2 => #/x?z=1&y=2
157                         var c = S.trim(getIframeDoc(iframe).body.innerText),
158                             ch = getHash();
159 
160                         // 后退时不等
161                         // 定时器调用 hashChange() 修改 iframe 同步调用过来的(手动改变 location)则相等
162                         if (c != ch) {
163                             S.log("set loc hash :" + c);
164                             location.hash = c;
165                             // 使lasthash为 iframe 历史， 不然重新写iframe，
166                             // 会导致最新状态（丢失前进状态）
167 
168                             // 后退则立即触发 hashchange，
169                             // 并更新定时器记录的上个 hash 值
170                             lastHash = c;
171                         }
172                         notifyHashChange();
173                     }
174                 }
175             };
176 
177             tearDown = function () {
178                 timer && clearTimeout(timer);
179                 timer = 0;
180                 Event.detach(iframe);
181                 DOM.remove(iframe);
182                 iframe = 0;
183             };
184         }
185 
186         special[HASH_CHANGE] = {
187             setup:function () {
188                 if (this !== win) {
189                     return;
190                 }
191                 // 第一次启动 hashchange 时取一下，不能类库载入后立即取
192                 // 防止类库嵌入后，手动修改过 hash，
193                 lastHash = getHash();
194                 // 不用注册 dom 事件
195                 setup();
196             },
197             tearDown:function () {
198                 if (this !== win) {
199                     return;
200                 }
201                 tearDown();
202             }
203         };
204     }
205 }, {
206     requires:["./base", "dom", "ua", "./special"]
207 });
208
