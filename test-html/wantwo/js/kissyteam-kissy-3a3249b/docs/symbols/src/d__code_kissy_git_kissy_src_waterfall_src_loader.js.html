1 /**
  2  * @fileOverview load content
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("waterfall/loader", function (S, Node, Waterfall) {
  6 
  7     var $ = Node.all,
  8         win=S.Env.host,
  9         SCROLL_TIMER = 50;
 10 
 11     /**
 12      * @class
 13      * @memberOf Waterfall
 14      */
 15     function Loader() {
 16         Loader.superclass.constructor.apply(this, arguments);
 17     }
 18 
 19 
 20     function doScroll() {
 21         var self = this;
 22         if (self.__pause) {
 23             return;
 24         }
 25         S.log("waterfall:doScroll");
 26         if (self.__loading) {
 27             return;
 28         }
 29         // 如果正在调整中，等会再看
 30         // 调整中的高度不确定，现在不适合判断是否到了加载新数据的条件
 31         if (self.isAdjusting()) {
 32             // 恰好 __onScroll 是 buffered . :)
 33             self.__onScroll();
 34             return;
 35         }
 36         var container = self.get("container"),
 37             colHeight = container.offset().top,
 38             diff = self.get("diff"),
 39             curColHeights = self.get("curColHeights");
 40         // 找到最小列高度
 41         if (curColHeights.length) {
 42             colHeight += Math.min.apply(Math, curColHeights);
 43         }
 44         // 动态载
 45         // 最小高度(或被用户看到了)低于预加载线
 46         if (diff + $(win).scrollTop() + $(win).height() > colHeight) {
 47             S.log("waterfall:loading");
 48             loadData.call(self);
 49         }
 50     }
 51 
 52     function loadData() {
 53         var self = this,
 54             container = this.get("container");
 55 
 56         self.__loading = 1;
 57 
 58         var load = self.get("load");
 59 
 60         load && load(success, end);
 61 
 62         function success(items) {
 63             self.__loading = 0;
 64             self.addItems(items);
 65         }
 66 
 67         function end() {
 68             self.end();
 69         }
 70 
 71     }
 72 
 73     Loader.ATTRS =
 74     /**
 75      * @lends Waterfall#
 76      */
 77     {
 78         diff:{
 79             getter:function (v) {
 80                 return v || 0;
 81                 // 默认一屏内加载
 82                 //return $(window).height() / 4;
 83             }
 84         }
 85     };
 86 
 87 
 88     S.extend(Loader, Waterfall,
 89         /**
 90          * @lends Waterfall#
 91          */
 92         {
 93             _init:function () {
 94                 var self = this;
 95                 Loader.superclass._init.apply(self, arguments);
 96                 self.__onScroll = S.buffer(doScroll, SCROLL_TIMER, self);
 97                 $(win).on("scroll", self.__onScroll);
 98                 doScroll.call(self);
 99             },
100 
101             end:function () {
102                 $(win).detach("scroll", this.__onScroll);
103             },
104 
105 
106             pause:function () {
107                 this.__pause = 1;
108             },
109 
110             resume:function () {
111                 this.__pause = 0;
112             },
113 
114             destroy:function () {
115                 var self = this;
116                 Loader.superclass.destroy.apply(self, arguments);
117                 $(win).detach("scroll", self.__onScroll);
118             }
119         });
120 
121     return Loader;
122 
123 }, {
124     requires:['node', './base']
125 });
