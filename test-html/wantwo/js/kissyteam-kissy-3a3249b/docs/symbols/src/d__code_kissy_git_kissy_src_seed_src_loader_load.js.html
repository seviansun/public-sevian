  5 (function(S, loader, utils, data) {
  6     if (typeof require !== 'undefined') {
  7         return;
  8     }
  9     var IE = utils.IE,
 10         LOADING = data.LOADING,
 11         LOADED = data.LOADED,
 12         ERROR = data.ERROR,
 13         ATTACHED = data.ATTACHED;
 14 
 15     S.mix(loader, {
 16         /**
 17          * Load a single module.
 18          */
 19         __load: function(mod, callback, cfg) {
 20 
 21             var self = this,
 22                 url = mod['fullpath'],
 23                 isCss = utils.isCss(url),
 24                 // 这个是全局的，防止多实例对同一模块的重复下载
 25                 loadQueque = S.Env._loadQueue,
 26                 status = loadQueque[url],
 27                 node = status;
 28 
 29             mod.status = mod.status || 0;
 30 
 31             // 可能已经由其它模块触发加载
 32             if (mod.status < LOADING && status) {
 33                 // 该模块是否已经载入到 global ?
 34                 mod.status = status === LOADED ? LOADED : LOADING;
 35             }
 36 
 37             // 1.20 兼容 1.1x 处理：加载 cssfullpath 配置的 css 文件
 38             // 仅发出请求，不做任何其它处理
 39             if (S.isString(mod["cssfullpath"])) {
 40                 S.getScript(mod["cssfullpath"]);
 41                 mod["cssfullpath"] = mod.csspath = LOADED;
 42             }
 43 
 44             if (mod.status < LOADING && url) {
 45                 mod.status = LOADING;
 46                 if (IE && !isCss) {
 47                     self.__startLoadModuleName = mod.name;
 48                     self.__startLoadTime = Number(+new Date());
 49                 }
 50                 node = S.getScript(url, {
 51                     success: function() {
 52                         if (isCss) {
 53 
 54                         } else {
 55                             //载入 css 不需要这步了
 56                             //标准浏览器下：外部脚本执行后立即触发该脚本的 load 事件,ie9 还是不行
 57                             if (self.__currentModule) {
 58                                 S.log("standard browser get modname after load : " + mod.name);
 59                                 self.__registerModule(mod.name, self.__currentModule.def,
 60                                     self.__currentModule.config);
 61                                 self.__currentModule = null;
 62                             }
 63                             // 模块载入后，如果需要也要混入对应 global 上模块定义
 64                             mixGlobal();
 65                             if (mod.fns && mod.fns.length > 0) {
 66 
 67                             } else {
 68                                 _modError();
 69                             }
 70                         }
 71                         if (mod.status != ERROR) {
 72                             S.log(mod.name + ' is loaded.', 'info');
 73                         }
 74                         _scriptOnComplete();
 75                     },
 76                     error: function() {
 77                         _modError();
 78                         _scriptOnComplete();
 79                     },
 80                     charset: mod.charset
 81                 });
 82 
 83                 loadQueque[url] = node;
 84             }
 85             // 已经在加载中，需要添加回调到 script onload 中
 86             // 注意：没有考虑 error 情形
 87             else if (mod.status === LOADING) {
 88                 utils.scriptOnload(node, function() {
 89                     // 模块载入后，如果需要也要混入对应 global 上模块定义
 90                     mixGlobal();
 91                     _scriptOnComplete();
 92                 });
 93             }
 94             // 是内嵌代码，或者已经 loaded
 95             else {
 96                 // 也要混入对应 global 上模块定义
 97                 mixGlobal();
 98                 callback();
 99             }
100 
101             function _modError() {
102                 S.log(mod.name + ' is not loaded! can not find module in path : ' + mod['fullpath'], 'error');
103                 mod.status = ERROR;
104             }
105 
106             function mixGlobal() {
107                 // 对于动态下载下来的模块，loaded 后，global 上有可能更新 mods 信息
108                 // 需要同步到 instance 上去
109                 // 注意：要求 mod 对应的文件里，仅修改该 mod 信息
110                 if (cfg.global) {
111                     self.__mixMod(mod.name, cfg.global);
112                 }
113             }
114 
115             function _scriptOnComplete() {
116                 loadQueque[url] = LOADED;
117 
118                 if (mod.status !== ERROR) {
119 
120                     // 注意：当多个模块依赖同一个下载中的模块A下，模块A仅需 attach 一次
121                     // 因此要加上下面的 !== 判断，否则会出现重复 attach,
122                     // 比如编辑器里动态加载时，被依赖的模块会重复
123                     if (mod.status !== ATTACHED) {
124                         mod.status = LOADED;
125                     }
126 
127                     callback();
128                 }
129             }
130         }
131     });
