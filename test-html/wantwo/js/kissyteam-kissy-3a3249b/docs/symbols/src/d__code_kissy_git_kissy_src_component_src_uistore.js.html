1 /**
  2  * @fileOverview storage for component's css
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("component/uistore", function(S) {
  6     var uis = {
  7         // 不带前缀 prefixCls
  8         /*
  9          "menu" :{
 10          priority:0,
 11          ui:Menu
 12          }
 13          */
 14     };
 15 
 16     function getUIByClass(cls) {
 17         var cs = cls.split(/\s+/),p = -1,ui = null;
 18         for (var i = 0; i < cs.length; i++) {
 19             var uic = uis[cs[i]];
 20             if (uic && uic.priority > p) {
 21                 ui = uic.ui;
 22             }
 23         }
 24         return ui;
 25     }
 26 
 27     function getClsByUI(constructor) {
 28         for (var u in uis) {
 29             var ui = uis[u];
 30             if (ui.ui == constructor) {
 31                 return u;
 32             }
 33         }
 34         return 0;
 35     }
 36 
 37     function getClsByInstance(self) {
 38         return getClsByUI(self.constructor);
 39     }
 40 
 41     function setUIByClass(cls, uic) {
 42         uis[cls] = uic;
 43     }
 44 
 45 
 46     function getCls(cls) {
 47         var cs = S.trim(cls).split(/\s+/);
 48         for (var i = 0; i < cs.length; i++) {
 49             if (cs[i]) {
 50                 cs[i] = this.get("prefixCls") + cs[i];
 51             }
 52         }
 53         return cs.join(" ");
 54     }
 55 
 56     return {
 57         getCls:getCls,
 58         getClsByUI:getClsByUI,
 59         getClsByInstance:getClsByInstance,
 60         getUIByClass:getUIByClass,
 61         setUIByClass:setUIByClass,
 62         PRIORITY:{
 63             LEVEL1:10,
 64             LEVEL2:20,
 65             LEVEL3:30,
 66             LEVEL4:40,
 67             LEVEL5:50,
 68             LEVEL6:60
 69         }
 70     };
 71 });
