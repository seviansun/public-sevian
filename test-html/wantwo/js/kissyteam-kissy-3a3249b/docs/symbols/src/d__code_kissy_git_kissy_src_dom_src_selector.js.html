 25     function query_each(f) {
 26         var self = this, el, i;
 27         for (i = 0; i < self.length; i++) {
 28             el = self[i];
 29             if (f(el, i) === false) {
 30                 break;
 31             }
 32         }

 35     /**
 36      * Accepts a string containing a CSS selector which is then used to match a set of elements.
 37      * @param {String|HTMLElement[]} selector
 38      * A string containing a selector expression.
 39      * or
 40      * array of HTMLElements.
 41      * @param {String|HTMLElement[]|Document} [context] context under which to find elements matching selector.
 42      * @return {HTMLElement[]} The array of found HTMLElements
 43      */
 44     function query(selector, context) {
 45         var ret,
 46             i,
 47             simpleContext,
 48             isSelectorString = isString(selector),
 49             // optimize common usage
 50             contexts = (context === undefined && (simpleContext = 1)) ?
 51                 [doc] :
 52                 tuneContext(context);
 53         // 常见的空
 54         if (!selector) {
 55             ret = [];
 56         }
 57         // 常见的选择器
 58         // DOM.query("#x")
 59         else if (isSelectorString) {
 60             selector = trim(selector);
 61             if (contexts.length == 1 && selector) {
 62                 ret = quickFindBySelectorStr(selector, contexts[0]);
 63             }
 64         }
 65         // 不写 context，就是包装一下
 66         else if (simpleContext) {
 67             // 1.常见的单个元素
 68             // DOM.query(document.getElementById("xx"))
 69             if (selector.nodeType || selector.setTimeout) {
 70                 ret = [selector];
 71             }
 72             // 2.KISSY NodeList 特殊点直接返回，提高性能
 73             else if (selector.getDOMNodes) {
 74                 return selector;
 75             }
 76             // 3.常见的数组
 77             // var x=DOM.query(".l");
 78             // DOM.css(x,"color","red");
 79             else if (isArray(selector)) {
 80                 ret = selector;
 81             }
 82             // 4.selector.item
 83             // DOM.query(document.getElementsByTagName("a"))
 84             // note:
 85             // document.createElement("select").item 已经在 1 处理了
 86             // S.all().item 已经在 2 处理了
 87             else if (isNodeList(selector)) {
 88                 ret = S.makeArray(selector);
 89             } else {
 90                 ret = [ selector ];
 91             }
 92         }
 93 
 94         if (!ret) {
 95             ret = [];
 96             if (selector) {
 97                 for (i = 0; i < contexts.length; i++) {
 98                     push.apply(ret, queryByContexts(selector, contexts[i]));
 99                 }
100                 //必要时去重排序
101                 if (ret.length > 1 &&
102                     // multiple contexts
103                     (contexts.length > 1 ||
104                         (isSelectorString &&
105                             // multiple selector
106                             selector.indexOf(COMMA) > -1))) {
107                     unique(ret);
108                 }
109             }
110         }
111 
112         // attach each method
113         ret.each = query_each;
114 
115         return ret;
116     }

118     function queryByContexts(selector, context) {
119         var ret = [],
120             isSelectorString = isString(selector);
121         if (isSelectorString && selector.match(REG_QUERY) ||
122             !isSelectorString) {
123             // 简单选择器自己处理
124             ret = queryBySimple(selector, context);
125         }
126         // 如果选择器有 , 分开递归一部分一部分来
127         else if (isSelectorString && selector.indexOf(COMMA) > -1) {
128             ret = queryBySelectors(selector, context);
129         }
130         else {
131             // 复杂了，交给 sizzle
132             ret = queryBySizzle(selector, context);
133         }
134         return ret;
135     }

151     function queryBySelectors(selector, context) {
152         var ret = [],
153             i,
154             selectors = selector.split(/\s*,\s*/);
155         for (i = 0; i < selectors.length; i++) {
156             push.apply(ret, queryByContexts(selectors[i], context));
157         }
158         // 多部分选择器可能得到重复结果
159         return ret;
160     }

162     function quickFindBySelectorStr(selector, context) {
163         var ret, t, match, id, tag, cls;
164         // selector 为 #id 是最常见的情况，特殊优化处理
165         if (REG_ID.test(selector)) {
166             t = getElementById(selector.slice(1), context);
167             if (t) {
168                 // #id 无效时，返回空数组
169                 ret = [t];
170             } else {
171                 ret = [];
172             }
173         }
174         // selector 为支持列表中的其它 6 种
175         else {
176             match = REG_QUERY.exec(selector);
177             if (match) {
178                 // 获取匹配出的信息
179                 id = match[1];
180                 tag = match[2];
181                 cls = match[3];
182                 // 空白前只能有 id ，取出来作为 context
183                 context = (id ? getElementById(id, context) : context);
184                 if (context) {
185                     // #id .cls | #id tag.cls | .cls | tag.cls | #id.cls
186                     if (cls) {
187                         if (!id || selector.indexOf(SPACE) != -1) { // 排除 #id.cls
188                             ret = [].concat(getElementsByClassName(cls, tag, context));
189                         }
190                         // 处理 #id.cls
191                         else {
192                             t = getElementById(id, context);
193                             if (t && hasClass(t, cls)) {
194                                 ret = [t];
195                             }
196                         }
197                     }
198                     // #id tag | tag
199                     else if (tag) { // 排除空白字符串
200                         ret = getElementsByTagName(tag, context);
201                     }
202                 }
203                 ret = ret || [];
204             }
205         }
206         return ret;
207     }

209     // 最简单情况了，单个选择器部分，单个上下文
210     function queryBySimple(selector, context) {
211         var ret,
212             isSelectorString = isString(selector);
213         if (isSelectorString) {
214             ret = quickFindBySelectorStr(selector, context) || [];
215         }
216         // 传入的 selector 是 NodeList 或已是 Array
217         else if (isArray(selector) || isNodeList(selector)) {
218             // 只能包含在 context 里面
219             // filter 会转换为 nodelist 为数组
220             ret = filter(selector, function (s) {
221                 return testByContext(s, context);
222             });
223         }
224         // 传入的 selector 是 HTMLNode 查看约束
225         // 否则 window/document，原样返回
226         else if (testByContext(selector, context)) {
227             ret = [selector];
228         }
229         return ret;
230     }

232     function testByContext(element, context) {
233         if (!element) {
234             return false;
235         }
236         // 防止 element 节点还没添加到 document ，但是也可以获取到 query(element) => [element]
237         // document 的上下文一律放行
238         // context == doc 意味着没有提供第二个参数，到这里只是想单纯包装原生节点，则不检测
239         if (context == doc) {
240             return true;
241         }
242         // 节点受上下文约束
243         return DOM.contains(context, element);
244     }

246     var unique = S.noop;
247     (function () {
248         var sortOrder,
249             hasDuplicate,
250             baseHasDuplicate = true;
251 
252         // Here we check if the JavaScript engine is using some sort of
253         // optimization where it does not always call our comparision
254         // function. If that is the case, discard the hasDuplicate value.
255         // Thus far that includes Google Chrome.
256         [0, 0].sort(function () {
257             baseHasDuplicate = false;
258             return 0;
259         });
260 
261         // 排序去重
262         unique = function (elements) {
263             if (sortOrder) {
264                 hasDuplicate = baseHasDuplicate;
265                 elements.sort(sortOrder);
266 
267                 if (hasDuplicate) {
268                     var i = 1, len = elements.length;
269                     while (i < len) {
270                         if (elements[i] === elements[ i - 1 ]) {
271                             elements.splice(i, 1);
272                         } else {
273                             i++;
274                         }
275                     }
276                 }
277             }
278             return elements;
279         };
280 
281         // 貌似除了 ie 都有了...
282         if (doc.documentElement.compareDocumentPosition) {
283             sortOrder = function (a, b) {
284                 if (a == b) {
285                     hasDuplicate = true;
286                     return 0;
287                 }
288 
289                 if (!a.compareDocumentPosition || !b.compareDocumentPosition) {
290                     return a.compareDocumentPosition ? -1 : 1;
291                 }
292 
293                 return a.compareDocumentPosition(b) & 4 ? -1 : 1;
294             };
295 
296         } else {
297             sortOrder = function (a, b) {
298                 // The nodes are identical, we can exit early
299                 if (a == b) {
300                     hasDuplicate = true;
301                     return 0;
302                     // Fallback to using sourceIndex (in IE) if it's available on both nodes
303                 } else if (a.sourceIndex && b.sourceIndex) {
304                     return a.sourceIndex - b.sourceIndex;
305                 }
306             };
307         }
308     })();

316     // query #id
317     function getElementById(id, context) {
318         var doc = context,
319             el;
320         if (context.nodeType !== DOM.DOCUMENT_NODE) {
321             doc = context.ownerDocument;
322         }
323         el = doc.getElementById(id);
324         if (el && el.id === id) {
325             // optimize for common usage
326         }
327         else if (el && el.parentNode) {
328             // ie opera confuse name with id
329             // https://github.com/kissyteam/kissy/issues/67
330             // 不能直接 el.id ，否则 input shadow form attribute
331             if (!idEq(el, id)) {
332                 // 直接在 context 下的所有节点找
333                 el = DOM.filter(ANY, "#" + id, context)[0] || null;
334             }
335             // ie 特殊情况下以及指明在 context 下找了，不需要再判断
336             // 如果指定了 context node , 还要判断 id 是否处于 context 内
337             else if (!testByContext(el, context)) {
338                 el = null;
339             }
340         } else {
341             el = null;
342         }
343         return el;
344     }

424     function hasClass(el, cls) {
425         var className;
426         return (className = el.className) && (" " + className + " ").indexOf(" " + cls + " ") !== -1;

473             /**
474              * Reduce the set of matched elements to those that match the selector or pass the function's test.
475              * @param {String|HTMLElement[]|NodeList} selector Matched elements
476              * @param {String|Function} filter Selector string or filter function
477              * @param {String|HTMLElement[]|Document} [context] Context under which to find matched elements
478              * @return {HTMLElement[]}
479              */
480             filter:function (selector, filter, context) {
481                 var elems = query(selector, context),
482                     sizzle = require("sizzle"),
483                     match,
484                     tag,
485                     id,
486                     cls,
487                     ret = [];
488 
489                 // 默认仅支持最简单的 tag.cls 或 #id 形式
490                 if (isString(filter) &&
491                     (filter = trim(filter)) &&
492                     (match = REG_QUERY.exec(filter))) {
493                     id = match[1];
494                     tag = match[2];
495                     cls = match[3];
496                     if (!id) {
497                         filter = function (elem) {
498                             var tagRe = true, clsRe = true;
499 
500                             // 指定 tag 才进行判断
501                             if (tag) {
502                                 tagRe = nodeName(elem, tag);
503                             }
504 
505                             // 指定 cls 才进行判断
506                             if (cls) {
507                                 clsRe = hasClass(elem, cls);
508                             }
509 
510                             return clsRe && tagRe;
511                         }
512                     } else if (id && !tag && !cls) {
513                         filter = function (elem) {
514                             return idEq(elem, id);
515                         };
516                     }
517                 }
518 
519                 if (S.isFunction(filter)) {
520                     ret = S.filter(elems, filter);
521                 }
522                 // 其它复杂 filter, 采用外部选择器
523                 else if (filter && sizzle) {
524                     ret = sizzle.matches(filter, elems);
525                 }
526                 // filter 为空或不支持的 selector
527                 else {
528                     error(filter);
529                 }
530 
531                 return ret;
532             },
533 
534             /**
535              * Returns true if the matched element(s) pass the filter test
536              * @param {String|HTMLElement[]} selector Matched elements
537              * @param {String|Function} filter Selector string or filter function
538              * @param {String|HTMLElement[]|Document} [context] Context under which to find matched elements
539              * @returns {Boolean}
540              */
541             test:function (selector, filter, context) {
542                 var elements = query(selector, context);
543                 return elements.length && (DOM.filter(elements, filter, context).length === elements.length);
544             }
545         });
