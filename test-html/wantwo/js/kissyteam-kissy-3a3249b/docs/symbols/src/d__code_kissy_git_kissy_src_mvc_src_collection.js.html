1 /**
  2  * @fileOverview collection of models
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("mvc/collection", function (S, Event, Model, mvc, Base) {
  6 
  7     function findModelIndex(mods, mod, comparator) {
  8         var i = mods.length;
  9         if (comparator) {
 10             var k = comparator(mod);
 11             for (i = 0; i < mods.length; i++) {
 12                 var k2 = comparator(mods[i]);
 13                 if (k < k2) {
 14                     break;
 15                 }
 16             }
 17         }
 18         return i;
 19     }
 20 
 21     /**
 22      * @class
 23      * @memberOf MVC
 24      */
 25     function Collection() {
 26         Collection.superclass.constructor.apply(this, arguments);
 27     }
 28 
 29     Collection.ATTRS =
 30     /**
 31      * @lends MVC.Collection#
 32      */
 33     {
 34         model:{
 35             value:Model
 36         },
 37         models:{
 38             /*
 39              normalize model list
 40              @param models
 41              */
 42             setter:function (models) {
 43                 var prev = this.get("models");
 44                 this.remove(prev, {silent:1});
 45                 this.add(models, {silent:1});
 46                 return this.get("models");
 47             },
 48             value:[]
 49         },
 50         url:{value:S.noop},
 51         comparator:{},
 52         sync:{
 53             value:function () {
 54                 mvc.sync.apply(this, arguments);
 55             }
 56         },
 57         parse:{
 58             value:function (resp) {
 59                 return resp;
 60             }
 61         }
 62     };
 63 
 64     S.extend(Collection, Base,
 65         /**
 66          * @lends MVC.Collection#
 67          */
 68         {
 69             sort:function () {
 70                 var comparator = this.get("comparator");
 71                 if (comparator) {
 72                     this.get("models").sort(function (a, b) {
 73                         return comparator(a) - comparator(b);
 74                     });
 75                 }
 76             },
 77 
 78             toJSON:function () {
 79                 return S.map(this.get("models"), function (m) {
 80                     return m.toJSON();
 81                 });
 82             },
 83 
 84             /**
 85              *
 86              * @param model
 87              * @param {object} opts
 88              * @param {Function} opts.success
 89              * @param {Function} opts.error
 90              * @param {Function} opts.complete
 91              * @return  boolean flag
 92              *          true:add success
 93              *          false:validate error
 94              */
 95             add:function (model, opts) {
 96                 var self = this,
 97                     ret = true;
 98                 if (S.isArray(model)) {
 99                     var orig = [].concat(model);
100                     S.each(orig, function (m) {
101                         var t = self._add(m, opts);
102                         ret = ret && t;
103                     });
104                 } else {
105                     ret = self._add(model, opts);
106                 }
107                 return ret;
108             },
109 
110             /**
111              *
112              * @param model
113              * @param {object} opts
114              * @param {Function} opts.success
115              * @param {Function} opts.error
116              * @param {Function} opts.complete
117              */
118             remove:function (model, opts) {
119                 var self = this;
120                 if (S.isArray(model)) {
121                     var orig = [].concat(model);
122                     S.each(orig, function (m) {
123                         self._remove(m, opts);
124                     });
125                 } else if (model) {
126                     self._remove(model, opts);
127                 }
128             },
129 
130             at:function (i) {
131                 return this.get("models")[i];
132             },
133 
134             _normModel:function (model) {
135                 var ret = true;
136                 if (!(model instanceof Model)) {
137                     var data = model,
138                         modelConstructor = this.get("model");
139                     model = new modelConstructor();
140                     ret = model.set(data, {
141                         silent:1
142                     });
143                 }
144                 return ret && model;
145             },
146 
147             /**
148              *
149              * @param {object} opts
150              * @param {Function} opts.success
151              * @param {Function} opts.error
152              * @param {Function} opts.complete
153              */
154             load:function (opts) {
155                 var self = this;
156                 opts = opts || {};
157                 var success = opts.success;
158                 /**
159                  * @ignore
160                  */
161                 opts.success = function (resp) {
162                     if (resp) {
163                         var v = self.get("parse").call(self, resp);
164                         if (v) {
165                             self.set("models", v, opts);
166                         }
167                     }
168                     success && success.apply(this, arguments);
169                 };
170                 self.get("sync").call(self, self, 'read', opts);
171                 return self;
172             },
173 
174             /**
175              *
176              * @param model
177              * @param {object} opts
178              * @param {Function} opts.success
179              * @param {Function} opts.error
180              * @param {Function} opts.complete
181              */
182             create:function (model, opts) {
183                 var self = this;
184                 opts = opts || {};
185                 model = this._normModel(model);
186                 if (model) {
187                     model.addToCollection(self);
188                     var success = opts.success;
189                     opts.success = function () {
190                         self.add(model, opts);
191                         success && success();
192                     };
193                     model.save(opts);
194                 }
195                 return model;
196             },
197 
198             _add:function (model, opts) {
199                 model = this._normModel(model);
200                 if (model) {
201                     opts = opts || {};
202                     var index = findModelIndex(this.get("models"), model, this.get("comparator"));
203                     this.get("models").splice(index, 0, model);
204                     model.addToCollection(this);
205                     if (!opts['silent']) {
206                         this.fire("add", {
207                             model:model
208                         });
209                     }
210                 }
211                 return model;
212             },
213 
214             /**
215              * not call model.destroy ,maybe model belongs to multiple collections
216              * @private
217              */
218             _remove:function (model, opts) {
219                 opts = opts || {};
220                 var index = S.indexOf(model, this.get("models"));
221                 if (index != -1) {
222                     this.get("models").splice(index, 1);
223                     model.removeFromCollection(this);
224                 }
225                 if (!opts['silent']) {
226                     this.fire("remove", {
227                         model:model
228                     });
229                 }
230             },
231 
232             getById:function (id) {
233                 var models = this.get("models");
234                 for (var i = 0; i < models.length; i++) {
235                     var model = models[i];
236                     if (model.getId() === id) {
237                         return model;
238                     }
239                 }
240                 return null;
241             },
242 
243             getByCid:function (cid) {
244                 var models = this.get("models");
245                 for (var i = 0; i < models.length; i++) {
246                     var model = models[i];
247                     if (model.get("clientId") === cid) {
248                         return model;
249                     }
250                 }
251                 return null;
252             }
253 
254         });
255 
256     return Collection;
257 
258 }, {
259     requires:['event', './model', './base', 'base']
260 });
