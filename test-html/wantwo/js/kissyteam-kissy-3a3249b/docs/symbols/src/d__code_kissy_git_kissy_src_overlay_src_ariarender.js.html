 20     var KEY_TAB = Node.KeyCodes.TAB;
 21 
 22     function _onKey(/*Normalized Event*/ evt) {
 23 
 24         var self = this,
 25             keyCode = evt.keyCode,
 26             firstFocusItem = self.get("el");
 27         if (keyCode != KEY_TAB) {
 28             return;
 29         }
 30         // summary:
 31         // Handles the keyboard events for accessibility reasons
 32 
 33         var node = $(evt.target); // get the target node of the keypress event
 34 
 35         // find the first and last tab focusable items in the hierarchy of the dialog container node
 36         // do this every time if the items may be added / removed from the the dialog may change visibility or state
 37 
 38         var lastFocusItem = self.__ariaArchor;
 39 
 40         // assumes firstFocusItem and lastFocusItem maintained by dialog object
 41 
 42         // see if we are shift-tabbing from first focusable item on dialog
 43         if (node.equals(firstFocusItem) && evt.shiftKey) {
 44             lastFocusItem[0].focus(); // send focus to last item in dialog
 45             evt.halt(); //stop the tab keypress event
 46         }
 47         // see if we are tabbing from the last focusable item
 48         else if (node.equals(lastFocusItem) && !evt.shiftKey) {
 49             firstFocusItem[0].focus(); // send focus to first item in dialog
 50             evt.halt(); //stop the tab keypress event
 51         }
 52         else {
 53             // see if the key is for the dialog
 54             if (node.equals(firstFocusItem) ||
 55                 firstFocusItem.contains(node)) {
 56                 return;
 57             }
 58         }
 59         // this key is for the document window
 60         // allow tabbing into the dialog
 61         evt.halt();//stop the event if not a tab keypress
 62     } // end of function
 63 
 64 
 65     Aria.prototype = {
 66 
 67         __renderUI:function() {
 68             var self = this,
 69                 el = self.get("el"),
 70                 header = self.get("header");
 71             if (self.get("aria")) {
 72                 el.attr("role", "dialog");
 73                 el.attr("tabindex", 0);
 74                 if (!header.attr("id")) {
 75                     header.attr("id", S.guid("ks-dialog-header"));
 76                 }
 77                 el.attr("aria-labelledby", header.attr("id"));
 78                 // 哨兵元素，从这里 tab 出去到弹窗根节点
 79                 // 从根节点 shift tab 出去到这里
 80                 self.__ariaArchor = $("<div tabindex='0'></div>").appendTo(el);
 81             }
 82         },
 83 
 84         __bindUI:function() {
 85 
 86             var self = this;
 87             if (self.get("aria")) {
 88                 var el = self.get("el"),
 89                     lastActive;
 90                 self.on("afterVisibleChange", function(ev) {
 91                     if (ev.newVal) {
 92                         lastActive = el[0].ownerDocument.activeElement;
 93                         el[0].focus();
 94                         el.attr("aria-hidden", "false");
 95                         el.on("keydown", _onKey, self);
 96                     } else {
 97                         el.attr("aria-hidden", "true");
 98                         el.detach("keydown", _onKey, self);
 99                         lastActive && lastActive.focus();
100                     }
101                 });
102             }
103         }
104     };
105 
106     return Aria;
107 }, {
108     requires:["node"]
