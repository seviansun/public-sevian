1 /**
  2  * @fileOverview manage a list of single-select options
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menubutton/select", function (S, Node, UIBase, Component, MenuButton, Menu, Option, undefined) {
  6 
  7     function getMenuChildren(self) {
  8         // 需要初始化 menu
  9         var m = self._getMenu(1);
 10         return m && m.get("children") || [];
 11     }
 12 
 13 
 14     var Select = UIBase.create(MenuButton, {
 15 
 16             /**
 17              * @protected
 18              */
 19             __bindMenu:function () {
 20                 var self = this,
 21                     menu = self._getMenu();
 22                 Select.superclass.__bindMenu.call(self);
 23                 if (menu) {
 24                     menu.on("show", self._handleMenuShow, self);
 25                 }
 26             },
 27             /**
 28              *  different from menubutton by highlighting the currently selected option
 29              *  on open menu.
 30              */
 31             _handleMenuShow:function () {
 32                 var self = this, m = self.get("menu");
 33                 m.set("highlightedItem", self.get("selectedItem") || m.getChildAt(0));
 34             },
 35             /**
 36              * @private
 37              */
 38             _updateCaption:function () {
 39                 var self = this,
 40                     item = self.get("selectedItem");
 41                 self.set("content", item ? item.get("content") : self.get("defaultCaption"));
 42             },
 43             _handleMenuClick:function (e) {
 44                 var self = this;
 45                 self.set("selectedItem", e.target);
 46                 self.set("collapsed", true);
 47                 Select.superclass._handleMenuClick.call(self, e);
 48             },
 49 
 50             removeItems:function () {
 51                 var self = this;
 52                 Select.superclass.removeItems.apply(self, arguments);
 53                 self.set("selectedItem", null);
 54             },
 55             removeItem:function (c) {
 56                 var self = this;
 57                 Select.superclass.removeItem.apply(self, arguments);
 58                 if (c == self.get("selectedItem")) {
 59                     self.set("selectedItem", null);
 60                 }
 61             },
 62             _uiSetSelectedItem:function (v, ev) {
 63                 if (ev && ev.prevVal) {
 64                     ev.prevVal.set("selected", false);
 65                 }
 66                 this._updateCaption();
 67             },
 68             _uiSetDefaultCaption:function () {
 69                 this._updateCaption();
 70             }
 71         },
 72         {
 73             ATTRS:{
 74 
 75                 // 也是 selectedItem 的一个视图
 76                 value:{
 77                     getter:function () {
 78                         var selectedItem = this.get("selectedItem");
 79                         return selectedItem && selectedItem.get("value");
 80                     },
 81                     setter:function (v) {
 82                         var self = this,
 83                             children = getMenuChildren(self);
 84                         for (var i = 0; i < children.length; i++) {
 85                             var item = children[i];
 86                             if (item.get("value") == v) {
 87                                 self.set("selectedItem", item);
 88                                 return;
 89                             }
 90                         }
 91                         self.set("selectedItem", null);
 92                         return null;
 93                     }
 94                 },
 95 
 96 
 97                 // @inheritedDoc  from button
 98                 // content :{}
 99 
100                 selectedItem:{
101                 },
102 
103                 // 只是 selectedItem 的一个视图，无状态
104                 selectedIndex:{
105                     setter:function (index) {
106                         var self = this,
107                             children = getMenuChildren(self);
108                         if (index < 0 || index >= children.length) {
109                             // 和原生保持一致
110                             return -1;
111                         }
112                         self.set("selectedItem", children[index]);
113                     },
114 
115                     getter:function () {
116                         return S.indexOf(this.get("selectedItem"),
117                             getMenuChildren(this));
118                     }
119                 },
120 
121                 defaultCaption:{
122                     value:""
123                 }
124             }
125         }
126     );
127 
128     Select.decorate = function (element, cfg) {
129         element = S.one(element);
130         cfg = cfg || {};
131         cfg.elBefore = element;
132 
133         var name,
134             allItems = [],
135             selectedItem = null,
136             curValue = element.val(),
137             options = element.all("option");
138 
139         S.mix(cfg, {
140             menu:function () {
141                 var m = this.constructMenu();
142                 for (var i = 0; i < allItems.length; i++) {
143                     m.addChild(new Option(allItems[i]));
144                 }
145                 return m;
146             }
147         });
148 
149         options.each(function (option) {
150             var item = {
151                 content:option.text(),
152                 prefixCls:cfg.prefixCls,
153                 elCls:option.attr("class"),
154                 value:option.val()
155             };
156             if (curValue == option.val()) {
157                 selectedItem = {
158                     content:item.content,
159                     value:item.value
160                 };
161             }
162             allItems.push(item);
163         });
164 
165         var select = new Select(S.mix(cfg, selectedItem));
166         select.render();
167 
168         if (name = element.attr("name")) {
169             var input = new Node("<input type='hidden' name='" + name
170                 + "' value='" + curValue + "'>").insertBefore(element, undefined);
171 
172             select.on("afterSelectedItemChange", function (e) {
173                 if (e.newVal) {
174                     input.val(e.newVal.get("value"));
175                 } else {
176                     input.val("");
177                 }
178             });
179         }
180         element.remove();
181         return select;
182     };
183 
184     Component.UIStore.setUIByClass("select", {
185         priority:Component.UIStore.PRIORITY.LEVEL3,
186         ui:Select
187     });
188 
189 
190     if (1 > 2) {
191         Select._uiSetDefaultCaption();
192     }
193 
194     return Select;
195 
196 }, {
197     requires:['node', 'uibase', 'component', './base', 'menu', './option']
198 });
