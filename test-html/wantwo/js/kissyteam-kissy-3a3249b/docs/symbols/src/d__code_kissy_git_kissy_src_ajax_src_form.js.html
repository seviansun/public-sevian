1 /**
  2  * @fileOverview process form config
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("ajax/form", function(S, io, DOM, FormSerializer) {
  6 
  7     io.on("start", function(e) {
  8         var xhrObject = e.xhr,
  9             c = xhrObject.config;
 10         // serialize form if needed
 11         if (c.form) {
 12             var form = DOM.get(c.form),
 13                 enctype = form['encoding'] || form.enctype;
 14             // 上传有其他方法
 15             if (enctype.toLowerCase() != "multipart/form-data") {
 16                 // when get need encode
 17                 var formParam = FormSerializer.serialize(form);
 18 
 19                 if (formParam) {
 20                     if (c.hasContent) {
 21                         // post 加到 data 中
 22                         c.data = c.data || "";
 23                         if (c.data) {
 24                             c.data += "&";
 25                         }
 26                         c.data += formParam;
 27                     } else {
 28                         // get 直接加到 url
 29                         c.url += ( /\?/.test(c.url) ? "&" : "?" ) + formParam;
 30                     }
 31                 }
 32             } else {
 33                 var d = c.dataType[0];
 34                 if (d == "*") {
 35                     d = "text";
 36                 }
 37                 c.dataType.length = 2;
 38                 c.dataType[0] = "iframe";
 39                 c.dataType[1] = d;
 40             }
 41         }
 42     });
 43 
 44     return io;
 45 
 46 }, {
 47         requires:['./base',"dom","./FormSerializer"]
 48     });
