1 /**
  2  * @fileOverview represent a menu option , just make it selectable and can have select status
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menubutton/option", function(S, UIBase, Component, Menu) {
  6     var MenuItem = Menu.Item;
  7     var Option = UIBase.create(MenuItem, {}, {
  8         ATTRS:{
  9             selectable:{
 10                 value:true
 11             }
 12         }
 13     });
 14     Component.UIStore.setUIByClass("option", {
 15         priority:10,
 16         ui:Option
 17     });
 18     return Option;
 19 
 20 }, {
 21     requires:['uibase','component','menu']
 22 });
