 12     /**
 13      *
 14      * @param host
 15      * @param method
 16      * @return method if fn or host[method]
 17      */
 18     function normalFn(host, method) {
 19         if (S.isString(method)) {
 20             return host[method];
 21         }
 22         return method;
 23     }

 26     /**
 27      * fire attribute value change
 28      */
 29     function __fireAttrChange(self, when, name, prevVal, newVal, subAttrName, attrName) {
 30         attrName = attrName || name;
 31         return self.fire(when + capitalFirst(name) + 'Change', {
 32             attrName:attrName,
 33             subAttrName:subAttrName,
 34             prevVal:prevVal,
 35             newVal:newVal
 36         });
 37     }

 40     /**
 41      *
 42      * @param obj
 43      * @param name
 44      * @param [create]
 45      * @return non-empty property value of obj
 46      */
 47     function ensureNonEmpty(obj, name, create) {
 48         var ret = obj[name] || {};
 49         if (create) {
 50             obj[name] = ret;
 51         }
 52         return ret;
 53     }

 55     /**
 56      *
 57      * @param self
 58      * @return non-empty attr config holder
 59      */
 60     function getAttrs(self) {
 61         /**
 62          * attribute meta information
 63          {
 64          attrName: {
 65          getter: function,
 66          setter: function,
 67          // 注意：只能是普通对象以及系统内置类型，而不能是 new Xx()，否则用 valueFn 替代
 68          value: v, // default value
 69          valueFn: function
 70          }
 71          }
 72          */
 73         return ensureNonEmpty(self, "__attrs", true);
 74     }

 76     /**
 77      *
 78      * @param self
 79      * @return non-empty attr value holder
 80      */
 81     function getAttrVals(self) {
 82         /**
 83          * attribute value
 84          {
 85          attrName: attrVal
 86          }
 87          */
 88         return ensureNonEmpty(self, "__attrVals", true);
 89     }

 91     /**
 92      * o, [x,y,z] => o[x][y][z]
 93      * @param o
 94      * @param path
 95      */
 96     function getValueByPath(o, path) {
 97         for (var i = 0, len = path.length;
 98              o != undef && i < len;
 99              i++) {
100             o = o[path[i]];
101         }
102         return o;
103     }
104 
105     /**
106      * o, [x,y,z], val => o[x][y][z]=val
107      * @param o
108      * @param path
109      * @param val
110      */
111     function setValueByPath(o, path, val) {
112         var rlen = path.length - 1,
113             s = o;
114         if (rlen >= 0) {
115             for (var i = 0; i < rlen; i++) {
116                 o = o[path[i]];
117             }
118             if (o != undef) {
119                 o[path[i]] = val;
120             } else {
121                 s = undef;
122             }
123         }
124         return s;
125     }

127     function setInternal(self, name, value, opts, attrs) {
128         var ret;
129         opts = opts || {};
130         var dot = ".",
131             path,
132             subVal,
133             prevVal,
134             fullName = name;
135 
136         if (name.indexOf(dot) !== -1) {
137             path = name.split(dot);
138             name = path.shift();
139         }
140 
141         prevVal = self.get(name);
142 
143         if (path) {
144             subVal = getValueByPath(prevVal, path);
145         }
146 
147         // if no change, just return
148         if (!path && prevVal === value) {
149             return undef;
150         } else if (path && subVal === value) {
151             return undef;
152         }
153 
154         if (path) {
155             var tmp = S.clone(prevVal);
156             setValueByPath(tmp, path, value);
157             value = tmp;
158         }
159 
160         // check before event
161         if (!opts['silent']) {
162             if (false === __fireAttrChange(self, 'before', name, prevVal, value, fullName)) {
163                 return false;
164             }
165         }
166         // set it
167         ret = self.__set(name, value, opts);
168 
169         if (ret === false) {
170             return ret;
171         }
172 
173         // fire after event
174         if (!opts['silent']) {
175             value = getAttrVals(self)[name];
176             __fireAttrChange(self, 'after', name, prevVal, value, fullName);
177             if (!attrs) {
178                 __fireAttrChange(self,
179                     '', '*',
180                     [prevVal], [value],
181                     [fullName], [name]);
182             } else {
183                 attrs.push({
184                     prevVal:prevVal,
185                     newVal:value,
186                     attrName:name,
187                     subAttrName:fullName
188                 });
189             }
190         }
191         return self;
192     }

194     /**
195      * @class <p>
196      * Attribute provides configurable attribute support along with attribute change events. It is designed to be
197      * augmented on to a host class, and provides the host with the ability to configure attributes to store and retrieve state,
198      * along with attribute change events.
199      * </p>
200      * <p>For example, attributes added to the host can be configured:</p>
201      * <ul>
202      *     <li>With a setter function, which can be used to manipulate
203      *     values passed to Attribute's {@link Attribute#set} method, before they are stored.</li>
204      *     <li>With a getter function, which can be used to manipulate stored values,
205      *     before they are returned by Attribute's {@link Attribute#get} method.</li>
206      *     <li>With a validator function, to validate values before they are stored.</li>
207      * </ul>
208      *
209      * <p>See the {@link Attribute#addAttr} method, for the complete set of configuration
210      * options available for attributes</p>.
211      *
212      * <p><strong>NOTE:</strong> Most implementations will be better off extending the {@link Base} class,
213      * instead of augmenting Attribute directly. Base augments Attribute and will handle the initial configuration
214      * of attributes for derived classes, accounting for values passed into the constructor.</p>
215      * @name Attribute
216      */
217     function Attribute() {
218     }

220     S.augment(Attribute,
221         /**
222          * @lends Attribute.prototype
223          */
224         {
225 
226             /**
227              * @return un-cloned attr config collections
228              */
229             getAttrs:function () {
230                 return getAttrs(this);
231             },
232 
233             /**
234              * @return un-cloned attr value collections
235              */
236             getAttrVals:function () {
237                 var self = this,
238                     o = {},
239                     a,
240                     attrs = getAttrs(self);
241                 for (a in attrs) {
242                     o[a] = self.get(a);
243                 }
244                 return o;
245             },
246 
247             /**
248              * Adds an attribute with the provided configuration to the host object.
249              * @param {String} name attrName
250              * @param {Object} attrConfig The config supports the following properties
251              * @param [attrConfig.value] simple object or system native object
252              * @param [attrConfig.valueFn] a function which can return current attribute's default value
253              * @param {Function(*)} [attrConfig.setter] call when set attribute's value
254              *                                          pass current attribute's value as parameter
255              *                                          if return value is not undefined,set returned value as real value
256              * @param {Function(*)} [attrConfig.getter] call when get attribute's value
257              *                                          pass current attribute's value as parameter
258              *                                          return getter's returned value to invoker
259              * @param {Function(*)} [attrConfig.validator]  call before set attribute's value
260              *                                              if return false,cancel this set action
261              * @param {boolean} [override] whether override existing attribute config ,default true
262              */
263             addAttr:function (name, attrConfig, override) {
264                 var self = this,
265                     attrs = getAttrs(self),
266                     cfg = S.clone(attrConfig);
267                 if (!attrs[name]) {
268                     attrs[name] = cfg;
269                 } else {
270                     S.mix(attrs[name], cfg, override);
271                 }
272                 return self;
273             },
274 
275             /**
276              * Configures a group of attributes, and sets initial values.
277              * @param {Object} attrConfigs  An object with attribute name/configuration pairs.
278              * @param {Object} initialValues user defined initial values
279              */
280             addAttrs:function (attrConfigs, initialValues) {
281                 var self = this;
282                 S.each(attrConfigs, function (attrConfig, name) {
283                     self.addAttr(name, attrConfig);
284                 });
285                 if (initialValues) {
286                     self.set(initialValues);
287                 }
288                 return self;
289             },
290 
291             /**
292              * Checks if the given attribute has been added to the host.
293              */
294             hasAttr:function (name) {
295                 return name && getAttrs(this).hasOwnProperty(name);
296             },
297 
298             /**
299              * Removes an attribute from the host object.
300              */
301             removeAttr:function (name) {
302                 var self = this;
303 
304                 if (self.hasAttr(name)) {
305                     delete getAttrs(self)[name];
306                     delete getAttrVals(self)[name];
307                 }
308 
309                 return self;
310             },

313             /**
314              * Sets the value of an attribute.
315              * @param {String|Object} name attribute's name or attribute name and value map
316              * @param [value] attribute's value
317              * @param {Object} [opts] some options
318              * @param {boolean} [opts.silent] whether fire change event
319              * @returns {boolean} whether pass validator
320              */
321             set:function (name, value, opts) {
322                 var self = this;
323                 if (S.isPlainObject(name)) {
324                     opts = value;
325                     var all = Object(name),
326                         attrs = [],
327                         e,
328                         errors = [];
329                     for (name in all) {
330                         // bulk validation
331                         // if any one failed,all values are not set
332                         if ((e = validate(self, name, all[name], all)) !== undef) {
333                             errors.push(e);
334                         }
335                     }
336                     if (errors.length) {
337                         if (opts.error) {
338                             opts.error(errors);
339                         }
340                         return false;
341                     }
342                     for (name in all) {
343                         setInternal(self, name, all[name], opts, attrs);
344                     }
345                     var attrNames = [],
346                         prevVals = [],
347                         newVals = [],
348                         subAttrNames = [];
349                     S.each(attrs, function (attr) {
350                         prevVals.push(attr.prevVal);
351                         newVals.push(attr.newVal);
352                         attrNames.push(attr.attrName);
353                         subAttrNames.push(attr.subAttrName);
354                     });
355                     if (attrNames.length) {
356                         __fireAttrChange(self,
357                             '',
358                             '*',
359                             prevVals,
360                             newVals,
361                             subAttrNames,
362                             attrNames);
363                     }
364                     return self;
365                 }
366                 return setInternal(self, name, value, opts);
367             },
368 
369             /**
370              * internal use, no event involved, just set.
371              * @protected overriden by mvc/model
372              */
373             __set:function (name, value, opts) {
374                 var self = this,
375                     setValue,
376                     // if host does not have meta info corresponding to (name,value)
377                     // then register on demand in order to collect all data meta info
378                     // 一定要注册属性元数据，否则其他模块通过 _attrs 不能枚举到所有有效属性
379                     // 因为属性在声明注册前可以直接设置值
380                     e,
381                     attrConfig = ensureNonEmpty(getAttrs(self), name, true),
382                     setter = attrConfig['setter'];
383 
384                 // validator check
385                 e = validate(self, name, value);
386 
387                 if (e !== undef) {
388                     if (opts.error) {
389                         opts.error(e);
390                     }
391                     return false;
392                 }
393 
394                 // if setter has effect
395                 if (setter && (setter = normalFn(self, setter))) {
396                     setValue = setter.call(self, value, name);
397                 }
398 
399                 if (setValue === INVALID) {
400                     return false;
401                 }
402 
403                 if (setValue !== undef) {
404                     value = setValue;
405                 }
406 
407                 // finally set
408                 getAttrVals(self)[name] = value;
409             },
410 
411             /**
412              * Gets the current value of the attribute.
413              * @param {String} name attribute's name
414              */
415             get:function (name) {
416                 var self = this,
417                     dot = ".",
418                     path,
419                     attrVals = getAttrVals(self),
420                     attrConfig,
421                     getter, ret;
422 
423                 if (name.indexOf(dot) !== -1) {
424                     path = name.split(dot);
425                     name = path.shift();
426                 }
427 
428                 attrConfig = ensureNonEmpty(getAttrs(self), name);
429                 getter = attrConfig['getter'];
430 
431                 // get user-set value or default value
432                 //user-set value takes privilege
433                 ret = name in attrVals ?
434                     attrVals[name] :
435                     self.__getDefAttrVal(name);
436 
437                 // invoke getter for this attribute
438                 if (getter && (getter = normalFn(self, getter))) {
439                     ret = getter.call(self, ret, name);
440                 }
441 
442                 if (path) {
443                     ret = getValueByPath(ret, path);
444                 }
445 
446                 return ret;
447             },
448 
449             /**
450              * get default attribute value from valueFn/value
451              * @private
452              * @param name
453              */
454             __getDefAttrVal:function (name) {
455                 var self = this,
456                     attrs = getAttrs(self),
457                     attrConfig = ensureNonEmpty(attrs, name),
458                     valFn = attrConfig.valueFn,
459                     val;
460 
461                 if (valFn && (valFn = normalFn(self, valFn))) {
462                     val = valFn.call(self);
463                     if (val !== undef) {
464                         attrConfig.value = val;
465                     }
466                     delete attrConfig.valueFn;
467                     attrs[name] = attrConfig;
468                 }
469 
470                 return attrConfig.value;
471             },
472 
473             /**
474              * Resets the value of an attribute.just reset what addAttr set  (not what invoker set when call new Xx(cfg))
475              * @param {String} name name of attribute
476              * @param {Object} [opts] some options
477              * @param {boolean} [opts.silent] whether fire change event
478              */
479             reset:function (name, opts) {
480                 var self = this;
481 
482                 if (S.isString(name)) {
483                     if (self.hasAttr(name)) {
484                         // if attribute does not have default value, then set to undefined.
485                         return self.set(name, self.__getDefAttrVal(name), opts);
486                     }
487                     else {
488                         return self;
489                     }
490                 }
491 
492                 opts = name;
493 
494                 var attrs = getAttrs(self),
495                     values = {};
496 
497                 // reset all
498                 for (name in attrs) {
499                     values[name] = self.__getDefAttrVal(name);
500                 }
501 
502                 self.set(values, opts);
503                 return self;
504             }
505         });
506 
507     function capitalFirst(s) {
508         return s.charAt(0).toUpperCase() + s.substring(1);
509     }
510 
511     function validate(self, name, value, all) {
512         var attrConfig = ensureNonEmpty(getAttrs(self), name, true),
513             e,
514             validator = attrConfig['validator'];
515         if (validator && (validator = normalFn(self, validator))) {
516             e = validator.call(self, value, name, all);
517             // undefined and true validate successfully
518             if (e !== undef && e !== true) {
519                 return e;
520             }
521         }
522         return undef;
523     }
524 
525     if (undef) {
526         Attribute.prototype.addAttrs = undef;
527     }
528     return Attribute;
