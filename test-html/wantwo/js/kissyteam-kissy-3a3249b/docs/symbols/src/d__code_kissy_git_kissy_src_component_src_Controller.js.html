  8     function wrapperViewSetter(attrName) {
  9         return function (ev) {
 10             var value = ev.newVal,
 11                 self = this,
 12                 view = self.get("view");
 13             view && view.set(attrName, value);
 14         };
 15     }
 16 
 17     function wrapperViewGetter(attrName) {
 18         return function (v) {
 19             var self = this,
 20                 view = self.get("view");
 21             return v === undefined ? view && view.get(attrName) : v;
 22         };
 23     }

 26     function initChild(self, c, elBefore) {
 27         // If this (parent) component doesn't have a DOM yet, call createDom now
 28         // to make sure we render the child component's element into the correct
 29         // parent element (otherwise render_ with a null first argument would
 30         // render the child into the document body, which is almost certainly not
 31         // what we want).
 32         self.create();
 33         var contentEl = self.getContentElement();
 34         c.__set("parent", self);
 35         // set 通知 view 也更新对应属性
 36         c.set("render", contentEl);
 37         c.set("elBefore", elBefore);
 38         // 如果 parent 已经渲染好了子组件也要立即渲染，就 创建 dom ，绑定事件
 39         if (self.get("rendered")) {
 40             c.render();
 41         }
 42         // 如果 parent 也没渲染，子组件 create 出来和 parent 节点关联
 43         // 子组件和 parent 组件一起渲染
 44         else {
 45             // 之前设好属性，view ，logic 同步还没 bind ,create 不是 render ，还没有 bindUI
 46             c.create();
 47             // 设置好，render 时插入到对应位置，这里不需要了
 48             // contentEl[0].insertBefore(c.get("el")[0], elBefore && elBefore[0] || null);
 49         }
 50     }

 52     /**
 53      * 不使用 valueFn
 54      * 只有 render 时需要找到默认，其他时候不需要，防止莫名其妙初始化
 55      */
 56     function getDefaultView() {
 57         // 逐层找默认渲染器
 58         var self = this,
 59             c = self.constructor,
 60             DefaultRender;
 61 
 62         while (c && !DefaultRender) {
 63             DefaultRender = c['DefaultRender'];
 64             c = c.superclass && c.superclass.constructor;
 65         }
 66         if (DefaultRender) {
 67             /**
 68              * 将渲染层初始化所需要的属性，直接构造器设置过去
 69              */
 70             var attrs = self['__attrs'] || {},
 71                 cfg = {};
 72             for (var attrName in attrs) {
 73                 if (attrs.hasOwnProperty(attrName)) {
 74                     var attrCfg = attrs[attrName], v;
 75                     if (attrCfg.view) {
 76                         // 只设置用户设置的值
 77                         // 考虑 c 上的默认值
 78                         if (( v = self.get(attrName) ) !== undefined) {
 79                             cfg[attrName] = v;
 80                         }
 81                     }
 82                 }
 83             }
 84             return new DefaultRender(cfg);
 85         }
 86         return 0;
 87     }

 89     function getClsByHierarchy(self) {
 90         if (self.__componentClasses) {
 91             return self.__componentClasses;
 92         }
 93         var constructor = self.constructor, re = [];
 94         while (constructor && constructor != Controller) {
 95             var cls = UIStore.getClsByUI(constructor);
 96             if (cls) {
 97                 re.push(cls);
 98             }
 99             constructor = constructor.superclass && constructor.superclass.constructor;
100         }
101         return self.__componentClasses = re.join(" ");
102     }

147             /**
148              * control 层的渲染 ui 就是 render view
149              * finally，不能被 override

151             renderUI:function () {
152                 var self = this, i, child;
153                 self.get("view").render();
154                 //then render my children
155                 var children = self.get("children");
156                 for (i = 0; i < children.length; i++) {
157                     child = children[i];
158                     // 不在 Base 初始化设置属性时运行，防止和其他初始化属性冲突
159                     initChild(self, child);
160                     child.render();
161                 }

164             /**
165              * 控制层的 createDom 实际上就是调用 view 层的 create 来创建真正的节点
166              */

168                 var self = this;
169                 var view = self.get("view") || getDefaultView.call(self);
170                 view.create();
171                 view._renderCls(getClsByHierarchy(self));
172                 if (!self.get("allowTextSelection_")) {
173                     view.get("el").unselectable();
174                 }
175                 self.__set("view", view);

236             _uiSetHandleMouseEvents:function (v) {
237                 var self = this,
238                     el = self.get("el");

247                     el.detach("mouseleave", self._handleMouseLeave, self);
248                     el.detach("mousedown", self._handleMouseDown, self);

261             _isMouseEventWithinElement:function (e, elem) {
262                 var relatedTarget = e.relatedTarget;
263                 if (!relatedTarget) {
264                     return false;
265                 }
266                 // 在里面或等于自身都不算 mouseenter/leave
267                 if (relatedTarget === elem[0]
268                     || elem.contains(relatedTarget)) {
269                     return true;
270                 }
271             },

273             _handleMouseOver:function (e) {
274                 var self = this,
275                     el = self.get("el");
276                 if (self.get("disabled")) {
277                     return true;
278                 }
279                 if (!self._isMouseEventWithinElement(e, el)) {
280                     self._handleMouseEnter(e);
281                 }

287                     el = self.get("el");
288                 if (self.get("disabled")) {

292                     self._handleMouseLeave(e);

326             /**
327              * root element handler for mouse down
328              * @param ev
329              */
330             _handleMouseDown:function (ev) {
331                 var self = this, el;
332                 if (self.get("disabled")) {
333                     return true;
334                 }
335                 if (ev.which == 1 && self.get("activeable")) {
336                     self.set("active", true);
337                 }
338                 el = self.getKeyEventTarget();
339                 // 左键，否则 unselectable 在 ie 下鼠标点击获得不到焦点
340                 if (ev.which == 1 && el.attr("tabindex") >= 0) {
341                     el[0].focus();
342                 }
343                 // Cancel the default action unless the control
344                 // allows text selection.
345                 if (ev.which == 1 && !self.get("allowTextSelection_")) {
346                     // firefox/chrome 不会引起焦点转移
347                     var n = ev.target.nodeName;
348                     n = n && n.toLowerCase();
349                     // do not prevent focus when click on editable element
350                     if (n != "input" && n != "textarea") {
351                         ev.preventDefault();
352                     }
353                 }
354             },

356             /**

359             _uiSetFocusable:function (v) {

361                     el = self.getKeyEventTarget();
362                 if (v) {

364                     el.on("blur", self._handleBlur, self);
365                     el.on("keydown", self._handleKeydown, self);
366                 } else {
367                     el.detach("focus", self._handleFocus, self);
368                     el.detach("blur", self._handleBlur, self);
369                     el.detach("keydown", self._handleKeydown, self);
370                 }
