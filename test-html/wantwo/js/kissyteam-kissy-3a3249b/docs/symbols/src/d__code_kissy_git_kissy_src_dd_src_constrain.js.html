1 KISSY.add("dd/constrain", function (S, Base, Node) {
  2 
  3     var $ = Node.all, WIN = S.Env.host;
  4 
  5     /**
  6      * @class Provide ability to constrain draggable to specified region
  7      * @memberOf DD
  8      */
  9     function Constrain() {
 10         Constrain.superclass.constructor.apply(this, arguments);
 11     }
 12 
 13     function onDragStart(e) {
 14         var self = this,
 15             drag = e.drag,
 16             l, t, lt,
 17             dragNode = drag.get("dragNode"),
 18             constrain = self.get("constrain");
 19         if (constrain) {
 20             if (constrain === true) {
 21                 var win = $(WIN);
 22                 self.__constrainRegion = {
 23                     left:l = win.scrollLeft(),
 24                     top:t = win.scrollTop(),
 25                     right:l + win.width(),
 26                     bottom:t + win.height()
 27                 };
 28             }
 29             if (constrain.nodeType || S.isString(constrain)) {
 30                 constrain = $(constrain);
 31             }
 32             if (constrain.getDOMNode) {
 33                 lt = constrain.offset();
 34                 self.__constrainRegion = {
 35                     left:lt.left,
 36                     top:lt.top,
 37                     right:lt.left + constrain.outerWidth(),
 38                     bottom:lt.top + constrain.outerHeight()
 39                 };
 40             } else if (S.isPlainObject(constrain)) {
 41                 self.__constrainRegion = constrain;
 42             }
 43             if (self.__constrainRegion) {
 44                 self.__constrainRegion.right -= dragNode.outerWidth();
 45                 self.__constrainRegion.bottom -= dragNode.outerHeight();
 46             }
 47         }
 48     }
 49 
 50     function onDragAlign(e) {
 51         var self = this,
 52             info = e.info,
 53             l = info.left,
 54             t = info.top,
 55             constrain = self.__constrainRegion;
 56         if (constrain) {
 57             info.left = Math.min(Math.max(constrain.left, l), constrain.right);
 58             info.top = Math.min(Math.max(constrain.top, t), constrain.bottom);
 59         }
 60     }
 61 
 62     function onDragEnd() {
 63         this.__constrainRegion = null;
 64     }
 65 
 66     S.extend(Constrain, Base,
 67         /**
 68          * @lends DD.Constrain#
 69          */
 70         {
 71             __constrainRegion:null,
 72 
 73             /**
 74              * start monitoring drag
 75              * @param {DD.Draggable} drag
 76              */
 77             attach:function (drag) {
 78                 var self = this;
 79                 drag.on("dragstart", onDragStart, self)
 80                     .on("dragend", onDragEnd, self)
 81                     .on("dragalign", onDragAlign, self);
 82             },
 83 
 84 
 85             /**
 86              * stop monitoring drag
 87              * @param {DD.Draggable} drag
 88              */
 89             unAttach:function (drag) {
 90                 var self = this;
 91                 drag.detach("dragstart", onDragStart, self)
 92                     .detach("dragend", onDragEnd, self)
 93                     .detach("dragalign", onDragAlign, self);
 94             }
 95         }, {
 96             ATTRS:/**
 97              * @lends DD.Constrain#
 98              */
 99             {
100                 /**
101                  * constrained container
102                  * @type {boolean|HTMLElement|String}
103                  */
104                 constrain:{
105                     value:true
106                 }
107             }
108         });
109 
110     return Constrain;
111 }, {
112     requires:['base', 'node']
113 });
