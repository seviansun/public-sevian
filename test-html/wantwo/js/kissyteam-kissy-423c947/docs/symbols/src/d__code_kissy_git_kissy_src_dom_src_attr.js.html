253                  * Get the value of a property for the first element in the set of matched elements.
254                  * or
255                  * Set one or more properties for the set of matched elements.
256                  * @param {Array<HTMLElement>|String} selector matched elements
257                  * @param {String|Object} name

332                 /**
333                  * Get the value of an attribute for the first element in the set of matched elements.
334                  * or
335                  * Set one or more attributes for the set of matched elements.
336                  * @param {HTMLElement[]|HTMLElement|String|Element} selector matched elements
337                  * @param {String|Object} name The name of the attribute to set. or A map of attribute-value pairs to set.
338                  * @param [val] A value to set for the attribute.
339                  * @returns {String}
340                  */
341                 attr:function (selector, name, val, pass) {
342                     /*
343                      Hazards From Caja Note:
344 
345                      - In IE[67], el.setAttribute doesn't work for attributes like
346                      'class' or 'for'.  IE[67] expects you to set 'className' or
347                      'htmlFor'.  Caja use setAttributeNode solves this problem.
348 
349                      - In IE[67], <input> elements can shadow attributes.  If el is a
350                      form that contains an <input> named x, then el.setAttribute(x, y)
351                      will set x's value rather than setting el's attribute.  Using
352                      setAttributeNode solves this problem.
353 
354                      - In IE[67], the style attribute can only be modified by setting
355                      el.style.cssText.  Neither setAttribute nor setAttributeNode will
356                      work.  el.style.cssText isn't bullet-proof, since it can be
357                      shadowed by <input> elements.
358 
359                      - In IE[67], you can never change the type of an <button> element.
360                      setAttribute('type') silently fails, but setAttributeNode
361                      throws an exception.  caja : the silent failure. KISSY throws error.
362 
363                      - In IE[67], you can never change the type of an <input> element.
364                      setAttribute('type') throws an exception.  We want the exception.
365 
366                      - In IE[67], setAttribute is case-sensitive, unless you pass 0 as a
367                      3rd argument.  setAttributeNode is case-insensitive.
368 
369                      - Trying to set an invalid name like ":" is supposed to throw an
370                      error.  In IE[678] and Opera 10, it fails without an error.
371                      */
372 
373                     var els = DOM.query(selector);
374 
375                     // supports hash
376                     if (S.isPlainObject(name)) {
377                         pass = val;
378                         for (var k in name) {
379                             if (name.hasOwnProperty(k)) {
380                                 DOM.attr(els, k, name[k], pass);
381                             }
382                         }
383                         return undefined;
384                     }
385 
386                     if (!(name = S.trim(name))) {
387                         return undefined;
388                     }
389 
390                     // attr functions
391                     if (pass && attrFn[name]) {
392                         return DOM[name](selector, val);
393                     }
394 
395                     // scrollLeft
396                     name = name.toLowerCase();
397 
398                     if (pass && attrFn[name]) {
399                         return DOM[name](selector, val);
400                     }
401 
402                     // custom attrs
403                     name = attrFix[name] || name;
404 
405                     var attrNormalizer,
406                         el = els[0],
407                         ret;
408 
409                     if (rboolean.test(name)) {
410                         attrNormalizer = boolHook;
411                     }
412                     // only old ie?
413                     else if (rinvalidChar.test(name)) {
414                         attrNormalizer = attrNodeHook;
415                     } else {
416                         attrNormalizer = attrHooks[name];
417                     }
418 
419 
420                     if (val === undefined) {
421                         if (el && el.nodeType === DOM.ELEMENT_NODE) {
422                             // browsers index elements by id/name on forms, give priority to attributes.
423                             if (nodeName(el, "form")) {
424                                 attrNormalizer = attrNodeHook;
425                             }
426                             if (attrNormalizer && attrNormalizer.get) {
427                                 return attrNormalizer.get(el, name);
428                             }
429 
430                             ret = el.getAttribute(name);
431 
432                             // standard browser non-existing attribute return null
433                             // ie<8 will return undefined , because it return property
434                             // so norm to undefined
435                             return ret === null ? undefined : ret;
436                         }
437                     } else {
438                         for (var i = els.length - 1; i >= 0; i--) {
439                             el = els[i];
440                             if (el && el.nodeType === DOM.ELEMENT_NODE) {
441                                 if (nodeName(el, "form")) {
442                                     attrNormalizer = attrNodeHook;
443                                 }
444                                 if (attrNormalizer && attrNormalizer.set) {
445                                     attrNormalizer.set(el, val, name);
446                                 } else {
447                                     // convert the value to a string (all browsers do this but IE)
448                                     el.setAttribute(name, EMPTY + val);
449                                 }
450                             }
451                         }
452                     }
453                     return undefined;
454                 },

514                 /**
515                  * Get the current value of the first element in the set of matched elements.
516                  * or
517                  * Set the value of each element in the set of matched elements.
518                  * @param {Array<HTMLElement>|String} selector matched elements
519                  * @param {String|Array<String>} [value] A string of text or an array of strings corresponding to the value of each matched element to set as selected/checked.
520                  * @returns {undefined|String|Array<String>|Number}
521                  */
522                 val:function (selector, value) {
523                     var hook, ret;
524 
525                     //getter
526                     if (value === undefined) {
527 
528                         var elem = DOM.get(selector);
529 
530                         if (elem) {
531                             hook = valHooks[ elem.nodeName.toLowerCase() ] || valHooks[ elem.type ];
532 
533                             if (hook && "get" in hook && (ret = hook.get(elem, "value")) !== undefined) {
534                                 return ret;
535                             }
536 
537                             ret = elem.value;
538 
539                             return typeof ret === "string" ?
540                                 // handle most common string cases
541                                 ret.replace(rreturn, "") :
542                                 // handle cases where value is null/undefined or number
543                                 ret == null ? "" : ret;
544                         }
545 
546                         return undefined;
547                     }
548 
549                     var els = DOM.query(selector), i;
550                     for (i = els.length - 1; i >= 0; i--) {
551                         elem = els[i];
552                         if (elem.nodeType !== 1) {
553                             return undefined;
554                         }
555 
556                         var val = value;
557 
558                         // Treat null/undefined as ""; convert numbers to string
559                         if (val == null) {
560                             val = "";
561                         } else if (typeof val === "number") {
562                             val += "";
563                         } else if (S.isArray(val)) {
564                             val = S.map(val, function (value) {
565                                 return value == null ? "" : value + "";
566                             });
567                         }
568 
569                         hook = valHooks[ elem.nodeName.toLowerCase() ] || valHooks[ elem.type ];
570 
571                         // If set returns undefined, fall back to normal setting
572                         if (!hook || !("set" in hook) || hook.set(elem, val, "value") === undefined) {
573                             elem.value = val;
574                         }
575                     }
576                     return undefined;
577                 },

584                  * Set the content of each element in the set of matched elements to the specified text.
585                  * @param {HTMLElement[]|HTMLElement|String} selector matched elements

592                         // supports css selector/Node/NodeList
593                         var el = DOM.get(selector);
594 
595                         // only gets value on supported nodes
596                         if (isElementNode(el)) {
597                             return el[TEXT] || EMPTY;
598                         }
599                         else if (isTextNode(el)) {
600                             return el.nodeValue;
601                         }
602                         return undefined;
603                     }
604                     // setter
605                     else {
606                         var els = DOM.query(selector), i;
607                         for (i = els.length - 1; i >= 0; i--) {
608                             el = els[i];
609                             if (isElementNode(el)) {
610                                 el[TEXT] = val;
611                             }
612                             else if (isTextNode(el)) {
613                                 el.nodeValue = val;
614                             }
615                         }
616                     }
617                     return undefined;
618                 }
