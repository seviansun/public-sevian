1 /**
  2  * @fileOverview non-refresh upload file with form by iframe
  3  * @author  yiminghe@gmail.com
  4  */
  5 KISSY.add("ajax/IframeTransport", function (S, DOM, Event, io) {
  6 
  7     var doc = S.Env.host.document,
  8         OK_CODE = 200,
  9         ERROR_CODE = 500,
 10         BREATH_INTERVAL = 30;
 11 
 12     // iframe 内的内容就是 body.innerText
 13     io.setupConfig({
 14         converters:{
 15             // iframe 到其他类型的转化和 text 一样
 16             iframe:io.getConfig().converters.text,
 17             text:{
 18                 iframe:function (text) {
 19                     return text;
 20                 }
 21             }
 22         }
 23     });
 24 
 25     function createIframe(xhr) {
 26         var id = S.guid("ajax-iframe");
 27         xhr.iframe = DOM.create("<iframe " +
 28             " id='" + id + "'" +
 29             // need name for target of form
 30             " name='" + id + "'" +
 31             " style='position:absolute;left:-9999px;top:-9999px;'/>");
 32         xhr.iframeId = id;
 33         DOM.prepend(xhr.iframe, doc.body || doc.documentElement);
 34     }
 35 
 36     function addDataToForm(data, form, serializeArray) {
 37         data = S.unparam(data);
 38         var ret = [];
 39         for (var d in data) {
 40             var isArray = S.isArray(data[d]),
 41                 vs = S.makeArray(data[d]);
 42             // 数组和原生一样对待，创建多个同名输入域
 43             for (var i = 0; i < vs.length; i++) {
 44                 var e = doc.createElement("input");
 45                 e.type = 'hidden';
 46                 e.name = d + (isArray && serializeArray ? "[]" : "");
 47                 e.value = vs[i];
 48                 DOM.append(e, form);
 49                 ret.push(e);
 50             }
 51         }
 52         return ret;
 53     }
 54 
 55 
 56     function removeFieldsFromData(fields) {
 57         DOM.remove(fields);
 58     }
 59 
 60     function IframeTransport(xhrObject) {
 61         this.xhrObject = xhrObject;
 62     }
 63 
 64     S.augment(IframeTransport, {
 65         send:function () {
 66 
 67             var self = this,
 68                 xhrObject = self.xhrObject,
 69                 c = xhrObject.config,
 70                 fields,
 71                 form = DOM.get(c.form);
 72 
 73             self.attrs = {
 74                 target:DOM.attr(form, "target") || "",
 75                 action:DOM.attr(form, "action") || ""
 76             };
 77             self.form = form;
 78 
 79             createIframe(xhrObject);
 80 
 81             // set target to iframe to avoid main page refresh
 82             DOM.attr(form, {
 83                 target:xhrObject.iframeId,
 84                 action:c.url
 85             });
 86 
 87             if (c.data) {
 88                 fields = addDataToForm(c.data, form, c.serializeArray);
 89             }
 90 
 91             self.fields = fields;
 92 
 93             var iframe = xhrObject.iframe;
 94 
 95             Event.on(iframe, "load error", self._callback, self);
 96 
 97             form.submit();
 98 
 99         },
100 
101         _callback:function (event, abort) {
102             //debugger
103             var self=this,
104                 form = self.form,
105                 xhrObject = self.xhrObject,
106                 eventType = event.type,
107                 iframe = xhrObject.iframe;
108 
109             // 防止重复调用 , 成功后 abort
110             if (!iframe) {
111                 return;
112             }
113 
114             DOM.attr(form, self.attrs);
115 
116             if (eventType == "load") {
117                 var iframeDoc = iframe.contentWindow.document;
118                 xhrObject.responseXML = iframeDoc;
119                 xhrObject.responseText = DOM.text(iframeDoc.body);
120                 xhrObject._xhrReady(OK_CODE, "success");
121             } else if (eventType == 'error') {
122                 xhrObject._xhrReady(ERROR_CODE, "error");
123             }
124 
125             removeFieldsFromData(this.fields);
126 
127             Event.detach(iframe);
128 
129             setTimeout(function () {
130                 // firefox will keep loading if not settimeout
131                 DOM.remove(iframe);
132             }, BREATH_INTERVAL);
133 
134             // nullify to prevent memory leak?
135             xhrObject.iframe = null;
136         },
137 
138         abort:function () {
139             this._callback({}, 1);
140         }
141     });
142 
143     io.setupTransport("iframe", IframeTransport);
144 
145     return io;
146 
147 }, {
148     requires:["dom", "event", "./base"]
149 });
