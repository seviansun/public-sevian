1 /**
  2  * @fileOverview delegate events for children
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("component/delegatechildren", function (S) {
  6 
  7     /**
  8      * @name Component.DelegateChildren
  9      */
 10     function DelegateChildren() {
 11 
 12     }
 13 
 14     S.augment(DelegateChildren, {
 15         __bindUI:function () {
 16             var self = this;
 17             self.get("el").on("mousedown mouseup mouseover mouseout dblclick",
 18                 self._handleChildMouseEvents, self);
 19         },
 20 
 21         _handleChildMouseEvents:function (e) {
 22             var control = this.getOwnerControl(e.target);
 23             if (control) {
 24                 // Child control identified; forward the event.
 25                 switch (e.type) {
 26                     case mousedown:
 27                         control._handleMouseDown(e);
 28                         break;
 29                     case mouseup:
 30                         control._handleMouseUp(e);
 31                         break;
 32                     case mouseover:
 33                         control._handleMouseOver(e);
 34                         break;
 35                     case mouseout:
 36                         control._handleMouseOut(e);
 37                         break;
 38                     case dblclick:
 39                         control._handleDblClick(e);
 40                         break;
 41                     default:
 42                         S.error(e.type + " unhandled!");
 43                 }
 44             }
 45         },
 46 
 47         getOwnerControl:function (node) {
 48             var self = this,
 49                 children = self.get("children"),
 50                 len = children.length,
 51                 elem = this.get("el")[0];
 52             while (node && node !== elem) {
 53                 for (var i = 0; i < len; i++) {
 54                     if (children[i].get("el")[0] === node) {
 55                         return children[i];
 56                     }
 57                 }
 58                 node = node.parentNode;
 59             }
 60             return null;
 61         }
 62     });
 63 
 64     return DelegateChildren;
 65 });
