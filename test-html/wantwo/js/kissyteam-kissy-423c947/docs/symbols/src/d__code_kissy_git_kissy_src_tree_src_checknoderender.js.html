1 /**
  2  * @fileOverview check node render
  3  * @author yiminghe@gmail.com

  5 KISSY.add("tree/checknoderender", function(S, Node, UIBase, Component, BaseNodeRender) {
  6     var $ = Node.all,
  7         ICON_CLS = "tree-icon",
  8         CHECK_CLS = "tree-item-check",
  9         ALL_STATES_CLS = "tree-item-checked0 tree-item-checked1 tree-item-checked2",
 10         INLINE_BLOCK = "inline-block";
 11     return UIBase.create(BaseNodeRender, {
 12 
 13         createDom:function() {
 14             var self = this;
 15             var expandIconEl = self.get("expandIconEl"),
 16                 checkEl = $("<div class='" + self.getCls(INLINE_BLOCK + " " + " "
 17                     + ICON_CLS) + "'/>").insertAfter(expandIconEl);
 18             self.__set("checkEl", checkEl);
 19         },
 20 
 21         _uiSetCheckState:function(s) {
 22             var self = this;
 23             var checkEl = self.get("checkEl");
 24             checkEl.removeClass(self.getCls(ALL_STATES_CLS))
 25                 .addClass(self.getCls(CHECK_CLS + "ed" + s));
 26         }
 27 
 28     }, {
 29         ATTRS:{
 30             checkEl:{},
 31             checkState:{}
 32         },
 33 
 34         CHECK_CLS:CHECK_CLS
 35     });
 36 }, {
 37     requires:['node','uibase','component','./basenoderender']
 38 });
