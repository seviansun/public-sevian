1 /**
  2  * @fileOverview decorate function for children render from markup
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("component/decoratechildren", function(S, UIStore) {
  6     function DecorateChildren() {
  7 
  8     }
  9 
 10     S.augment(DecorateChildren, {
 11         decorateInternal:function(el) {
 12             var self = this;
 13             // 不用 __set , 通知 view 更新
 14             self.set("el", el);
 15             self.decorateChildren(el);
 16         },
 17 
 18         /**
 19          * 生成一个组件
 20          */
 21         decorateChildrenInternal:function(UI, c, prefixCls) {
 22             this.addChild(new UI({
 23                 srcNode:c,
 24                 prefixCls:prefixCls
 25             }));
 26         },
 27 
 28         /**
 29          * 得到适合装饰该节点的组件类
 30          * @param c
 31          */
 32         _findUIByClass:function(c) {
 33             var self = this,
 34                 cls = c.attr("class") || "",
 35                 prefixCls = self.get("prefixCls");
 36             // 过滤掉特定前缀
 37             cls = cls.replace(new RegExp("\\b" + prefixCls, "ig"), "");
 38             var UI = UIStore.getUIByClass(cls);
 39             if (!UI) {
 40                 S.log(c);
 41                 S.log("can not find ui " + cls + " from this markup");
 42             }
 43             return UI;
 44         },
 45 
 46         /**
 47          * container 需要在装饰时对儿子特殊处理，递归装饰
 48          */
 49         decorateChildren:function(el) {
 50             var self = this,children = el.children(),
 51                 prefixCls = self.get("prefixCls");
 52             children.each(function(c) {
 53                 var UI = self._findUIByClass(c);
 54                 self.decorateChildrenInternal(UI, c, prefixCls);
 55             });
 56         }
 57     });
 58 
 59     return DecorateChildren;
 60 
 61 }, {
 62     requires:['./uistore']
 63 });
