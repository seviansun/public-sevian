1 /**
  2  * @fileOverview  inspired by yui3 :
  3  *
  4  * Synthetic event that fires when the <code>value</code> property of an input
  5  * field or textarea changes as a result of a keystroke, mouse operation, or
  6  * input method editor (IME) input event.
  7  *
  8  * Unlike the <code>onchange</code> event, this event fires when the value
  9  * actually changes and not when the element loses focus. This event also
 10  * reports IME and multi-stroke input more reliably than <code>oninput</code> or
 11  * the various key events across browsers.
 12  *
 13  * @author yiminghe@gmail.com
 14  */
 15 KISSY.add('event/valuechange', function (S, Event, DOM, special) {
 16     var VALUE_CHANGE = "valuechange",
 17         nodeName = DOM._nodeName,
 18         KEY = "event/valuechange",
 19         HISTORY_KEY = KEY + "/history",
 20         POLL_KEY = KEY + "/poll",
 21         interval = 50;
 22 
 23     function clearPollTimer(target) {
 24         if (DOM.hasData(target, POLL_KEY)) {
 25             var poll = DOM.data(target, POLL_KEY);
 26             clearTimeout(poll);
 27             DOM.removeData(target, POLL_KEY);
 28         }
 29     }
 30 
 31     function stopPoll(target) {
 32         DOM.removeData(target, HISTORY_KEY);
 33         clearPollTimer(target);
 34     }
 35 
 36     function stopPollHandler(ev) {
 37         clearPollTimer(ev.target);
 38     }
 39 
 40     function checkChange(target) {
 41         var v = target.value,
 42             h = DOM.data(target, HISTORY_KEY);
 43         if (v !== h) {
 44             // 只触发自己绑定的 handler
 45             Event.fire(target, VALUE_CHANGE, {
 46                 prevVal:h,
 47                 newVal:v
 48             }, true);
 49             DOM.data(target, HISTORY_KEY, v);
 50         }
 51     }
 52 
 53     function startPoll(target) {
 54         if (DOM.hasData(target, POLL_KEY)) {
 55             return;
 56         }
 57         DOM.data(target, POLL_KEY, setTimeout(function () {
 58             checkChange(target);
 59             DOM.data(target, POLL_KEY, setTimeout(arguments.callee, interval));
 60         }, interval));
 61     }
 62 
 63     function startPollHandler(ev) {
 64         var target = ev.target;
 65         // when focus ,record its current value immediately
 66         if (ev.type == "focus") {
 67             DOM.data(target, HISTORY_KEY, target.value);
 68         }
 69         startPoll(target);
 70     }
 71 
 72     function webkitSpeechChangeHandler(e) {
 73         checkChange(e.target);
 74     }
 75 
 76     function monitor(target) {
 77         unmonitored(target);
 78         Event.on(target, "blur", stopPollHandler);
 79         // fix #94
 80         // see note 2012-02-08
 81         Event.on(target, "webkitspeechchange", webkitSpeechChangeHandler);
 82         Event.on(target, "mousedown keyup keydown focus", startPollHandler);
 83     }
 84 
 85     function unmonitored(target) {
 86         stopPoll(target);
 87         Event.remove(target, "blur", stopPollHandler);
 88         Event.remove(target, "webkitspeechchange", webkitSpeechChangeHandler);
 89         Event.remove(target, "mousedown keyup keydown focus", startPollHandler);
 90     }
 91 
 92     special[VALUE_CHANGE] = {
 93         setup:function () {
 94             var target = this;
 95             if (nodeName(target, "input")
 96                 || nodeName(target, "textarea")) {
 97                 monitor(target);
 98             }
 99         },
100         tearDown:function () {
101             var target = this;
102             unmonitored(target);
103         }
104     };
105     return Event;
106 }, {
107     requires:["./base", "dom", "./special"]
108 });
