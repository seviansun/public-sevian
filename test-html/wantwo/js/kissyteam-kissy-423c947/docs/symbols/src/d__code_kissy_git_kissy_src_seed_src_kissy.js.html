 22     var host = this,
 23         hasEnumBug = !({toString:1}.propertyIsEnumerable('toString')),
 24         hasOwn = Object.prototype.hasOwnProperty,
 25         emumProperties = [
 26             'hasOwnProperty',
 27             'isPrototypeOf',
 28             'propertyIsEnumerable',
 29             'toString',
 30             'toLocaleString',
 31             'valueOf'
 32         ],
 33         meta = {
 34             /**
 35              * Copies all the properties of s to r.
 36              * @name KISSY.mix
 37              * @function
 38              * @param {Object} r the augmented object
 39              * @param {Object} s the object need to augment
 40              * @param {boolean} [ov=true] whether overwrite existing property
 41              * @param {String[]} [wl] array of white-list properties
 42              * @param deep {boolean} whether recursive mix if encounter object,
 43              * if deep is set true,then ov should be set true too!
 44              * @return {Object} the augmented object
 45              * @example
 46              * <code>
 47              * var t={};
 48              * S.mix({x:{y:2,z:4}},{x:{y:3,a:t}},1,0,1) =>{x:{y:3,z:4,a:{}}} , a!==t
 49              * S.mix({x:{y:2,z:4}},{x:{y:3,a:t}},1) => {x:{y:3,a:t}}
 50              * </code>
 51              */
 52             mix:function (r, s, ov, wl, deep) {
 53                 if (!s || !r) {
 54                     return r;
 55                 }
 56                 if (ov === undefined) {
 57                     ov = true;
 58                 }
 59                 var i, p, len;
 60 
 61                 if (wl && (len = wl.length)) {
 62                     for (i = 0; i < len; i++) {
 63                         p = wl[i];
 64                         if (p in s) {
 65                             _mix(p, r, s, ov, deep);
 66                         }
 67                     }
 68                 } else {
 69                     for (p in s) {
 70                         // no hasOwnProperty judge !
 71                         _mix(p, r, s, ov, deep);
 72                     }
 73 
 74                     // fix #101
 75                     if (hasEnumBug) {
 76                         for (var j = 0; j < emumProperties.length; j++) {
 77                             p = emumProperties[j];
 78                             if (ov && hasOwn.call(s, p)) {
 79                                 r[p] = s[p];
 80                             }
 81                         }
 82                     }
 83 
 84                 }
 85                 return r;
 86             }
 87         },
 88 
 89         _mix = function (p, r, s, ov, deep) {
 90             if (ov || !(p in r)) {
 91                 var target = r[p], src = s[p];
 92                 // prevent never-end loop
 93                 if (target === src) {
 94                     return;
 95                 }
 96                 // 来源是数组和对象，并且要求深度 mix
 97                 if (deep && src && (S.isArray(src) || S.isPlainObject(src))) {
 98                     // 目标值为对象或数组，直接 mix
 99                     // 否则 新建一个和源值类型一样的空数组/对象，递归 mix
100                     var clone = target && (S.isArray(target) || S.isPlainObject(target)) ?
101                         target :
102                         (S.isArray(src) ? [] : {});
103                     r[p] = S.mix(clone, src, ov, undefined, true);
104                 } else if (src !== undefined) {
105                     r[p] = s[p];
106                 }
107             }
108         },

111         // be overwritten so that defined namespaces are preserved.
112         seed = (host && host[S]) || {},
113

118     // compatibled for  '<this> is null' in unknown engine.
119     seed.Env = seed.Env || {};
120     host = seed.Env.host || (seed.Env.host = host || {});

151             /**
152              * Returns a new object containing all of the properties of
153              * all the supplied objects. The properties from later objects
154              * will overwrite those in earlier objects. Passing in a
155              * single object will create a shallow copy of it.
156              * @param {...} m1 objects need to be merged
157              * @return {Object} the new merged object
158              */
159             merge:function (m1) {
160                 var o = {}, i, l = arguments.length;
161                 for (i = 0; i < l; i++) {
162                     S.mix(o, arguments[i]);
163                 }
164                 return o;

167             /**
168              * Applies prototype properties from the supplier to the receiver.
169              * @param   {Object} r received object
170              * @param   {...Object} s1 object need to  augment
171              *          {boolean} [ov=true] whether overwrite existing property
172              *          {String[]} [wl] array of white-list properties
173              * @return  {Object} the augmented object
174              */
175             augment:function (r, s1) {
176                 var args = S.makeArray(arguments),
177                     len = args.length - 2,
178                     i = 1,
179                     ov = args[len],
180                     wl = args[len + 1];
181 
182                 if (!S.isArray(wl)) {
183                     ov = wl;
184                     wl = undefined;
185                     len++;
186                 }
187                 if (!S.isBoolean(ov)) {
188                     ov = undefined;
189                     len++;
190                 }
191 
192                 for (; i < len; i++) {
193                     S.mix(r.prototype, args[i].prototype || args[i], ov, wl);
194                 }
195 
196                 return r;
197             },

199             /**
200              * Utility to set up the prototype, constructor and superclass properties to
201              * support an inheritance strategy that can chain constructors and methods.
202              * Static members will not be inherited.
203              * @param r {Function} the object to modify
204              * @param s {Function} the object to inherit
205              * @param px {Object} prototype properties to add/override
206              * @param {Object} [sx] static properties to add/override
207              * @return r {Object}
208              */
209             extend:function (r, s, px, sx) {
210                 if (!s || !r) {
211                     return r;
212                 }
213 
214                 var create = Object.create ?
215                     function (proto, c) {
216                         return Object.create(proto, {
217                             constructor:{
218                                 value:c
219                             }
220                         });
221                     } :
222                     function (proto, c) {
223                         function F() {
224                         }
225 
226                         F.prototype = proto;
227 
228                         var o = new F();
229                         o.constructor = c;
230                         return o;
231                     },
232                     sp = s.prototype,
233                     rp;
234 
235                 // add prototype chain
236                 rp = create(sp, r);
237                 r.prototype = S.mix(rp, r.prototype);
238                 r.superclass = create(sp, s);
239 
240                 // add prototype overrides
241                 if (px) {
242                     S.mix(rp, px);
243                 }
244 
245                 // add object overrides
246                 if (sx) {
247                     S.mix(r, sx);
248                 }
249 
250                 return r;

260             /**
261              * Returns the namespace specified and creates it if it doesn't exist. Be careful
262              * when naming packages. Reserved words may work in some browsers and not others.
263              * <code>
264              * S.namespace('KISSY.app'); // returns KISSY.app
265              * S.namespace('app.Shop'); // returns KISSY.app.Shop
266              * S.namespace('TB.app.Shop', true); // returns TB.app.Shop
267              * </code>
268              * @return {Object}  A reference to the last namespace object created
269              */
270             namespace:function () {
271                 var args = S.makeArray(arguments),
272                     l = args.length,
273                     o = null, i, j, p,
274                     global = (args[l - 1] === true && l--);
275 
276                 for (i = 0; i < l; i++) {
277                     p = (EMPTY + args[i]).split('.');
278                     o = global ? host : this;
279                     for (j = (host[p[0]] === o) ? 1 : 0; j < p.length; ++j) {
280                         o = o[p[j]] = o[p[j]] || { };
281                     }
282                 }
283                 return o;

286             /**
287              * create app based on KISSY.
288              * @param name {String} the app name
289              * @param sx {Object} static properties to add/override
290              * <code>
291              * S.app('TB');
292              * TB.namespace('app'); // returns TB.app
293              * </code>
294              * @return {Object}  A reference to the app global object
295              * @deprecated recommended using packages
296              */
297             app:function (name, sx) {
298                 var isStr = S.isString(name),
299                     O = isStr ? host[name] || {} : name,
300                     i = 0,
301                     __APP_INIT_METHODS = S.__APP_INIT_METHODS,
302                     len = __APP_INIT_METHODS.length;
303 
304                 S.mix(O, this, true, S.__APP_MEMBERS);
305                 for (; i < len; i++) {
306                     __APP_INIT_METHODS[i].call(O);
307                 }
308 
309                 S.mix(O, S.isFunction(sx) ? sx() : sx);
310                 isStr && (host[name] = O);
311 
312                 return O;
313             },

316              * set KISSY configuration

318              * @param {Object[]} c.packages

320              * @param {String} c.packages.0.path package path
321              * @param {String} c.packages.0.tag timestamp for this package's module file
322              * @param {Array[]} c.map file map configs
323              * @param {Array[]} c.map.0 a single map rule
324              * @param {RegExp} c.map.0.0 a regular expression to match url
325              * @param {String|Function} c.map.0.1 provide replacement for String.replace
326              * @param {boolean} c.combine whether to enable combo
327              * @param {String} c.base set base for kissy loader.use with caution!
328              * @param {boolean} c.debug whether to enable debug mod

332              * KISSY.config({
