1 /**
  2  * render aria properties to input element
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("autocomplete/inputRender", function (S, UIBase, Component) {
  6     return UIBase.create(Component.Render, [], {
  7         renderUI:function () {
  8             var el = this.get("el");
  9             el.attr({
 10                 aria-haspopup:true,
 11                 aria-autocomplete:"list",
 12                 autocomplete:"off",
 13                 role:"combobox"
 14             });
 15         },
 16 
 17         _uiSetAriaOwns:function (v) {
 18             this.get("el").attr("aria-owns", v);
 19         },
 20 
 21         _uiSetAriaExpanded:function (v) {
 22             this.get("el").attr("aria-expanded", v);
 23         }
 24     }, {
 25         ATTRS:{
 26             ariaOwns:{
 27             },
 28             ariaExpanded:{
 29             },
 30             elTagName:{
 31                 value:'input'
 32             }
 33         }
 34     });
 35 }, {
 36     requires:['uibase', 'component']
 37 });
