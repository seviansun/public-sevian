1 /**

  3  * @author  lifesinger@gmail.com,yiminghe@gmail.com
  4  * @description this code can only run at browser environment
  5  */
  6 (function (S, undefined) {
  7 
  8     var win = S.Env.host,
  9 
 10         doc = win['document'],
 11 
 12         docElem = doc.documentElement,
 13 
 14         EMPTY = '',
 15 
 16         readyDefer = new S.Defer(),
 17 
 18         readyPromise = readyDefer.promise,
 19 
 20         // The number of poll times.
 21         POLL_RETRYS = 500,
 22 
 23         // The poll interval in milliseconds.
 24         POLL_INTERVAL = 40,
 25 
 26         // #id or id
 27         RE_IDSTR = /^#?([\w-]+)$/,
 28 
 29         RE_NOT_WHITE = /\S/;
 30 
 31     S.mix(S,
 32         /**
 33          * @lends KISSY
 34          */
 35         {
 36 
 37 
 38             /**
 39              * A crude way of determining if an object is a window
 40              */
 41             isWindow:function (o) {
 42                 return S.type(o) === 'object'
 43                     && 'setInterval' in o
 44                     && 'document' in o
 45                     && o.document.nodeType == 9;
 46             },
 47 
 48 
 49             /**
 50              * get xml representation of data
 51              * @param {String} data
 52              */
 53             parseXML:function (data) {
 54                 var xml;
 55                 try {
 56                     // Standard
 57                     if (win.DOMParser) {
 58                         xml = new DOMParser().parseFromString(data, "text/xml");
 59                     } else { // IE
 60                         xml = new ActiveXObject("Microsoft.XMLDOM");
 61                         xml.async = "false";
 62                         xml.loadXML(data);
 63                     }
 64                 } catch (e) {
 65                     S.log("parseXML error : ");
 66                     S.log(e);
 67                     xml = undefined;
 68                 }
 69                 if (!xml || !xml.documentElement || xml.getElementsByTagName("parsererror").length) {
 70                     S.error("Invalid XML: " + data);
 71                 }
 72                 return xml;
 73             },
 74 
 75             /**
 76              * Evalulates a script in a global context.
 77              */
 78             globalEval:function (data) {
 79                 if (data && RE_NOT_WHITE.test(data)) {
 80                     // http://weblogs.java.net/blog/driscoll/archive/2009/09/08/eval-javascript-global-context
 81                     ( win.execScript || function (data) {
 82                         win[ "eval" ].call(win, data);
 83                     } )(data);
 84                 }
 85             },
 86 
 87             /**
 88              * Specify a function to execute when the DOM is fully loaded.
 89              * @param fn {Function} A function to execute after the DOM is ready
 90              * <code>
 91              * KISSY.ready(function(S){ });
 92              * </code>
 93              * @return {KISSY}
 94              */
 95             ready:function (fn) {
 96 
 97                 function f() {
 98                     try {
 99                         fn(S);
100                     } catch (e) {
101                         // print stack info for firefox/chrome
102                         S.log(e.stack, "error");
103                         if (S.Config.debug) {
104                             throw e;
105                         }
106                     }
107                 }
108 
109                 readyPromise.then(f);
110 
111                 return this;
112             },
113 
114             /**
115              * Executes the supplied callback when the item with the supplied id is found.
116              * @param id <String> The id of the element, or an array of ids to look for.
117              * @param fn <Function> What to execute when the element is found.
118              */
119             available:function (id, fn) {
120                 id = (id + EMPTY).match(RE_IDSTR)[1];
121                 if (!id || !S.isFunction(fn)) {
122                     return;
123                 }
124 
125                 var retryCount = 1,
126                     node,
127                     timer = S.later(function () {
128                         if ((node = doc.getElementById(id)) && (fn(node) || 1) ||
129                             ++retryCount > POLL_RETRYS) {
130                             timer.cancel();
131                         }
132                     }, POLL_INTERVAL, true);
133             }
134         });
135 
136 
137     /**
138      * Binds ready events.
139      */
140     function _bindReady() {
141         var doScroll = docElem.doScroll,
142             eventType = doScroll ? 'onreadystatechange' : 'DOMContentLoaded',
143             COMPLETE = 'complete',
144             fire = function () {
145                 readyDefer.resolve(S)
146             };
147 
148         // Catch cases where ready() is called after the
149         // browser event has already occurred.
150         if (doc.readyState === COMPLETE) {
151             return fire();
152         }
153 
154         // w3c mode
155         if (doc.addEventListener) {
156             function domReady() {
157                 doc.removeEventListener(eventType, domReady, false);
158                 fire();
159             }
160 
161             doc.addEventListener(eventType, domReady, false);
162 
163             // A fallback to window.onload, that will always work
164             win.addEventListener('load', fire, false);
165         }
166         // IE event model is used
167         else {
168             function stateChange() {
169                 if (doc.readyState === COMPLETE) {
170                     doc.detachEvent(eventType, stateChange);
171                     fire();
172                 }
173             }
174 
175             // ensure firing before onload (but completed after all inner iframes is loaded)
176             // maybe late but safe also for iframes
177             doc.attachEvent(eventType, stateChange);
178 
179             // A fallback to window.onload, that will always work.
180             win.attachEvent('onload', fire);
181 
182             // If IE and not a frame
183             // continually check to see if the document is ready
184             var notframe;
185 
186             try {
187                 notframe = (win['frameElement'] === null);
188             } catch (e) {
189                 S.log("get frameElement error : ");
190                 S.log(e);
191                 notframe = false;
192             }
193 
194             // can not use in iframe,parent window is dom ready so doScoll is ready too
195             if (doScroll && notframe) {
196                 function readyScroll() {
197                     try {
198                         // Ref: http://javascript.nwbox.com/IEContentLoaded/
199                         doScroll('left');
200                         fire();
201                     } catch (ex) {
202                         //S.log("detect document ready : " + ex);
203                         setTimeout(readyScroll, POLL_INTERVAL);
204                     }
205                 }
206 
207                 readyScroll();
208             }
209         }
210         return 0;
211     }
212 
213     // If url contains '?ks-debug', debug mode will turn on automatically.
214     if (location && (location.search || EMPTY).indexOf('ks-debug') !== -1) {
215         S.Config.debug = true;
216     }
217 
218     /**
219      * bind on start
220      * in case when you bind but the DOMContentLoaded has triggered
221      * then you has to wait onload
222      * worst case no callback at all
223      */
224     _bindReady();
225 
226     if (navigator.userAgent.match(/MSIE/)) {
227         try {
228             doc.execCommand("BackgroundImageCache", false, true);
229         } catch (e) {
230         }
231     }
232 
233 })(KISSY, undefined);
