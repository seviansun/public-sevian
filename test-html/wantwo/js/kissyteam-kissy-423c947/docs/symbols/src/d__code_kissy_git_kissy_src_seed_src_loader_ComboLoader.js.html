1 /**
  2  * using combo to load module files
  3  * @author yiminghe@gmail.com
  4  */
  5 (function (S) {
  6 
  7     if (typeof require !== 'undefined') {
  8         return;
  9     }
 10 
 11     function loadScripts(urls, callback, charset) {
 12         var count = urls && urls.length;
 13         if (!count) {
 14             callback();
 15             return;
 16         }
 17         for (var i = 0; i < urls.length; i++) {
 18             var url = urls[i];
 19             S.getScript(url, function () {
 20                 if (!(--count)) {
 21                     callback();
 22                 }
 23             }, charset || "utf-8");
 24         }
 25     }
 26 
 27     var MAX_URL_LENGTH = 1024,
 28         Loader = S.Loader,
 29         data = Loader.STATUS,
 30         utils = Loader.Utils;
 31 
 32     /**
 33      * using combo to load module files
 34      * @class
 35      * @param SS KISSY
 36      */
 37     function ComboLoader(SS) {
 38         S.mix(this, {
 39             SS:SS,
 40             queue:[],
 41             loading:0
 42         });
 43     }
 44 
 45     S.augment(ComboLoader,
 46         Loader.Target,
 47         /**
 48          * @lends ComboLoader#
 49          */
 50         {
 51             next:function () {
 52                 var self = this;
 53                 if (self.queue.length) {
 54                     var args = self.queue.shift();
 55                     self._use(args.modNames, args.fn);
 56                 }
 57             },
 58 
 59             enqueue:function (modNames, fn) {
 60                 var self = this;
 61                 self.queue.push({
 62                     modNames:modNames,
 63                     fn:fn
 64                 });
 65             },
 66 
 67             _use:function (modNames, fn) {
 68                 var self = this, SS = self.SS;
 69 
 70                 self.loading = 1;
 71 
 72                 modNames = utils.getModNamesAsArray(modNames);
 73 
 74                 modNames= utils.normalizeModNamesWithAlias(SS,modNames);
 75 
 76                 var unaliasModNames = utils.normalizeModNames(SS, modNames);
 77 
 78                 var allModNames = self.calculate(unaliasModNames);
 79 
 80                 var comboUrls = self.getComboUrls(allModNames);
 81 
 82                 // load css first to avoid page blink
 83                 var css = comboUrls.css,
 84                     countCss = 0;
 85 
 86                 for (var p in css) {
 87                     countCss++;
 88                 }
 89 
 90                 if (!countCss) {
 91                     self._useJs(comboUrls, fn, modNames);
 92                     return;
 93                 }
 94 
 95                 for (p in css) {
 96                     loadScripts(css[p], function () {
 97                         if (!(--countCss)) {
 98                             S.each(unaliasModNames, function (name) {
 99                                 utils.attachMod(self.SS, self.getModInfo(name));
100                             });
101                             self._useJs(comboUrls, fn, modNames);
102                         }
103                     }, css[p].charset);
104                 }
105             },
106 
107             use:function (modNames, callback) {
108                 var self = this,
109                     fn = function () {
110                         // KISSY.use in callback will be queued
111                         if (callback) {
112                             callback.apply(this, arguments);
113                         }
114                         self.loading = 0;
115                         self.next();
116                     };
117 
118                 self.enqueue(modNames, fn);
119 
120                 if (!self.loading) {
121                     self.next();
122                 }
123             },
124 
125             _useJs:function (comboUrls, fn, modNames) {
126                 var self = this,
127                     jss = comboUrls.js,
128                     countJss = 0;
129 
130 
131                 for (var p in jss) {
132                     countJss++;
133                 }
134 
135                 if (!countJss) {
136                     fn.apply(null, utils.getModules(self.SS, modNames));
137                     return;
138                 }
139                 var success = 1;
140                 for (p in jss) {
141                     (function (p) {
142                         loadScripts(jss[p], function () {
143                             var mods = jss[p].mods;
144                             for (var i = 0; i < mods.length; i++) {
145                                 var mod = mods[i];
146                                 // fix #111
147                                 // https://github.com/kissyteam/kissy/issues/111
148                                 if (!mod.fn) {
149                                     S.log(mod.name + ' is not loaded! can not find module in path : ' + jss[p], 'error');
150                                     mod.status = data.ERROR;
151                                     success = 0;
152                                     return;
153                                 }
154                             }
155                             if (success && !(--countJss)) {
156                                 var unaliasModNames = utils.normalizeModNames(self.SS, modNames);
157                                 self.attachMods(unaliasModNames);
158                                 if (utils.isAttached(self.SS, unaliasModNames)) {
159                                     fn.apply(null, utils.getModules(self.SS, modNames))
160                                 } else {
161                                     // new require is introduced by KISSY.add
162                                     // run again
163                                     self._use(modNames, fn)
164                                 }
165                             }
166                         }, jss[p].charset);
167                     })(p);
168                 }
169             },
170 
171             add:function (name, def, config) {
172                 var self = this,
173                     requires,
174                     SS = self.SS;
175 
176                 if (utils.normAdd(SS, name, def, config)) {
177                     return;
178                 }
179                 if (config && (requires = config.requires)) {
180                     utils.normalDepModuleName(name, requires);
181                 }
182                 utils.registerModule(SS, name, def, config);
183             },
184 
185 
186             attachMods:function (modNames) {
187                 var self = this;
188                 S.each(modNames, function (modName) {
189                     self.attachMod(modName);
190                 });
191             },
192 
193             attachMod:function (modName) {
194                 var SS = this.SS,
195                     mod = this.getModInfo(modName);
196                 if (
197                 // new require after add
198                 // not register yet!
199                     !mod || utils.isAttached(SS, modName)) {
200                     return;
201                 }
202                 var requires = utils.normalizeModNames(SS, mod.requires, modName);
203                 for (var i = 0; i < requires.length; i++) {
204                     this.attachMod(requires[i]);
205                 }
206                 for (i = 0; i < requires.length; i++) {
207                     if (!utils.isAttached(SS, requires[i])) {
208                         return false;
209                     }
210                 }
211                 utils.attachMod(SS, mod);
212             },
213 
214             calculate:function (modNames) {
215                 var ret = {}, SS = this.SS;
216                 for (var i = 0; i < modNames.length; i++) {
217                     var m = modNames[i];
218                     if (!utils.isAttached(SS, m)) {
219                         if (!utils.isLoaded(SS, m)) {
220                             ret[m] = 1;
221                         }
222                         S.mix(ret, this.getRequires((m)));
223                     }
224                 }
225                 var ret2 = [];
226                 for (var r in ret) {
227                     ret2.push(r);
228                 }
229                 return ret2;
230             },
231 
232             getComboUrls:function (modNames) {
233                 var self = this,
234                     i,
235                     mod,
236                     packagePath,
237                     combos = {};
238 
239                 S.each(modNames, function (modName) {
240                     mod = self.getModInfo(modName);
241                     packagePath = utils.getPackagePath(self.SS, mod);
242                     var type = utils.isCss(mod.path) ? "css" : "js";
243                     combos[packagePath] = combos[packagePath] || {};
244                     combos[packagePath][type] = combos[packagePath][type] || [];
245                     combos[packagePath][type].tag = combos[packagePath][type].tag || mod.tag;
246                     combos[packagePath][type].packageTag = mod.packageTag;
247                     combos[packagePath][type].charset = mod.charset;
248                     combos[packagePath][type].push(mod);
249                 });
250 
251                 S.each(combos, function (v) {
252                     var js, css;
253                     if (js = v["js"]) {
254                         // module level tag is superior to package level tag
255                         js.tag = js.tag || js.packageTag;
256                     }
257                     if (css = v["css"]) {
258                         css.tag = css.tag || css.packageTag;
259                     }
260                 });
261 
262                 var res = {
263                     js:{},
264                     css:{}
265                 }, t;
266 
267                 var comboPrefix = S.Config.comboPrefix,
268                     comboSep = S.Config.comboSep,
269                     maxUrlLength = S.Config['comboMaxUrlLength'] || MAX_URL_LENGTH;
270 
271                 for (packagePath in combos) {
272                     for (var type in combos[packagePath]) {
273                         t = [];
274                         var jss = combos[packagePath][type];
275                         res[type][packagePath] = [];
276                         res[type][packagePath].charset = jss.charset;
277                         // current package's mods
278                         res[type][packagePath].mods = [];
279                         var prefix = packagePath + comboPrefix,
280                             l = prefix.length;
281                         for (i = 0; i < jss.length; i++) {
282                             t.push(jss[i].path);
283                             res[type][packagePath].mods.push(jss[i]);
284                             if (l + t.join(comboSep).length > maxUrlLength) {
285                                 t.pop();
286                                 res[type][packagePath].push(self.getComboUrl(
287                                     t.length > 1 ? prefix : packagePath,
288                                     t, comboSep, jss.tag
289                                 ));
290                                 t = [];
291                                 i--;
292                             }
293                         }
294                         if (t.length) {
295                             res[type][packagePath].push(self.getComboUrl(
296                                 t.length > 1 ? prefix : packagePath,
297                                 t, comboSep, jss.tag
298                             ));
299                         }
300                     }
301                 }
302 
303                 return res;
304             },
305 
306             getComboUrl:function (prefix, t, comboSep, tag) {
307                 return utils.getMappedPath(
308                     this.SS,
309                     prefix + t.join(comboSep) + (tag ? ("?t=" + tag) : "")
310                 );
311             },
312 
313             getModInfo:function (modName) {
314                 var SS = this.SS, mods = SS.Env.mods;
315                 return mods[modName];
316             },
317 
318             // get requires mods need to be loaded dynamically
319             getRequires:function (modName) {
320                 var self = this,
321                     SS = self.SS,
322                     mod = self.getModInfo(modName),
323                     ret = {};
324                 // if this mod is attached then its require is attached too!
325                 if (mod && !utils.isAttached(SS, modName)) {
326                     var requires = utils.normalizeModNames(SS, mod.requires, modName);
327                     // circular dependency check
328                     if (S.Config.debug) {
329                         var allRequires = mod.__allRequires || (mod.__allRequires = {});
330                         if (allRequires[modName]) {
331                             S.error("detect circular dependency among : ");
332                             S.error(allRequires);
333                         }
334                     }
335                     for (var i = 0; i < requires.length; i++) {
336                         var r = requires[i];
337                         if (S.Config.debug) {
338                             // circular dependency check
339                             var rMod = self.getModInfo(r);
340                             allRequires[r] = 1;
341                             if (rMod && rMod.__allRequires) {
342                                 S.each(rMod.__allRequires, function (_, r2) {
343                                     allRequires[r2] = 1;
344                                 });
345                             }
346                         }
347                         // if not load into page yet
348                         if (!utils.isLoaded(SS, r)
349                             // and not attached
350                             && !utils.isAttached(SS, r)) {
351                             ret[r] = 1;
352                         }
353                         var ret2 = self.getRequires(r);
354                         S.mix(ret, ret2);
355                     }
356                 }
357                 return ret;
358             }
359         });
360 
361     Loader.Combo = ComboLoader;
