  5 KISSY.add('ua/base', function (S, undefined) {
  6 
  7     var ua = navigator.userAgent,
  8         EMPTY = '',
  9         MOBILE = 'mobile',
 10         core = EMPTY,
 11         shell = EMPTY, m,
 12         IE_DETECT_RANGE = [6, 9],
 13         v,
 14         end,
 15         VERSION_PLACEHOLDER = '{{version}}',
 16         IE_DETECT_TPL = '<!--[if IE ' + VERSION_PLACEHOLDER + ']><' + 's></s><![endif]-->',
 17         div = S.Env.host.document.createElement('div'),
 18         s,
 19         o = {
 20             // browser core type
 21             webkit:undefined,
 22             trident:undefined,
 23             gecko:undefined,
 24             presto:undefined,
 25 
 26             // browser type
 27             chrome:undefined,
 28             safari:undefined,
 29             firefox:undefined,
 30             ie:undefined,
 31             opera:undefined,
 32 
 33             mobile:undefined,
 34             core:undefined,
 35             shell:undefined
 36         },
 37         numberify = function (s) {
 38             var c = 0;
 39             // convert '1.2.3.4' to 1.234
 40             return parseFloat(s.replace(/\./g, function () {
 41                 return (c++ === 0) ? '.' : '';
 42             }));
 43         };
 44 
 45     // try to use IE-Conditional-Comment detect IE more accurately
 46     // IE10 doesn't support this method, @ref: http://blogs.msdn.com/b/ie/archive/2011/07/06/html5-parsing-in-ie10.aspx
 47     div.innerHTML = IE_DETECT_TPL.replace(VERSION_PLACEHOLDER, '');
 48     s = div.getElementsByTagName('s');
 49 
 50     if (s.length > 0) {
 51 
 52         shell = 'ie';
 53         o[core = 'trident'] = 0.1; // Trident detected, look for revision
 54 
 55         // Get the Trident's accurate version
 56         if ((m = ua.match(/Trident\/([\d.]*)/)) && m[1]) {
 57             o[core] = numberify(m[1]);
 58         }
 59 
 60         // Detect the accurate version
 61         // 注意：
 62         //  o.shell = ie, 表示外壳是 ie
 63         //  但 o.ie = 7, 并不代表外壳是 ie7, 还有可能是 ie8 的兼容模式
 64         //  对于 ie8 的兼容模式，还要通过 documentMode 去判断。但此处不能让 o.ie = 8, 否则
 65         //  很多脚本判断会失误。因为 ie8 的兼容模式表现行为和 ie7 相同，而不是和 ie8 相同
 66         for (v = IE_DETECT_RANGE[0], end = IE_DETECT_RANGE[1]; v <= end; v++) {
 67             div.innerHTML = IE_DETECT_TPL.replace(VERSION_PLACEHOLDER, v);
 68             if (s.length > 0) {
 69                 o[shell] = v;
 70                 break;
 71             }
 72         }
 73 
 74     } else {
 75 
 76         // WebKit
 77         if ((m = ua.match(/AppleWebKit\/([\d.]*)/)) && m[1]) {
 78             o[core = 'webkit'] = numberify(m[1]);
 79 
 80             // Chrome
 81             if ((m = ua.match(/Chrome\/([\d.]*)/)) && m[1]) {
 82                 o[shell = 'chrome'] = numberify(m[1]);
 83             }
 84             // Safari
 85             else if ((m = ua.match(/\/([\d.]*) Safari/)) && m[1]) {
 86                 o[shell = 'safari'] = numberify(m[1]);
 87             }
 88 
 89             // Apple Mobile
 90             if (/ Mobile\//.test(ua)) {
 91                 o[MOBILE] = 'apple'; // iPad, iPhone or iPod Touch
 92             }
 93             // Other WebKit Mobile Browsers
 94             else if ((m = ua.match(/NokiaN[^\/]*|Android \d\.\d|webOS\/\d\.\d/))) {
 95                 o[MOBILE] = m[0].toLowerCase(); // Nokia N-series, Android, webOS, ex: NokiaN95
 96             }
 97         }
 98         // NOT WebKit
 99         else {
100             // Presto
101             // ref: http://www.useragentstring.com/pages/useragentstring.php
102             if ((m = ua.match(/Presto\/([\d.]*)/)) && m[1]) {
103                 o[core = 'presto'] = numberify(m[1]);
104 
105                 // Opera
106                 if ((m = ua.match(/Opera\/([\d.]*)/)) && m[1]) {
107                     o[shell = 'opera'] = numberify(m[1]); // Opera detected, look for revision
108 
109                     if ((m = ua.match(/Opera\/.* Version\/([\d.]*)/)) && m[1]) {
110                         o[shell] = numberify(m[1]);
111                     }
112 
113                     // Opera Mini
114                     if ((m = ua.match(/Opera Mini[^;]*/)) && m) {
115                         o[MOBILE] = m[0].toLowerCase(); // ex: Opera Mini/2.0.4509/1316
116                     }
117                     // Opera Mobile
118                     // ex: Opera/9.80 (Windows NT 6.1; Opera Mobi/49; U; en) Presto/2.4.18 Version/10.00
119                     // issue: 由于 Opera Mobile 有 Version/ 字段，可能会与 Opera 混淆，同时对于 Opera Mobile 的版本号也比较混乱
120                     else if ((m = ua.match(/Opera Mobi[^;]*/)) && m) {
121                         o[MOBILE] = m[0];
122                     }
123                 }
124 
125                 // NOT WebKit or Presto
126             } else {
127                 // MSIE
128                 // 由于最开始已经使用了 IE 条件注释判断，因此落到这里的唯一可能性只有 IE10+
129                 if ((m = ua.match(/MSIE\s([^;]*)/)) && m[1]) {
130                     o[core = 'trident'] = 0.1; // Trident detected, look for revision
131                     o[shell = 'ie'] = numberify(m[1]);
132 
133                     // Get the Trident's accurate version
134                     if ((m = ua.match(/Trident\/([\d.]*)/)) && m[1]) {
135                         o[core] = numberify(m[1]);
136                     }
137 
138                     // NOT WebKit, Presto or IE
139                 } else {
140                     // Gecko
141                     if ((m = ua.match(/Gecko/))) {
142                         o[core = 'gecko'] = 0.1; // Gecko detected, look for revision
143                         if ((m = ua.match(/rv:([\d.]*)/)) && m[1]) {
144                             o[core] = numberify(m[1]);
145                         }
146 
147                         // Firefox
148                         if ((m = ua.match(/Firefox\/([\d.]*)/)) && m[1]) {
149                             o[shell = 'firefox'] = numberify(m[1]);
150                         }
151                     }
152                 }
153             }
154         }
155     }
156 
157     o.core = core;
158     o.shell = shell;
159     o._numberify = numberify;
160     return o;
161 });

171  *    if(document.addEventListener) 假设 IE9 支持标准事件，则代码不用修改，就自适应了“未来浏览器”。
172  *    对于未知浏览器也是如此。但是，这并不意味着浏览器嗅探就得彻底抛弃。当代码很明确就是针对已知特定浏览器的，
173  *    同时并非是某个特性探测可以解决时，用浏览器嗅探反而能带来代码的简洁，同时也也不会有什么后患。总之，一切
174  *    皆权衡。
175  *  - UA.ie && UA.ie < 8 并不意味着浏览器就不是 IE8, 有可能是 IE8 的兼容模式。进一步的判断需要使用 documentMode.
