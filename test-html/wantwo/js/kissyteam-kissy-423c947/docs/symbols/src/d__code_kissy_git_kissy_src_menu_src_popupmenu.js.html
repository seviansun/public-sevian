1 /**
  2  * @fileOverview positionable and not focusable menu
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/popupmenu", function (S, UIBase, Component, Menu, PopupMenuRender) {
  6 
  7     function getParentMenu(self) {
  8         var subMenuItem = self.get("parent"),
  9             parentMenu;
 10         if (subMenuItem && subMenuItem.get("menu") == self) {
 11             parentMenu = subMenuItem.get("parent");
 12         }
 13         return parentMenu;
 14     }
 15 
 16     function getAutoHideParentMenu(self) {
 17         var parentMenu = getParentMenu(self);
 18         if (parentMenu && parentMenu.get(autoHideOnMouseLeave)) {
 19             return parentMenu;
 20         }
 21         return 0;
 22     }
 23 
 24     var autoHideOnMouseLeave = "autoHideOnMouseLeave";
 25 
 26     function clearOwn(self) {
 27         var l;
 28         if (l = self._leaveHideTimer) {
 29             clearTimeout(l);
 30             self._leaveHideTimer = 0;
 31         }
 32     }
 33 
 34     /**
 35      * Popup Menu
 36      * @name PopupMenu
 37      * @memberOf Menu
 38      * @constructor
 39      */
 40     var PopupMenu = UIBase.create(Menu, [
 41         UIBase.Position,
 42         UIBase.Align
 43     ], {
 44         _clearLeaveHideTimers:function () {
 45             var self = this, i, item, menu;
 46             if (!self.get(autoHideOnMouseLeave)) {
 47                 return;
 48             }
 49             // 清除自身
 50             clearOwn(self);
 51             var cs = self.get("children");
 52             for (i = 0; i < cs.length; i++) {
 53                 item = cs[i];
 54                 // 递归清除子菜单
 55                 if ((menu = item.get("menu")) &&
 56                     // 不是懒加载函数
 57                     !S.isFunction(menu) &&
 58                     menu.get(autoHideOnMouseLeave)) {
 59                     menu._clearLeaveHideTimers();
 60                 }
 61             }
 62         },
 63         _handleMouseLeave:function () {
 64             var self = this,
 65                 parent;
 66             if (!self.get(autoHideOnMouseLeave)) {
 67                 return;
 68             }
 69             self._leaveHideTimer = setTimeout(function () {
 70                 self.hide();
 71                 var subMenuItem = self.get("parent"), m;
 72                 if (subMenuItem) {
 73                     m = getParentMenu(self);
 74                     // 过段时间仍然是父亲 submenu 仍然是他的兄弟中高亮，证明已经离开
 75                     // 否则
 76                     // 1.鼠标移到 submenu 的话，submenu mouseenter clearTimers,
 77                     //   这个 timer 执行不了！
 78                     //
 79                     // 2.鼠标移到了 submenu 并列的其他 menuitem，
 80                     //   导致其他 menuitem highlighted
 81                     //   那么 当前所属 submenu unhighlighted
 82                     //   执行 clearTimers ，这个 timer 仍然不执行
 83 
 84                     // 那么只剩下一种情况，移除了整个嵌套的 menu，
 85                     // 那么执行该 timer
 86                     // menu hide 并且将其所属的 submenu 高亮去掉！
 87                     if (m && m.get("highlightedItem") === subMenuItem) {
 88                         m.set("highlightedItem", null);
 89                     }
 90                 }
 91             }, self.get("autoHideDelay"));
 92             parent = getAutoHideParentMenu(self);
 93             if (parent) {
 94                 parent._handleMouseLeave();
 95             }
 96         },
 97 
 98         _handleMouseEnter:function () {
 99             var self = this;
100             if (!self.get(autoHideOnMouseLeave)) {
101                 return;
102             }
103             var parent = getAutoHideParentMenu(self);
104             if (parent) {
105                 parent._handleMouseEnter();
106             }
107             self._clearLeaveHideTimers();
108         },
109 
110 
111         /**
112          *  suppose it has focus (as a context menu),
113          *  then it must hide when click document
114          */
115         _handleBlur:function () {
116             var self = this;
117             PopupMenu.superclass._handleBlur.apply(self, arguments);
118             self.hide();
119         }
120     }, {
121         ATTRS:{
122             // 弹出菜单一般不可聚焦，焦点在使它弹出的元素上
123             focusable:{
124                 value:false
125             },
126             visibleMode:{
127                 value:"visibility"
128             },
129             autoHideOnMouseLeave:{},
130             autoHideDelay:{
131                 value:100
132             }
133         },
134         DefaultRender:PopupMenuRender
135     });
136 
137     Component.UIStore.setUIByClass("popupmenu", {
138         priority:Component.UIStore.PRIORITY.LEVEL2,
139         ui:PopupMenu
140     });
141 
142     return PopupMenu;
143 
144 }, {
145     requires:['uibase', 'component', './base', './popupmenurender']
146 });
