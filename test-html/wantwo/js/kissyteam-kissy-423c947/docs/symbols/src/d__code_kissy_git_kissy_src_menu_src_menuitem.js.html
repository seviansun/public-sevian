1 /**
  2  * @fileOverview menu item ,child component for menu
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/menuitem", function (S, UIBase, Component, MenuItemRender) {
  6 
  7     var $ = S.all;
  8 
  9     var MenuItem = UIBase.create(Component.Controller, [UIBase.ContentBox], {
 10 
 11         _handleMouseEnter:function (e) {
 12             // 父亲不允许自己处理
 13             if (MenuItem.superclass._handleMouseEnter.call(this, e)) {
 14                 return true;
 15             }
 16             this.get("parent").set("highlightedItem", this);
 17         },
 18 
 19         _handleMouseLeave:function (e) {
 20             // 父亲不允许自己处理
 21             if (MenuItem.superclass._handleMouseLeave.call(this, e)) {
 22                 return true;
 23             }
 24             this.get("parent").set("highlightedItem", undefined);
 25         },
 26 
 27         _performInternal:function () {
 28             var self = this;
 29             // 可选
 30             if (self.get("selectable")) {
 31                 self.set("selected", true);
 32             }
 33             // 可选中，取消选中
 34             if (self.get("checkable")) {
 35                 self.set("checked", !self.get("checked"));
 36             }
 37             self.get("parent").fire("click", {
 38                 // 使用熟悉的 target，而不是自造新词！
 39                 target:self
 40             });
 41             return true;
 42         },
 43 
 44         _uiSetChecked:function (v) {
 45             this._forwardSetAttrToView("checked", v);
 46         },
 47 
 48         _uiSetSelected:function (v) {
 49             this._forwardSetAttrToView("selected", v);
 50         },
 51 
 52         _uiSetHighlighted:function (v) {
 53             MenuItem.superclass._uiSetHighlighted.apply(this, arguments);
 54             // 是否要滚动到当前菜单项(横向，纵向)
 55             if (v) {
 56                 var el = this.get("el"),
 57                     // 找到向上路径上第一个可以滚动的容器，直到父组件节点（包括）
 58                     // 找不到就放弃，为效率考虑不考虑 parent 的嵌套可滚动 div
 59                     p = el.parent(function (e) {
 60                         return $(e).css("overflow") != "visible";
 61                     }, this.get("parent").get("el").parent());
 62                 if (!p) {
 63                     return;
 64                 }
 65                 el.scrollIntoView(p, undefined, undefined, true);
 66             }
 67         },
 68 
 69         containsElement:function (element) {
 70             return this.get('view') && this.get('view').containsElement(element);
 71         }
 72 
 73     }, {
 74         ATTRS:{
 75 
 76             /**
 77              * 是否支持焦点处理
 78              * @override
 79              */
 80             focusable:{
 81                 value:false
 82             },
 83 
 84             visibleMode:{
 85                 value:"display"
 86             },
 87 
 88             /**
 89              * 是否绑定鼠标事件
 90              * @override
 91              */
 92             handleMouseEvents:{
 93                 value:false
 94             },
 95 
 96             selectable:{
 97                 view:true
 98             },
 99 
100             checkable:{
101                 view:true
102             },
103 
104             // @inheritedDoc
105             // option.text
106             // content:{},
107 
108             // option.value
109             value:{},
110 
111             checked:{},
112             selected:{}
113         },
114 
115         HTML_PARSER:{
116             selectable:function (el) {
117                 var cls = this.getCls("menuitem-selectable");
118                 return el.hasClass(cls);
119             }
120         }
121     });
122 
123     MenuItem.DefaultRender = MenuItemRender;
124 
125     Component.UIStore.setUIByClass("menuitem", {
126         priority:Component.UIStore.PRIORITY.LEVEL1,
127         ui:MenuItem
128     });
129 
130     return MenuItem;
131 }, {
132     requires:['uibase', 'component', './menuitemrender']
133 });
