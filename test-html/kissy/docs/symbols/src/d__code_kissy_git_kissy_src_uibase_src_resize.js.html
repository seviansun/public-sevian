1 /**
  2  * @fileOverview resize extension using resizable
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("uibase/resize", function (S) {
  6 
  7     /**
  8      * make component resizable
  9      * @class
 10      * @memberOf UIBase
 11      */
 12     function Resize() {
 13     }
 14 
 15     Resize.ATTRS =
 16     /**
 17      * @lends UIBase.Resize.prototype
 18      */
 19     {
 20         /**
 21          * 调整大小的配置
 22          * @example
 23          * <code>
 24          *  {
 25          *    minWidth:100, //类型整数, 表示拖动调整大小的最小宽度
 26          *    maxWidth:1000, //类型整数, 表示拖动调整大小的最大宽度
 27          *    minHeight:100, //类型整数, 表示拖动调整大小的最小高度
 28          *    maxHeight:1000, //类型整数, 表示拖动调整大小的最大高度
 29          *    handlers:["b","t","r","l","tr","tl","br","bl"] //类型字符串数组, 取自上述 8 个值的集合.
 30          *    // handlers 配置表示的数组元素可取上述八种值之一, t,b,l,r 分别表示 top,bottom,left,right,
 31          *    // 加上组合共八种取值, 可在上, 下, 左, 右以及左上, 左下, 右上, 右下进行拖动.
 32          *  }
 33          * </code>
 34          * @type Object
 35          */
 36         resize:{
 37             value:{
 38             }
 39         }
 40     };
 41 
 42     Resize.prototype = {
 43         __destructor:function () {
 44             var r = this.resizer;
 45             r && r.destroy();
 46         },
 47         _uiSetResize:function (v) {
 48             var Resizable = S.require("resizable"),
 49                 self = this;
 50             if (Resizable) {
 51                 self.resizer && self.resizer.destroy();
 52                 v.node = self.get("el");
 53                 if (v.handlers) {
 54                     self.resizer = new Resizable(v);
 55                 }
 56             }
 57         }
 58     };
 59 
 60 
 61     return Resize;
 62 });
