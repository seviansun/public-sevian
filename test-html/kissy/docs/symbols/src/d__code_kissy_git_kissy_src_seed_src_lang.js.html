965     function cloneInternal(input, f, memory) {
966         var destination = input,
967             isArray,
968             isPlainObject,
969             k,
970             stamp;
971         if (!input) {
972             return destination;
973         }
974 
975         // If input is the source object of a pair of objects in memory,
976         // then return the destination object in that pair of objects .
977         // and abort these steps.
978         if (input[CLONE_MARKER]) {
979             // 对应的克隆后对象
980             return memory[input[CLONE_MARKER]].destination;
981         } else if (typeof input === "object") {
982             // 引用类型要先记录
983             var constructor = input.constructor;
984             if (S.inArray(constructor, [Boolean, String, Number, Date, RegExp])) {
985                 destination = new constructor(input.valueOf());
986             }
987             // ImageData , File, Blob , FileList .. etc
988             else if (isArray = S.isArray(input)) {
989                 destination = f ? S.filter(input, f) : input.concat();
990             } else if (isPlainObject = S.isPlainObject(input)) {
991                 destination = {};
992             }
993             // Add a mapping from input (the source object)
994             // to output (the destination object) to memory.
995             // 做标记
996             input[CLONE_MARKER] = (stamp = S.guid());
997             // 存储源对象以及克隆后的对象
998             memory[stamp] = {destination:destination, input:input};
999         }
1000         // If input is an Array object or an Object object,
1001         // then, for each enumerable property in input,
1002         // add a new property to output having the same name,
1003         // and having a value created from invoking the internal structured cloning algorithm recursively
1004         // with the value of the property as the "input" argument and memory as the "memory" argument.
1005         // The order of the properties in the input and output objects must be the same.
1006 
1007         // clone it
1008         if (isArray) {
1009             for (var i = 0; i < destination.length; i++) {
1010                 destination[i] = cloneInternal(destination[i], f, memory);
1011             }
1012         } else if (isPlainObject) {
1013             for (k in input) {
1014                 if (input.hasOwnProperty(k)) {
1015                     if (k !== CLONE_MARKER &&
1016                         (!f || (f.call(input, input[k], k, input) !== FALSE))) {
1017                         destination[k] = cloneInternal(input[k], f, memory);
1018                     }
1019                 }
1020             }
1021         }
1022 
1023         return destination;
1024     }
