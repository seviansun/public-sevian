1 /**
  2  * @fileOverview constrain extension for kissy
  3  * @author yiminghe@gmail.com, 乔花<qiaohua@taobao.com>
  4  */
  5 KISSY.add("uibase/constrain", function (S, DOM, Node) {
  6 
  7     /**
  8      * constrain component to specified region
  9      * @class
 10      * @memberOf UIBase
 11      */
 12     function Constrain() {
 13 
 14     }
 15 
 16     Constrain.ATTRS =
 17     /**
 18      * @lends UIBase.Constrain.prototype
 19      */
 20     {
 21         /**
 22          * <br>true:viewport限制 <br> node:限制在该节点范围
 23          * @type HTMLElement|boolean
 24          */
 25         constrain:{
 26             //不限制
 27             //true:viewport限制
 28             //node:限制在节点范围
 29             value:false
 30         }
 31     };
 32 
 33     /**
 34      * 获取受限区域的宽高, 位置
 35      * @return {Object | undefined} {left: 0, top: 0, maxLeft: 100, maxTop: 100}
 36      */
 37     function _getConstrainRegion(constrain) {
 38         var ret;
 39         if (!constrain) {
 40             return ret;
 41         }
 42         var el = this.get("el");
 43         if (constrain !== true) {
 44             constrain = Node.one(constrain);
 45             ret = constrain.offset();
 46             S.mix(ret, {
 47                 maxLeft:ret.left + constrain.outerWidth() - el.outerWidth(),
 48                 maxTop:ret.top + constrain.outerHeight() - el.outerHeight()
 49             });
 50         }
 51         // 没有指定 constrain, 表示受限于可视区域
 52         else {
 53             //不要使用 viewportWidth()
 54             //The innerWidth attribute, on getting,
 55             //must return the viewport width including the size of a rendered scroll bar (if any).
 56             //On getting, the clientWidth attribute returns the viewport width
 57             //excluding the size of a rendered scroll bar (if any)
 58             //  if the element is the root element 
 59             var vWidth = S.Env.host.document.documentElement.clientWidth;
 60             ret = { left:DOM.scrollLeft(), top:DOM.scrollTop() };
 61             S.mix(ret, {
 62                 maxLeft:ret.left + vWidth - el.outerWidth(),
 63                 maxTop:ret.top + DOM.viewportHeight() - el.outerHeight()
 64             });
 65         }
 66 
 67         return ret;
 68     }
 69 
 70     Constrain.prototype = {
 71 
 72         __renderUI:function () {
 73             var self = this,
 74                 attrs = self.getAttrs(),
 75                 xAttr = attrs["x"],
 76                 yAttr = attrs["y"],
 77                 oriXSetter = xAttr["setter"],
 78                 oriYSetter = yAttr["setter"];
 79             xAttr.setter = function (v) {
 80                 var r = oriXSetter && oriXSetter.call(self, v);
 81                 if (r === undefined) {
 82                     r = v;
 83                 }
 84                 if (!self.get("constrain")) {
 85                     return r;
 86                 }
 87                 var _ConstrainExtRegion = _getConstrainRegion.call(
 88                     self, self.get("constrain"));
 89                 return Math.min(Math.max(r,
 90                     _ConstrainExtRegion.left),
 91                     _ConstrainExtRegion.maxLeft);
 92             };
 93             yAttr.setter = function (v) {
 94                 var r = oriYSetter && oriYSetter.call(self, v);
 95                 if (r === undefined) {
 96                     r = v;
 97                 }
 98                 if (!self.get("constrain")) {
 99                     return r;
100                 }
101                 var _ConstrainExtRegion = _getConstrainRegion.call(
102                     self, self.get("constrain"));
103                 return Math.min(Math.max(r,
104                     _ConstrainExtRegion.top),
105                     _ConstrainExtRegion.maxTop);
106             };
107             self.addAttr("x", xAttr);
108             self.addAttr("y", yAttr);
109         }
110     };
111 
112 
113     return Constrain;
114 
115 }, {
116     requires:["dom", "node"]
117 });
