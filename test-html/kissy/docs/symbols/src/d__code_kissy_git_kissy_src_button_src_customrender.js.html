1 /**
  2  * @fileOverview view for button , double div for pseudo-round corner
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("button/customrender", function (S, Node, UIBase, ButtonRender) {
  6 
  7     //双层 div 模拟圆角
  8     var CONTENT_CLS = "button-outer-box",
  9         INNER_CLS = "button-inner-box";
 10 
 11 
 12     return UIBase.create(ButtonRender, [UIBase.ContentBox.Render], {
 13 
 14             /**
 15              *  Controller 会在 create 后进行 unselectable，
 16              *  需要所有的节点创建工作放在 createDom 中
 17              */
 18             createDom:function () {
 19                 var self = this,
 20                     el = self.get("el"),
 21                     contentEl = self.get("contentEl"),
 22                     id = S.guid('ks-button-labelby');
 23                 el.attr("aria-labelledby", id);
 24                 //按钮的描述节点在最内层，其余都是装饰
 25                 contentEl.addClass(self.getCls(CONTENT_CLS));
 26                 var elChildren = S.makeArray(contentEl[0].childNodes),
 27                     innerEl = new Node("<div id='" + id + "' " +
 28                         "class='" + self.getCls(INNER_CLS) + "'/>")
 29                         .appendTo(contentEl);
 30                 // content 由 contentboxrender 处理
 31                 for (var i = 0; i < elChildren.length; i++) {
 32                     innerEl.append(elChildren[i]);
 33                 }
 34                 self.__set("innerEl", innerEl);
 35             },
 36 
 37             /**
 38              * 内容移到内层
 39              * @override
 40              * @param v
 41              */
 42             _uiSetContent:function (v) {
 43                 this.get("innerEl").html(v);
 44             }
 45         }, {
 46             /**
 47              * @inheritedDoc
 48              * content:{}
 49              */
 50             innerEl:{}
 51         }
 52     );
 53 }, {
 54     requires:['node', 'uibase', './buttonrender']
 55 });
