1 /**
  2  *  @fileOverview menu where items can be filtered based on user keyboard input
  3  *  @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/filtermenu", function (S, UIBase, Component, Menu, FilterMenuRender) {
  6 
  7     var HIT_CLS = "menuitem-hit";
  8 
  9     // 转义正则特殊字符,返回字符串用来构建正则表达式
 10     function regExpEscape(s) {
 11         return s.replace(/([-()\[\]{}+?*.$\^|,:#<!\\])/g, '\\$1').
 12             replace(/\x08/g, '\\x08');
 13     }
 14 
 15     var FilterMenu = UIBase.create(Menu, {
 16             bindUI:function () {
 17                 var self = this,
 18                     view = self.get("view"),
 19                     filterInput = view.get("filterInput");
 20                 /*监控键盘事件*/
 21                 filterInput.on("keyup", self.handleFilterEvent, self);
 22             },
 23 
 24             _handleMouseEnter:function () {
 25                 var self = this;
 26                 FilterMenu.superclass._handleMouseEnter.apply(self, arguments);
 27                 // 权益解决,filter input focus 后会滚动到牌聚焦处,select 则不会
 28                 // 如果 filtermenu 的菜单项被滚轮滚到后面,点击触发不了,会向前滚动到 filter input
 29                 self.getKeyEventTarget()[0].select();
 30             },
 31 
 32             handleFilterEvent:function () {
 33                 var self = this,
 34                     view = self.get("view"),
 35                     filterInput = view.get("filterInput"),
 36                     highlightedItem = self.get("highlightedItem");
 37                 /* 根据用户输入过滤 */
 38                 self.set("filterStr", filterInput.val());
 39 
 40                 // 如果没有高亮项或者高亮项因为过滤被隐藏了
 41                 // 默认选择符合条件的第一项
 42                 if (!highlightedItem || !highlightedItem.get("visible")) {
 43                     self.set("highlightedItem",
 44                         self._getNextEnabledHighlighted(0, 1));
 45                 }
 46             },
 47 
 48             _uiSetFilterStr:function (v) {
 49                 // 过滤条件变了立即过滤
 50                 this.filterItems(v);
 51             },
 52 
 53             filterItems:function (str) {
 54                 var self = this,
 55                     view = self.get("view"),
 56                     _labelEl = view.get("labelEl"),
 57                     filterInput = view.get("filterInput");
 58 
 59                 // 有过滤条件提示隐藏,否则提示显示
 60                 _labelEl[str ? "hide" : "show"]();
 61 
 62                 if (self.get("allowMultiple")) {
 63                     var enteredItems = [],
 64                         lastWord;
 65                     // \uff0c => ，
 66                     var match = str.match(/(.+)[,\uff0c]\s*([^\uff0c,]*)/);
 67                     // 已经确认的项
 68                     // , 号之前的项必定确认
 69 
 70                     var items = [];
 71 
 72                     if (match) {
 73                         items = match[1].split(/[,\uff0c]/);
 74                     }
 75 
 76                     // 逗号结尾
 77                     // 如果可以补全,那么补全最后一项为第一个高亮项
 78                     if (/[,\uff0c]$/.test(str)) {
 79                         enteredItems = [];
 80                         if (match) {
 81                             enteredItems = items;
 82                             //待补全的项
 83                             lastWord = items[items.length - 1];
 84                             var item = self.get("highlightedItem"),
 85                                 content = item && item.get("content");
 86                             // 有高亮而且最后一项不为空补全
 87                             if (content && content.indexOf(lastWord) > -1
 88                                 && lastWord) {
 89                                 enteredItems[enteredItems.length - 1] = content;
 90                             }
 91                             filterInput.val(enteredItems.join(",") + ",");
 92                         }
 93                         str = '';
 94                     } else {
 95                         // 需要菜单过滤的过滤词,在最后一个 , 后面
 96                         if (match) {
 97                             str = match[2] || "";
 98                         }
 99                         // 没有 , 则就是当前输入的
100                         // else{ str=str}
101 
102                         //记录下
103                         enteredItems = items;
104                     }
105                     var oldEnteredItems = self.get("enteredItems");
106                     // 发生变化,长度变化和内容变化等同
107                     if (oldEnteredItems.length != enteredItems.length) {
108                         S.log("enteredItems : ");
109                         S.log(enteredItems);
110                         self.set("enteredItems", enteredItems);
111                     }
112                 }
113 
114                 var children = self.get("children"),
115                     strExp = str && new RegExp(regExpEscape(str), "ig"),
116                     // 匹配项样式类
117                     hit = this.getCls(HIT_CLS);
118 
119                 // 过滤所有子组件
120                 S.each(children, function (c) {
121                     var content = c.get("content");
122                     if (!str) {
123                         // 没有过滤条件
124                         // 恢复原有内容
125                         // 显示出来
126                         c.get("contentEl").html(content);
127                         c.set("visible", true);
128                     } else {
129                         if (content.indexOf(str) > -1) {
130                             // 如果符合过滤项
131                             // 显示
132                             c.set("visible", true);
133                             // 匹配子串着重 wrap
134                             c.get("contentEl").html(content.replace(strExp, function (m) {
135                                 return "<span class='" + hit + "'>" + m + "<" + "/span>";
136                             }));
137                         } else {
138                             // 不符合
139                             // 隐藏
140                             c.set("visible", false);
141                         }
142                     }
143                 });
144             },
145 
146             decorateInternal:function (el) {
147                 var self = this;
148                 self.set("el", el);
149                 var menuContent = el.one("." + self.getCls("menu-content"));
150                 self.decorateChildren(menuContent);
151             },
152 
153             /**
154              * 重置状态,用于重用
155              */
156             reset:function () {
157                 var self = this,
158                     view = self.get("view");
159                 self.set("filterStr", "");
160                 self.set("enteredItems", []);
161                 var filterInput = view && view.get("filterInput");
162                 filterInput && filterInput.val("");
163             },
164 
165             destructor:function () {
166                 var view = this.get("view");
167                 var filterInput = view && view.get("filterInput");
168                 filterInput && filterInput.detach();
169             }
170 
171         },
172         {
173             ATTRS:{
174                 label:{
175                     view:true
176                 },
177 
178                 filterStr:{
179                 },
180 
181                 enteredItems:{
182                     value:[]
183                 },
184 
185                 allowMultiple:{
186                     value:false
187                 }
188             },
189             DefaultRender:FilterMenuRender
190         }
191     );
192 
193     Component.UIStore.setUIByClass("filtermenu", {
194         priority:Component.UIStore.PRIORITY.LEVEL2,
195         ui:FilterMenu
196     });
197 
198     return FilterMenu;
199 }, {
200     requires:['uibase', 'component', './base', './filtermenurender']
201 });
