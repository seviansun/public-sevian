  5 KISSY.add("overlay/aria", function(S,Event) {
  6     function Aria() {
  7     }
  8 
  9     Aria.ATTRS = {
 10         aria:{
 11             view:true
 12         }
 13     };
 14 
 15     Aria.prototype = {
 16 
 17         __bindUI:function() {
 18             var self = this,el = self.get("el");
 19             if (self.get("aria")) {
 20                 el.on("keydown", function(e) {
 21                     if (e.keyCode === Event.KeyCodes.ESC) {
 22                         self.hide();
 23                         e.halt();
 24                     }
 25                 });
 26             }
 27         }
 28     };
 29     return Aria;
 30 },{
 31     requires:['event']
 32 });
