1 /**
  2  * @fileOverview animate on single property
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("anim/fx", function (S, DOM, undefined) {
  6 
  7     /**
  8      * basic animation about single css property or element attribute
  9      * @param cfg
 10      */
 11     function Fx(cfg) {
 12         this.load(cfg);
 13     }
 14 
 15     S.augment(Fx, {
 16 
 17         load:function (cfg) {
 18             var self = this;
 19             S.mix(self, cfg);
 20             self.startTime = S.now();
 21             self.pos = 0;
 22             self.unit = self.unit || "";
 23         },
 24 
 25         frame:function (end) {
 26             var self = this,
 27                 endFlag = 0,
 28                 elapsedTime;
 29             if (self.finished) {
 30                 return 1;
 31             }
 32             var t = S.now();
 33             if (end || t >= self.duration + self.startTime) {
 34                 self.pos = 1;
 35                 endFlag = 1;
 36             } else {
 37                 elapsedTime = t - self.startTime;
 38                 self.pos = self.easing(elapsedTime / self.duration);
 39             }
 40             self.update();
 41             self.finished = self.finished || endFlag;
 42             return endFlag;
 43         },
 44 
 45         /**
 46          * 数值插值函数
 47          * @param {Number} from 源值
 48          * @param {Number} to 目的值
 49          * @param {Number} pos 当前位置，从 easing 得到 0~1
 50          * @return {Number} 当前值
 51          */
 52         interpolate:function (from, to, pos) {
 53             // 默认只对数字进行 easing
 54             if (S.isNumber(from) &&
 55                 S.isNumber(to)) {
 56                 return (from + (to - from) * pos).toFixed(3);
 57             } else {
 58                 return undefined;
 59             }
 60         },
 61 
 62         update:function () {
 63             var self = this,
 64                 prop = self.prop,
 65                 elem = self.elem,
 66                 from = self.from,
 67                 to = self.to,
 68                 val = self.interpolate(from, to, self.pos);
 69 
 70             if (val === undefined) {
 71                 // 插值出错，直接设置为最终值
 72                 if (!self.finished) {
 73                     self.finished = 1;
 74                     DOM.css(elem, prop, to);
 75                     S.log(self.prop + " update directly ! : " + val + " : " + from + " : " + to);
 76                 }
 77             } else {
 78                 val += self.unit;
 79                 if (isAttr(elem, prop)) {
 80                     DOM.attr(elem, prop, val, 1);
 81                 } else {
 82                     DOM.css(elem, prop, val);
 83                 }
 84             }
 85         },
 86 
 87         /**
 88          * current value
 89          */
 90         cur:function () {
 91             var self = this,
 92                 prop = self.prop,
 93                 elem = self.elem;
 94             if (isAttr(elem, prop)) {
 95                 return DOM.attr(elem, prop, undefined, 1);
 96             }
 97             var parsed,
 98                 r = DOM.css(elem, prop);
 99             // Empty strings, null, undefined and "auto" are converted to 0,
100             // complex values such as "rotate(1rad)" or "0px 10px" are returned as is,
101             // simple values such as "10px" are parsed to Float.
102             return isNaN(parsed = parseFloat(r)) ?
103                 !r || r === "auto" ? 0 : r
104                 : parsed;
105         }
106     });
107 
108     function isAttr(elem, prop) {
109         // support scrollTop/Left now!
110         if ((!elem.style || elem.style[ prop ] == null) &&
111             DOM.attr(elem, prop, undefined, 1) != null) {
112             return 1;
113         }
114         return 0;
115     }
116 
117     Fx.Factories = {};
118 
119     Fx.getFx = function (cfg) {
120         var Constructor = Fx.Factories[cfg.prop] || Fx;
121         return new Constructor(cfg);
122     };
123 
124     return Fx;
125 
126 }, {
127     requires:['dom']
128 });
