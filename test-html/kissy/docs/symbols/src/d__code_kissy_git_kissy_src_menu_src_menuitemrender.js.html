1 /**
  2  * @fileOverview simple menuitem render
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/menuitemrender", function (S, Node, UIBase, Component) {
  6 
  7     var CHECK_CLS = "menuitem-checkbox",
  8         CONTENT_CLS = "menuitem-content";
  9 
 10     function setUpCheckEl(self) {
 11         var el = self.get("el"),
 12             cls = self.getCls(CHECK_CLS),
 13             checkEl = el.one("." + cls);
 14         if (!checkEl) {
 15             checkEl = new Node("<div class='" + cls + "'/>").prependTo(el);
 16             // if not ie will lose focus when click
 17             checkEl.unselectable();
 18         }
 19         return checkEl;
 20     }
 21 
 22     return UIBase.create(Component.Render, [UIBase.ContentBox.Render], {
 23 
 24         _setSelected:function (v, componentCls) {
 25             var self = this,
 26                 tag = "-selected",
 27                 el = self.get("el"),
 28                 cls = self._completeClasses(componentCls, tag);
 29             el[v ? 'addClass' : 'removeClass'](cls);
 30         },
 31 
 32         _setChecked:function (v, componentCls) {
 33             var self = this,
 34                 tag = "-checked",
 35                 el = self.get("el"),
 36                 cls = self._completeClasses(componentCls, tag);
 37             el[v ? 'addClass' : 'removeClass'](cls);
 38         },
 39 
 40         _uiSetSelectable:function (v) {
 41             this.get("el").attr("role", v ? 'menuitemradio' : 'menuitem');
 42         },
 43 
 44         _uiSetCheckable:function (v) {
 45             if (v) {
 46                 setUpCheckEl(this);
 47             }
 48             this.get("el").attr("role", v ? 'menuitemcheckbox' : 'menuitem');
 49         },
 50 
 51         containsElement:function (element) {
 52             var el = this.get("el");
 53             return el[0] == element || el.contains(element);
 54         }
 55     }, {
 56         ATTRS:{
 57             contentElCls:{
 58                 valueFn:function () {
 59                     return this.getCls(CONTENT_CLS);
 60                 }
 61             },
 62             elAttrs:{
 63                 valueFn:function () {
 64                     return {
 65                         role:"menuitem",
 66                         id:S.guid("ks-menuitem")
 67                     };
 68                 }
 69             },
 70             selected:{},
 71             // @inheritedDoc
 72             // content:{},
 73             // 属性必须声明，否则无法和 _uiSetChecked 绑定在一起
 74             checked:{}
 75         }
 76     });
 77 }, {
 78     requires:['node', 'uibase', 'component']
 79 });
