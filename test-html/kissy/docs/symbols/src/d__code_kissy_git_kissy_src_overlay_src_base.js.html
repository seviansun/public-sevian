1 /**
  2  * @fileOverview model and control for overlay
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("overlay/base", function (S, UIBase, Component, OverlayRender, Effect) {
  6 
  7     function require(s) {
  8         return S.require("uibase/" + s);
  9     }
 10 
 11     /**
 12      * KISSY Overlay Component
 13      * @class
 14      * @namespace
 15      * @name Overlay
 16      * @extends Component.Controller
 17      * @extends UIBase.ContentBox
 18      * @extends UIBase.Position
 19      * @extends UIBase.Loading
 20      * @extends UIBase.Align
 21      * @extends UIBase.Close
 22      * @extends UIBase.Resize
 23      * @extends UIBase.Mask
 24      * @param {Object} config config object to set properties of its parent class
 25      */
 26     var Overlay = UIBase.create(Component.Controller, [
 27         require("contentbox"),
 28         require("position"),
 29         require("loading"),
 30         require("align"),
 31         require("close"),
 32         require("resize"),
 33         require("mask"),
 34         Effect
 35     ],
 36         /**
 37          * @lends Overlay#
 38          */
 39         {
 40             /**
 41              * see {@link UIBase.Box#show}
 42              * @function
 43              * @name Overlay#show
 44              */
 45         }, {
 46             ATTRS:/**
 47              * @lends Overlay#
 48              */
 49             {
 50                 /**
 51                  * whether this component can be focused. Default:false
 52                  * @type Boolean
 53                  */
 54                 focusable:{
 55                     value:false
 56                 },
 57 
 58                 /**
 59                  * whether this component can be closed. Default:false
 60                  * @type Boolean
 61                  */
 62                 closable:{
 63                     value:false
 64                 },
 65 
 66                 /**
 67                  * whether this component can be responsive to mouse. Default:false
 68                  * @type Boolean
 69                  */
 70                 handleMouseEvents:{
 71                     value:false
 72                 },
 73 
 74                 /**
 75                  * whether this component's text content can be selected. Default:true
 76                  * @type Boolean
 77                  */
 78                 allowTextSelection_:{
 79                     value:true
 80                 },
 81 
 82                 /**
 83                  * see {@linl UIBase.Box#visibleMode}. Default:"visibility"
 84                  */
 85                 visibleMode:{
 86                     value:"visibility"
 87                 }
 88             }
 89         });
 90 
 91     Overlay.DefaultRender = OverlayRender;
 92 
 93     Component.UIStore.setUIByClass("overlay", {
 94         priority:Component.UIStore.PRIORITY.LEVEL1,
 95         ui:Overlay
 96     });
 97 
 98     return Overlay;
 99 }, {
100     requires:['uibase', 'component', './overlayrender', './effect']
101 });
