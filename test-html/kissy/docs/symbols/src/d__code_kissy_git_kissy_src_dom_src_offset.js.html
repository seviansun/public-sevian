 41             /**
 42              * Get the current coordinates of the first element in the set of matched elements, relative to the document.
 43              * or
 44              * Set the current coordinates of every element in the set of matched elements, relative to the document.
 45              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
 46              * @param {Object} [coordinates ] An object containing the properties top and left,
 47              * which are integers indicating the new top and left coordinates for the elements.
 48              * @param {Number} [coordinates.left ] the new top and left coordinates for the elements.
 49              * @param {Number} [coordinates.top ] the new top and top coordinates for the elements.
 50              * @param {window} [relativeWin] The window to measure relative to. If relativeWin
 51              *     is not in the ancestor frame chain of the element, we measure relative to
 52              *     the top-most window.
 53              * @returns {Object|undefined} if Get, the format of returned value is same with coordinates.
 54              */
 55             offset:function (selector, coordinates, relativeWin) {
 56                 // getter
 57                 if (coordinates === undefined) {
 58                     var elem = DOM.get(selector), ret;
 59                     if (elem) {
 60                         ret = getOffset(elem, relativeWin);
 61                     }
 62                     return ret;
 63                 }
 64                 // setter
 65                 var els = DOM.query(selector), i;
 66                 for (i = els.length - 1; i >= 0; i--) {
 67                     elem = els[i];
 68                     setOffset(elem, coordinates);
 69                 }
 70                 return undefined;
 71             },
 72 
 73             /**
 74              * Makes the first of matched elements visible in the container
 75              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
 76              * @param {String|HTMLElement} [container=window] Container element
 77              * @param {Boolean} [top=true] Whether align with top of container.
 78              * @param {Boolean} [hscroll=true] Whether trigger horizontal scroll.
 79              * @param {Boolean} [auto=false] Whether adjust element automatically
 80              * (only scrollIntoView when element is out of view)
 81              * @see http://www.w3.org/TR/2009/WD-html5-20090423/editing.html#scrollIntoView
 82              *        http://www.sencha.com/deploy/dev/docs/source/Element.scroll-more.html#scrollIntoView
 83              *        http://yiminghe.javaeye.com/blog/390732
 84              */
 85             scrollIntoView:function (selector, container, top, hscroll, auto) {
 86                 var elem;
 87 
 88                 if (!(elem = DOM.get(selector))) {
 89                     return;
 90                 }
 91 
 92                 if (container) {
 93                     container = DOM.get(container);
 94                 }
 95 
 96                 if (!container) {
 97                     container = elem.ownerDocument;
 98                 }
 99 
100                 if (auto !== true) {
101                     hscroll = hscroll === undefined ? true : !!hscroll;
102                     top = top === undefined ? true : !!top;
103                 }
104 
105                 // document 归一化到 window
106                 if (nodeTypeIs(container, DOM.DOCUMENT_NODE)) {
107                     container = getWin(container);
108                 }
109 
110                 var isWin = !!getWin(container),
111                     elemOffset = DOM.offset(elem),
112                     eh = DOM.outerHeight(elem),
113                     ew = DOM.outerWidth(elem),
114                     containerOffset,
115                     ch,
116                     cw,
117                     containerScroll,
118                     diffTop,
119                     diffBottom,
120                     win,
121                     winScroll,
122                     ww,
123                     wh;
124 
125                 if (isWin) {
126                     win = container;
127                     wh = DOM.height(win);
128                     ww = DOM.width(win);
129                     winScroll = {
130                         left:DOM.scrollLeft(win),
131                         top:DOM.scrollTop(win)
132                     };
133                     // elem 相对 container 可视视窗的距离
134                     diffTop = {
135                         left:elemOffset[LEFT] - winScroll[LEFT],
136                         top:elemOffset[TOP] - winScroll[TOP]
137                     };
138                     diffBottom = {
139                         left:elemOffset[LEFT] + ew - (winScroll[LEFT] + ww),
140                         top:elemOffset[TOP] + eh - (winScroll[TOP] + wh)
141                     };
142                     containerScroll = winScroll;
143                 }
144                 else {
145                     containerOffset = DOM.offset(container);
146                     ch = container.clientHeight;
147                     cw = container.clientWidth;
148                     containerScroll = {
149                         left:DOM.scrollLeft(container),
150                         top:DOM.scrollTop(container)
151                     };
152                     // elem 相对 container 可视视窗的距离
153                     // 注意边框 , offset 是边框到根节点
154                     diffTop = {
155                         left:elemOffset[LEFT] - containerOffset[LEFT] -
156                             (PARSEINT(DOM.css(container, 'borderLeftWidth')) || 0),
157                         top:elemOffset[TOP] - containerOffset[TOP] -
158                             (PARSEINT(DOM.css(container, 'borderTopWidth')) || 0)
159                     };
160                     diffBottom = {
161                         left:elemOffset[LEFT] + ew -
162                             (containerOffset[LEFT] + cw +
163                                 (PARSEINT(DOM.css(container, 'borderRightWidth')) || 0)),
164                         top:elemOffset[TOP] + eh -
165                             (containerOffset[TOP] + ch +
166                                 (PARSEINT(DOM.css(container, 'borderBottomWidth')) || 0))
167                     };
168                 }
169 
170                 if (diffTop.top < 0 || diffBottom.top > 0) {
171                     // 强制向上
172                     if (top === true) {
173                         DOM.scrollTop(container, containerScroll.top + diffTop.top);
174                     } else if (top === false) {
175                         DOM.scrollTop(container, containerScroll.top + diffBottom.top);
176                     } else {
177                         // 自动调整
178                         if (diffTop.top < 0) {
179                             DOM.scrollTop(container, containerScroll.top + diffTop.top);
180                         } else {
181                             DOM.scrollTop(container, containerScroll.top + diffBottom.top);
182                         }
183                     }
184                 }
185 
186                 if (hscroll) {
187                     if (diffTop.left < 0 || diffBottom.left > 0) {
188                         // 强制向上
189                         if (top === true) {
190                             DOM.scrollLeft(container, containerScroll.left + diffTop.left);
191                         } else if (top === false) {
192                             DOM.scrollLeft(container, containerScroll.left + diffBottom.left);
193                         } else {
194                             // 自动调整
195                             if (diffTop.left < 0) {
196                                 DOM.scrollLeft(container, containerScroll.left + diffTop.left);
197                             } else {
198                                 DOM.scrollLeft(container, containerScroll.left + diffBottom.left);
199                             }
200                         }
201                     }
202                 }
203             },

248     // http://old.jr.pl/www.quirksmode.org/viewport/compatibility.html
249     // http://www.quirksmode.org/dom/w3c_cssom.html
250     // add ScrollLeft/ScrollTop getter/setter methods
251     S.each(['Left', 'Top'], function (name, i) {
252         var method = SCROLL + name;
253 
254         DOM[method] = function (elem, v) {
255             if (isNumber(elem)) {
256                 return arguments.callee(win, elem);
257             }
258             elem = DOM.get(elem);
259             var ret,
260                 w = getWin(elem),
261                 d;
262             if (w) {
263                 if (v !== undefined) {
264                     v = parseFloat(v);
265                     // 注意多 windw 情况，不能简单取 win
266                     var left = name == "Left" ? v : DOM.scrollLeft(w),
267                         top = name == "Top" ? v : DOM.scrollTop(w);
268                     w['scrollTo'](left, top);
269                 } else {
270                     //标准
271                     //chrome == body.scrollTop
272                     //firefox/ie9 == documentElement.scrollTop
273                     ret = w[ 'page' + (i ? 'Y' : 'X') + 'Offset'];
274                     if (!isNumber(ret)) {
275                         d = w[DOCUMENT];
276                         //ie6,7,8 standard mode
277                         ret = d[DOC_ELEMENT][method];
278                         if (!isNumber(ret)) {
279                             //quirks mode
280                             ret = d[BODY][method];
281                         }
282                     }
283                 }
284             } else if (isElementNode(elem)) {
285                 if (v !== undefined) {
286                     elem[method] = parseFloat(v)
287                 } else {
288                     ret = elem[method];
289                 }
290             }
291             return ret;
292         }

332     function getClientPosition(elem) {
333         var box, x = 0, y = 0,
334             body = doc.body,
335             w = getWin(elem[OWNER_DOCUMENT]);
336 
337         // 根据 GBS 最新数据，A-Grade Browsers 都已支持 getBoundingClientRect 方法，不用再考虑传统的实现方式
338         if (elem[GET_BOUNDING_CLIENT_RECT]) {
339             box = elem[GET_BOUNDING_CLIENT_RECT]();
340 
341             // 注：jQuery 还考虑减去 docElem.clientLeft/clientTop
342             // 但测试发现，这样反而会导致当 html 和 body 有边距/边框样式时，获取的值不正确
343             // 此外，ie6 会忽略 html 的 margin 值，幸运地是没有谁会去设置 html 的 margin
344 
345             x = box[LEFT];
346             y = box[TOP];
347 
348             // ie 下应该减去窗口的边框吧，毕竟默认 absolute 都是相对窗口定位的
349             // 窗口边框标准是设 documentElement ,quirks 时设置 body
350             // 最好禁止在 body 和 html 上边框 ，但 ie < 9 html 默认有 2px ，减去
351             // 但是非 ie 不可能设置窗口边框，body html 也不是窗口 ,ie 可以通过 html,body 设置
352             // 标准 ie 下 docElem.clientTop 就是 border-top
353             // ie7 html 即窗口边框改变不了。永远为 2
354 
355             // 但标准 firefox/chrome/ie9 下 docElem.clientTop 是窗口边框，即使设了 border-top 也为 0
356             var clientTop = isIE && doc['documentMode'] != 9
357                 && (isStrict ? docElem.clientTop : body.clientTop)
358                 || 0,
359                 clientLeft = isIE && doc['documentMode'] != 9
360                     && (isStrict ? docElem.clientLeft : body.clientLeft)
361                     || 0;
362             if (1 > 2) {
363             }
364             x -= clientLeft;
365             y -= clientTop;
366 
367             // iphone/ipad/itouch 下的 Safari 获取 getBoundingClientRect 时，已经加入 scrollTop
368             if (UA.mobile == 'apple') {
369                 x -= DOM[SCROLL_LEFT](w);
370                 y -= DOM[SCROLL_TOP](w);
371             }
372         }
373 
374         return { left:x, top:y };
375     }

378     function getPageOffset(el) {
379         var pos = getClientPosition(el);
380         var w = getWin(el[OWNER_DOCUMENT]);
381         pos.left += DOM[SCROLL_LEFT](w);
382         pos.top += DOM[SCROLL_TOP](w);
383         return pos;
384     }

386     // 获取 elem 相对 elem.ownerDocument 的坐标
387     function getOffset(el, relativeWin) {
388         var position = {left:0, top:0};
389 
390         // Iterate up the ancestor frame chain, keeping track of the current window
391         // and the current element in that window.
392         var currentWin = getWin(el[OWNER_DOCUMENT]);
393         var currentEl = el;
394         relativeWin = relativeWin || currentWin;
395         do {
396             // if we're at the top window, we want to get the page offset.
397             // if we're at an inner frame, we only want to get the window position
398             // so that we can determine the actual page offset in the context of
399             // the outer window.
400             var offset = currentWin == relativeWin ?
401                 getPageOffset(currentEl) :
402                 getClientPosition(currentEl);
403             position.left += offset.left;
404             position.top += offset.top;
405         } while (currentWin && currentWin != relativeWin &&
406             (currentEl = currentWin['frameElement']) &&
407             (currentWin = currentWin.parent));
408 
409         return position;
410     }
411 
412     // 设置 elem 相对 elem.ownerDocument 的坐标
413     function setOffset(elem, offset) {
414         // set position first, in-case top/left are set even on static elem
415         if (DOM.css(elem, POSITION) === 'static') {
416             elem.style[POSITION] = RELATIVE;
417         }
418         var old = getOffset(elem), ret = { }, current, key;
419 
420         for (key in offset) {
421             current = PARSEINT(DOM.css(elem, key), 10) || 0;
422             ret[key] = current + offset[key] - old[key];
423         }
424         DOM.css(elem, ret);
425     }
