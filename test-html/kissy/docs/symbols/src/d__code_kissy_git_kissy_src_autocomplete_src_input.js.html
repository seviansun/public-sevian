1 /**
  2  * input wrapper for autoComplete component
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("autocomplete/input", function (S, Event, UIBase, Component, Menu, AutoCompleteRender) {
  6     var AutoComplete,
  7         KeyCodes = Event.KeyCodes;
  8 
  9     /**
 10      * Input/Textarea Wrapper for autoComplete
 11      * @name AutoComplete
 12      * @extends Component.Controller
 13      * @class
 14      */
 15     AutoComplete = UIBase.create(Component.Controller, [],
 16         /**
 17          * @lends AutoComplete
 18          */
 19         {
 20             __CLASS:"AutoComplete",
 21 
 22             // user's input text
 23             _savedInputValue:null,
 24 
 25             _stopNotify:0,
 26 
 27             bindUI:function () {
 28                 var self = this,
 29                     el = self.get("el");
 30                 el.on("valuechange", self._onValueChange, self);
 31             },
 32             /**
 33              * fetch autoComplete list by value and show autoComplete list
 34              * @param {String} value value for fetching autoComplete list
 35              */
 36             sendRequest:function (value) {
 37                 var self = this,
 38                     dataSource = self.get("dataSource");
 39                 dataSource.fetchData(value, self._renderData, self);
 40             },
 41             _onValueChange:function () {
 42                 var self = this;
 43                 if (self._stopNotify) {
 44                     return;
 45                 }
 46                 self._savedInputValue = self.get('el').val();
 47                 self.sendRequest(self._savedInputValue);
 48             },
 49 
 50             _handleBlur:function () {
 51                 AutoComplete.superclass._handleBlur.apply(this, arguments);
 52                 var autoCompleteMenu = this.get("menu");
 53                 // S.log("input blur!!!!!!!");
 54                 if (autoCompleteMenu) {
 55                     // 通知 menu
 56                     autoCompleteMenu._onInputBlur();
 57                 }
 58             },
 59 
 60             _renderData:function (data) {
 61                 var self = this,
 62                     autoCompleteMenu = self.get("menu");
 63                 autoCompleteMenu.removeChildren(true);
 64                 if (data && data.length) {
 65                     data = data.slice(0, self.get("maxItemCount"));
 66                     var menuCfg = self.get("menuCfg") || {};
 67                     var v;
 68                     // 同步当前 width
 69                     autoCompleteMenu.set("width", menuCfg.width || self.get("el").css("width"));
 70                     var contents = [];
 71                     if (self.get("format")) {
 72                         contents = self.get("format").call(self,
 73                             self.get("el").val(),
 74                             data);
 75                     }
 76                     for (var i = 0; i < data.length; i++) {
 77                         v = data[i];
 78                         autoCompleteMenu.addChild(new Menu.Item(S.mix({
 79                             prefixCls:self.get("prefixCls") + "autocomplete-",
 80                             content:v,
 81                             textContent:v,
 82                             value:v
 83                         }, contents[i])))
 84                     }
 85                     autoCompleteMenu._showForAutoComplete(self);
 86                 } else {
 87                     autoCompleteMenu.hide();
 88                 }
 89             },
 90 
 91             _handleKeyEventInternal:function (e) {
 92                 var self = this,
 93                     el = self.get("el"),
 94                     autoCompleteMenu = self.get("menu");
 95                 if (!autoCompleteMenu) {
 96                     return;
 97                 }
 98 
 99                 // autocomplete will change input value
100                 // but it does not need to reload data
101                 if (S.inArray(e.keyCode, [
102                     KeyCodes.UP,
103                     KeyCodes.DOWN,
104                     KeyCodes.ESC
105                 ])) {
106                     self._stopNotify = 1;
107                 } else {
108                     self._stopNotify = 0;
109                 }
110                 var activeItem;
111                 if (autoCompleteMenu.get("visible")) {
112                     var handledByMenu = autoCompleteMenu._handleKeydown(e);
113 
114                     if (S.inArray(e.keyCode, [KeyCodes.DOWN, KeyCodes.UP])) {
115                         // update menu's active value to input just for show
116                         el.val(autoCompleteMenu.get("activeItem").get("textContent"))
117                     }
118                     // esc
119                     if (e.keyCode == KeyCodes.ESC) {
120                         autoCompleteMenu.hide();
121                         // restore original user's input text
122                         el.val(self._savedInputValue);
123                         return true;
124                     }
125                     // tab
126                     // if menu is open and an menuitem is highlighted, see as click/enter
127                     if (e.keyCode == KeyCodes.TAB) {
128                         if (activeItem = autoCompleteMenu.get("activeItem")) {
129                             activeItem._performInternal();
130                             return true;
131                         }
132                     }
133                     return handledByMenu;
134                 } else if (e.keyCode == KeyCodes.DOWN || e.keyCode == KeyCodes.UP) {
135                     if (autoCompleteMenu.get("children").length) {
136                         autoCompleteMenu._showForAutoComplete(self);
137                         return true;
138                     } else {
139                         self.sendRequest(el.val());
140                     }
141                 }
142             },
143 
144             _uiSetSelectedItem:function (item) {
145                 var self = this;
146                 if (item) {
147                     var textContent = item.get("textContent");
148                     self.get("el").val(textContent);
149                     self._savedInputValue = textContent;
150                     /**
151                      * @name AutoComplete#select
152                      * @description fired when user select from suggestion list
153                      * @event
154                      * @param e
155                      * @param e.value value of selected menuItem
156                      * @param e.content content of selected menuItem
157                      * @param e.input current active input
158                      */
159                     self.fire("select", {
160                         value:item.get("value"),
161                         content:item.get("content"),
162                         textContent:textContent
163                     });
164                 }
165             },
166 
167             destructor:function () {
168                 var self = this,
169                     autoCompleteMenu = self.get("menu");
170                 autoCompleteMenu.detachInput(self, self.get("destroyMenu"));
171                 self.__set("menu", null);
172             }
173         }, {
174             ATTRS:/**
175              * @lends AutoComplete
176              */
177             {
178                 focusable:{
179                     value:true
180                 },
181                 handleMouseEvents:{
182                     value:false
183                 },
184                 allowTextSelection_:{
185                     value:true
186                 },
187 
188                 /**
189                  * Whether destroy menu when this destroys.Default false
190                  * @type Boolean
191                  */
192                 destroyMenu:{
193                     value:false
194                 },
195 
196                 /**
197                  * AutoComplete dropDown menuList
198                  * @type AutoComplete.Menu
199                  */
200                 menu:{
201                     setter:function (m) {
202                         if (m) {
203                             m.attachInput(this);
204                         }
205                     }
206                 },
207                 /**
208                  * aria-owns.ReadOnly.
209                  * @type String
210                  */
211                 ariaOwns:{
212                     view:true
213                 },
214                 /**
215                  * aria-expanded.ReadOnly.
216                  * @type String
217                  */
218                 ariaExpanded:{
219                     view:true
220                 },
221                 /**
222                  * dataSource for autoComplete.For Configuration when new.
223                  * @type AutoComplete.LocalDataSource|AutoComplete.RemoteDataSource
224                  */
225                 dataSource:{
226                     // 和 input 关联起来，input可以有很多，每个数据源可以不一样，但是 menu 共享
227                 },
228                 /**
229                  * maxItemCount max count of data to be shown
230                  * @type Number
231                  */
232                 maxItemCount:{value:99999},
233                 /**
234                  * Config autoComplete menu list.For Configuration when new.
235                  * {Number} menuCfg.width :
236                  * Config autoComplete menu list's alignment.
237                  * Default to current active input's width.<br/>
238                  * {Object} menuCfg.align :
239                  * Config autoComplete menu list's alignment.
240                  * Same with {@link UIBase.Align#align} .
241                  * Default : align current input's bottom left edge
242                  * with autoComplete list's top left edge.
243                  * @type Object
244                  */
245                 menuCfg:{
246                     value:{
247                         // width
248                         // align
249                     }
250                 },
251                 /**
252                  * Format function to return array of
253                  * html/text/menu item attributes from array of data.
254                  * @type {Function}
255                  */
256                 format:{},
257 
258                 /**
259                  * Readonly.User selected menu item by click or enter on highlighted suggested menu item
260                  * @type Menu.Item
261                  */
262                 selectedItem:{}
263             },
264 
265             DefaultRender:AutoCompleteRender
266         });
267 
268     Component.UIStore.setUIByClass("autocomplete-input", {
269         priority:Component.UIStore.PRIORITY.LEVEL1,
270         ui:AutoComplete
271     });
272 
273     return AutoComplete;
274 }, {
275     requires:[
276         'event',
277         'uibase', 'component',
278         'menu',
279         './inputRender']
280 });
