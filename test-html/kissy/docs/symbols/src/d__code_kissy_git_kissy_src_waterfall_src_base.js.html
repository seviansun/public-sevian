  7     var $ = Node.all,
  8         win = S.Env.host,

 11     /**
 12      * @class
 13      * @namespace
 14      * @name Waterfall
 15      */
 16     function Waterfall() {
 17         Waterfall.superclass.constructor.apply(this, arguments);
 18         this._init();
 19     }

 22     function timedChunk(items, process, context, callback) {
 23         var todo = [].concat(S.makeArray(items)),
 24             stopper = {},
 25             timer;
 26         if (todo.length > 0) {
 27             timer = setTimeout(function () {
 28                 var start = +new Date();
 29                 do {
 30                     var item = todo.shift();
 31                     process.call(context, item);
 32                 } while (todo.length > 0 && (+new Date() - start < 50));
 33 
 34                 if (todo.length > 0) {
 35                     timer = setTimeout(arguments.callee, 25);
 36                 } else {
 37                     callback && callback.call(context, items);
 38                 }
 39             }, 25);
 40         } else {
 41             callback && S.later(callback, 0, false, context, [items]);
 42         }
 43 
 44         stopper.stop = function () {
 45             if (timer) {
 46                 clearTimeout(timer);
 47                 todo = [];
 48                 items.each(function (item) {
 49                     item.stop();
 50                 });
 51             }
 52         };
 53 
 54         return stopper;
 55     }

 85         effect:{
 86             value:{
 87                 effect:"fadeIn",
 88                 duration:1

110     function doResize() {
111         var self = this,
112             containerRegion = self._containerRegion;
113         // 宽度没变就没必要调整
114         if (containerRegion &&
115             self.get("container").width() === containerRegion.width) {
116             return
117         }
118         self.adjust();
119     }
120 
121     function recalculate() {
122         var self = this,
123             container = self.get("container"),
124             containerWidth = container.width(),
125             curColHeights = self.get("curColHeights");
126         curColHeights.length = Math.max(parseInt(containerWidth / self.get("colWidth")),
127             self.get("minColCount"));
128         self._containerRegion = {
129             width:containerWidth
130         };
131         S.each(curColHeights, function (v, i) {
132             curColHeights[i] = 0;
133         });
134         self.set("colItems", []);
135     }
136 
137     function adjustItemAction(self, add, itemRaw, callback) {
138         var effect = self.get("effect"),
139             item = $(itemRaw),
140             curColHeights = self.get("curColHeights"),
141             container = self.get("container"),
142             curColCount = curColHeights.length,
143             dest = 0,
144             containerRegion = self._containerRegion,
145             guard = Number.MAX_VALUE;
146 
147         for (var i = 0; i < curColCount; i++) {
148             if (curColHeights[i] < guard) {
149                 guard = curColHeights[i];
150                 dest = i;
151             }
152         }
153 
154         if (!curColCount) {
155             guard = 0;
156         }
157 
158         // 元素保持间隔不变，居中
159         var margin = Math.max(containerRegion.width - curColCount * self.get("colWidth"), 0) / 2,
160             destProp = {
161                 // left:dest * Math.max(containerRegion.width / curColCount, self.get("colWidth"))
162                 //    + containerRegion.left,
163                 // 元素间固定间隔好点
164                 left:dest * self.get("colWidth") + margin,
165                 top:guard
166             };
167 
168         /*
169          不在容器里，就加上
170          */
171         if (add) {
172             // 初始需要动画，那么先把透明度换成 0
173             item.css(destProp);
174             if (effect && effect.effect) {
175                 item.css("visibility", "hidden");
176             }
177             container.append(item);
178             callback && callback();
179         }
180         // 否则调整，需要动画
181         else {
182             var adjustEffect = self.get("adjustEffect");
183             if (adjustEffect) {
184                 item.animate(destProp, adjustEffect.duration, adjustEffect.easing, callback);
185             } else {
186                 item.css(destProp);
187                 callback && callback();
188             }
189         }
190 
191         // 加入到 dom 树才能取得高度
192         curColHeights[dest] += item.outerHeight(true);
193         var colItems = self.get("colItems");
194         colItems[dest] = colItems[dest] || [];
195         colItems[dest].push(item);
196         item.attr("data-waterfall-col", dest);
197 
198         return item;
199     }
200 
201     function addItem(itemRaw) {
202         var self = this,
203             item = adjustItemAction(self, true, itemRaw),
204             effect = self.get("effect");
205         if (effect && effect.effect) {
206             // 先隐藏才能调用 fadeIn slideDown
207             item.hide();
208             item.css("visibility", "");
209             item[effect.effect](
210                 effect.duration,
211                 0,
212                 effect.easing
213             );
214         }
215     }

217     S.extend(Waterfall, Base,
218         /**
219          * @lends Waterfall
220          */
221         {
222             isAdjusting:function () {
223                 return !!this._adjuster;
224             },
225 
226             _init:function () {
227                 var self = this;
228                 // 一开始就 adjust 一次，可以对已有静态数据处理
229                 doResize.call(self);
230                 self.__onResize = S.buffer(doResize, RESIZE_DURATION, self);
231                 $(win).on("resize", self.__onResize);
232             },

234 
235             /**
236              * 调整一个元素的高度
237              * @param {Node} item 待调整的元素
238              * @param {Object} cfg 控制配置
239              * @param {Function} cfg.callback 调整结束后的回调
240              * @param {Function} cfg.process
241              * 用于做调整操作的函数，
242              * 可以返回数字表示调整后的高度，
243              * 不返回的话直接取调整后元素的高度
244              * @param {Object} cfg.effect 其他元素配合调整位置的特效配置
245              * @param {Number} cfg.effect.duration
246              * @param {String} cfg.effect.easing
247              */
248             adjustItem:function (item, cfg) {
249                 var self = this;
250 
251                 if (self.isAdjusting()) {
252                     return;
253                 }
254 
255                 var originalOuterHeight = item.outerHeight(true),
256                     outerHeight;
257 
258                 if (cfg.process) {
259                     outerHeight = cfg.process.call(self);
260                 }
261 
262                 if (outerHeight === undefined) {
263                     outerHeight = item.outerHeight(true);
264                 }
265 
266                 var diff = outerHeight - originalOuterHeight,
267                     curColHeights = self.get("curColHeights"),
268                     dest = parseInt(item.attr("data-waterfall-col")),
269                     colItems = self.get("colItems")[dest],
270                     items = [],
271                     original = Math.max.apply(Math, curColHeights),
272                     now;
273 
274                 for (var i = 0; i < colItems.length; i++) {
275                     if (colItems[i][0] === item[0]) {
276                         break;
277                     }
278                 }
279 
280                 i++;
281 
282                 while (i < colItems.length) {
283                     items.push(colItems[i]);
284                     i++;
285                 }
286 
287                 curColHeights[dest] += diff;
288 
289                 now = Math.max.apply(Math, curColHeights);
290 
291                 if (now != original) {
292                     self.get("container").height(now);
293                 }
294 
295                 var effect = cfg.effect,
296                     num = items.length;
297 
298                 if (effect === undefined) {
299                     effect = self.get("adjustEffect");
300                 }
301 
302                 function check() {
303                     num--;
304                     if (num <= 0) {
305                         self._adjuster = 0;
306                         cfg && cfg.callback && cfg.callback.call(self);
307                     }
308                 }
309 
310                 if (!num) {
311                     return check();
312                 }
313 
314                 return self._adjuster = timedChunk(items, function (item) {
315                     if (effect) {
316                         item.animate({
317                                 top:parseInt(item.css("top")) + diff
318                             },
319                             effect.duration,
320                             effect.easing,
321                             check);
322                     } else {
323                         item.css("top", parseInt(item.css("top")) + diff);
324                         check();
325                     }
326                 });
327             },
328 
329             /**
330              * 删除一个元素
331              * @param {Node} item 待删除的元素
332              * @param {Object} cfg 控制配置
333              * @param {Function} cfg.callback 删除结束后的回调
334              * @param {Function} cfg.process 用于做删除操作的函数
335              * @param {Object} cfg.effect 删除后其他元素调整位置特效配置
336              * @param {Number} cfg.effect.duration
337              * @param {String} cfg.effect.easing
338              */
339             removeItem:function (item, cfg) {
340                 cfg = cfg || {};
341                 var self = this,
342                     callback = cfg.callback;
343                 self.adjustItem(item, S.mix(cfg, {
344                     process:function () {
345                         item.remove();
346                         return 0;
347                     },
348                     callback:function () {
349                         var dest = parseInt(item.attr("data-waterfall-col")),
350                             colItems = self.get("colItems")[dest];
351                         for (var i = 0; i < colItems.length; i++) {
352                             if (colItems[i][0] == item[0]) {
353                                 colItems.splice(i, 1);
354                                 break;
355                             }
356                         }
357                         callback && callback();
358                     }
359                 }));
360             },
361 
362             /**
363              * 调整所有的元素位置
364              * @param [callback]
365              */
366             adjust:function (callback) {
367                 S.log("waterfall:adjust");
368                 var self = this,
369                     items = self.get("container").all(".ks-waterfall");
370                 /* 正在加，直接开始这次调整，剩余的加和正在调整的一起处理 */
371                 /* 正在调整中，取消上次调整，开始这次调整 */
372                 if (self.isAdjusting()) {
373                     self._adjuster.stop();
374                     self._adjuster = 0;
375                 }
376                 /*计算容器宽度等信息*/
377                 recalculate.call(self);
378                 var num = items.length;
379 
380                 function check() {
381                     num--;
382                     if (num <= 0) {
383                         self.get("container").height(Math.max.apply(Math, self.get("curColHeights")));
384                         self._adjuster = 0;
385                         callback && callback.call(self);
386                         items.length && self.fire('adjustComplete', {
387                             items:items
388                         });
389                     }
390                 }
391 
392                 if (!num) {
393                     return check();
394                 }
395 
396                 return self._adjuster = timedChunk(items, function (item) {
397                     adjustItemAction(self, false, item, check);
398                 });
399             },
400 
401             addItems:function (items, callback) {
402                 var self = this;
403 
404                 /* 正在调整中，直接这次加，和调整的节点一起处理 */
405                 /* 正在加，直接这次加，一起处理 */
406                 self._adder = timedChunk(items,
407                     addItem,
408                     self,
409                     function () {
410                         self.get("container").height(Math.max.apply(Math,
411                             self.get("curColHeights")));
412                         self._adder = 0;
413                         callback && callback.call(self);
414                         items.length && self.fire('addComplete', {
415                             items:items
416                         });
417                     });
418 
419                 return self._adder;
420             },
421 
422             destroy:function () {
423                 $(win).detach("resize", this.__onResize);
424             }
425         });
426 
427 
428     return Waterfall;
