1 /**
  2  * @fileOverview ua-extra
  3  * @author gonghao<gonghao@ghsky.com>
  4  */
  5 KISSY.add('ua/extra', function(S, UA) {
  6     var ua = navigator.userAgent,
  7         win=S.Env.host,
  8         m, external, shell,
  9         o = { },
 10         numberify = UA._numberify;
 11 
 12     /**
 13      * 说明：
 14      * @子涯总结的各国产浏览器的判断依据: http://spreadsheets0.google.com/ccc?key=tluod2VGe60_ceDrAaMrfMw&hl=zh_CN#gid=0
 15      * 根据 CNZZ 2009 年度浏览器占用率报告，优化了判断顺序：http://www.tanmi360.com/post/230.htm
 16      * 如果检测出浏览器，但是具体版本号未知用 0.1 作为标识
 17      * 世界之窗 & 360 浏览器，在 3.x 以下的版本都无法通过 UA 或者特性检测进行判断，所以目前只要检测到 UA 关键字就认为起版本号为 3
 18      */
 19 
 20     // 360Browser
 21     if (m = ua.match(/360SE/)) {
 22         o[shell = 'se360'] = 3; // issue: 360Browser 2.x cannot be recognised, so if recognised default set verstion number to 3
 23     }
 24     // Maxthon
 25     else if ((m = ua.match(/Maxthon/)) && (external = win.external)) {
 26         // issue: Maxthon 3.x in IE-Core cannot be recognised and it doesn't have exact version number
 27         // but other maxthon versions all have exact version number
 28         shell = 'maxthon';
 29         try {
 30             o[shell] = numberify(external['max_version']);
 31         } catch(ex) {
 32             o[shell] = 0.1;
 33         }
 34     }
 35     // TT
 36     else if (m = ua.match(/TencentTraveler\s([\d.]*)/)) {
 37         o[shell = 'tt'] = m[1] ? numberify(m[1]) : 0.1;
 38     }
 39     // TheWorld
 40     else if (m = ua.match(/TheWorld/)) {
 41         o[shell = 'theworld'] = 3; // issue: TheWorld 2.x cannot be recognised, so if recognised default set verstion number to 3
 42     }
 43     // Sougou
 44     else if (m = ua.match(/SE\s([\d.]*)/)) {
 45         o[shell = 'sougou'] = m[1] ? numberify(m[1]) : 0.1;
 46     }
 47 
 48     // If the browser has shell(no matter IE-core or Webkit-core or others), set the shell key
 49     shell && (o.shell = shell);
 50 
 51     S.mix(UA, o);
 52     return UA;
 53 }, {
 54     requires:["ua/base"]
 55 });
