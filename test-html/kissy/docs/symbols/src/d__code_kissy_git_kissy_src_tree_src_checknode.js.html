1 /**
  2  * @fileOverview checkable tree node
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("tree/checknode", function(S, Node, UIBase, Component, BaseNode, CheckNodeRender) {
  6     var $ = Node.all,
  7         PARTIAL_CHECK = 2,
  8         CHECK_CLS = "tree-item-check",
  9         CHECK = 1,
 10         EMPTY = 0;
 11 
 12     var CheckNode = UIBase.create(BaseNode, {
 13         _performInternal:function(e) {
 14             var self=this;
 15             // 需要通知 tree 获得焦点
 16             self.get("tree").get("el")[0].focus();
 17             var target = $(e.target),
 18                 view = self.get("view"),
 19                 tree = self.get("tree");
 20 
 21             if (e.type == "dblclick") {
 22                 // 双击在 +- 号上无效
 23                 if (target.equals(view.get("expandIconEl"))) {
 24                     return;
 25                 }
 26                 // 双击在 checkbox 上无效
 27                 if (target.equals(view.get("checkEl"))) {
 28                     return;
 29                 }
 30                 // 双击在字或者图标上，切换 expand 装tai
 31                 self.set("expanded", !self.get("expanded"));
 32             }
 33 
 34             // 点击在 +- 号，切换状态
 35             if (target.equals(view.get("expandIconEl"))) {
 36                 self.set("expanded", !self.get("expanded"));
 37                 return;
 38             }
 39 
 40             // 单击任何其他地方都切换 check 状态
 41             var checkState = self.get("checkState");
 42             if (checkState == CHECK) {
 43                 checkState = EMPTY;
 44             } else {
 45                 checkState = CHECK;
 46             }
 47             self.set("checkState", checkState);
 48             tree.fire("click", {
 49                 target:self
 50             });
 51         },
 52 
 53         _uiSetCheckState:function(s) {
 54             var self=this;
 55             if (s == CHECK || s == EMPTY) {
 56                 S.each(self.get("children"), function(c) {
 57                     c.set("checkState", s);
 58                 });
 59             }
 60             // 每次状态变化都通知 parent 沿链检查，一层层向上通知
 61             // 效率不高，但是结构清晰
 62             var parent = self.get("parent");
 63             if (parent) {
 64                 var checkCount = 0;
 65                 var cs = parent.get("children");
 66                 for (var i = 0; i < cs.length; i++) {
 67                     var c = cs[i],cState = c.get("checkState");
 68                     // 一个是部分选，父亲必定是部分选，立即结束
 69                     if (cState == PARTIAL_CHECK) {
 70                         parent.set("checkState", PARTIAL_CHECK);
 71                         return;
 72                     } else if (cState == CHECK) {
 73                         checkCount++;
 74                     }
 75                 }
 76 
 77                 // 儿子全都选了，父亲也全选
 78                 if (checkCount == cs.length) {
 79                     parent.set("checkState", CHECK);
 80                 }
 81                 // 儿子都没选，父亲也不选
 82                 else if (checkCount === 0) {
 83                     parent.set("checkState", EMPTY);
 84                 }
 85                 // 有的儿子选了，有的没选，父亲部分选
 86                 else {
 87                     parent.set("checkState", PARTIAL_CHECK);
 88                 }
 89             }
 90         }
 91     }, {
 92         ATTRS:{
 93             checkState:{
 94                 view:true,
 95                 // check 的三状态
 96                 // 0 一个不选
 97                 // 1 儿子有选择
 98                 // 2 全部都选了
 99                 value:0
100             }
101         },
102         CHECK_CLS :CHECK_CLS,
103         DefaultRender:CheckNodeRender,
104         PARTIAL_CHECK:PARTIAL_CHECK,
105         CHECK:CHECK,
106         EMPTY:EMPTY
107     });
108 
109     Component.UIStore.setUIByClass(CHECK_CLS, {
110         priority:Component.UIStore.PRIORITY.LEVEL2,
111         ui:CheckNode
112     });
113 
114     return CheckNode;
115 }, {
116     requires:['node','uibase','component','./basenode','./checknoderender']
117 });
