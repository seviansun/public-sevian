 28         function cleanData(els) {
 29             var Event = S.require("event");
 30             if (Event) {
 31                 Event.detach(els);
 32             }
 33             DOM.removeData(els);
 34         }

 42                 /**
 43                  * Creates DOM elements on the fly from the provided string of raw HTML.
 44                  * @param {String} html A string of HTML to create on the fly. Note that this parses HTML, not XML.
 45                  * @param {Object} [props] An map of attributes on the newly-created element.
 46                  * @param {Document} [ownerDoc] A document in which the new elements will be created
 47                  * @returns {DocumentFragment|HTMLElement}
 48                  */
 49                 create:function (html, props, ownerDoc, _trim/*internal*/) {
 50 
 51                     var ret = null;
 52 
 53                     if (!html) {
 54                         return ret;
 55                     }
 56 
 57                     if (html.nodeType) {
 58                         return DOM.clone(html);
 59                     }
 60 
 61 
 62                     if (!isString(html)) {
 63                         return ret;
 64                     }
 65 
 66                     if (_trim === undefined) {
 67                         _trim = true;
 68                     }
 69 
 70                     if (_trim) {
 71                         html = S.trim(html);
 72                     }
 73 
 74 
 75                     var creators = DOM._creators,
 76                         holder,
 77                         whitespaceMatch,
 78                         context = ownerDoc || doc,
 79                         m,
 80                         tag = DIV,
 81                         k,
 82                         nodes;
 83 
 84                     if (!rhtml.test(html)) {
 85                         ret = context.createTextNode(html);
 86                     }
 87                     // 简单 tag, 比如 DOM.create('<p>')
 88                     else if ((m = RE_SIMPLE_TAG.exec(html))) {
 89                         ret = context.createElement(m[1]);
 90                     }
 91                     // 复杂情况，比如 DOM.create('<img src="sprite.png" />')
 92                     else {
 93                         // Fix "XHTML"-style tags in all browsers
 94                         html = html.replace(rxhtmlTag, "<$1><" + "/$2>");
 95 
 96                         if ((m = RE_TAG.exec(html)) && (k = m[1])) {
 97                             tag = k.toLowerCase();
 98                         }
 99 
100                         holder = (creators[tag] || creators[DIV])(html, context);
101                         // ie 把前缀空白吃掉了
102                         if (lostLeadingWhitespace && (whitespaceMatch = html.match(rleadingWhitespace))) {
103                             holder.insertBefore(context.createTextNode(whitespaceMatch[0]), holder.firstChild);
104                         }
105                         nodes = holder.childNodes;
106 
107                         if (nodes.length === 1) {
108                             // return single node, breaking parentNode ref from "fragment"
109                             ret = nodes[0][PARENT_NODE].removeChild(nodes[0]);
110                         }
111                         else if (nodes.length) {
112                             // return multiple nodes as a fragment
113                             ret = nl2frag(nodes, context);
114                         } else {
115                             S.error(html + " : create node error");
116                         }
117                     }
118 
119                     return attachProps(ret, props);
120                 },
121 
122                 _creators:{
123                     div:function (html, ownerDoc) {
124                         var frag = ownerDoc && ownerDoc != doc ? ownerDoc.createElement(DIV) : DEFAULT_DIV;
125                         // html 为 <style></style> 时不行，必须有其他元素？
126                         frag['innerHTML'] = "m<div>" + html + "<" + "/div>";
127                         return frag.lastChild;
128                     }

131                 /**
132                  * Get the HTML contents of the first element in the set of matched elements.
133                  * or
134                  * Set the HTML contents of each element in the set of matched elements.
135                  * @param {HTMLElement|String|HTMLElement[]} [selector] matched elements
136                  * @param {String} htmlString  A string of HTML to set as the content of each matched element.
137                  * @param {Boolean} [loadScripts=false] True to look for and process scripts
138                  */
139                 html:function (selector, htmlString, loadScripts, callback) {
140                     // supports css selector/Node/NodeList
141                     var els = DOM.query(selector),
142                         el = els[0];
143                     if (!el) {
144                         return
145                     }
146                     // getter
147                     if (htmlString === undefined) {
148                         // only gets value on the first of element nodes
149                         if (isElementNode(el)) {
150                             return el['innerHTML'];
151                         } else {
152                             return null;
153                         }
154                     }
155                     // setter
156                     else {
157 
158                         var success = false, i, elem;
159                         htmlString += "";
160 
161                         // faster
162                         // fix #103,some html element can not be set through innerHTML
163                         if (!htmlString.match(/<(?:script|style|link)/i) &&
164                             (!lostLeadingWhitespace || !htmlString.match(rleadingWhitespace)) &&
165                             !creatorsMap[ (htmlString.match(RE_TAG) || ["", ""])[1].toLowerCase() ]) {
166 
167                             try {
168                                 for (i = els.length - 1; i >= 0; i--) {
169                                     elem = els[i];
170                                     if (isElementNode(elem)) {
171                                         cleanData(getElementsByTagName(elem, "*"));
172                                         elem.innerHTML = htmlString;
173                                     }
174                                 }
175                                 success = true;
176                             } catch (e) {
177                                 // a <= "<a>"
178                                 // a.innerHTML='<p>1</p>';
179                             }
180 
181                         }
182 
183                         if (!success) {
184                             var valNode = DOM.create(htmlString, 0, el.ownerDocument, 0);
185                             for (i = els.length - 1; i >= 0; i--) {
186                                 elem = els[i];
187                                 if (isElementNode(elem)) {
188                                     DOM.empty(elem);
189                                     DOM.append(valNode, elem, loadScripts);
190                                 }
191                             }
192                         }
193                         callback && callback();
194                     }
195                 },
196 
197                 /**
198                  * Remove the set of matched elements from the DOM.
199                  * @param {HTMLElement|String|HTMLElement[]} [selector] matched elements
200                  * @param {Boolean} [keepData=false] whether keep bound events and jQuery data associated with the elements from removed.
201                  */
202                 remove:function (selector, keepData) {
203                     var el, els = DOM.query(selector), i;
204                     for (i = els.length - 1; i >= 0; i--) {
205                         el = els[i];
206                         if (!keepData && isElementNode(el)) {
207                             // 清理数据
208                             var elChildren = getElementsByTagName(el, "*");
209                             cleanData(elChildren);
210                             cleanData(el);
211                         }
212 
213                         if (el.parentNode) {
214                             el.parentNode.removeChild(el);
215                         }
216                     }
217                 },
218 
219                 /**
220                  * Create a deep copy of the first of matched elements.
221                  * @param {HTMLElement|String|HTMLElement[]} [selector] matched elements
222                  * @param {Boolean} [deep=false] whether perform deep copy
223                  * @param {Boolean} [withDataAndEvent=false] A Boolean indicating
224                  * whether event handlers and data should be copied along with the elements.
225                  * @param {Boolean} [deepWithDataAndEvent=false]
226                  * A Boolean indicating whether event handlers and data for all children of the cloned element should be copied.
227                  * if set true then deep argument must be set true as well.
228                  * @see https://developer.mozilla.org/En/DOM/Node.cloneNode
229                  * @returns {HTMLElement}
230                  */
231                 clone:function (selector, deep, withDataAndEvent, deepWithDataAndEvent) {
232                     var elem = DOM.get(selector);
233 
234                     if (!elem) {
235                         return null;
236                     }
237 
238                     // TODO
239                     // ie bug :
240                     // 1. ie<9 <script>xx</script> => <script></script>
241                     // 2. ie will execute external script
242                     var clone = elem.cloneNode(deep);
243 
244                     if (isElementNode(elem) ||
245                         nodeTypeIs(elem, DOM.DOCUMENT_FRAGMENT_NODE)) {
246                         // IE copies events bound via attachEvent when using cloneNode.
247                         // Calling detachEvent on the clone will also remove the events
248                         // from the original. In order to get around this, we use some
249                         // proprietary methods to clear the events. Thanks to MooTools
250                         // guys for this hotness.
251                         if (isElementNode(elem)) {
252                             fixAttributes(elem, clone);
253                         }
254 
255                         if (deep) {
256                             processAll(fixAttributes, elem, clone);
257                         }
258                     }
259                     // runtime 获得事件模块
260                     if (withDataAndEvent) {
261                         cloneWidthDataAndEvent(elem, clone);
262                         if (deep && deepWithDataAndEvent) {
263                             processAll(cloneWidthDataAndEvent, elem, clone);
264                         }
265                     }
266                     return clone;
267                 },
268 
269                 /**
270                  * Remove(include data and event handlers) all child nodes of the set of matched elements from the DOM.
271                  * @param {HTMLElement|String|HTMLElement[]} [selector] matched elements
272                  */
273                 empty:function (selector) {
274                     var els = DOM.query(selector), el, i;
275                     for (i = els.length - 1; i >= 0; i--) {
276                         el = els[i];
277                         DOM.remove(el.childNodes);
278                     }
279                 },
280 
281                 _nl2frag:nl2frag
282             });

284         function processAll(fn, elem, clone) {
285             if (nodeTypeIs(elem, DOM.DOCUMENT_FRAGMENT_NODE)) {
286                 var eCs = elem.childNodes,
287                     cloneCs = clone.childNodes,
288                     fIndex = 0;
289                 while (eCs[fIndex]) {
290                     if (cloneCs[fIndex]) {
291                         processAll(fn, eCs[fIndex], cloneCs[fIndex]);
292                     }
293                     fIndex++;
294                 }
295             } else if (isElementNode(elem)) {
296                 var elemChildren = getElementsByTagName(elem, "*"),
297                     cloneChildren = getElementsByTagName(clone, "*"),
298                     cIndex = 0;
299                 while (elemChildren[cIndex]) {
300                     if (cloneChildren[cIndex]) {
301                         fn(elemChildren[cIndex], cloneChildren[cIndex]);
302                     }
303                     cIndex++;
304                 }
305             }
306         }

309         // 克隆除了事件的 data
310         function cloneWidthDataAndEvent(src, dest) {
311             var Event = S.require('event');
312 
313             if (isElementNode(dest) && !DOM.hasData(src)) {
314                 return;
315             }
316 
317             var srcData = DOM.data(src);
318 
319             // 浅克隆，data 也放在克隆节点上
320             for (var d in srcData) {
321                 DOM.data(dest, d, srcData[d]);
322             }
323 
324             // 事件要特殊点
325             if (Event) {
326                 // _removeData 不需要？刚克隆出来本来就没
327                 Event._removeData(dest);
328                 Event._clone(src, dest);
329             }
330         }

332         // wierd ie cloneNode fix from jq
333         function fixAttributes(src, dest) {
334 
335             // clearAttributes removes the attributes, which we don't want,
336             // but also removes the attachEvent events, which we *do* want
337             if (dest.clearAttributes) {
338                 dest.clearAttributes();
339             }
340 
341             // mergeAttributes, in contrast, only merges back on the
342             // original attributes, not the events
343             if (dest.mergeAttributes) {
344                 dest.mergeAttributes(src);
345             }
346 
347             var nodeName = dest.nodeName.toLowerCase(),
348                 srcChilds = src.childNodes;
349 
350             // IE6-8 fail to clone children inside object elements that use
351             // the proprietary classid attribute value (rather than the type
352             // attribute) to identify the type of content to display
353             if (nodeName === "object" && !dest.childNodes.length) {
354                 for (var i = 0; i < srcChilds.length; i++) {
355                     dest.appendChild(srcChilds[i].cloneNode(true));
356                 }
357                 // dest.outerHTML = src.outerHTML;
358             } else if (nodeName === "input" && (src.type === "checkbox" || src.type === "radio")) {
359                 // IE6-8 fails to persist the checked state of a cloned checkbox
360                 // or radio button. Worse, IE6-7 fail to give the cloned element
361                 // a checked appearance if the defaultChecked value isn't also set
362                 if (src.checked) {
363                     dest['defaultChecked'] = dest.checked = src.checked;
364                 }
365 
366                 // IE6-7 get confused and end up setting the value of a cloned
367                 // checkbox/radio button to an empty string instead of "on"
368                 if (dest.value !== src.value) {
369                     dest.value = src.value;
370                 }
371 
372                 // IE6-8 fails to return the selected option to the default selected
373                 // state when cloning options
374             } else if (nodeName === "option") {
375                 dest.selected = src.defaultSelected;
376                 // IE6-8 fails to set the defaultValue to the correct value when
377                 // cloning other types of input fields
378             } else if (nodeName === "input" || nodeName === "textarea") {
379                 dest.defaultValue = src.defaultValue;
380             }
381 
382             // Event data gets referenced instead of copied if the expando
383             // gets copied too
384             // 自定义 data 根据参数特殊处理，expando 只是个用于引用的属性
385             dest.removeAttribute(DOM.__EXPANDO);
386         }

388         // 添加成员到元素中
389         function attachProps(elem, props) {
390             if (S.isPlainObject(props)) {
391                 if (isElementNode(elem)) {
392                     DOM.attr(elem, props, true);
393                 }
394                 // document fragment
395                 else if (nodeTypeIs(elem, DOM.DOCUMENT_FRAGMENT_NODE)) {
396                     DOM.attr(elem.childNodes, props, true);
397                 }
398             }
399             return elem;
400         }

402         // 将 nodeList 转换为 fragment
403         function nl2frag(nodes, ownerDoc) {
404             var ret = null, i, len;
405 
406             if (nodes
407                 && (nodes.push || nodes.item)
408                 && nodes[0]) {
409                 ownerDoc = ownerDoc || nodes[0].ownerDocument;
410                 ret = ownerDoc.createDocumentFragment();
411                 nodes = S.makeArray(nodes);
412                 for (i = 0, len = nodes.length; i < len; i++) {
413                     ret.appendChild(nodes[i]);
414                 }
415             }
416             else {
417                 S.log('Unable to convert ' + nodes + ' to fragment.');
418             }
419             return ret;
420         }
