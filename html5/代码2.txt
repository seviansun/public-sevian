        function copy(d) {
            var e = document.createElement("div");
            for (var key in d.style) {
                if (d.style[key] != undefined) {
                    e.style[key] = d.style[key];
                }
            }
            document.getElementsByTagName("body")[0].appendChild(e);
        }
    </script>
