  5 KISSY.add("ajax/base", function (S, JSON, Event, XhrObject, undefined) {

  7     var rlocalProtocol = /^(?:about|app|app\-storage|.+\-extension|file|widget):$/,
  8         rspace = /\s+/,
  9         rurl = /^([\w\+\.\-]+:)(?:\/\/([^\/?#:]*)(?::(\d+))?)?/,
 10         mirror = function (s) {
 11             return s;
 12         },
 13         HTTP_PORT = 80,
 14         HTTPS_PORT = 443,
 15         rnoContent = /^(?:GET|HEAD)$/,
 16         curLocation,
 17         doc = S.Env.host.document,
 18         curLocationParts;
 19 
 20     try {
 21         curLocation = location.href;
 22     } catch (e) {
 23         S.log("ajax/base get curLocation error : ");
 24         S.log(e);
 25         // Use the href attribute of an A element
 26         // since IE will modify it given document.location
 27         curLocation = doc.createElement("a");
 28         curLocation.href = "";
 29         curLocation = curLocation.href;
 30     }
 31 
 32     // fix on nodejs , curLocation == "/xx/yy/kissy-nodejs.js"
 33     curLocationParts = rurl.exec(curLocation) || ["", "", "", ""];
 34 
 35     var isLocal = rlocalProtocol.test(curLocationParts[1]),
 36         transports = {},
 37         defaultConfig = {
 38             type:"GET",
 39             contentType:"application/x-www-form-urlencoded; charset=UTF-8",
 40             async:true,
 41             serializeArray:true,
 42             processData:true,
 43             accepts:{
 44                 xml:"application/xml, text/xml",
 45                 html:"text/html",
 46                 text:"text/plain",
 47                 json:"application/json, text/javascript",
 48                 *:"*/*"
 49             },
 50             converters:{
 51                 text:{
 52                     json:JSON.parse,
 53                     html:mirror,
 54                     text:mirror,
 55                     xml:S.parseXML
 56                 }
 57             },
 58             contents:{
 59                 xml:/xml/,
 60                 html:/html/,
 61                 json:/json/
 62             }
 63         };
 64 
 65     defaultConfig.converters.html = defaultConfig.converters.text;
 66 
 67     function setUpConfig(c) {
 68         // deep mix,exclude context!
 69         var context = c.context;
 70         delete c.context;
 71         c = S.mix(S.clone(defaultConfig), c || {}, undefined, undefined, true);
 72         c.context = context;
 73 
 74         if (!("crossDomain" in c)) {
 75             var parts = rurl.exec(c.url.toLowerCase());
 76             c.crossDomain = !!( parts &&
 77                 ( parts[ 1 ] != curLocationParts[ 1 ] || parts[ 2 ] != curLocationParts[ 2 ] ||
 78                     ( parts[ 3 ] || ( parts[ 1 ] === "http:" ? HTTP_PORT : HTTPS_PORT ) )
 79                         !=
 80                         ( curLocationParts[ 3 ] || ( curLocationParts[ 1 ] === "http:" ? HTTP_PORT : HTTPS_PORT ) ) )
 81                 );
 82         }
 83 
 84         if (c.processData && c.data && !S.isString(c.data)) {
 85             // 必须 encodeURIComponent 编码 utf-8
 86             c.data = S.param(c.data, undefined, undefined, c.serializeArray);
 87         }
 88 
 89         // fix #90 ie7 about "//x.htm"
 90         c.url = c.url.replace(/^\/\//, curLocationParts[1] + "//");
 91         c.type = c.type.toUpperCase();
 92         c.hasContent = !rnoContent.test(c.type);
 93 
 94         // 数据类型处理链，一步步将前面的数据类型转化成最后一个
 95         c.dataType = S.trim(c.dataType || "*").split(rspace);
 96 
 97         if (!("cache" in c) && S.inArray(c.dataType[0], ["script", "jsonp"])) {
 98             c.cache = false;
 99         }
100 
101         if (!c.hasContent) {
102             if (c.data) {
103                 c.url += ( /\?/.test(c.url) ? "&" : "?" ) + c.data;
104                 delete c.data;
105             }
106             if (c.cache === false) {
107                 c.url += ( /\?/.test(c.url) ? "&" : "?" ) + "_ksTS=" + (S.now() + "_" + S.guid());
108             }
109         }
110 
111         c.context = c.context || c;
112         return c;
113     }

131          * @param {IO.XhrObject} e.xhr current xhr object

133 
134         /**
135          * @name IO#error
136          * @description fired after request occurs error
137          * @event
138          * @param {Event.Object} e

141          */
142         io.fire(eventType, { ajaxConfig:xhrObject.config, xhr:xhrObject});
143     }

145     /**
146      * @name IO
147      * @namespace Provides utility that brokers HTTP requests through a simplified interface
148      * @function
149      *
150      * @param {Object} c <br/>name-value of object to config this io request.<br/>
151      *  all values are optional.<br/>
152      *  default value can be set through {@link io.setupConfig}<br/>
153      *
154      * @param {String} c.url <br/>request destination
155      *
156      * @param {String} c.type <br/>request type.
157      * eg: "get","post"<br/>
158      * Default: "get"<br/>
159      *
160      * @param {String} c.contentType <br/>
161      * Default: "application/x-www-form-urlencoded; charset=UTF-8"<br/>
162      * Data will always be transmitted to the server using UTF-8 charset<br/>
163      *
164      * @param {Object} c.accepts <br/>
165      * Default: depends on DataType.<br/>
166      * The content type sent in request header that tells the server<br/>
167      * what kind of response it will accept in return.<br/>
168      * It is recommended to do so once in the {@link io.setupConfig}
169      *
170      * @param {Boolean} c.async <br/>
171      * Default: true<br/>
172      * whether request is sent asynchronously<br/>
173      *
174      * @param {Boolean} c.cache <br/>
175      * Default: true ,false for dataType "script" and "jsonp"<br/>
176      * if set false,will append _ksTs=Date.now() to url automatically<br/>
177      *
178      * @param {Object} c.contents <br/>
179      * a name-regexp map to determine request data's dataType<br/>
180      * It is recommended to do so once in the {@link io.setupConfig}<br/>
181      *
182      * @param {Object} c.context <br/>
183      * specify the context of this request's callback (success,error,complete)
184      *
185      * @param {Object} c.converters <br/>
186      * Default:{text:{json:JSON.parse,html:mirror,text:mirror,xml:KISSY.parseXML}}<br/>
187      * specified how to transform one dataType to another dataType<br/>
188      * It is recommended to do so once in the {@link io.setupConfig}
189      *
190      * @param {Boolean} c.crossDomain <br/>
191      * Default: false for same-domain request,true for cross-domain request<br/>
192      * if server-side jsonp redirect to another domain ,you should set this to true
193      *
194      * @param {Object} c.data <br/>
195      * Data sent to server.if processData is true,data will be serialized to String type.<br/>
196      * if value if an Array, serialization will be based on serializeArray.
197      *
198      * @param {String} c.dataType <br/>
199      * return data as a specified type<br/>
200      * Default: Based on server contentType header<br/>
201      * "xml" : a XML document<br/>
202      * "text"/"html": raw server data <br/>
203      * "script": evaluate the return data as script<br/>
204      * "json": parse the return data as json and return the result as final data<br/>
205      * "jsonp": load json data via jsonp
206      *
207      * @param {Object} c.headers <br/>
208      * additional name-value header to send along with this request.
209      *
210      * @param {String} c.jsonp <br/>
211      * Default: "callback"<br/>
212      * Override the callback function name in a jsonp request. eg:<br/>
213      * set "callback2" , then jsonp url will append  "callback2=?".
214      *
215      * @param {String} c.jsonpCallback <br/>
216      * Specify the callback function name for a jsonp request.<br/>
217      * set this value will replace the auto generated function name.<br/>
218      * eg:<br/>
219      * set "customCall" , then jsonp url will append "callback=customCall"
220      *
221      * @param {String} c.mimeType <br/>
222      * override xhr's mime type
223      *
224      * @param {Boolean} c.processData <br/>
225      * Default: true<br/>
226      * whether data will be serialized as String
227      *
228      * @param {String} c.scriptCharset <br/>
229      * only for dataType "jsonp" and "script" and "get" type.<br/>
230      * force the script to certain charset.
231      *
232      * @param {Function} c.beforeSend <br/>
233      * beforeSend(xhrObject,config)<br/>
234      * callback function called before the request is sent.this function has 2 arguments<br/>
235      * 1. current KISSY xhrObject<br/>
236      * 2. current io config<br/>
237      * note: can be used for add progress event listener for native xhr's upload attribute
238      * see <a href="http://www.w3.org/TR/XMLHttpRequest/#event-xhr-progress">XMLHttpRequest2</a>
239      *
240      * @param {Function} c.success <br/>
241      * success(data,textStatus,xhr)<br/>
242      * callback function called if the request succeeds.this function has 3 arguments<br/>
243      * 1. data returned from this request with type specified by dataType<br/>
244      * 2. status of this request with type String<br/>
245      * 3. XhrObject of this request , for details {@link IO.XhrObject}
246      *
247      * @param {Function} c.error <br/>
248      * success(data,textStatus,xhr) <br/>
249      * callback function called if the request occurs error.this function has 3 arguments<br/>
250      * 1. null value<br/>
251      * 2. status of this request with type String,such as "timeout","Not Found","parsererror:..."<br/>
252      * 3. XhrObject of this request , for details {@link IO.XhrObject}
253      *
254      * @param {Function} c.complete <br/>
255      * success(data,textStatus,xhr)<br/>
256      * callback function called if the request finished(success or error).this function has 3 arguments<br/>
257      * 1. null value if error occurs or data returned from server<br/>
258      * 2. status of this request with type String,such as success:"ok",
259      * error:"timeout","Not Found","parsererror:..."<br/>
260      * 3. XhrObject of this request , for details {@link IO.XhrObject}
261      *
262      * @param {Number} c.timeout <br/>
263      * Set a timeout(in seconds) for this request.if will call error when timeout
264      *
265      * @param {Boolean} c.serializeArray <br/>
266      * whether add [] to data's name when data's value is array in serialization
267      *
268      * @param {Object} c.xhrFields <br/>
269      * name-value to set to native xhr.set as xhrFields:{withCredentials:true}
270      *
271      * @param {String} c.username <br/>
272      * a username tobe used in response to HTTP access authentication request
273      *
274      * @param {String} c.password <br/>
275      * a password tobe used in response to HTTP access authentication request
276      *
277      * @param {Object} c.xdr <br/>
278      * cross domain request config object
279      *
280      * @param {String} c.xdr.src <br/>
281      * Default: kissy's flash url
282      * flash sender url
283      *
284      * @param {String} c.xdr.use <br/>
285      * if set to "use", it will always use flash for cross domain request even in chrome/firefox
286      *
287      * @param {Object} c.xdr.subDomain <br/>
288      * cross sub domain request config object
289      *
290      * @param {String} c.xdr.subDomain.proxy <br/>
291      * proxy page,eg:<br/>
292      * a.t.cn/a.htm send request to b.t.cn/b.htm: <br/>
293      * 1. a.htm set document.domain='t.cn'<br/>
294      * 2. b.t.cn/proxy.htm 's content is <script>document.domain='t.cn'</script><br/>
295      * 3. in a.htm , call io({xdr:{subDomain:{proxy:'/proxy.htm'}}})
296      *
297      * @returns {IO.XhrObject} current request object
298      */
299     function io(c) {
300 
301         if (!c.url) {
302             return undefined;
303         }
304 
305         c = setUpConfig(c);
306 
307         var xhrObject = new XhrObject(c);
308 
309         /**
310          * @name IO#start
311          * @description fired before generating request object
312          * @event
313          * @param {Event.Object} e
314          * @param {Object} e.ajaxConfig current request 's config
315          * @param {IO.XhrObject} e.xhr current xhr object
316          */
317 
318         fire("start", xhrObject);
319 
320         var transportContructor = transports[c.dataType[0]] || transports["*"],
321             transport = new transportContructor(xhrObject);
322         xhrObject.transport = transport;
323 
324         if (c.contentType) {
325             xhrObject.setRequestHeader("Content-Type", c.contentType);
326         }
327         var dataType = c.dataType[0],
328             accepts = c.accepts;
329         // Set the Accepts header for the server, depending on the dataType
330         xhrObject.setRequestHeader(
331             "Accept",
332             dataType && accepts[dataType] ?
333                 accepts[ dataType ] + (dataType === "*" ? "" : ", */*; q=0.01"  ) :
334                 accepts[ "*" ]
335         );
336 
337         // Check for headers option
338         for (var i in c.headers) {
339             xhrObject.setRequestHeader(i, c.headers[ i ]);
340         }
341 
342 
343         // allow setup native listener
344         // such as xhr.upload.addEventListener('progress', function (ev) {})
345         if (c.beforeSend && ( c.beforeSend.call(c.context || c, xhrObject, c) === false)) {
346             return undefined;
347         }
348 
349         function genHandler(handleStr) {
350             return function (v) {
351                 if (xhrObject.timeoutTimer) {
352                     clearTimeout(xhrObject.timeoutTimer);
353                     xhrObject.timeoutTimer = 0;
354                 }
355                 var h = c[handleStr];
356                 h && h.apply(c.context, v);
357                 fire(handleStr, xhrObject);
358             };
359         }
360 
361         xhrObject.then(genHandler("success"), genHandler("error"));
362 
363         xhrObject.fin(genHandler("complete"));
364 
365         xhrObject.readyState = 1;
366 
367         /**
368          * @name IO#send
369          * @description fired before sending request
370          * @event
371          * @param {Event.Object} e
372          * @param {Object} e.ajaxConfig current request 's config
373          * @param {IO.XhrObject} e.xhr current xhr object
374          */
375 
376         fire("send", xhrObject);
377 
378         // Timeout
379         if (c.async && c.timeout > 0) {
380             xhrObject.timeoutTimer = setTimeout(function () {
381                 xhrObject.abort("timeout");
382             }, c.timeout * 1000);
383         }
384 
385         try {
386             // flag as sending
387             xhrObject.state = 1;
388             transport.send();
389         } catch (e) {
390             // Propagate exception as error if not done
391             if (xhrObject.state < 2) {
392                 xhrObject._xhrReady(-1, e);
393                 // Simply rethrow otherwise
394             } else {
395                 S.error(e);
396             }
397         }
398 
399         return xhrObject;
400     }

422             },

427                 transports[name] = fn;
428             },
429             /**
430              * @private
431              */
432             getTransport:function (name) {
433                 return transports[name];
434             },
435             /**
436              * get default config value for io request
437              * @returns {Object}
438              */
439             getConfig:function () {
440                 return defaultConfig;
441             }
442         });
443 
444     return io;
