 23     function indexMap(s) {
 24         if (/(.+\/)(\?t=.+)?$/.test(s)) {
 25             return RegExp.$1 + "index" + RegExp.$2;
 26         }
 27         return s;

 31     function removeSuffixAndTagFromModName(modName) {
 32         var tag = undefined,
 33             m,
 34             withTagReg = /([^?]+)(?:\?t=(.+))/;
 35 
 36         if (m = modName.match(withTagReg)) {
 37             modName = m[1];
 38             tag = m[2];
 39         }
 40 
 41         // js do not need suffix
 42         modName = modName.replace(/\.js$/i, "");
 43         return {
 44             modName:modName,
 45             tag:tag
 46         };
 47     }

 67         /**
 68          * resolve relative part of path
 69          * x/../y/z -> y/z
 70          * x/./y/z -> x/y/z
 71          * @param path uri path
 72          * @return {string} resolved path
 73          * @description similar to path.normalize in nodejs
 74          */
 75         normalizePath:function (path) {
 76             var paths = path.split("/"),
 77                 re = [],
 78                 p;
 79             for (var i = 0; i < paths.length; i++) {
 80                 p = paths[i];
 81                 if (p == ".") {
 82                 } else if (p == "..") {
 83                     re.pop();
 84                 } else {
 85                     re.push(p);
 86                 }
 87             }
 88             return re.join("/");
 89         },

 91         /**
 92          * 根据当前模块以及依赖模块的相对路径，得到依赖模块的绝对路径
 93          * @param moduleName 当前模块
 94          * @param depName 依赖模块
 95          * @return {string|Array} 依赖模块的绝对路径
 96          * @description similar to path.resolve in nodejs
 97          */
 98         normalDepModuleName:function (moduleName, depName) {
 99             if (!depName) {
100                 return depName;
101             }
102             if (S.isArray(depName)) {
103                 for (var i = 0; i < depName.length; i++) {
104                     depName[i] = utils.normalDepModuleName(moduleName, depName[i]);
105                 }
106                 return depName;
107             }
108             if (startsWith(depName, "../") || startsWith(depName, "./")) {
109                 var anchor = "", index;
110                 // x/y/z -> x/y/
111                 if ((index = moduleName.lastIndexOf("/")) != -1) {
112                     anchor = moduleName.substring(0, index + 1);
113                 }
114                 return normalizePath(anchor + depName);
115             } else if (depName.indexOf("./") != -1
116                 || depName.indexOf("../") != -1) {
117                 return normalizePath(depName);
118             } else {
119                 return depName;
120             }
121         },

128         /**
129          * 路径正则化，不能是相对地址
130          * 相对地址则转换成相对页面的绝对地址
131          * 用途:
132          * package path 相对地址则相对于当前页面获取绝对地址
133          */
134         normalBasePath:function (path) {
135             path = S.trim(path);
136 
137             // path 为空时，不能变成 "/"
138             if (path && path.charAt(path.length - 1) != '/') {
139                 path += "/";
140             }
141 
142             /**
143              * 一定要正则化，防止出现 ../ 等相对路径
144              * 考虑本地路径
145              */
146             if (!path.match(/^(http(s)?)|(file):/i)
147                 && !startsWith(path, "/")) {
148                 path = __pagePath + path;
149             }
150             return normalizePath(path);
151         },

162         getPackagePath:function (self, mod) {
163             //缓存包路径，未申明的包的模块都到核心模块中找
164             if (mod.packagePath) {
165                 return mod.packagePath;
166             }
167 
168             var //一个模块合并到了另一个模块文件中去
169                 modName = mod.name,
170                 packages = self.Config.packages || {},
171                 pName = "",
172                 p_def;
173 
174             for (var p in packages) {
175                 if (packages.hasOwnProperty(p)) {
176                     if (S.startsWith(modName, p) &&
177                         p.length > pName) {
178                         pName = p;
179                     }
180                 }
181             }
182 
183             p_def = packages[pName];
184 
185             mod.charset = p_def && p_def.charset || mod.charset;
186 
187             if (p_def) {
188                 mod.packageTag = p_def.tag;
189             } else {
190                 // kissy 自身组件的事件戳后缀
191                 mod.packageTag = encodeURIComponent(self.Config.tag);
192             }
193 
194             return mod.packagePath = (p_def && p_def.path) || self.Config.base;
195         },

247         attachMod:function (self, mod) {
248             if (mod.status == data.ATTACHED) {
249                 return;
250             }

253                 value;
254 
255             // 需要解开 index，相对路径，去除 tag，但是需要保留 alias，防止值不对应

261                     value = fn.apply(mod, utils.getModules(self, mod.requires));

264                 }

266             }
267

271                 mod:mod

282         /**
283          * Three effects:
284          * 1. add index : / => /index
285          * 2. unalias : core => dom,event,ua
286          * 3. relative to absolute : ./x => y/x
287          * 4. create module info with tag : core.js?t=xx => core , .tag=xx         *
288          * @param {KISSY} self Global KISSY instance
289          * @param {String|String[]} modNames Array of module names or module names string separated by comma
290          */
291         normalizeModNames:function (self, modNames, refModName) {
292             var ret = [],
293                 mods = self.Env.mods;
294             S.each(modNames, function (name) {
295                 var alias, m;
296                 // 1. index map
297                 name = indexMap(name);
298                 // 2. un alias
299                 if ((m = mods[name]) && (alias = m.alias)) {
300                     ret.push.apply(ret, indexMap(alias));
301                 } else {
302                     ret.push(name);
303                 }
304             });
305             // 3. relative to absolute (optional)
306             if (refModName) {
307                 ret = utils.normalDepModuleName(refModName, ret);
308             }
309             // 4. create module info with tag
310             S.each(ret, function (name, i) {
311                 ret[i] = utils.createModuleInfo(self, name).name;
312             });
313             return ret;
314         },
315 
316         normalizeModNamesWithAlias:function (self,modNames, refModName) {
317             var ret = [];
318             S.each(modNames, function (name) {
319                 // 1. index map
320                 name = indexMap(name);
321                 ret.push(name);
322             });
323             // 2. relative to absolute (optional)
324             if (refModName) {
325                 ret = utils.normalDepModuleName(refModName, ret);
326             }
327             // 3. create module info with tag
328             S.each(ret, function (name, i) {
329                 ret[i] = utils.createModuleInfo(self, name).name;
330             });
331             return ret;
332         },

334         //注册模块，将模块和定义 factory 关联起来
335         registerModule:function (self, name, def, config) {
336             config = config || {};
337 
338             utils.createModuleInfo(self, name);
339 
340             var mods = self.Env.mods,
341                 mod = mods[name];
342 
343             // 注意：通过 S.add(name[, fn[, config]]) 注册的代码，无论是页面中的代码，
344             // 还是 js 文件里的代码，add 执行时，都意味着该模块已经 LOADED
345             S.mix(mod, { name:name, status:data.LOADED });
346 
347             if (mod.fn) {
348                 S.log(name + " is defined more than once");
349                 return;
350             }
351 
352             mod.fn = def;
353 
354             S.mix((mods[name] = mod), config);
355 
356             S.log(name + " is loaded");
357         },

359         normAdd:function (self, name, def, config) {
360             var mods = self.Env.mods,
361                 o;
362 
363             // S.add(name, config) => S.add( { name: config } )
364             if (S.isString(name)
365                 && !config
366                 && S.isPlainObject(def)) {
367                 o = {};
368                 o[name] = def;
369                 name = o;
370             }
371 
372             // S.add( { name: config } )
373             if (S.isPlainObject(name)) {
374                 S.each(name, function (modCfg, modName) {
375                     utils.createModuleInfo(self, modName);
376                     S.mix(mods[modName], modCfg);
377                 });
378                 return true;
379             }
380         },

395     function defaultComponentJsName(m) {
396         var suffix = ".js", match;
397         if (match = m.match(/(.+)(\.css)$/i)) {
398             suffix = match[2];
399             m = match[1];
400         }
401         return m + (S.Config.debug ? '' : '-min') + suffix;
402     }

404     function isStatus(self, modNames, status) {
405         var mods = self.Env.mods,
406             ret = true;
407         modNames = S.makeArray(modNames);
408         S.each(modNames, function (name) {
409             var mod = mods[name];
410             if (!mod || mod.status !== status) {
411                 ret = false;
412                 return ret;
413             }
414         });
415         return ret;
416     }
