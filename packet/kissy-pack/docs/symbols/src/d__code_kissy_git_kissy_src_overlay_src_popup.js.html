1 /**
  2  * @fileOverview KISSY.Popup
  3  * @author  乔花<qiaohua@taobao.com> , yiminghe@gmail.com
  4  */
  5 KISSY.add('overlay/popup', function (S, Component, Overlay, undefined) {
  6 
  7     /**
  8      * KISSY Popup Component
  9      * @class
 10      * @memberOf Overlay
 11      * @extends Overlay
 12      * @param {NodeList} [container] existing dom node
 13      * @param {Object} config see {@link Overlay}
 14      */
 15     function Popup(container, config) {
 16         var self = this;
 17 
 18         // 支持 Popup(config)
 19         if (S.isUndefined(config)) {
 20             config = container;
 21         } else {
 22             config.srcNode = container;
 23         }
 24 
 25         Popup.superclass.constructor.call(self, config);
 26     }
 27 
 28     Popup.ATTRS =
 29     /**
 30      * @lends Overlay.Popup#
 31      */
 32     {
 33         /**
 34          * trigger element to show popup
 35          * @type NodeList
 36          */
 37         trigger:{                          // 触发器
 38             setter:function (v) {
 39                 if (S.isString(v)) {
 40                     v = S.all(v);
 41                 }
 42                 return v;
 43             }
 44         },
 45         /**
 46          * how do activate trigger element. "click" or "mouse",Default:"click"
 47          * @type String
 48          */
 49         triggerType:{value:'click'}, // 触发类型
 50         currentTrigger:{},
 51         /**
 52          * when trigger type is mouse, the delayed time to show popup. Default:100,in milliseconds
 53          * @type Number
 54          */
 55         mouseDelay:{
 56             value:100                      // triggerType 为 mouse 时, Popup 显示的延迟时间, 默认为 100ms
 57         },
 58         /**
 59          * when trigger type is click, whether support toggle. Default:false
 60          * @type Boolean
 61          */
 62         toggle:{
 63             value:false                     // triggerType 为 click 时, Popup 是否有toggle功能
 64         }
 65     };
 66 
 67     S.extend(Popup, Overlay,
 68         /**
 69          * @lends Overlay.Popup#
 70          */
 71         {
 72             /**
 73              * see {@link UIBase.Box#show}
 74              * @name Overlay.Popup#show
 75              * @function
 76              */
 77 
 78 
 79 
 80             initializer:function () {
 81                 var self = this;
 82                 // 获取相关联的 DOM 节点
 83                 var trigger = self.get("trigger");
 84                 if (trigger) {
 85                     if (self.get("triggerType") === 'mouse') {
 86                         self._bindTriggerMouse();
 87 
 88                         self.on('bindUI', function () {
 89                             self._bindContainerMouse();
 90                         });
 91                     } else {
 92                         self._bindTriggerClick();
 93                     }
 94                 }
 95             },
 96 
 97             _bindTriggerMouse:function () {
 98                 var self = this,
 99                     trigger = self.get("trigger"),
100                     timer;
101 
102                 self.__mouseEnterPopup = function (ev) {
103                     self._clearHiddenTimer();
104 
105                     timer = S.later(function () {
106                         self._showing(ev);
107                         timer = undefined;
108                     }, self.get('mouseDelay'));
109                 };
110 
111                 S.each(trigger, function (el) {
112                     S.one(el).on('mouseenter', self.__mouseEnterPopup);
113                 });
114 
115                 self._mouseLeavePopup = function () {
116                     if (timer) {
117                         timer.cancel();
118                         timer = undefined;
119                     }
120 
121                     self._setHiddenTimer();
122                 };
123 
124                 S.each(trigger, function (el) {
125                     S.one(el).on('mouseleave', self._mouseLeavePopup);
126                 });
127             },
128 
129             _bindContainerMouse:function () {
130                 var self = this;
131 
132                 self.get('el').on('mouseleave', self._setHiddenTimer, self)
133                     .on('mouseenter', self._clearHiddenTimer, self);
134             },
135 
136             _setHiddenTimer:function () {
137                 var self = this;
138                 self._hiddenTimer = S.later(function () {
139                     self._hiding();
140                 }, self.get('mouseDelay'));
141             },
142 
143             _clearHiddenTimer:function () {
144                 var self = this;
145                 if (self._hiddenTimer) {
146                     self._hiddenTimer.cancel();
147                     self._hiddenTimer = undefined;
148                 }
149             },
150 
151             _bindTriggerClick:function () {
152                 var self = this;
153                 self.__clickPopup = function (ev) {
154                     ev.halt();
155                     if (self.get('toggle')) {
156                         self[self.get('visible') ? '_hiding' : '_showing'](ev);
157                     }
158                     else  self._showing(ev);
159                 };
160                 S.each(self.get("trigger"), function (el) {
161                     S.one(el).on('click', self.__clickPopup);
162                 });
163             },
164             _showing:function (ev) {
165                 var self = this;
166                 self.set('currentTrigger', S.one(ev.target));
167                 self.show();
168             },
169             _hiding:function () {
170                 this.set('currentTrigger', undefined);
171                 this.hide();
172             },
173 
174             destructor:function () {
175                 var self = this;
176                 var t = self.get("trigger");
177                 if (t) {
178                     if (self.__clickPopup) {
179                         S.each(t, function (el) {
180                             S.one(el).detach('click', self.__clickPopup);
181                         });
182                     }
183                     if (self.__mouseEnterPopup) {
184                         S.each(t, function (el) {
185                             S.one(el).detach('mouseenter', self.__mouseEnterPopup);
186                         });
187                     }
188 
189                     if (self._mouseLeavePopup) {
190                         S.each(t, function (el) {
191                             S.one(el).detach('mouseleave', self._mouseLeavePopup);
192                         });
193                     }
194                 }
195                 if (self.get('el')) {
196                     self.get('el').detach('mouseleave', self._setHiddenTimer, self)
197                         .detach('mouseenter', self._clearHiddenTimer, self);
198                 }
199             }
200         });
201 
202 
203     Component.UIStore.setUIByClass("popup", {
204         priority:Component.UIStore.PRIORITY.LEVEL1,
205         ui:Popup
206     });
207 
208     return Popup;
209 }, {
210     requires:[ "component", "./base"]
211 });
