  5 (function(S, loader,data) {
  6     if (typeof require !== 'undefined') {
  7         return;
  8     }
  9     var LOADED = data.LOADED,
 10         mix = S.mix;
 11 
 12     mix(loader, {
 13         //注册模块，将模块和定义 factory 关联起来
 14         __registerModule:function(name, def, config) {
 15             config = config || {};
 16             var self = this,
 17                 mods = self.Env.mods,
 18                 mod = mods[name] || {};
 19 
 20             // 注意：通过 S.add(name[, fn[, config]]) 注册的代码，无论是页面中的代码，
 21             // 还是 js 文件里的代码，add 执行时，都意味着该模块已经 LOADED
 22             mix(mod, { name: name, status: LOADED });
 23 
 24             if (mod.fns && mod.fns.length) {
 25                 S.log(name + " is defined more than once");
 26                 //S.error(name + " is defined more than once");
 27             }
 28 
 29             //支持 host，一个模块多个 add factory
 30             mod.fns = mod.fns || [];
 31             mod.fns.push(def);
 32             mix((mods[name] = mod), config);
 33         }
 34     });
