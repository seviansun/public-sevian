1 /**
  2  * remote datasource
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("autocomplete/remoteDataSource", function (S, IO) {
  6 
  7     /**
  8      * dataSource which wrap {@link IO} utility.
  9      * @class
 10      * @memberOf AutoComplete
 11      * @param {Object} cfg configs
 12      * @param {Object} cfg.xhrCfg IO configuration.same as {@link} IO
 13      * @param {Object} cfg.dataSourceCfg dataSource configs
 14      * @param {String} cfg.dataSourceCfg.paramName
 15      * Used as parameter name to send autoS=Complete input's value to server
 16      * @param {String} cfg.dataSourceCfg.cache Whether server response data is cached
 17      * @param {Boolean} cfg.dataSourceCfg.allowEmpty whether send empty to server when input val is empty.default:false
 18      * @param {Function} cfg.dataSourceCfg.parse Serve as a parse function to parse server
 19      * response to return a valid array of data for autoComplete.
 20      */
 21     function RemoteDataSource(cfg) {
 22         var self = this;
 23         RemoteDataSource.superclass.constructor.apply(self, arguments);
 24         self.io = null;
 25         self.caches = {};
 26     }
 27 
 28     RemoteDataSource.ATTRS = {
 29         dataSourceCfg:{
 30             value:{}
 31         },
 32         xhrCfg:{
 33             value:{}
 34         }
 35     };
 36 
 37     S.extend(RemoteDataSource, S.Base, {
 38         /**
 39          * Datasource interface. How to get data for autoComplete
 40          * @function
 41          * @name AutoComplete.RemoteDataSource#fetchData
 42          * @param {String} inputVal current active input's value
 43          * @param {Function} callback callback to notify autoComplete when data is ready
 44          * @param {Object} context callback's execution context
 45          */
 46         fetchData:function (inputVal, callback, context) {
 47             var self = this,
 48                 v,
 49                 dataSourceCfg = self.get("dataSourceCfg");
 50             if (self.io) {
 51                 // abort previous request
 52                 self.io.abort();
 53                 self.io = null;
 54             }
 55             if (!inputVal && dataSourceCfg.allowEmpty !== true) {
 56                 return callback.call(context, []);
 57             }
 58             if (dataSourceCfg.cache) {
 59                 if (v = self.caches[inputVal]) {
 60                     return callback.call(context, v);
 61                 }
 62             }
 63             var xhrCfg = self.get("xhrCfg");
 64             xhrCfg.data = xhrCfg.data || {};
 65             xhrCfg.data[dataSourceCfg['paramName']] = inputVal;
 66             xhrCfg.success = function (data) {
 67                 if (dataSourceCfg.parse) {
 68                     data = dataSourceCfg.parse(inputVal, data);
 69                 }
 70                 self.__set("data", data);
 71                 if (dataSourceCfg.cache) {
 72                     self.caches[inputVal] = data;
 73                 }
 74                 callback.call(context, data);
 75             };
 76             self.io = IO(xhrCfg);
 77         }
 78     });
 79 
 80     return RemoteDataSource;
 81 }, {
 82     requires:['ajax']
 83 });
