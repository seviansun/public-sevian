  5 KISSY.add('dom/traversal', function (S, DOM, undefined) {
  6 
  7     var isElementNode = DOM._isElementNode,
  8         doc = S.Env.host.document,
  9         CONTAIN_MASK = 16,
 10         __contains = doc.documentElement.contains ?
 11             function (a, b) {
 12                 if (a.nodeType == DOM.TEXT_NODE) {
 13                     return false;
 14                 }
 15                 var precondition;
 16                 if (b.nodeType == DOM.TEXT_NODE) {
 17                     b = b.parentNode;
 18                     // a 和 b父亲相等也就是返回 true
 19                     precondition = true;
 20                 } else if (b.nodeType == DOM.DOCUMENT_NODE) {
 21                     // b === document
 22                     // 没有任何元素能包含 document
 23                     return false;
 24                 } else {
 25                     // a 和 b 相等返回 false
 26                     precondition = a !== b;
 27                 }
 28                 // !a.contains => a===document
 29                 // 注意原生 contains 判断时 a===b 也返回 true
 30                 return precondition && (a.contains ? a.contains(b) : true);
 31             } : (
 32             doc.documentElement.compareDocumentPosition ?
 33                 function (a, b) {
 34                     return !!(a.compareDocumentPosition(b) & CONTAIN_MASK);
 35                 } :
 36                 // it can not be true , pathetic browser
 37                 0
 38             );
 39 
 40 
 41     S.mix(DOM,
 42         /**
 43          * @lends DOM
 44          */
 45         {
 46 
 47             /**
 48              * Get the first element that matches the filter,
 49              * beginning at the first element of matched elements and progressing up through the DOM tree.
 50              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
 51              * @param {String|Function} filter Selector string or filter function
 52              * @param {HTMLElement|String|Document|HTMLElement[]} [context] Search bound element
 53              * @returns {HTMLElement}
 54              */
 55             closest:function (selector, filter, context) {
 56                 return nth(selector, filter, 'parentNode', function (elem) {
 57                     return elem.nodeType != DOM.DOCUMENT_FRAGMENT_NODE;
 58                 }, context, true);
 59             },
 60 
 61             /**
 62              * Get the parent of the first element in the current set of matched elements, optionally filtered by a selector.
 63              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
 64              * @param {String|Function} [filter] Selector string or filter function
 65              * @param {HTMLElement|String|Document|HTMLElement[]} [context] Search bound element
 66              * @returns {HTMLElement}
 67              */
 68             parent:function (selector, filter, context) {
 69                 return nth(selector, filter, 'parentNode', function (elem) {
 70                     return elem.nodeType != DOM.DOCUMENT_FRAGMENT_NODE;
 71                 }, context);
 72             },
 73 
 74             /**
 75              * Get the first child of the first element in the set of matched elements.
 76              * If a filter is provided, it retrieves the next child only if it matches that filter.
 77              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
 78              * @param {String|Function} [filter] Selector string or filter function
 79              * @returns {HTMLElement}
 80              */
 81             first:function (selector, filter) {
 82                 var elem = DOM.get(selector);
 83                 return nth(elem && elem.firstChild, filter, 'nextSibling',
 84                     undefined, undefined, true);
 85             },
 86 
 87             /**
 88              * Get the last child of the first element in the set of matched elements.
 89              * If a filter is provided, it retrieves the previous child only if it matches that filter.
 90              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
 91              * @param {String|Function} [filter] Selector string or filter function
 92              * @returns {HTMLElement}
 93              */
 94             last:function (selector, filter) {
 95                 var elem = DOM.get(selector);
 96                 return nth(elem && elem.lastChild, filter, 'previousSibling',
 97                     undefined, undefined, true);
 98             },
 99 
100             /**
101              * Get the immediately following sibling of the first element in the set of matched elements.
102              * If a filter is provided, it retrieves the next child only if it matches that filter.
103              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
104              * @param {String|Function} [filter] Selector string or filter function
105              * @returns {HTMLElement}
106              */
107             next:function (selector, filter) {
108                 return nth(selector, filter, 'nextSibling', undefined);
109             },
110 
111             /**
112              * Get the immediately preceding  sibling of the first element in the set of matched elements.
113              * If a filter is provided, it retrieves the previous child only if it matches that filter.
114              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
115              * @param {String|Function} [filter] Selector string or filter function
116              * @returns {HTMLElement}
117              */
118             prev:function (selector, filter) {
119                 return nth(selector, filter, 'previousSibling', undefined);
120             },
121 
122             /**
123              * Get the siblings of the first element in the set of matched elements, optionally filtered by a filter.
124              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
125              * @param {String|Function} [filter] Selector string or filter function
126              * @returns {HTMLElement[]}
127              */
128             siblings:function (selector, filter) {
129                 return getSiblings(selector, filter, true);
130             },
131 
132             /**
133              * Get the children of the first element in the set of matched elements, optionally filtered by a filter.
134              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
135              * @param {String|Function} [filter] Selector string or filter function
136              * @returns {HTMLElement[]}
137              */
138             children:function (selector, filter) {
139                 return getSiblings(selector, filter, undefined);
140             },
141 
142             /**
143              * Get the childrNodes of the first element in the set of matched elements (includes text and comment nodes),
144              * optionally filtered by a filter.
145              * @param {HTMLElement[]|String|HTMLElement} selector Matched elements
146              * @param {String|Function} [filter] Selector string or filter function
147              * @returns {Node[]}
148              */
149             contents:function (selector, filter) {
150                 return getSiblings(selector, filter, undefined, 1);
151             },
152 
153             /**
154              * Check to see if a DOM node is within another DOM node.
155              * @param {HTMLElement|String|Element} container The DOM element that may contain the other element.
156              * @param {HTMLElement|String|Element} contained The DOM element that may be contained by the other element.
157              * @returns {Boolean}
158              */
159             contains:function (container, contained) {
160                 container = DOM.get(container);
161                 contained = DOM.get(contained);
162                 if (container && contained) {
163                     return __contains(container, contained);
164                 }
165                 return false;
166             },
167 
168             /**
169              * Check to see if a DOM node is equal with another DOM node.
170              * @param {HTMLElement|String|Element} n1
171              * @param {HTMLElement|String|Element} n2
172              * @returns {Boolean}
173              */
174             equals:function (n1, n2) {
175                 n1 = DOM.query(n1);
176                 n2 = DOM.query(n2);
177                 if (n1.length != n2.length) {
178                     return false;
179                 }
180                 for (var i = n1.length; i >= 0; i--) {
181                     if (n1[i] != n2[i]) {
182                         return false;
183                     }
184                 }
185                 return true;
186             }
187         });

189     // 获取元素 elem 在 direction 方向上满足 filter 的第一个元素
190     // filter 可为 number, selector, fn array ，为数组时返回多个
191     // direction 可为 parentNode, nextSibling, previousSibling
192     // context : 到某个阶段不再查找直接返回
193     function nth(elem, filter, direction, extraFilter, context, includeSef) {
194         if (!(elem = DOM.get(elem))) {
195             return null;
196         }
197         if (filter === 0) {
198             return elem;
199         }
200         if (!includeSef) {
201             elem = elem[direction];
202         }
203         if (!elem) {
204             return null;
205         }
206         context = (context && DOM.get(context)) || null;
207 
208         if (filter === undefined) {
209             // 默认取 1
210             filter = 1;
211         }
212         var ret = [],
213             isArray = S.isArray(filter),
214             fi,
215             flen;
216 
217         if (S.isNumber(filter)) {
218             fi = 0;
219             flen = filter;
220             filter = function () {
221                 return ++fi === flen;
222             };
223         }
224 
225         // 概念统一，都是 context 上下文，只过滤子孙节点，自己不管
226         while (elem && elem != context) {
227             if (isElementNode(elem)
228                 && testFilter(elem, filter)
229                 && (!extraFilter || extraFilter(elem))) {
230                 ret.push(elem);
231                 if (!isArray) {
232                     break;
233                 }
234             }
235             elem = elem[direction];
236         }
237 
238         return isArray ? ret : ret[0] || null;
239     }
240 
241     function testFilter(elem, filter) {
242         if (!filter) {
243             return true;
244         }
245         if (S.isArray(filter)) {
246             for (var i = 0; i < filter.length; i++) {
247                 if (DOM.test(elem, filter[i])) {
248                     return true;
249                 }
250             }
251         } else if (DOM.test(elem, filter)) {
252             return true;
253         }
254         return false;
255     }
256 
257     // 获取元素 elem 的 siblings, 不包括自身
258     function getSiblings(selector, filter, parent, allowText) {
259         var ret = [],
260             elem = DOM.get(selector),
261             parentNode = elem;
262 
263         if (elem && parent) {
264             parentNode = elem.parentNode;
265         }
266 
267         if (parentNode) {
268             ret = S.makeArray(parentNode.childNodes);
269             if (!allowText) {
270                 ret = DOM.filter(ret, function (el) {
271                     return el.nodeType == 1;
272                 });
273             }
274             if (filter) {
275                 ret = DOM.filter(ret, filter);
276             }
277         }
278 
279         return ret;
280     }
281 
282     return DOM;
