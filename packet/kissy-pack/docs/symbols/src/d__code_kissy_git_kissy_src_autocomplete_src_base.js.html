1 /**
  2  * autoComplete logic . control many inputs and only use one menu for performance
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("autocomplete/base", function (S, Event, UIBase, Menu) {
  6     var ALIGN = {
  7         points:["bl", "tl"],
  8         overflow:{
  9             failX:1,
 10             failY:1,
 11             adjustX:1,
 12             adjustY:1
 13         }
 14     },
 15         KeyCodes = Event.KeyCodes;
 16 
 17     function reAlign2() {
 18         var self = this,
 19             menu = self._menu,
 20             _input = self._input;
 21         if (_input && menu && menu.get("visible")) {
 22             var menuCfg = self.get("menuCfg") || {};
 23             menu.set("align", S.merge({
 24                 node:_input.get("el")
 25             }, ALIGN, menuCfg.align));
 26         }
 27     }
 28 
 29     var reAlign = S.buffer(reAlign2, 50);
 30 
 31     /**
 32      * AutoComplete provides autocomplete logic
 33      * for many text input fields or textareas and one menu
 34      * @name AutoComplete
 35      * @extends UIBase
 36      * @class
 37      */
 38     return UIBase.create([],
 39         /**
 40          * @lends AutoComplete#
 41          */
 42         {
 43             __CLASS:"AutoComplete",
 44             // drop down menu
 45             _menu:null,
 46 
 47             // current input
 48             _input:null,
 49 
 50             _inputs:null,
 51 
 52             // user's input text
 53             _savedInputValue:null,
 54 
 55             /**
 56              * attach one input or textarea to this autoComplete logic
 57              * @param {AutoComplete.Input} input input or textarea wrapper instance
 58              */
 59             attachInput:function (input) {
 60                 this._inputs = this._inputs || [];
 61                 input.autoComplete = this;
 62                 this._inputs.push(input);
 63             },
 64 
 65             /**
 66              * detach existing input or textarea from this autoComplete logic
 67              * @param {AutoComplete.Input} input previous attached input or textarea instance
 68              */
 69             detachInput:function (input) {
 70                 var index = S.indexOf(input, this._inputs);
 71                 if (index != -1) {
 72                     this._inputs.splice(index, 1);
 73                 }
 74                 input.autoComplete = null;
 75                 input.destroy();
 76             },
 77 
 78             /**
 79              * fetch autoComplete list by value and show autoComplete list
 80              * @param {String} value value for fetching autoComplete list
 81              */
 82             sendRequest:function (value) {
 83                 if (!this._input) {
 84                     return;
 85                 }
 86                 var dataSource = this.get("dataSource");
 87                 this._savedInputValue = value;
 88                 dataSource.fetchData(value, this._renderData, this);
 89             },
 90 
 91             _renderData:function (data) {
 92                 var self = this;
 93                 if (data && data.length) {
 94                     data = data.slice(0, self.get("maxItemCount"));
 95                     var _input = self._input;
 96                     var menu = self._getMenu();
 97                     menu.removeChildren(true);
 98                     if (_input) {
 99                         var menuCfg = self.get("menuCfg") || {};
100                         if (!menuCfg.width) {
101                             // 需要放在这里
102                             // note width :
103                             // show 之前，addChild 之后，元素已经建立，改了不能同步到 render
104                             menu.set("width", _input.get("width"));
105                         }
106                     }
107                     var contents = data;
108                     if (self.get("format")) {
109                         contents = self.get("format").call(self, data);
110                     }
111                     for (var i = 0; i < data.length; i++) {
112                         menu.addChild(new Menu.Item({
113                             prefixCls:self.get("prefixCls") + "autocomplete-",
114                             content:contents[i],
115                             value:data[i]
116                         }))
117                     }
118                     self.set("open", true);
119                 } else {
120                     self.set("open", false);
121                 }
122             },
123 
124             _bindMenu:function () {
125                 var self = this, menu = self._menu;
126                 menu.on("show", function () {
127                     var input = self._input;
128                     input.set("ariaOwns", menu.get("el").attr("id"));
129                     input.set("ariaExpanded", true);
130                 });
131                 menu.on("hide", function () {
132                     var input = self._input;
133                     input.set("ariaOwns", menu.get("el").attr("id"));
134                     input.set("ariaExpanded", false);
135                 });
136                 menu.on("click", function (e) {
137                     var item = e.target;
138                     var content = item.get("content");
139                     var input = self._input;
140                     // stop valuechange event
141                     input._stopNotify = 1;
142                     input.get("el").val(content);
143                     self.set("open", false);
144                     setTimeout(function () {
145                             input._stopNotify = 0;
146 
147                             /**
148                              * @name AutoComplete#select
149                              * @description fired when user select from suggestion list
150                              * @event
151                              * @param e
152                              * @param e.value value of selected menuItem
153                              * @param e.content content of selected menuItem
154                              * @param e.input current active input
155                              */
156 
157                             self.fire("select", {
158                                 value:item.get("value"),
159                                 content:content,
160                                 input:input
161                             })
162                         },
163                         // valuechange interval
164                         50);
165                 });
166                 Event.on(window, "resize", reAlign, self);
167             },
168 
169             _getMenu:function () {
170                 var self = this;
171                 if (!self._menu) {
172                     var menuCfg = self.get("menuCfg") || {};
173                     self._menu = new Menu.PopupMenu({
174                         prefixCls:self.get("prefixCls") + "autocomplete-",
175                         width:menuCfg.width
176                     });
177                     self._bindMenu();
178                 }
179                 return self._menu;
180             },
181 
182             _onInputFocus:function (input) {
183                 this._input = input;
184             },
185 
186             _onPrepareCollapse:function () {
187                 var self = this;
188                 self.dismissTimer = setTimeout(function () {
189                     self.set("open", false);
190                 }, 100);
191             },
192 
193             _handleKeyEventInternal:function (e) {
194                 var menu = this._menu,
195                     input = this._input.get("el");
196                 // 转发给 menu 处理
197                 if (menu && menu.get("visible")) {
198                     var handledByMenu = menu._handleKeydown(e);
199                     if (S.inArray(e.keyCode, [KeyCodes.DOWN, KeyCodes.UP])) {
200                         // update menu's active value to input just for show
201                         input.val(menu.get("activeItem").get("content"))
202                     }
203                     // esc
204                     if (e.keyCode == KeyCodes.ESC) {
205                         this.set("open", false);
206                         // restore original user's input text
207                         input.val(this._savedInputValue);
208                         return true;
209                     }
210                     return handledByMenu;
211                 } else if (menu) {
212                     // display the list if the list is hidden
213                     if (e.keyCode == KeyCodes.DOWN || e.keyCode == KeyCodes.UP) {
214                         this.set("open", true);
215                         return true;
216                     }
217                 } else {
218                     // display the list if the list is hidden
219                     if (e.keyCode == KeyCodes.DOWN || e.keyCode == KeyCodes.UP) {
220                         this.sendRequest(input.val());
221                         return true;
222                     }
223                 }
224             },
225 
226             _uiSetOpen:function (d) {
227                 var self = this,
228                     menu = self._menu,
229                     _input = self._input;
230                 if (menu) {
231                     if (d && _input) {
232                         var menuCfg = self.get("menuCfg") || {};
233                         menu.set("align", S.merge({
234                             node:_input.get("el")
235                         }, ALIGN, menuCfg.align));
236                         menu.show();
237                     } else {
238                         menu.hide();
239                     }
240                 }
241             },
242 
243             destructor:function () {
244                 var self = this;
245                 Event.remove(window, "resize", reAlign, self);
246                 if (self._menu) {
247                     self._menu.destroy();
248                     self._menu = null;
249                 }
250                 S.each(self._inputs, function (inp) {
251                     inp.destroy();
252                 });
253                 self._inputs = null;
254             }
255         }, {
256             ATTRS:/**
257              * @lends AutoComplete#
258              */
259             {
260                 /**
261                  * AutoComplete Component's prefix class.For Configuration when new.
262                  * @type String
263                  */
264                 prefixCls:{
265                     value:'ks-'
266                 },
267                 /**
268                  * maxItemCount max count of data to be shown
269                  * @type Number
270                  */
271                 maxItemCount:{value:99999},
272                 /**
273                  * dataSource for autoComplete.For Configuration when new.
274                  * @type AutoComplete.LocalDataSource|AutoComplete.RemoteDataSource
275                  */
276                 dataSource:{},
277                 /**
278                  * Indicate whether autoComplete list is shown.
279                  * @type Boolean
280                  */
281                 open:{
282                 },
283                 /**
284                  * Config autoComplete menu list.For Configuration when new.
285                  * {Number} menuCfg.width :
286                  * Config autoComplete menu list's alignment.
287                  * Default to current active input's width.
288                  * <br/> menuCfg.align :
289                  * Config autoComplete menu list's alignment.
290                  * Same with {@link UIBase.Align#align} .
291                  * Default : align current input's bottom left edge
292                  * with autoComplete list's top left edge.
293                  * {Object}
294                  * @type Object
295                  */
296                 menuCfg:{
297                     value:{
298                         // width
299                         // align
300                     }
301                 },
302                 /**
303                  * Format function to return  array of html from array of data.
304                  * @type {Function}
305                  */
306                 format:{}
307             }
308         });
309 }, {
310     requires:['event', 'uibase', 'menu']
311 });
