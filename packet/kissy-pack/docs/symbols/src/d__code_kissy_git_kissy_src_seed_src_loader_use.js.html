 20     S.augment(Loader, {
 21         /**
 22          * Start load specific mods, and fire callback when these mods and requires are attached.
 23          * @example
 24          * <code>
 25          * S.use('mod-name', callback, config);
 26          * S.use('mod1,mod2', callback, config);
 27          * </code>
 28          * @param {String|String[]} modNames names of mods to be loaded,if string then separated by space
 29          * @param {Function} callback callback when modNames are all loaded,
 30          *                   with KISSY as first argument and mod's value as the following arguments
 31          */
 32         use:function (modNames, callback) {
 33             var self = this,
 34                 SS = self.SS;
 35 
 36             modNames = utils.getModNamesAsArray(modNames);
 37             modNames = utils.normalizeModNamesWithAlias(SS,modNames);
 38 
 39             var normalizedModNames = utils.normalizeModNames(SS, modNames),
 40                 count = normalizedModNames.length,
 41                 currentIndex = 0;
 42 
 43             // 已经全部 attached, 直接执行回调即可
 44             if (utils.isAttached(SS, normalizedModNames)) {
 45                 var mods = utils.getModules(SS, modNames);
 46                 callback && callback.apply(SS, mods);
 47                 return;
 48             }
 49 
 50             // 有尚未 attached 的模块
 51             S.each(normalizedModNames, function (modName) {
 52                 // 从 name 开始调用，防止不存在模块
 53                 attachModByName(self, modName, function () {
 54                     currentIndex++;
 55                     if (currentIndex == count) {
 56                         var mods = utils.getModules(SS, modNames);
 57                         callback && callback.apply(SS, mods);
 58                     }
 59                 });
 60             });
 61 
 62             return self;
 63         }
 64     });

 67     function buildModPath(self, mod, base) {
 68         var SS = self.SS,
 69             Config = SS.Config;
 70 
 71         base = base || Config.base;
 72 
 73         build("fullpath", "path");
 74 
 75         if (mod["cssfullpath"] !== data.LOADED) {
 76             build("cssfullpath", "csspath");
 77         }

 79         function build(fullpath, path) {
 80             var flag = "__" + fullpath + "Ready",
 81                 t,
 82                 p = mod[fullpath],
 83                 sp = mod[path];
 84             if (mod[flag]) {
 85                 return;
 86             }
 87             if (!p && sp) {
 88                 //如果是 ./ 或 ../ 则相对当前模块路径
 89                 sp = mod[path] = utils.normalDepModuleName(mod.name, sp);
 90                 p = base + sp;
 91             }
 92             // debug 模式下，加载非 min 版
 93             if (p) {
 94                 mod[fullpath] = utils.getMappedPath(SS, p +
 95                     ((t = mod.getUrlTag()) ? ("?t=" + t) : ""));
 96                 mod[flag] = 1;
 97             }
 98         }

101     // 加载指定模块名模块，如果不存在定义默认定义为内部模块
102     function attachModByName(self, modName, callback) {
103         var SS = self.SS,
104             mod = SS.Env.mods[modName];
105         if (mod.status === ATTACHED) {
106             callback();
107             return;
108         }
109         attachModRecursive(self, mod, callback);

113     /**
114      * Attach a module and all required modules.
115      */
116     function attachModRecursive(self, mod, callback) {
117         var SS = self.SS,
118             r,
119             rMod,
120             i,
121             callbackBeCalled = 0,
122             // 最终有效的 require ，add 处声明为准
123             newRequires,
124             mods = SS.Env.mods;
125 
126         // 复制一份当前的依赖项出来，防止 add 后修改！
127         // 事先配置的 require ，同 newRequires 有区别
128         var requires = utils.normalizeModNames(SS, mod.requires, mod.name);
129 
130         /**
131          * check cyclic dependency between mods
132          * @private
133          */
134         function cyclicCheck() {
135             // one mod's all requires mods to run its callback
136             var __allRequires = mod.__allRequires = mod.__allRequires || {},
137                 myName = mod.name,
138                 rmod,
139                 r__allRequires;
140 
141             S.each(requires, function (r) {
142                 rmod = mods[r];
143                 __allRequires[r] = 1;
144                 if (rmod && (r__allRequires = rmod.__allRequires)) {
145                     S.mix(__allRequires, r__allRequires);
146                 }
147             });
148 
149             if (__allRequires[myName]) {
150                 S.log(__allRequires, "error");
151                 var JSON = window.JSON,
152                     error = "";
153                 if (JSON) {
154                     error = JSON.stringify(__allRequires);
155                 }
156                 S.error("find cyclic dependency by mod " + myName + " between mods : " + error);
157 
158             }
159         }
160 
161         if (S.Config.debug) {
162             cyclicCheck();
163         }
164 
165         // attach all required modules
166         for (i = 0; i < requires.length; i++) {
167             r = requires[i];
168             rMod = mods[r];
169             if (rMod && rMod.status === ATTACHED) {
170                 //no need
171             } else {
172                 attachModByName(self, r, fn);
173             }
174         }
175 
176         // load and attach this module
177         buildModPath(self, mod, utils.getPackagePath(SS, mod));
178 
179         loadModByScript(self, mod, function () {
180 
181             // KISSY.add 可能改了 config，这里重新取下
182             newRequires = utils.normalizeModNames(SS, mod.requires, mod.name);
183 
184             var needToLoad = [];
185 
186             //本模块下载成功后串行下载 require
187             for (i = 0; i < newRequires.length; i++) {
188                 var r = newRequires[i],
189                     rMod = mods[r],
190                     inA = S.inArray(r, requires);
191                 //已经处理过了或将要处理
192                 if (rMod &&
193                     rMod.status === ATTACHED ||
194                     //已经正在处理了
195                     inA) {
196                     //no need
197                 } else {
198                     //新增的依赖项
199                     needToLoad.push(r);
200                 }
201             }
202 
203             if (needToLoad.length) {
204                 for (i = 0; i < needToLoad.length; i++) {
205                     attachModByName(self, needToLoad[i], fn);
206                 }
207             } else {
208                 fn();
209             }
210         });
211 
212         function fn() {
213             if (
214             // 前提条件，本模块 script onload 已经调用
215             // ie 下 add 与 script onload 并不连续！！
216             // attach 以 newRequires 为准
217                 newRequires &&
218                     !callbackBeCalled &&
219                     // 2012-03-16 by yiminghe@gmail.com
220                     // add 与 onload ie 下不连续
221                     // c 依赖 a
222                     // a 模块 add 时进行 attach
223                     // a add 后 c 模块 onload 触发
224                     // 检测到 a 已经 attach 则调用该函数
225                     // a onload 后又调用该函数则需要用 callbackBeCalled 来把门
226                     utils.isAttached(SS, newRequires)) {
227                 if (mod.status == LOADED) {
228                     utils.attachMod(SS, mod);
229                 }
230                 if (mod.status == ATTACHED) {
231                     callbackBeCalled = 1;
232                     callback();
233                 }
234             }
235         }
236     }

239     /**
240      * Load a single module.
241      */
242     function loadModByScript(self, mod, callback) {
243         var SS = self.SS,
244             cssfullpath,
245             url = mod['fullpath'],
246             isCss = utils.isCss(url)
247 
248         mod.status = mod.status || INIT;
249 
250         // 1.20 兼容 1.1x 处理：加载 cssfullpath 配置的 css 文件
251         // 仅发出请求，不做任何其它处理
252         if (cssfullpath = mod["cssfullpath"]) {
253             S.getScript(cssfullpath);
254             mod["cssfullpath"] = mod.csspath = LOADED;
255         }
256 
257         if (mod.status < LOADING) {
258             mod.status = LOADING;
259             if (IE && !isCss) {
260                 self.__startLoadModuleName = mod.name;
261                 self.__startLoadTime = Number(+new Date());
262             }
263             S.getScript(url, {
264                 // syntaxError in all browser will trigger this
265                 // same as #111 : https://github.com/kissyteam/kissy/issues/111
266                 success:function () {
267                     if (!isCss) {
268                         // 载入 css 不需要这步了
269                         // 标准浏览器下：外部脚本执行后立即触发该脚本的 load 事件,ie9 还是不行
270                         if (self.__currentModule) {
271                             S.log("standard browser get modname after load : " + mod.name);
272                             utils.registerModule(SS,
273                                 mod.name, self.__currentModule.def,
274                                 self.__currentModule.config);
275                             self.__currentModule = null;
276                         }
277                     }
278                     checkAndHandle();
279                 },
280                 error:checkAndHandle,
281                 // source:mod.name + "-init",
282                 charset:mod.charset
283             });
284         }
285         // 已经在加载中，需要添加回调到 script onload 中
286         // 注意：没有考虑 error 情形，只在第一次处理即可
287         // 交给 getScript 排队
288         else if (mod.status == LOADING) {
289             S.getScript(url, {
290                 success:checkAndHandle,
291                 // source:mod.name + "-loading",
292                 charset:mod.charset
293             });
294         }
295         // loaded/attached/error
296         else {
297             checkAndHandle();
298         }
299 
300         function checkAndHandle() {
301             if (isCss || mod.fn) {
302                 // css 不会设置 LOADED! 必须外部设置
303                 if (isCss && mod.status != ATTACHED) {
304                     mod.status = LOADED;
305                 }
306                 callback();
307             } else {
308                 // ie will call success even when getScript error(404)
309                 _modError();
310             }
311         }
312 
313         function _modError() {
314             S.log(mod.name + ' is not loaded! can not find module in path : ' + mod['fullpath'], 'error');
315             mod.status = ERROR;
316         }
