1 /**
  2  * @fileOverview close extension for kissy dialog
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("uibase/close", function () {
  6 
  7     /**
  8      * config detail of close action
  9      * @class
 10      * @memberOf UIBase
 11      */
 12     function Close() {
 13     }
 14 
 15     var HIDE = "hide";
 16     Close.ATTRS =
 17     /**
 18      * @lends UIBase.Close.prototype
 19      */
 20     {
 21         /**
 22          * 是否自带关闭按钮
 23          * @type boolean
 24          */
 25         closable:{
 26             view:true
 27         },
 28 
 29         /**
 30          * 点击关闭按钮的动作，销毁("destroy")或隐藏("hide")
 31          * @type string
 32          */
 33         closeAction:{
 34             value:HIDE
 35         }
 36     };
 37 
 38     var actions = {
 39         hide:HIDE,
 40         destroy:"destroy"
 41     };
 42 
 43     Close.prototype = {
 44 
 45         __bindUI:function () {
 46 
 47             var self = this,
 48                 closeBtn = self.get("view").get("closeBtn");
 49             closeBtn && closeBtn.on("click", function (ev) {
 50                 self[actions[self.get("closeAction")] || HIDE]();
 51                 ev.preventDefault();
 52             });
 53         }
 54     };
 55     return Close;
 56 
 57 });
