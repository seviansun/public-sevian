  6 
  7     if (typeof require !== 'undefined') {
  8         return;

 11     var Loader = S.Loader,
 12         utils = Loader.Utils;
 13 
 14 
 15     S.augment(Loader,
 16         Loader.Target,
 17         {
 18 
 19             //firefox,ie9,chrome 如果add没有模块名，模块定义先暂存这里
 20             __currentModule:null,
 21 
 22             //ie6,7,8开始载入脚本的时间
 23             __startLoadTime:0,
 24 
 25             //ie6,7,8开始载入脚本对应的模块名
 26             __startLoadModuleName:null,
 27 
 28             /**
 29              * Registers a module.
 30              * @param {String|Object} [name] module name
 31              * @param {Function|Object} [def] entry point into the module that is used to bind module to KISSY
 32              * @param {Object} [config] special config for this add
 33              * @param {String[]} [config.requires] array of mod's name that current module requires
 34              * @example
 35              * <code>
 36              * KISSY.add('module-name', function(S){ }, {requires: ['mod1']});
 37 
 38              * KISSY.add({
 39              *     'mod-name': {
 40              *         fullpath: 'url',
 41              *         requires: ['mod1','mod2']
 42              *     }
 43              * });
 44              * </code>
 45              */
 46             add:function (name, def, config) {
 47                 var self = this,
 48                     SS = self.SS,
 49                     mod,
 50                     requires,
 51                     mods = SS.Env.mods,
 52                     o;
 53 
 54                 if (utils.normAdd(SS, name, def, config)) {
 55                     return;
 56                 }
 57 
 58                 // S.add(name[, fn[, config]])
 59                 if (S.isString(name)) {
 60 
 61                     utils.registerModule(SS, name, def, config);
 62 
 63                     mod = mods[name];
 64 
 65                     // 显示指定 add 不 attach
 66                     if (config && config['attach'] === false) {
 67                         return;
 68                     }
 69 
 70                     requires = utils.normalizeModNames(SS, mod.requires, name);
 71 
 72                     if (utils.isAttached(SS, requires)) {
 73                         utils.attachMod(SS, mod);
 74                     }
 75                     return;
 76                 }
 77                 // S.add(fn,config);
 78                 else if (S.isFunction(name)) {
 79                     config = def;
 80                     def = name;
 81                     if (utils.IE) {
 82                         /*
 83                          Kris Zyp
 84                          2010年10月21日, 上午11时34分
 85                          We actually had some discussions off-list, as it turns out the required
 86                          technique is a little different than described in this thread. Briefly,
 87                          to identify anonymous modules from scripts:
 88                          * In non-IE browsers, the onload event is sufficient, it always fires
 89                          immediately after the script is executed.
 90                          * In IE, if the script is in the cache, it actually executes *during*
 91                          the DOM insertion of the script tag, so you can keep track of which
 92                          script is being requested in case define() is called during the DOM
 93                          insertion.
 94                          * In IE, if the script is not in the cache, when define() is called you
 95                          can iterate through the script tags and the currently executing one will
 96                          have a script.readyState == "interactive"
 97                          See RequireJS source code if you need more hints.
 98                          Anyway, the bottom line from a spec perspective is that it is
 99                          implemented, it works, and it is possible. Hope that helps.
100                          Kris
101                          */
102                         // http://groups.google.com/group/commonjs/browse_thread/thread/5a3358ece35e688e/43145ceccfb1dc02#43145ceccfb1dc02
103                         // use onload to get module name is not right in ie
104                         name = findModuleNameByInteractive(self);
105                         S.log("old_ie get modname by interactive : " + name);
106                         utils.registerModule(SS, name, def, config);
107                         self.__startLoadModuleName = null;
108                         self.__startLoadTime = 0;
109                     } else {
110                         // 其他浏览器 onload 时，关联模块名与模块定义
111                         self.__currentModule = {
112                             def:def,
113                             config:config
114                         };
115                     }
116                     return;
117                 }
118                 S.log("invalid format for KISSY.add !", "error");
119             }
120         });
121 
122 
123 
124 
125     // ie 特有，找到当前正在交互的脚本，根据脚本名确定模块名
126     // 如果找不到，返回发送前那个脚本
127     function findModuleNameByInteractive(self) {
128         var self = this,
129             SS = self.SS,
130             base,
131             scripts = S.Env.host.document.getElementsByTagName("script"),
132             re,
133             script;
134 
135         for (var i = 0; i < scripts.length; i++) {
136             script = scripts[i];
137             if (script.readyState == "interactive") {
138                 re = script;
139                 break;
140             }
141         }
142         if (!re) {
143             // sometimes when read module file from cache , interactive status is not triggered
144             // module code is executed right after inserting into dom
145             // i has to preserve module name before insert module script into dom , then get it back here
146             S.log("can not find interactive script,time diff : " + (+new Date() - self.__startLoadTime), "error");
147             S.log("old_ie get modname from cache : " + self.__startLoadModuleName);
148             return self.__startLoadModuleName;
149             //S.error("找不到 interactive 状态的 script");
150         }
151 
152         // src 必定是绝对路径
153         // or re.hasAttribute ? re.src :  re.getAttribute('src', 4);
154         // http://msdn.microsoft.com/en-us/library/ms536429(VS.85).aspx
155         var src = utils.absoluteFilePath(re.src);
156         // 注意：模块名不包含后缀名以及参数，所以去除
157         // 系统模块去除系统路径
158         // 需要 base norm , 防止 base 被指定为相对路径
159         // configs 统一处理
160         // SS.Config.base = SS.normalBasePath(self.Config.base);
161         if (src.lastIndexOf(base = SS.Config.base, 0) === 0) {
162             return utils.removePostfix(src.substring(base.length));
163         }
164         var packages = SS.Config.packages;
165         //外部模块去除包路径，得到模块名
166         for (var p in packages) {
167             if (packages.hasOwnProperty(p)) {
168                 var p_path = packages[p].path;
169                 if (packages.hasOwnProperty(p) &&
170                     src.lastIndexOf(p_path, 0) === 0) {
171                     return utils.removePostfix(src.substring(p_path.length));
172                 }
173             }
174         }
175         S.log("interactive script does not have package config ：" + src, "error");
176     }
177 
178 })(KISSY);
