1 /**
  2  * @fileOverview deletable menuitem
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/delmenuitem", function(S, Node, UIBase, Component, MenuItem, DelMenuItemRender) {
  6     var $ = Node.all;
  7     var CLS = DelMenuItemRender.CLS,
  8         DEL_CLS = DelMenuItemRender.DEL_CLS;
  9 
 10     function del(self) {
 11         var parent = self.get("parent");
 12         if (parent.fire("beforeDelete", {
 13             target:self
 14         }) === false) {
 15             return;
 16         }
 17         parent.removeChild(self, true);
 18         parent.set("highlightedItem", null);
 19         parent.fire("delete", {
 20             target:self
 21         });
 22     }
 23 
 24     var DelMenuItem = UIBase.create(MenuItem, {
 25         _performInternal:function(e) {
 26             var target = $(e.target);
 27             // 点击了删除
 28             if (target.hasClass(this.getCls(DEL_CLS))) {
 29                 del(this);
 30                 return true;
 31             }
 32             return MenuItem.prototype._performInternal.call(this, e);
 33         },
 34         _handleKeydown:function(e) {
 35             // d 键
 36             if (e.keyCode === Node.KeyCodes.D) {
 37                 del(this);
 38                 return true;
 39             }
 40         }
 41     }, {
 42         ATTRS:{
 43             delTooltip:{
 44                 view:true
 45             }
 46         },
 47         DefaultRender:DelMenuItemRender
 48     });
 49 
 50 
 51     Component.UIStore.setUIByClass(CLS, {
 52         priority:Component.UIStore.PRIORITY.LEVEL4,
 53         ui:DelMenuItem
 54     });
 55     return DelMenuItem;
 56 }, {
 57     requires:['node','uibase','component','./menuitem','./delmenuitemrender']
 58 });
