1 /**
  2  *  @fileOverview mix loader into S and infer KISSy baseUrl if not set
  3  *  @author  lifesinger@gmail.com,yiminghe@gmail.com
  4  */
  5 (function (S) {
  6 
  7     if (typeof require !== 'undefined') {
  8         return;
  9     }
 10 
 11     var Loader = S.Loader,
 12         utils = Loader.Utils,
 13         ComboLoader = S.Loader.Combo;
 14 
 15     S.mix(S,
 16         /**
 17          * @lends KISSY
 18          */
 19         {
 20             /**
 21              * Registers a module with the KISSY global.
 22              * @param {String} [name] module name.
 23              * it must be set if combine is true in {@link KISSY.config}
 24              * @param {Function} def module definition function that is used to return
 25              * this module value
 26              * @param {KISSY} def.S KISSY global instance
 27              * @param def.x... this module's required modules' value
 28              * @param {Object} [cfg] module optional config data
 29              * @param {String[]} cfg.requires this module's required module name list
 30              * @example
 31              * // dom module's definition
 32              * <code>
 33              * KISSY.add("dom",function(S,UA){
 34              *  return { css:function(el,name,val){} };
 35              * },{
 36              *  requires:["ua"]
 37              * });
 38              * </code>
 39              */
 40             add:function (name, def, cfg) {
 41                 this.getLoader().add(name, def, cfg);
 42             },
 43             /**
 44              * Attached one or more modules to global KISSY instance.
 45              * @param {String|String[]} names moduleNames. 1-n modules to bind(use comma to separate)
 46              * @param {Function} callback callback function executed
 47              * when KISSY has the required functionality.
 48              * @param {KISSY} callback.S KISSY instance
 49              * @param callback.x... used module values
 50              * @example
 51              * // loads and attached overlay,dd and its dependencies
 52              * KISSY.use("overlay,dd",function(S,Overlay){});
 53              */
 54             use:function (names, callback) {
 55                 this.getLoader().use(names, callback);
 56             },
 57             /**
 58              * get KISSY's loader instance
 59              * @returns {KISSY.Loader}
 60              */
 61             getLoader:function () {
 62                 var self = this;
 63                 if (self.Config.combine) {
 64                     return self.Env._comboLoader;
 65                 } else {
 66                     return self.Env._loader;
 67                 }
 68             },
 69             /**
 70              * get module value defined by define function
 71              * @param {string} moduleName
 72              * @private
 73              */
 74             require:function (moduleName) {
 75                 var self = this,
 76                     mods = self.Env.mods,
 77                     mod = mods[moduleName];
 78                 return mod && mod.value;
 79             }
 80         });
 81 
 82 
 83     // notice: timestamp
 84     var baseReg = /^(.*)(seed|kissy)(-aio)?(-min)?\.js[^/]*/i,
 85         baseTestReg = /(seed|kissy)(-aio)?(-min)?\.js/i;
 86 
 87     /**
 88      * get base from src
 89      * @param script script node
 90      * @return base for kissy
 91      * @private
 92      * @example
 93      * <pre>
 94      *   http://a.tbcdn.cn/s/kissy/1.1.6/??kissy-min.js,suggest/suggest-pkg-min.js
 95      *   http://a.tbcdn.cn/??s/kissy/1.1.6/kissy-min.js,s/kissy/1.1.5/suggest/suggest-pkg-min.js
 96      *   http://a.tbcdn.cn/??s/kissy/1.1.6/suggest/suggest-pkg-min.js,s/kissy/1.1.5/kissy-min.js
 97      *   http://a.tbcdn.cn/s/kissy/1.1.6/kissy-min.js?t=20101215.js
 98      *  note about custom combo rules, such as yui3:
 99      *   <script src="path/to/kissy" data-combo-prefix="combo?" data-combo-sep="&"></script>
100      * <pre>
101      */
102     function getBaseUrl(script) {
103         var src = utils.absoluteFilePath(script.src),
104             prefix = S.Config.comboPrefix = script.getAttribute('data-combo-prefix') || '??',
105             sep = S.Config.comboSep = script.getAttribute('data-combo-sep') || ',',
106             parts = src.split(sep),
107             base,
108             part0 = parts[0],
109             index = part0.indexOf(prefix);
110 
111         // no combo
112         if (index == -1) {
113             base = src.replace(baseReg, '$1');
114         } else {
115             base = part0.substring(0, index);
116             var part01 = part0.substring(index + 2, part0.length);
117             // combo first
118             // notice use match better than test
119             if (part01.match(baseTestReg)) {
120                 base += part01.replace(baseReg, '$1');
121             }
122             // combo after first
123             else {
124                 S.each(parts, function (part) {
125                     if (part.match(baseTestReg)) {
126                         base += part.replace(baseReg, '$1');
127                         return false;
128                     }
129                 });
130             }
131         }
132         return base;
133     }
134 
135     /**
136      * Initializes loader.
137      */
138     function initLoader() {
139         var self = this, env = self.Env;
140         env.mods = env.mods || {}; // all added mods
141         env._loader = new Loader(self);
142         env._comboLoader = new ComboLoader(self);
143     }
144 
145     // get base from current script file path
146     var scripts = S.Env.host.document.getElementsByTagName('script');
147 
148     S.config({
149         base:getBaseUrl(scripts[scripts.length - 1])
150     });
151 
152     // the default timeout for getScript
153     S.Config.timeout = 10;
154 
155     S.Config.tag = S.__BUILD_TIME;
156 
157     initLoader.call(S);
158 
159     // for S.app working properly
160     S.__APP_MEMBERS.push("add", "use", "require");
161 
162     S.__APP_INIT_METHODS.push(initLoader);
163 
164 })(KISSY);
