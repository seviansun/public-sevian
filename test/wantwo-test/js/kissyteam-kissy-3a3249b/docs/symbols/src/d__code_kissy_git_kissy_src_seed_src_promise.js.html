  7     var S = KISSY;

  9     function nextTick(fn) {
 10         // for debug
 11         // fn();
 12         // make parallel call in production
 13         // setTimeout(fn, 0);
 14         // sync,same with event
 15         fn();
 16     }
 17 
 18     /**
 19      * @class Defer constructor For KISSY,implement Promise specification.
 20      * @memberOf KISSY
 21      */
 22     function Defer(promise) {
 23         var self = this;
 24         if (!(self instanceof Defer)) {
 25             return new Defer(promise);
 26         }
 27         // http://en.wikipedia.org/wiki/Object-capability_model
 28         // principal of least authority
 29         /**
 30          * @description defer object's promise
 31          * @type KISSY.Promise
 32          * @memberOf KISSY.Defer#
 33          * @name promise
 34          */
 35         self.promise = promise || new Promise();
 36     }
 37 
 38     Defer.prototype =
 39     /**
 40      * @lends KISSY.Defer.prototype
 41      */
 42     {
 43         constructor:Defer,
 44         /**
 45          * fulfill defer object's promise
 46          * note: can only be called once
 47          * @param value defer object's value
 48          * @returns defer object's promise
 49          */
 50         resolve:function (value) {
 51             var promise = this.promise,
 52                 pendings;
 53             if (!(pendings = promise._pendings)) {
 54                 return undefined;
 55             }
 56             // set current promise's resolved value
 57             // maybe a promise or instant value
 58             promise._value = value;
 59             pendings = [].concat(pendings);
 60             promise._pendings = undefined;
 61             for (var i = 0; i < pendings.length; i++) {
 62                 (function (p) {
 63                     nextTick(function () {
 64                         promise._when(p[0], p[1]);
 65                     });
 66                 })(pendings[i]);
 67             }
 68             return value;
 69         },
 70         /**
 71          * reject defer object's promise
 72          * @param reason
 73          * @returns defer object's promise
 74          */
 75         reject:function (reason) {
 76             return this.resolve(new Reject(reason));
 77         }
 78     };
 79 
 80     function isPromise(obj) {
 81         return  obj && obj instanceof Promise;
 82     }
 83 
 84     /**
 85      * @class Promise constructor.
 86      * This class should not be instantiated manually.
 87      * Instances will be created and returned as needed by {@link KISSY.Defer#promise}
 88      * @namespace
 89      * @param v promise's resolved value
 90      * @memberOf KISSY
 91      */
 92     function Promise(v) {
 93         var self = this;
 94         // maybe internal value is also a promise
 95         self._value = v;
 96         if (!arguments.length) {
 97             self._pendings = [];
 98         }
 99     }
100 
101     Promise.prototype =
102     /**
103      * @lends KISSY.Promise.prototype
104      */
105     {
106         constructor:Promise,
107         /**
108          * two effects:
109          * 1. call fulfilled with immediate value
110          * 2. push fulfilled in right promise
111          * @private
112          * @param fulfilled
113          * @param rejected
114          */
115         _when:function (fulfilled, rejected) {
116             var promise = this,
117                 v = promise._value,
118                 pendings = promise._pendings;
119             // unresolved
120             // pushed to pending list
121             if (pendings) {
122                 pendings.push([fulfilled, rejected]);
123             }
124             // rejected or nested promise
125             else if (isPromise(v)) {
126                 nextTick(function () {
127                     v._when(fulfilled, rejected);
128                 });
129             } else {
130                 // fulfilled value
131                 // normal value represents ok
132                 // need return user's return value
133                 // if return promise then forward
134                 return fulfilled && fulfilled(v);
135             }
136             return undefined;
137         },
138         /**
139          * register callbacks when this promise object is resolved
140          * @param {Function(*)} fulfilled called when resolved successfully,pass a resolved value to this function and
141          *                      return a value (could be promise object) for the new promise's resolved value.
142          * @param {Function(*)} [rejected] called when error occurs,pass error reason to this function and
143          *                      return a new reason for the new promise's error reason
144          * @returns {KISSY.Promise} a new promise object
145          */
146         then:function (fulfilled, rejected) {
147             return when(this, fulfilled, rejected);
148         },
149         /**
150          * call rejected callback when this promise object is rejected
151          * @param {Function(*)} rejected called with rejected reason
152          * @returns {KISSY.Promise} a new promise object
153          */
154         fail:function (rejected) {
155             return when(this, 0, rejected);
156         },
157         /**
158          * call callback when this promise object is rejected or resolved
159          * @param {Function(*,Boolean)} callback the second parameter is
160          * true when resolved and false when rejected
161          * @@returns {KISSY.Promise} a new promise object
162          */
163         fin:function (callback) {
164             return when(this, function (value) {
165                 return callback(value, true);
166             }, function (reason) {
167                 return callback(reason, false);
168             });
169         },
170         /**
171          * whether the given object is a resolved promise
172          * if it is resolved with another promise,
173          * then that promise needs to be resolved as well.
174          */
175         isResolved:function () {
176             return isResolved(this);
177         },
178         /**
179          * whether the given object is a rejected promise
180          */
181         isRejected:function () {
182             return isRejected(this);
183         }
184     };
185 
186     function Reject(reason) {
187         if (reason instanceof Reject) {
188             return reason;
189         }
190         Promise.apply(this, arguments);
191         if (this._value instanceof Promise) {
192             S.error('assert.not(this._value instanceof promise) in Reject constructor');
193         }
194         return undefined;
195     }
196 
197     S.extend(Reject, Promise, {
198         // override,simply call rejected
199         _when:function (fulfilled, rejected) {
200             // if there is a rejected , should always has! see when()
201             if (!rejected) {
202                 S.error("no rejected callback!");
203             }
204             return rejected ? rejected(this._value) : new Reject(this._value);
205         }
206     });
207 
208     function resolve(value) {
209         if (value instanceof Promise) {
210             return value;
211         }
212         return new Promise(value);
213     }
214 
215     /**
216      * wrap for promise._when
217      * @param value
218      * @param fulfilled
219      * @param [rejected]
220      */
221     function when(value, fulfilled, rejected) {
222         var defer = new Defer(),
223             done = 0;
224 
225         // wrap user's callback to catch exception
226         function _fulfilled(value) {
227             try {
228                 return fulfilled ? fulfilled(value) : value;
229             } catch (e) {
230                 S.log(e,"error");
231                 return new Reject(e);
232             }
233         }
234 
235         function _rejected(reason) {
236             try {
237                 return rejected ? rejected(reason) : new Reject(reason);
238             } catch (e) {
239                 S.log(e,"error");
240                 return new Reject(e);
241             }
242         }
243 
244         nextTick(function () {
245             resolve(value)._when(function (value) {
246                 if (done) {
247                     S.error("already done at fulfilled");
248                     return;
249                 }
250                 if (value instanceof Promise) {
251                     S.error("assert.not(value instanceof Promise) in when")
252                 }
253                 done = 1;
254                 defer.resolve(
255                     // may return another promise
256                     resolve(value)._when(_fulfilled, _rejected)
257                 );
258             }, function (reason) {
259                 if (done) {
260                     S.error("already done at rejected");
261                     return;
262                 }
263                 done = 1;
264                 // _reject may return non-Reject object for error recovery
265                 defer.resolve(_rejected(reason));
266             });
267         });
268 
269         // chained and leveled
270         // wait for value's resolve
271         return defer.promise;
272     }
273 
274     function isResolved(obj) {
275         // exclude Reject at first
276         return !isRejected(obj) &&
277             isPromise(obj) &&
278             (obj._pendings === undefined) &&
279             (
280                 // immediate value
281                 !isPromise(obj._value) ||
282                     // resolved with a resolved promise !!! :)
283                     // Reject._value is string
284                     isResolved(obj._value)
285                 );
286     }
287 
288     function isRejected(obj) {
289         return isPromise(obj) &&
290             (obj._pendings === undefined) &&
291             (obj._value instanceof Reject);
292     }

294     KISSY.Defer = Defer;
295     KISSY.Promise = Promise;

297     S.mix(Promise,
298         /**
299          * @lends KISSY.Promise
300          */
301         {
302             /**
303              * register callbacks when obj as a promise is resolved
304              * or call fulfilled callback directly when obj is not a promise object
305              * @param {KISSY.Promise|*} obj a promise object or value of any type
306              * @param {Function(*)} fulfilled called when obj resolved successfully,pass a resolved value to this function and
307              *                      return a value (could be promise object) for the new promise's resolved value.
308              * @param {Function(*)} [rejected] called when error occurs in obj,pass error reason to this function and
309              *                      return a new reason for the new promise's error reason
310              * @returns {KISSY.Promise} a new promise object
311              * @example
312              * <code>
313              * function check(p){
314              *   S.Promise.when(p,function(v){
315              *     alert(v===1);
316              *   });
317              * }
318              *
319              * var defer=S.Defer();
320              * defer.resolve(1);
321              *
322              * check(1); // => alert(true)
323              *
324              * check(defer.promise); //=> alert(true);
325              * </code>
326              * @function
327              */
328             when:when,
329             /**
330              * whether the given object is a promise
331              * @function
332              * @param obj the tested object
333              */
334             isPromise:isPromise,
335             /**
336              * whether the given object is a resolved promise
337              * @function
338              * @param obj the tested object
339              */
340             isResolved:isResolved,
341             /**
342              * whether the given object is a rejected promise
343              * @function
344              * @param obj the tested object
345              */
346             isRejected:isRejected,
347             /**
348              * return a new promise
349              * which is resolved when all promises is resolved
350              * and rejected when any one of promises is rejected
351              * @param {KISSY.Promise[]} promises list of promises
352              */
353             all:function (promises) {
354                 return when([].concat(promises), function (promises) {
355                     var count = promises.length;
356                     if (!count) {
357                         return promises;
358                     }
359                     var defer = Defer();
360                     for (var i = 0; i < promises.length; i++) {
361                         (function (promise, i) {
362                             when(promise, function (value) {
363                                 promises[i] = value;
364                                 if (--count === 0) {
365                                     // if all is resolved
366                                     // then resolve final returned promise with all value
367                                     defer.resolve(promises);
368                                 }
369                             }, function (r) {
370                                 // if any one is rejected
371                                 // then reject final return promise with first reason
372                                 defer.reject(r);
373                             });
374                         })(promises[i], i);
375                     }
376                     return defer.promise;
377                 });
378             }
379         });
