1 /**
  2  * @fileOverview abstraction of tree node ,root and other node will extend it
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("tree/basenode", function(S, Node, UIBase, Component, BaseNodeRender) {
  6     var $ = Node.all,
  7         ITEM_CLS = BaseNodeRender.ITEM_CLS,
  8         KeyCodes = Node.KeyCodes;
  9 
 10 
 11     /**
 12      * 基类树节点
 13      * @constructor
 14      */
 15     var BaseNode = UIBase.create(Component.Controller,
 16         /*
 17          * 可多继承从某个子节点开始装饰儿子组件
 18          */
 19         [Component.DecorateChild],
 20         {
 21             _keyNav:function(e) {
 22                 var self = this,
 23                     processed = true,
 24                     n,
 25                     children = self.get("children"),
 26                     keyCode = e.keyCode;
 27 
 28                 // 顺序统统为前序遍历顺序
 29                 switch (keyCode) {
 30                     // home
 31                     // 移到树的顶层节点
 32                     case KeyCodes.HOME:
 33                         n = self.get("tree");
 34                         break;
 35 
 36                     // end
 37                     // 移到最后一个可视节点
 38                     case KeyCodes.END:
 39                         n = self.get("tree").getLastVisibleDescendant();
 40                         break;
 41 
 42                     // 上
 43                     // 当前节点的上一个兄弟节点的最后一个可显示节点
 44                     case KeyCodes.UP:
 45                         n = self.getPreviousVisibleNode();
 46                         break;
 47 
 48                     // 下
 49                     // 当前节点的下一个可显示节点
 50                     case KeyCodes.DOWN:
 51                         n = self.getNextVisibleNode();
 52                         break;
 53 
 54                     // 左
 55                     // 选择父节点或 collapse 当前节点
 56                     case KeyCodes.LEFT:
 57                         if (self.get("expanded") && (children.length || self.get("isLeaf") === false)) {
 58                             self.set("expanded", false);
 59                         } else {
 60                             n = self.get("parent");
 61                         }
 62                         break;
 63 
 64                     // 右
 65                     // expand 当前节点
 66                     case KeyCodes.RIGHT:
 67                         if (children.length || self.get("isLeaf") === false) {
 68                             if (!self.get("expanded")) {
 69                                 self.set("expanded", true);
 70                             } else {
 71                                 children[0].select();
 72                             }
 73                         }
 74                         break;
 75 
 76                     default:
 77                         processed = false;
 78                         break;
 79 
 80                 }
 81                 if (n) {
 82                     n.select();
 83                 }
 84                 return processed;
 85             },
 86 
 87             getLastVisibleDescendant:function() {
 88                 var self = this,children = self.get("children");
 89                 // 没有展开或者根本没有儿子节点，可视的只有自己
 90                 if (!self.get("expanded") || !children.length) {
 91                     return self;
 92                 }
 93                 // 可视的最后一个子孙
 94                 return children[children.length - 1].getLastVisibleDescendant();
 95             },
 96 
 97             getNextVisibleNode:function() {
 98                 var self = this,
 99                     children = self.get("children"),
100                     parent = self.get("parent");
101                 if (self.get("expanded") && children.length) {
102                     return children[0];
103                 }
104                 // 没有展开或者根本没有儿子节点
105                 // 深度遍历的下一个
106                 var n = self.next();
107                 while (parent && !n) {
108                     n = parent.next();
109                     parent = parent.get("parent");
110                 }
111                 return n;
112             },
113 
114             getPreviousVisibleNode:function() {
115                 var self = this,prev = self.prev();
116                 if (!prev) {
117                     prev = self.get("parent");
118                 } else {
119                     prev = prev.getLastVisibleDescendant();
120                 }
121                 return prev;
122             },
123 
124             next:function() {
125                 var self = this,parent = self.get("parent");
126                 if (!parent) {
127                     return null;
128                 }
129                 var siblings = parent.get('children');
130                 var index = S.indexOf(self, siblings);
131                 if (index == siblings.length - 1) {
132                     return null;
133                 }
134                 return siblings[index + 1];
135             },
136 
137             prev:function() {
138                 var self = this, parent = self.get("parent");
139                 if (!parent) {
140                     return null;
141                 }
142                 var siblings = parent.get('children');
143                 var index = S.indexOf(self, siblings);
144                 if (index === 0) {
145                     return null;
146                 }
147                 return siblings[index - 1];
148             },
149 
150             /**
151              * 选中当前节点
152              * @public
153              */
154             select : function() {
155                 var self = this;
156                 self.get("tree").set("selectedItem", self);
157             },
158 
159             _performInternal:function(e) {
160                 var self = this,
161                     target = $(e.target),
162                     tree = self.get("tree"),
163                     view = self.get("view");
164                 tree.get("el")[0].focus();
165                 if (target.equals(view.get("expandIconEl"))) {
166                     // 忽略双击
167                     if (e.type != 'dblclick') {
168                         self.set("expanded", !self.get("expanded"));
169                     }
170                 } else if (e.type == 'dblclick') {
171                     self.set("expanded", !self.get("expanded"));
172                 }
173                 else {
174                     self.select();
175                     tree.fire("click", {
176                         target:self
177                     });
178                 }
179             },
180 
181             // 默认 addChild，这里需要设置 tree 属性
182             decorateChildrenInternal:function(ui, c, cls) {
183                 var self = this;
184                 self.addChild(new ui({
185                     srcNode:c,
186                     tree:self.get("tree"),
187                     prefixCls:cls
188                 }));
189             },
190 
191             addChild:function(c) {
192                 var self = this,tree = self.get("tree");
193                 c.__set("tree", tree);
194                 c.__set("depth", self.get('depth') + 1);
195                 BaseNode.superclass.addChild.call(self, c);
196                 self._updateRecursive();
197                 tree._register(c);
198                 S.each(c.get("children"), function(cc) {
199                     tree._register(cc);
200                 });
201             },
202 
203             /*
204              每次添加/删除节点，都检查自己以及自己子孙 class
205              每次 expand/collapse，都检查
206              */
207             _computeClass:function(cause) {
208                 var self = this,view = self.get("view");
209                 view._computeClass(self.get('children'), self.get("parent"), cause);
210             },
211 
212             _updateRecursive:function() {
213                 var self = this,len = self.get('children').length;
214                 self._computeClass("_updateRecursive");
215                 S.each(self.get("children"), function(c, index) {
216                     c._computeClass("_updateRecursive_children");
217                     c.get("view").set("ariaPosInSet", index + 1);
218                     c.get("view").set("ariaSize", len);
219                 });
220             },
221 
222             removeChild:function(c) {
223                 var self = this,tree = self.get("tree");
224                 tree._unregister(c);
225                 S.each(c.get("children"), function(cc) {
226                     tree._unregister(cc);
227                 });
228                 BaseNode.superclass.removeChild.apply(self, S.makeArray(arguments));
229                 self._updateRecursive();
230             },
231 
232             _uiSetExpanded:function(v) {
233                 var self = this,
234                     tree = self.get("tree");
235                 self._computeClass("expanded-" + v);
236                 if (v) {
237                     tree.fire("expand", {
238                         target:self
239                     });
240                 } else {
241                     tree.fire("collapse", {
242                         target:self
243                     });
244                 }
245             },
246 
247             _uiSetSelected:function(v) {
248                 this._forwardSetAttrToView("selected", v);
249             },
250 
251 
252             expandAll:function() {
253                 var self = this;
254                 self.set("expanded", true);
255                 S.each(self.get("children"), function(c) {
256                     c.expandAll();
257                 });
258             },
259 
260             collapseAll:function() {
261                 var self = this;
262                 self.set("expanded", false);
263                 S.each(self.get("children"), function(c) {
264                     c.collapseAll();
265                 });
266             }
267         },
268 
269         {
270             DefaultRender:BaseNodeRender,
271             ATTRS:{
272                 /*事件代理*/
273                 handleMouseEvents:{
274                     value:false
275                 },
276                 id:{
277                     getter:function() {
278                         var self = this,
279                             id = self.get("el").attr("id");
280                         if (!id) {
281                             self.get("el").attr("id", id = S.guid("tree-node"));
282                         }
283                         return id;
284                     }
285                 },
286                 /**
287                  * 节点字内容
288                  * @type String
289                  */
290                 content:{view:true},
291 
292                 /**
293                  * 强制指明该节点是否具备子孙，影响样式，不配置默认样式自动变化
294                  * @type Boolean
295                  */
296                 isLeaf:{
297                     view:true
298                 },
299 
300                 expandIconEl:{ view:true},
301 
302                 iconEl:{ view:true},
303 
304                 /**
305                  * 是否选中
306                  * @type Boolean
307                  */
308                 selected:{},
309 
310                 expanded:{
311                     value:false,
312                     view:true
313                 },
314 
315                 /**
316                  * html title
317                  * @type String
318                  */
319                 tooltip:{view:true},
320                 tree:{
321                 },
322 
323                 /**
324                  * depth of node
325                  */
326                 depth:{
327                     value:0,
328                     view:true
329                 },
330                 focusable:{value:false},
331                 decorateChildCls:{
332                     value:"tree-children"
333                 }
334             },
335 
336             HTML_PARSER:{
337                 expanded:function(el) {
338                     var children = el.one("." + this.getCls("tree-children"));
339                     if (!children) {
340                         return false;
341                     }
342                     return children.css("display") != "none";
343                 }
344             }
345         });
346 
347     Component.UIStore.setUIByClass(ITEM_CLS, {
348         priority:10,
349         ui:BaseNode
350     });
351 
352     return BaseNode;
353 
354 }, {
355     requires:['node','uibase','component','./basenoderender']
356 });
