1 /**
  2  * @fileOverview loading mask support for overlay
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("uibase/loading", function () {
  6 
  7     /**
  8      * make component can mask loading
  9      * @class
 10      * @memberOf UIBase
 11      */
 12     function Loading() {
 13     }
 14 
 15     Loading.prototype =
 16     /**
 17      * @lends UIBase.Loading.prototype
 18      */
 19     {
 20         /**
 21          * mask component as loading
 22          */
 23         loading:function () {
 24             this.get("view").loading();
 25         },
 26 
 27         /**
 28          * unmask component as loading
 29          */
 30         unloading:function () {
 31             this.get("view").unloading();
 32         }
 33     };
 34 
 35     return Loading;
 36 
 37 });
