  5 KISSY.add("ua/css", function (S, UA) {
  6     var o = [
  7         // browser core type
  8         "webkit",
  9         "trident",
 10         "gecko",
 11         "presto",
 12         // browser type
 13         "chrome",
 14         "safari",
 15         "firefox",
 16         "ie",
 17         "opera"
 18     ],
 19         documentElement = S.Env.host.document.documentElement,
 20         className = "",
 21         v;
 22     S.each(o, function (key) {
 23         if (v = UA[key]) {
 24             className += " ks-" + key + ((v + "").replace(/\./g, "_"));
 25             className += " ks-" + key;
 26         }
 27     });
 28     documentElement.className = S.trim(documentElement.className + className);
 29 }, {
 30     requires:['./base']
 31 });
