1 /**
  2  * @fileOverview delegate all draggable nodes to one draggable object
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("dd/draggable-delegate", function (S, DDM, Draggable, DOM, Node) {
  6 
  7     /**
  8      * drag multiple nodes under a container element using only one draggable instance as a delegate.
  9      * @memberOf DD
 10      * @class
 11      */
 12     function DraggableDelegate() {
 13         DraggableDelegate.superclass.constructor.apply(this, arguments);
 14     }
 15 
 16 
 17     /**
 18      * 父容器监听 mousedown，找到合适的拖动 handlers 以及拖动节点
 19      * @param ev
 20      */
 21     function _handleMouseDown(ev) {
 22         var self = this,
 23             handler,
 24             node;
 25 
 26         if (!self._checkMouseDown(ev)) {
 27             return;
 28         }
 29 
 30         var handlers = self.get("handlers"),
 31             target = new Node(ev.target);
 32 
 33         // 不需要像 Draggble 一样，判断 target 是否在 handler 内
 34         // 委托时，直接从 target 开始往上找 handler
 35         if (handlers.length) {
 36             handler = self._getHandler(target);
 37         } else {
 38             handler = target;
 39         }
 40 
 41         if (handler) {
 42             node = self._getNode(handler);
 43         }
 44 
 45         // can not find handler or can not find matched node from handler
 46         // just return !
 47         if (!node) {
 48             return;
 49         }
 50 
 51         self.__set("activeHandler", handler);
 52 
 53         // 找到 handler 确定 委托的 node ，就算成功了
 54         self.__set("node", node);
 55         self.__set("dragNode", node);
 56         self._prepare(ev);
 57     }
 58 
 59     S.extend(DraggableDelegate, Draggable, {
 60 
 61             _uiSetDisabledChange:function (d) {
 62                 this.get("container")[d ? 'addClass' :
 63                     'removeClass'](DDM.get("prefixCls") + '-disabled');
 64             },
 65 
 66             _init:function () {
 67                 var self = this,
 68                     node = self.get('container');
 69                 node.on('mousedown', _handleMouseDown, self)
 70                     .on('dragstart', self._fixDragStart);
 71             },
 72 
 73             /**
 74              * 得到适合 handler，从这里开始启动拖放，对于 handlers 选择器字符串数组
 75              * @param target
 76              */
 77             _getHandler:function (target) {
 78                 var self = this,
 79                     ret = undefined,
 80                     node = self.get("container"),
 81                     handlers = self.get('handlers');
 82                 while (target && target[0] !== node[0]) {
 83                     S.each(handlers, function (h) {
 84                         if (DOM.test(target[0], h)) {
 85                             ret = target;
 86                             return false;
 87                         }
 88                     });
 89                     if (ret) {
 90                         break;
 91                     }
 92                     target = target.parent();
 93                 }
 94                 return ret;
 95             },
 96 
 97             /**
 98              * 找到真正应该移动的节点，对应 selector 属性选择器字符串
 99              * @param h
100              */
101             _getNode:function (h) {
102                 return h.closest(this.get("selector"), this.get("container"));
103             },
104 
105             destroy:function () {
106                 var self = this;
107                 self.get("container")
108                     .detach('mousedown',
109                     _handleMouseDown,
110                     self)
111                     .detach('dragstart', self._fixDragStart);
112                 self.detach();
113             }
114         },
115         {
116             ATTRS:/**
117              * @lends DD.DraggableDelegate#
118              */
119             {
120                 /**
121                  * a selector query to get the container to listen for mousedown events on.
122                  * All "draggable selector" should be a child of this container
123                  * @type {HTMLElement|String}
124                  */
125                 container:{
126                     setter:function (v) {
127                         return Node.one(v);
128                     }
129                 },
130 
131                 /**
132                  * a selector query to get the children of container to make draggable elements from.
133                  * usually as for tag.cls.
134                  * @type {String}
135                  */
136                 selector:{
137                 },
138 
139                 /**
140                  * handlers to initiate drag operation.
141                  * can only be as form of tag.cls.
142                  * default:[selector]
143                  * @type {String[]}
144                  **/
145                 handlers:{
146                     value:[],
147                     // 覆盖父类的 getter ，这里 normalize 成节点
148                     getter:0
149                 }
150             }
151         });
152 
153     return DraggableDelegate;
154 }, {
155     requires:['./ddm', './draggable', 'dom', 'node']
156 });
