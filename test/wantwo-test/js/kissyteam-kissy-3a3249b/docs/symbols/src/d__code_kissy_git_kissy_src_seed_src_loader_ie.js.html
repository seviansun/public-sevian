  5 (function (S, loader, utils) {
  6     if (typeof require !== 'undefined') {
  7         return;
  8     }
  9     S.mix(loader, {
 10         //ie 特有，找到当前正在交互的脚本，根据脚本名确定模块名
 11         // 如果找不到，返回发送前那个脚本
 12         __findModuleNameByInteractive:function () {
 13             var self = this,
 14                 scripts = document.getElementsByTagName("script"),
 15                 re,
 16                 script;
 17 
 18             for (var i = 0; i < scripts.length; i++) {
 19                 script = scripts[i];
 20                 if (script.readyState == "interactive") {
 21                     re = script;
 22                     break;
 23                 }
 24             }
 25             if (!re) {
 26                 // sometimes when read module file from cache , interactive status is not triggered
 27                 // module code is executed right after inserting into dom
 28                 // i has to preserve module name before insert module script into dom , then get it back here
 29                 S.log("can not find interactive script,time diff : " + (+new Date() - self.__startLoadTime), "error");
 30                 S.log("old_ie get modname from cache : " + self.__startLoadModuleName);
 31                 return self.__startLoadModuleName;
 32                 //S.error("找不到 interactive 状态的 script");
 33             }
 34 
 35             // src 必定是绝对路径
 36             // or re.hasAttribute ? re.src :  re.getAttribute('src', 4);
 37             // http://msdn.microsoft.com/en-us/library/ms536429(VS.85).aspx
 38             var src = utils.absoluteFilePath(re.src);
 39             // S.log("interactive src :" + src);
 40             // 注意：模块名不包含后缀名以及参数，所以去除
 41             // 系统模块去除系统路径
 42             // 需要 base norm , 防止 base 被指定为相对路径
 43             self.Config.base = utils.normalBasePath(self.Config.base);
 44             if (src.lastIndexOf(self.Config.base, 0)
 45                 === 0) {
 46                 return utils.removePostfix(src.substring(self.Config.base.length));
 47             }
 48             var packages = self.Config.packages;
 49             //外部模块去除包路径，得到模块名
 50             for (var p in packages) {
 51                 if (packages.hasOwnProperty(p)) {
 52                     var p_path = packages[p].path;
 53                     if (packages.hasOwnProperty(p) &&
 54                         src.lastIndexOf(p_path, 0) === 0) {
 55                         return utils.removePostfix(src.substring(p_path.length));
 56                     }
 57                 }
 58             }
 59             S.log("interactive script does not have package config ：" + src, "error");
 60         }
 61     });
