  6 
  7     /**
  8      * make component positionable
  9      * @class
 10      * @memberOf UIBase
 11      */
 12     function Position() {
 13     }
 14 
 15     Position.ATTRS =
 16     /**
 17      * @lends UIBase.Position.prototype
 18      */
 19     {
 20         /**
 21          * 横坐标值
 22          * @type Number
 23          */
 24         x:{
 25             view:true
 26         },
 27         /**
 28          * 纵坐标值
 29          * @type Number
 30          */
 31         y:{
 32             view:true
 33         },
 34         /**
 35          * 横纵坐标值
 36          * @type Number[]
 37          */
 38         xy:{
 39             // 相对 page 定位, 有效值为 [n, m], 为 null 时, 选 align 设置
 40             setter:function (v) {
 41 
 42                 var self = this,
 43                     xy = S.makeArray(v);
 44 
 45                 /*
 46                  属性内分发特别注意：
 47                  xy -> x,y
 48 
 49                  */
 50                 if (xy.length) {
 51                     xy[0] && self.set("x", xy[0]);
 52                     xy[1] && self.set("y", xy[1]);
 53                 }
 54                 return v;
 55             },
 56             /**
 57              * xy 纯中转作用
 58              */
 59             getter:function () {
 60                 return [this.get("x"), this.get("y")];
 61             }
 62         },
 63         /**
 64          * z-index 值
 65          * @type Number
 66          */
 67         zIndex:{
 68             view:true
 69         }
 70     };
 71

101         /**
102          * 移动到绝对位置上, move(x, y) or move(x) or move([x, y])
103          * @param {Number|Number[]} x
104          * @param {Number} [y]
105          */
106         move:function (x, y) {
107             var self = this;
108             if (S.isArray(x)) {
109                 y = x[1];
110                 x = x[0];
111             }
112             self.set("xy", [x, y]);
