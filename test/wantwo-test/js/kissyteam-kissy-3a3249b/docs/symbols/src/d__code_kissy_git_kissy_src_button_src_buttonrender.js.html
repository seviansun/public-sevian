1 /**
  2  * @fileOverview abstract view for button
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("button/buttonrender", function(S, UIBase, Component) {
  6     // http://www.w3.org/TR/wai-aria-practices/
  7     return UIBase.create(Component.Render, {
  8         createDom:function() {
  9             //set wai-aria role
 10             this.get("el").attr("role", "button")
 11                 .addClass(this.getCls("inline-block button"));
 12         },
 13         _uiSetTooltip:function(title) {
 14             this.get("el").attr("title", title);
 15         },
 16         _uiSetDescribedby:function(describedby) {
 17             this.get("el").attr("aria-describedby", describedby);
 18         },
 19 
 20         _uiSetCollapseSide:function(side) {
 21             var self = this,
 22                 cls = self.getCls("button-collapse-"),
 23                 el = self.get("el");
 24             el.removeClass(cls + "left " + cls + "right");
 25             if (side) {
 26                 el.addClass(cls + side);
 27             }
 28         }
 29     }, {
 30         ATTRS:{
 31             /**
 32              * @inheritedDoc
 33              * disabled:{}
 34              */
 35 
 36             /**
 37              * @inheritedDoc
 38              * prefixCls:{}
 39              */
 40 
 41                 // aria-describledby support
 42             describedby:{},
 43             tooltip:{},
 44             collapseSide:{}
 45         }
 46     });
 47 }, {
 48     requires:['uibase','component']
 49 });
