384      * 防止 html5 draggable 元素的拖放默认行为

388         e.preventDefault();

399     /**
400      * 鼠标按下时，查看触发源是否是属于 handler 集合，
401      * 保存当前状态
402      * 通知全局管理器开始作用
403      * @param ev
404      */
405     function _handleMouseDown(ev) {
406         var self = this,
407             t = ev.target;
408 
409         if (self._checkMouseDown(ev)) {
410 
411             if (!self._check(t)) {
412                 return;
413             }
414 
415             self._prepare(ev);
416         }
417     }
418 
419     S.extend(Draggable, Base,
420         /**
421          * @lends DD.Draggable#
422          */
423         {
424 
425             /**
426              * 开始拖时鼠标所在位置，例如
427              *  {x:100,y:200}
428              * @type Object
429              * @private
430              */
431             startMousePos:NULL,
432 
433             /**
434              * 开始拖时节点所在位置，例如
435              *  {x:100,y:200}
436              * @type Object
437              * @private
438              */
439             startNodePos:NULL,
440 
441             /**
442              * 开始拖时鼠标和节点所在位置的差值
443              */
444             _diff:NULL,
445 
446             /**
447              * mousedown 1秒后自动开始拖的定时器
448              */
449             _bufferTimer:NULL,
450 
451             _uiSetDisabledChange:function (d) {
452                 this.get("dragNode")[d ? 'addClass' :
453                     'removeClass'](DDM.get("prefixCls") + '-disabled');
454             },
455 
456             _init:function () {
457                 var self = this,
458                     node = self.get('node');
459                 node.on('mousedown', _handleMouseDown, self)
460                     .on('dragstart', self._fixDragStart);
461             },
462 
463             _fixDragStart:fixDragStart,
464 
465             /**
466              *
467              * @param {HTMLElement} t
468              */
469             _check:function (t) {
470                 var self = this,
471                     handlers = self.get('handlers'),
472                     ret = 0;
473                 each(handlers, function (handler) {
474                     //子区域内点击也可以启动
475                     if (handler.contains(t) ||
476                         handler[0] == t) {
477                         ret = 1;
478                         self.__set("activeHandler", handler);
479                         return false;
480                     }
481                 });
482                 return ret;
483             },
484 
485             _checkMouseDown:function (ev) {
486                 if (this.get('primaryButtonOnly') && ev.button > 1 ||
487                     this.get("disabled")) {
488                     return 0;
489                 }
490                 return 1;
491             },
492 
493             _prepare:function (ev) {
494 
495                 var self = this;
496 
497                 if (ie) {
498                     fixIEMouseDown();
499                 }
500 
501                 // 防止 firefox/chrome 选中 text
502                 if (self.get("halt")) {
503                     ev.halt();
504                 } else {
505                     ev.preventDefault();
506                 }
507 
508                 var node = self.get("node"),
509                     mx = ev.pageX,
510                     my = ev.pageY,
511                     nxy = node.offset();
512                 self.startMousePos = self.mousePos = {
513                     left:mx,
514                     top:my
515                 };
516                 self.startNodePos = nxy;
517                 self._diff = {
518                     left:mx - nxy.left,
519                     top:my - nxy.top
520                 };
521                 DDM._regToDrag(self);
522 
523                 var bufferTime = self.get("bufferTime");
524 
525                 // 是否中央管理，强制限制拖放延迟
526                 if (bufferTime) {
527                     self._bufferTimer = setTimeout(function () {
528                         // 事件到了，仍然是 mousedown 触发！
529                         //S.log("drag start by timeout");
530                         self._start();
531                     }, bufferTime);
532                 }
533 
534             },
535 
536             _clearBufferTimer:function () {
537                 var self = this;
538                 if (self._bufferTimer) {
539                     clearTimeout(self._bufferTimer);
540                     self._bufferTimer = 0;
541                 }
542             },
543 
544             _move:function (ev) {
545                 var self = this,
546                     ret,
547                     diff = self._diff,
548                     startMousePos = self.startMousePos,
549                     pageX = ev.pageX,
550                     pageY = ev.pageY,
551                     left = pageX - diff.left,
552                     top = pageY - diff.top;
553 
554 
555                 if (!self.get("dragging")) {
556                     var clickPixelThresh = self.get("clickPixelThresh");
557                     // 鼠标经过了一定距离，立即开始
558                     if (Math.abs(pageX - startMousePos.left) >= clickPixelThresh ||
559                         Math.abs(pageY - startMousePos.top) >= clickPixelThresh) {
560                         //S.log("start drag by pixel : " + l1 + " : " + l2);
561                         self._start();
562                     }
563                     // 开始后，下轮 move 开始触发 drag 事件
564                     return;
565                 }
566 
567                 self.mousePos = {
568                     left:pageX,
569                     top:pageY
570                 };
571 
572                 ret = {
573                     left:left,
574                     top:top,
575                     pageX:pageX,
576                     pageY:pageY,
577                     drag:self
578                 };
579 
580                 self.fire("dragalign", {
581                     info:ret
582                 });
583 
584                 var def = 1;
585 
586                 if (self.fire("drag", ret) === false) {
587                     def = 0;
588                 }
589 
590                 if (def && self.get("move")) {
591                     // 取 'node' , 改 node 可能是代理哦
592                     self.get('node').offset(ret);
593                 }
594             },
595 
596             _end:function () {
597                 var self = this,
598                     activeDrop;
599 
600                 // 否则清除定时器即可
601                 self._clearBufferTimer();
602                 if (ie) {
603                     fixIEMouseUp();
604                 }
605                 // 如果已经开始，收尾工作
606                 if (self.get("dragging")) {
607                     self.get("node").removeClass(DDM.get("prefixCls") + "drag-over");
608                     if (activeDrop = DDM.get("activeDrop")) {
609                         self.fire('dragdrophit', {
610                             drag:self,
611                             drop:activeDrop
612                         });
613                     } else {
614                         self.fire('dragdropmiss', {
615                             drag:self
616                         });
617                     }
618                     self.__set("dragging", 0);
619                     self.fire("dragend", {
620                         drag:self
621                     });
622                 }
623             },
624 
625             _handleOut:function () {
626                 var self = this;
627                 self.get("node").removeClass(DDM.get("prefixCls") + "drag-over");
628                 /**
629                  *  html5 => dragleave
630                  */
631                 self.fire("dragexit", {
632                     drag:self,
633                     drop:DDM.get("activeDrop")
634                 });
635             },
636 
637             /**
638              *
639              * @param e
640              */
641             _handleEnter:function (e) {
642                 var self = this;
643                 self.get("node").addClass(DDM.get("prefixCls") + "drag-over");
644                 //第一次先触发 dropenter,dragenter
645                 self.fire("dragenter", e);
646             },
647 
648 
649             _handleOver:function (e) {
650                 this.fire("dragover", e);
651             },
652 
653             _start:function () {
654                 var self = this;
655                 self._clearBufferTimer();
656                 self.__set("dragging", 1);
657                 DDM._start();
658                 self.fire("dragstart", {
659                     drag:self
660                 });
661             },
662 
663             /**
664              * make the drag node undraggable
665              */
666             destroy:function () {
667                 var self = this,
668                     node = self.get('dragNode');
669                 node.detach('mousedown', _handleMouseDown, self)
670                     .detach('dragstart', self._fixDragStart);
671                 self.detach();
672             }
673         });
674 
675     return Draggable;
