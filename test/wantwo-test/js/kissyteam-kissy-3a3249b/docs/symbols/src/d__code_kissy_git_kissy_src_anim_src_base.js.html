  7     var camelCase = DOM._camelCase,
  8         _isElementNode = DOM._isElementNode,
  9         specialVals = ["hide", "show", "toggle"],
 10         // shorthand css properties
 11         SHORT_HANDS = {
 12             // http://www.w3.org/Style/CSS/Tracker/issues/9
 13             // http://snook.ca/archives/html_and_css/background-position-x-y
 14             // backgroundPositionX  backgroundPositionY does not support
 15             background:[
 16                 "backgroundPosition"
 17             ],
 18             border:[
 19                 "borderBottomWidth",
 20                 "borderLeftWidth",
 21                 'borderRightWidth',
 22                 // 'borderSpacing', 组合属性？
 23                 'borderTopWidth'
 24             ],
 25             borderBottom:["borderBottomWidth"],
 26             borderLeft:["borderLeftWidth"],
 27             borderTop:["borderTopWidth"],
 28             borderRight:["borderRightWidth"],
 29             font:[
 30                 'fontSize',
 31                 'fontWeight'
 32             ],
 33             margin:[
 34                 'marginBottom',
 35                 'marginLeft',
 36                 'marginRight',
 37                 'marginTop'
 38             ],
 39             padding:[
 40                 'paddingBottom',
 41                 'paddingLeft',
 42                 'paddingRight',
 43                 'paddingTop'
 44             ]
 45         },
 46         defaultConfig = {
 47             duration:1,
 48             easing:'easeNone'
 49         },
 50         rfxnum = /^([+\-]=)?([\d+.\-]+)([a-z%]*)$/i;

 52     Anim.SHORT_HANDS = SHORT_HANDS;

 55     /**
 56      * @class A class for constructing animation instances.
 57      * @param {HTMLElement|window} elem Html dom node or window
 58      * (window can only animate scrollTop/scrollLeft)
 59      * @param {Object} props style map
 60      * @param {Number|Object} [duration] duration(s) or anim config
 61      * @param {String|Function} [duration.easing] easing fn or string
 62      * @param {Function} [duration.complete] callback function when this animation is complete
 63      * @param {Number} [duration.duration] duration(s)
 64      * @param {String|Boolean} [duration.queue] current animation's queue, if false then no queue
 65      * @param {Function|String} [easing] easing fn or string
 66      * @param {Function} [callback] callback function when this animation is complete
 67      * @extends Event.Target
 68      * @name Anim
 69      *
 70      */
 71     function Anim(elem, props, duration, easing, callback) {
 72         var self = this, config;
 73 
 74         // ignore non-exist element
 75         if (!(elem = DOM.get(elem))) {
 76             return;
 77         }
 78 
 79         // factory or constructor
 80         if (!(self instanceof Anim)) {
 81             return new Anim(elem, props, duration, easing, callback);
 82         }
 83 
 84         /**
 85          * the transition properties
 86          */
 87         if (S.isString(props)) {
 88             props = S.unparam(String(props), ";", ":");
 89         } else {
 90             // clone to prevent collision within multiple instance
 91             props = S.clone(props);
 92         }
 93 
 94         /**
 95          * 驼峰属性名
 96          */
 97         for (var prop in props) {
 98             var camelProp = camelCase(S.trim(prop));
 99             if (prop != camelProp) {
100                 props[camelProp] = props[prop];
101                 delete props[prop];
102             }
103         }
104 
105         /**
106          * animation config
107          */
108         if (S.isPlainObject(duration)) {
109             config = S.clone(duration);
110         } else {
111             config = {
112                 duration:parseFloat(duration) || undefined,
113                 easing:easing,
114                 complete:callback
115             };
116         }
117 
118         config = S.merge(defaultConfig, config);
119         self.config = config;
120         config.duration *= 1000;
121 
122         // domEl deprecated!
123         self.elem = self['domEl'] = elem;
124         self.props = props;
125 
126         // 实例属性
127         self._backupProps = {};
128         self._fxs = {};
129 
130         // register callback
131         self.on("complete", onComplete);
132     }
133 
134 
135     function onComplete(e) {
136         var self = this,
137             _backupProps,
138             config = self.config;
139 
140         // only recover after complete anim
141         if (!S.isEmptyObject(_backupProps = self._backupProps)) {
142             DOM.css(self.elem, _backupProps);
143         }
144 
145         if (config.complete) {
146             config.complete.call(self, e);
147         }
148     }
149 
150     function runInternal() {
151         var self = this,
152             config = self.config,
153             _backupProps = self._backupProps,
154             elem = self.elem,
155             hidden,
156             val,
157             prop,
158             specialEasing = (config['specialEasing'] || {}),
159             fxs = self._fxs,
160             props = self.props;
161 
162         // 进入该函数即代表执行（q[0] 已经是 ...）
163         saveRunning(self);
164 
165         if (self.fire("start") === false) {
166             // no need to invoke complete
167             self.stop(0);
168             return;
169         }
170 
171         if (_isElementNode(elem)) {
172             hidden = (DOM.css(elem, "display") === "none");
173             for (prop in props) {
174                 val = props[prop];
175                 // 直接结束
176                 if (val == "hide" && hidden || val == 'show' && !hidden) {
177                     // need to invoke complete
178                     self.stop(1);
179                     return;
180                 }
181             }
182         }
183 
184         // 分离 easing
185         S.each(props, function (val, prop) {
186             if (!props.hasOwnProperty(prop)) {
187                 return;
188             }
189             var easing;
190             if (S.isArray(val)) {
191                 easing = specialEasing[prop] = val[1];
192                 props[prop] = val[0];
193             } else {
194                 easing = specialEasing[prop] = (specialEasing[prop] || config.easing);
195             }
196             if (S.isString(easing)) {
197                 easing = specialEasing[prop] = Easing[easing];
198             }
199             specialEasing[prop] = easing || Easing['easeNone'];
200         });
201 
202 
203         // 扩展分属性
204         S.each(SHORT_HANDS, function (shortHands, p) {
205             var sh,
206                 origin,
207                 val;
208             if (val = props[p]) {
209                 origin = {};
210                 S.each(shortHands, function (sh) {
211                     // 得到原始分属性之前值
212                     origin[sh] = DOM.css(elem, sh);
213                     specialEasing[sh] = specialEasing[p];
214                 });
215                 DOM.css(elem, p, val);
216                 for (sh in origin) {
217                     // 得到期待的分属性最后值
218                     props[sh] = DOM.css(elem, sh);
219                     // 还原
220                     DOM.css(elem, sh, origin[sh]);
221                 }
222                 // 删除复合属性
223                 delete props[p];
224             }
225         });
226 
227         // 取得单位，并对单个属性构建 Fx 对象
228         for (prop in props) {
229             if (!props.hasOwnProperty(prop)) {
230                 continue;
231             }
232 
233             val = S.trim(props[prop]);
234 
235             var to,
236                 from,
237                 propCfg = {
238                     elem:elem,
239                     prop:prop,
240                     duration:config.duration,
241                     easing:specialEasing[prop]
242                 },
243                 fx = Fx.getFx(propCfg);
244 
245             // hide/show/toggle : special treat!
246             if (S.inArray(val, specialVals)) {
247                 // backup original value
248                 _backupProps[prop] = DOM.style(elem, prop);
249                 if (val == "toggle") {
250                     val = hidden ? "show" : "hide";
251                 }
252                 if (val == "hide") {
253                     to = 0;
254                     from = fx.cur();
255                     // 执行完后隐藏
256                     _backupProps.display = 'none';
257                 } else {
258                     from = 0;
259                     to = fx.cur();
260                     // prevent flash of content
261                     DOM.css(elem, prop, from);
262                     DOM.show(elem);
263                 }
264                 val = to;
265             } else {
266                 to = val;
267                 from = fx.cur();
268             }
269 
270             val += "";
271 
272             var unit = "",
273                 parts = val.match(rfxnum);
274 
275             if (parts) {
276                 to = parseFloat(parts[2]);
277                 unit = parts[3];
278 
279                 // 有单位但单位不是 px
280                 if (unit && unit !== "px") {
281                     DOM.css(elem, prop, val);
282                     from = (to / fx.cur()) * from;
283                     DOM.css(elem, prop, from + unit);
284                 }
285 
286                 // 相对
287                 if (parts[1]) {
288                     to = ( (parts[ 1 ] === "-=" ? -1 : 1) * to ) + from;
289                 }
290             }
291 
292             propCfg.from = from;
293             propCfg.to = to;
294             propCfg.unit = unit;
295             fx.load(propCfg);
296             fxs[prop] = fx;
297         }
298 
299         if (_isElementNode(elem) &&
300             (props.width || props.height)) {
301             // Make sure that nothing sneaks out
302             // Record all 3 overflow attributes because IE does not
303             // change the overflow attribute when overflowX and
304             // overflowY are set to the same value
305             S.mix(_backupProps, {
306                 overflow:DOM.style(elem, "overflow"),
307                 overflow-x:DOM.style(elem, "overflowX"),
308                 overflow-y:DOM.style(elem, "overflowY")
309             });
310             DOM.css(elem, "overflow", "hidden");
311             // inline element should has layout/inline-block
312             if (DOM.css(elem, "display") === "inline" &&
313                 DOM.css(elem, "float") === "none") {
314                 if (UA['ie']) {
315                     DOM.css(elem, "zoom", 1);
316                 } else {
317                     DOM.css(elem, "display", "inline-block");
318                 }
319             }
320         }
321 
322         AM.start(self);
323     }

332             /**
333              * @return {boolean} whether this animation is running
334              */
335             isRunning:function () {
336                 return isRunning(this);
337             },
338 
339             _runInternal:runInternal,
340 
341             /**
342              * start this animation
343              */
344             run:function () {
345                 var self = this,
346                     queueName = self.config.queue;
347 
348                 if (queueName === false) {
349                     runInternal.call(self);
350                 } else {
351                     // 当前动画对象加入队列
352                     Q.queue(self);
353                 }
354 
355                 return self;
356             },
357 
358             _frame:function () {
359 
360                 var self = this,
361                     prop,
362                     config = self.config,
363                     end = 1,
364                     c,
365                     fx,
366                     fxs = self._fxs;
367 
368                 for (prop in fxs) {
369                     if (fxs.hasOwnProperty(prop) &&
370                         // 当前属性没有结束
371                         !((fx = fxs[prop]).finished)) {
372                         // 非短路
373                         if (config.frame) {
374                             c = config.frame(fx);
375                         }
376                         // 结束
377                         if (c == 1 ||
378                             // 不执行自带
379                             c == 0) {
380                             fx.finished = c;
381                             end &= c;
382                         }
383                         else {
384                             end &= fx.frame();
385                         }
386                     }
387                 }
388 
389                 if ((self.fire("step") === false) ||
390                     end) {
391                     // complete 事件只在动画到达最后一帧时才触发
392                     self.stop(end);
393                 }
394             },
395 
396             /**
397              * stop this animation
398              * @param {Boolean} [finish] whether jump to the last position of this animation
399              */
400             stop:function (finish) {
401                 var self = this,
402                     config = self.config,
403                     queueName = config.queue,
404                     prop,
405                     fx,
406                     fxs = self._fxs;
407 
408                 // already stopped
409                 if (!self.isRunning()) {
410                     // 从自己的队列中移除
411                     if (queueName !== false) {
412                         Q.remove(self);
413                     }
414                     return;
415                 }
416 
417                 if (finish) {
418                     for (prop in fxs) {
419                         if (fxs.hasOwnProperty(prop) &&
420                             // 当前属性没有结束
421                             !((fx = fxs[prop]).finished)) {
422                             // 非短路
423                             if (config.frame) {
424                                 config.frame(fx, 1);
425                             } else {
426                                 fx.frame(1);
427                             }
428                         }
429                     }
430                     self.fire("complete");
431                 }
432 
433                 AM.stop(self);
434 
435                 removeRunning(self);
436 
437                 if (queueName !== false) {
438                     // notify next anim to run in the same queue
439                     Q.dequeue(self);
440                 }
441 
442                 return self;
443             }
444         });

446     var runningKey = S.guid("ks-anim-unqueued-" + S.now() + "-");
447 
448     function saveRunning(anim) {
449         var elem = anim.elem,
450             allRunning = DOM.data(elem, runningKey);
451         if (!allRunning) {
452             DOM.data(elem, runningKey, allRunning = {});
453         }
454         allRunning[S.stamp(anim)] = anim;
455     }
456 
457     function removeRunning(anim) {
458         var elem = anim.elem,
459             allRunning = DOM.data(elem, runningKey);
460         if (allRunning) {
461             delete allRunning[S.stamp(anim)];
462             if (S.isEmptyObject(allRunning)) {
463                 DOM.removeData(elem, runningKey);
464             }
465         }
466     }
467 
468     function isRunning(anim) {
469         var elem = anim.elem,
470             allRunning = DOM.data(elem, runningKey);
471         if (allRunning) {
472             return !!allRunning[S.stamp(anim)];
473         }
474         return 0;
475     }
476 
477     /**
478      * stop all the anims currently running
479      * @param {HTMLElement} elem element which anim belongs to
480      * @param {boolean} end whether jump to last position
481      * @param {boolean} clearQueue whether clean current queue
482      * @param {String|Boolean} queueName current queue's name to be cleared
483      * @private
484      */
485     Anim.stop = function (elem, end, clearQueue, queueName) {
486         if (
487         // default queue
488             queueName === null ||
489                 // name of specified queue
490                 S.isString(queueName) ||
491                 // anims not belong to any queue
492                 queueName === false
493             ) {
494             return stopQueue.apply(undefined, arguments);
495         }
496         // first stop first anim in queues
497         if (clearQueue) {
498             Q.removeQueues(elem);
499         }
500         var allRunning = DOM.data(elem, runningKey),
501             // can not stop in for/in , stop will modified allRunning too
502             anims = S.merge(allRunning);
503         for (var k in anims) {
504             anims[k].stop(end);
505         }
506     };
507 
508     /**
509      *
510      * @param elem element which anim belongs to
511      * @param queueName queue'name if set to false only remove
512      * @param end
513      * @param clearQueue
514      * @private
515      */
516     function stopQueue(elem, end, clearQueue, queueName) {
517         if (clearQueue && queueName !== false) {
518             Q.removeQueue(elem, queueName);
519         }
520         var allRunning = DOM.data(elem, runningKey),
521             anims = S.merge(allRunning);
522         for (var k in anims) {
523             var anim = anims[k];
524             if (anim.config.queue == queueName) {
525                 anim.stop(end);
526             }
527         }
528     }

530     /**
531      * whether elem is running anim
532      * @param {HTMLElement} elem
533      * @private
534      */
535     Anim.isRunning = function (elem) {
536         var allRunning = DOM.data(elem, runningKey);
537         return allRunning && !S.isEmptyObject(allRunning);
538     };
