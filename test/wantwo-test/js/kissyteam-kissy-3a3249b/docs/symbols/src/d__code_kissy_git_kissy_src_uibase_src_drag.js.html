1 /**
  2  * @fileOverview drag extension for position
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("uibase/drag", function (S) {
  6 
  7 
  8     /**
  9      * config drag options
 10      * @class
 11      * @memberOf UIBase
 12      */
 13     function Drag() {
 14     }
 15 
 16     Drag.ATTRS =
 17 
 18     /**
 19      * @lends UIBase.Drag
 20      */
 21     {
 22         /**
 23          * see {@link DD.Draggable#handlers}
 24          */
 25         handlers:{
 26             value:[]
 27         },
 28         /**
 29          * 是否可拖放
 30          * @type boolean
 31          */
 32         draggable:{value:true}
 33     };
 34 
 35     Drag.prototype = {
 36 
 37         _uiSetHandlers:function (v) {
 38             if (v && v.length > 0 && this.__drag) {
 39                 this.__drag.set("handlers", v);
 40             }
 41         },
 42 
 43         __bindUI:function () {
 44             var Draggable = S.require("dd/draggable");
 45             var self = this,
 46                 el = self.get("el");
 47             if (self.get("draggable") && Draggable) {
 48                 self.__drag = new Draggable({
 49                     node:el
 50                 });
 51             }
 52         },
 53 
 54         _uiSetDraggable:function (v) {
 55 
 56             var self = this,
 57                 d = self.__drag;
 58             if (!d) {
 59                 return;
 60             }
 61             if (v) {
 62                 d.detach("drag");
 63                 d.on("drag", self._dragExtAction, self);
 64             } else {
 65                 d.detach("drag");
 66             }
 67         },
 68 
 69         _dragExtAction:function (offset) {
 70             this.set("xy", [offset.left, offset.top])
 71         },
 72         /**
 73          *
 74          */
 75         __destructor:function () {
 76             //S.log("DragExt __destructor");
 77             var d = this.__drag;
 78             d && d.destroy();
 79         }
 80 
 81     };
 82     return Drag;
 83 
 84 });
