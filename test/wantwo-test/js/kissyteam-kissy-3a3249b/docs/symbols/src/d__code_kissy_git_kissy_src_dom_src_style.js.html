 62     // modified from jquery : bullet-proof method of getting default display
 63     // fix domain problem in ie>6 , ie6 still access denied
 64     function getDefaultDisplay(tagName) {
 65         var body,
 66             elem;
 67         if (!defaultDisplay[ tagName ]) {
 68             body = doc.body || doc.documentElement;
 69             elem = doc.createElement(tagName);
 70             DOM.prepend(elem, body);
 71             var oldDisplay = DOM.css(elem, "display");
 72             body.removeChild(elem);
 73             // If the simple way fails,
 74             // get element's real default display by attaching it to a temp iframe
 75             if (oldDisplay === "none" || oldDisplay === "") {
 76                 // No iframe to use yet, so create it
 77                 if (!defaultDisplayDetectIframe) {
 78                     defaultDisplayDetectIframe = doc.createElement("iframe");
 79 
 80                     defaultDisplayDetectIframe.frameBorder =
 81                         defaultDisplayDetectIframe.width =
 82                             defaultDisplayDetectIframe.height = 0;
 83 
 84                     DOM.prepend(defaultDisplayDetectIframe, body);
 85                     var iframeSrc;
 86                     if (iframeSrc = DOM._genEmptyIframeSrc()) {
 87                         defaultDisplayDetectIframe.src = iframeSrc;
 88                     }
 89                 } else {
 90                     DOM.prepend(defaultDisplayDetectIframe, body);
 91                 }
 92 
 93                 // Create a cacheable copy of the iframe document on first call.
 94                 // IE and Opera will allow us to reuse the iframeDoc without re-writing the fake HTML
 95                 // document to it; WebKit & Firefox won't allow reusing the iframe document.
 96                 if (!defaultDisplayDetectIframeDoc || !defaultDisplayDetectIframe.createElement) {
 97 
 98                     try {
 99                         defaultDisplayDetectIframeDoc = defaultDisplayDetectIframe.contentWindow.document;
100                         defaultDisplayDetectIframeDoc.write(( doc.compatMode === "CSS1Compat" ? "<!doctype html>" : "" )
101                             + "<html><head>" +
102                             (UA['ie'] && DOM._isCustomDomain() ?
103                                 "<script>document.domain = '" +
104                                     doc.domain
105                                     + "';</script>" : "")
106                             +
107                             "</head><body>");
108                         defaultDisplayDetectIframeDoc.close();
109                     } catch (e) {
110                         // ie6 need a breath , such as alert(8) or setTimeout;
111                         // 同时需要同步，所以无解，勉强返回
112                         return "block";
113                     }
114                 }
115 
116                 elem = defaultDisplayDetectIframeDoc.createElement(tagName);
117 
118                 defaultDisplayDetectIframeDoc.body.appendChild(elem);
119 
120                 oldDisplay = DOM.css(elem, "display");
121 
122                 body.removeChild(defaultDisplayDetectIframe);
123             }
124 
125             // Store the correct default display
126             defaultDisplay[ tagName ] = oldDisplay;
127         }
128 
129         return defaultDisplay[ tagName ];
130     }

498     S.each(["left", "top"], function (name) {
499         /**
500          * @ignore
501          */
502         CUSTOM_STYLES[ name ] = {
503             get:function (elem, computed) {
504                 if (computed) {
505                     var val = DOM._getComputedStyle(elem, name), offset;
506 
507                     // 1. 当没有设置 style.left 时，getComputedStyle 在不同浏览器下，返回值不同
508                     //    比如：firefox 返回 0, webkit/ie 返回 auto
509                     // 2. style.left 设置为百分比时，返回值为百分比
510                     // 对于第一种情况，如果是 relative 元素，值为 0. 如果是 absolute 元素，值为 offsetLeft - marginLeft
511                     // 对于第二种情况，大部分类库都未做处理，属于“明之而不 fix”的保留 bug
512                     if (val === AUTO) {
513                         val = 0;
514                         if (S.inArray(DOM.css(elem, 'position'), ['absolute', 'fixed'])) {
515                             offset = elem[name === 'left' ? 'offsetLeft' : 'offsetTop'];
516 
517                             // old-ie 下，elem.offsetLeft 包含 offsetParent 的 border 宽度，需要减掉
518                             if (isIE && doc['documentMode'] != 9 || UA['opera']) {
519                                 // 类似 offset ie 下的边框处理
520                                 // 如果 offsetParent 为 html ，需要减去默认 2 px == documentElement.clientTop
521                                 // 否则减去 borderTop 其实也是 clientTop
522                                 // http://msdn.microsoft.com/en-us/library/aa752288%28v=vs.85%29.aspx
523                                 // ie<9 注意有时候 elem.offsetParent 为 null ...
524                                 // 比如 DOM.append(DOM.create("<div class='position:absolute'></div>"),document.body)
525                                 offset -= elem.offsetParent && elem.offsetParent['client' + (name == 'left' ? 'Left' : 'Top')]
526                                     || 0;
527                             }
528                             val = offset - (PARSEINT(DOM.css(elem, 'margin-' + name)) || 0);
529                         }
530                         val += "px";
531                     }
532                     return val;
533                 }
534             }
535         };
536     });

539     function swap(elem, options, callback) {
540         var old = {};
541 
542         // Remember the old values, and insert the new ones
543         for (var name in options) {
544             old[ name ] = elem[STYLE][ name ];
545             elem[STYLE][ name ] = options[ name ];
546         }
547 
548         callback.call(elem);
549 
550         // Revert the old values
551         for (name in options) {
552             elem[STYLE][ name ] = old[ name ];
553         }
554     }

557     function style(elem, name, val) {
558         var style;
559         if (elem.nodeType === 3 || elem.nodeType === 8 || !(style = elem[STYLE])) {
560             return undefined;
561         }
562         name = camelCase(name);
563         var ret, hook = CUSTOM_STYLES[name];
564         name = cssProps[name] || name;
565         // setter
566         if (val !== undefined) {
567             // normalize unsetting
568             if (val === null || val === EMPTY) {
569                 val = EMPTY;
570             }
571             // number values may need a unit
572             else if (!isNaN(Number(val)) && !cssNumber[name]) {
573                 val += DEFAULT_UNIT;
574             }
575             if (hook && hook.set) {
576                 val = hook.set(elem, val);
577             }
578             if (val !== undefined) {
579                 // ie 无效值报错
580                 try {
581                     style[name] = val;
582                 } catch (e) {
583                     S.log("css set error :" + e);
584                 }
585                 // #80 fix,font-family
586                 if (val === EMPTY && style.removeAttribute) {
587                     style.removeAttribute(name);
588                 }
589             }
590 
591             if (!style.cssText) {
592                 elem.removeAttribute('style');
593             }
594             return undefined;
595         }
596         //getter
597         else {
598             // If a hook was provided get the non-computed value from there
599             if (hook && "get" in hook && (ret = hook.get(elem, false)) !== undefined) {
600 
601             } else {
602                 // Otherwise just get the value from the style object
603                 ret = style[ name ];
604             }
605             return ret === undefined ? "" : ret;
606         }
607 
608     }

611     /**
612      * 得到元素的大小信息
613      * @param elem
614      * @param name
615      * @param {String} [extra]  "padding" : (css width) + padding
616      *                          "border" : (css width) + padding + border
617      *                          "margin" : (css width) + padding + border + margin
618      */
619     function getWH(elem, name, extra) {
620         if (S.isWindow(elem)) {
621             return name == WIDTH ? DOM.viewportWidth(elem) : DOM.viewportHeight(elem);
622         } else if (elem.nodeType == 9) {
623             return name == WIDTH ? DOM.docWidth(elem) : DOM.docHeight(elem);
624         }
625         var which = name === WIDTH ? ['Left', 'Right'] : ['Top', 'Bottom'],
626             val = name === WIDTH ? elem.offsetWidth : elem.offsetHeight;
627 
628         if (val > 0) {
629             if (extra !== "border") {
630                 S.each(which, function (w) {
631                     if (!extra) {
632                         val -= parseFloat(DOM.css(elem, "padding" + w)) || 0;
633                     }
634                     if (extra === "margin") {
635                         val += parseFloat(DOM.css(elem, extra + w)) || 0;
636                     } else {
637                         val -= parseFloat(DOM.css(elem, "border" + w + "Width")) || 0;
638                     }
639                 });
640             }
641 
642             return val
643         }
644 
645         // Fall back to computed then uncomputed css if necessary
646         val = DOM._getComputedStyle(elem, name);
647         if (val == null || (Number(val)) < 0) {
648             val = elem.style[ name ] || 0;
649         }
650         // Normalize "", auto, and prepare for extra
651         val = parseFloat(val) || 0;
652 
653         // Add padding, border, margin
654         if (extra) {
655             S.each(which, function (w) {
656                 val += parseFloat(DOM.css(elem, "padding" + w)) || 0;
657                 if (extra !== "padding") {
658                     val += parseFloat(DOM.css(elem, "border" + w + "Width")) || 0;
659                 }
660                 if (extra === "margin") {
661                     val += parseFloat(DOM.css(elem, extra + w)) || 0;
662                 }
663             });
664         }
665 
666         return val;
667     }
