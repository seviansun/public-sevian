1 /**
  2  * @fileOverview KISSY Scalable Event Framework
  3  */
  4 KISSY.add("event", function (S, _data, KeyCodes, Event, Target, Object) {
  5     S.mix(Event,
  6         /**
  7          * @lends Event
  8          */
  9         {
 10             KeyCodes:KeyCodes,
 11             Target:Target,
 12             Object:Object,
 13             on:Event.add,
 14             detach:Event.remove,
 15             /**
 16              *
 17              * @param targets KISSY selector
 18              * @param {String} [eventType] The type of event to delegate.
 19              * use space to separate multiple event types.
 20              * @param {String|Function} selector filter selector string or function to find right element
 21              * @param {Function} [fn] The event handler/listener.
 22              * @param {Object} [scope] The scope (this reference) in which the handler function is executed.
 23              */
 24             delegate:function (targets, eventType, selector, fn, scope) {
 25                 return Event.add(targets, eventType, {
 26                     fn:fn,
 27                     scope:scope,
 28                     selector:selector
 29                 });
 30             },
 31             /**
 32              * @param targets KISSY selector
 33              * @param {String} [eventType] The type of event to undelegate.
 34              * use space to separate multiple event types.
 35              * @param {String|Function} selector filter selector string or function to find right element
 36              * @param {Function} [fn] The event handler/listener.
 37              * @param {Object} [scope] The scope (this reference) in which the handler function is executed.
 38              */
 39             undelegate:function (targets, eventType, selector, fn, scope) {
 40                 return Event.remove(targets, eventType, {
 41                     fn:fn,
 42                     scope:scope,
 43                     selector:selector
 44                 });
 45             }
 46         });
 47 
 48     S.mix(Event, _data);
 49 
 50     S.mix(S, {
 51         Event:Event,
 52         EventTarget:Event.Target,
 53         EventObject:Event.Object
 54     });
 55 
 56     return Event;
 57 }, {
 58     requires:[
 59         "event/data",
 60         "event/keycodes",
 61         "event/base",
 62         "event/target",
 63         "event/object",
 64         "event/focusin",
 65         "event/hashchange",
 66         "event/valuechange",
 67         "event/mouseenter",
 68         "event/submit",
 69         "event/change",
 70         "event/mousewheel",
 71         "event/add",
 72         "event/remove"
 73     ]
 74 });
 75 
 76 /**
 77  *  2012-02-12 yiminghe@gmail.com note:
 78  *   - 普通 remove() 不管 selector 都会查，如果 fn scope 相等就移除
 79  *   - undelegate() selector 为 ""，那么去除所有委托绑定的 handler
 80  */
