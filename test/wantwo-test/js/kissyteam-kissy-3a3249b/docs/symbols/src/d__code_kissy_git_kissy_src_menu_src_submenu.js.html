  5 KISSY.add("menu/submenu", function (S, Event, UIBase, Component, MenuItem, SubMenuRender) {

 10     function bindMenu(self, menu) {
 11         /**
 12          * 自己不是 menu，自己只是 menuitem，其所属的 menu 为 get("parent")
 13          */
 14         var parentMenu = self.get("parent");
 15 
 16         //当改菜单项所属的菜单隐藏后，该菜单项关联的子菜单也要隐藏
 17         if (parentMenu) {
 18 
 19             parentMenu.on("hide", self._onParentHide, self);
 20 
 21             // 子菜单选中后也要通知父级菜单
 22             // 不能使用 afterSelectedItemChange ，多个 menu 嵌套，可能有缓存
 23             // 单个 menu 来看可能 selectedItem没有变化
 24             menu.on("click", function (ev) {
 25                 parentMenu.fire("click", {
 26                     target:ev.target
 27                 });
 28             });
 29 
 30             // if not bind doc click for parent menu
 31             // if already bind, then if parent menu hide, menu will hide too
 32             if (!parentMenu.__bindDocClickToHide) {
 33                 Event.on(doc, "click", _onDocClick, self);
 34                 parentMenu.__bindDocClickToHide = 1;
 35                 menu.__bindDocClickToHide = 1;
 36             }
 37 
 38             // 通知父级菜单
 39             menu.on("afterActiveItemChange", function (ev) {
 40                 parentMenu.set("activeItem", ev.newVal);
 41             });
 42         }
 43         // 访问子菜单，当前 submenu 不隐藏 menu
 44         // leave submenuitem -> enter menuitem -> menu item highlight ->
 45         // -> menu highlight -> onChildHighlight_ ->
 46 
 47         // menu render 后才会注册 afterHighlightedItemChange 到 _uiSet
 48         // 这里的 onChildHighlight_ 比 afterHighlightedItemChange 先执行
 49         // 保险点用 beforeHighlightedItemChange
 50         menu.on("beforeHighlightedItemChange", self.onChildHighlight_, self);
 51     }
 52 
 53     function getMenu(self, init) {
 54         var m = self.get("menu");
 55         if (S.isFunction(m)) {
 56             if (init) {
 57                 m = m.call(self);
 58                 self.__set("menu", m);
 59             } else {
 60                 return null;
 61             }
 62         }
 63         if (m && m.get("parent") !== self) {
 64             m.__set("parent", self);
 65             bindMenu(self, m);
 66         }
 67         return m;
 68     }
 69 
 70     function _onDocClick(e) {
 71         var self = this,
 72             menu = getMenu(self),
 73             target = e.target,
 74             parentMenu = self.get("parent"),
 75             el = self.get("el");
 76 
 77         // only hide this menu, if click outside this menu and this menu's submenus
 78         if (!parentMenu.containsElement(target)) {
 79             menu && menu.hide();
 80             // submenuitem should also hide
 81             self.get("parent").set("highlightedItem", null);
 82         }
 83     }
 84 
 85     var KeyCodes = Event.KeyCodes,
 86         doc = S.Env.host.document,
 87         MENU_DELAY = 300;
 88     /**
 89      * Class representing a submenu that can be added as an item to other menus.
 90      */
 91     var SubMenu = UIBase.create(MenuItem, [Component.DecorateChild], {
 92 
 93             _onParentHide:function () {
 94                 var menu = getMenu(this);
 95                 menu && menu.hide();
 96             },
 97             /**
 98              * @inheritDoc
 99              * Sets a timer to show the submenu
100              **/
101             _handleMouseEnter:function (e) {
102                 var self = this;
103                 if (SubMenu.superclass._handleMouseEnter.call(self, e)) {
104                     return true;
105                 }
106                 self.clearTimers();
107                 self.showTimer_ = S.later(self.showMenu,
108                     self.get("menuDelay"), false, self);
109             },
110 
111             showMenu:function () {
112                 var self = this,
113                     menu = getMenu(self, 1);
114                 if (menu) {
115                     menu.set("align", S.mix({
116                         node:self.get("el"),
117                         points:['tr', 'tl']
118                     }, self.get("menuAlign")));
119                     menu.render();
120                     /**
121                      * If activation of your menuitem produces a popup menu,
122                      then the menuitem should have aria-haspopup set to the ID of the corresponding menu
123                      to allow the assistive technology to follow the menu hierarchy
124                      and assist the user in determining context during menu navigation.
125                      */
126                     self.get("el").attr("aria-haspopup",
127                         menu.get("el").attr("id"));
128                     menu.show();
129                 }
130             },
131 
132 
133             /**
134              * Clears the show and hide timers for the sub menu.
135              */
136             clearTimers:function () {
137                 var self = this;
138                 if (self.dismissTimer_) {
139                     self.dismissTimer_.cancel();
140                     self.dismissTimer_ = null;
141                 }
142                 if (self.showTimer_) {
143                     self.showTimer_.cancel();
144                     self.showTimer_ = null;
145                 }
146                 var menu = getMenu(self);
147                 // TODO 耦合 popmenu.js
148                 if (menu && menu._leaveHideTimer) {
149                     clearTimeout(menu._leaveHideTimer);
150                     menu._leaveHideTimer = 0;
151                 }
152             },
153 
154             /**
155              * Listens to the sub menus items and ensures that this menu item is selected
156              * while dismissing the others.  This handles the case when the user mouses
157              * over other items on their way to the sub menu.
158              * @param  e Highlight event to handle.
159              * @private
160              */
161             onChildHighlight_:function (e) {
162                 var self = this;
163                 if (e.newVal) {
164                     self.clearTimers();
165                     // superclass(menuitem)._handleMouseLeave 已经把自己 highlight 去掉了
166                     // 导致本类 _uiSetHighlighted 调用，又把子菜单隐藏了
167                     self.get("parent").set("highlightedItem", self);
168                 }
169             },
170 
171             hideMenu:function () {
172                 var menu = getMenu(this);
173                 if (menu) {
174                     menu.hide();
175                 }
176             },
177 
178             // click ，立即显示
179             _performInternal:function () {
180                 var self = this;
181                 self.clearTimers();
182                 self.showMenu();
183                 //  trigger click event from menuitem
184                 SubMenu.superclass._performInternal.apply(self, arguments);
185             },
186 
187             /**
188              * Handles a key event that is passed to the menu item from its parent because
189              * it is highlighted.  If the right key is pressed the sub menu takes control
190              * and delegates further key events to its menu until it is dismissed OR the
191              * left key is pressed.
192              * @param e A key event.
193              * @return {boolean} Whether the event was handled.
194              */
195             _handleKeydown:function (e) {
196                 var self = this,
197                     menu = getMenu(self),
198                     hasKeyboardControl_ = menu && menu.get("visible"),
199                     keyCode = e.keyCode;
200 
201                 if (!hasKeyboardControl_) {
202                     // right
203                     if (keyCode == KeyCodes.RIGHT) {
204                         self.showMenu();
205                         menu = getMenu(self);
206                         if (menu) {
207                             var menuChildren = menu.get("children");
208                             if (menuChildren[0]) {
209                                 menu.set("highlightedItem", menuChildren[0]);
210                             }
211                         }
212                     }
213                     // enter as click
214                     else if (e.keyCode == Event.KeyCodes.ENTER) {
215                         return this._performInternal(e);
216                     }
217                     else {
218                         return undefined;
219                     }
220                 } else if (menu._handleKeydown(e)) {
221                 }
222                 // The menu has control and the key hasn't yet been handled, on left arrow
223                 // we turn off key control.
224                 // left
225                 else if (keyCode == KeyCodes.LEFT) {
226                     self.hideMenu();
227                     // 隐藏后，当前激活项重回
228                     self.get("parent").set("activeItem", self);
229                 } else {
230                     return undefined;
231                 }
232                 return true;
233             },
234 
235             /**
236              * @inheritDoc
237              * Dismisses the submenu on a delay, with the result that the user needs less
238              * accuracy when moving to submenus.
239              **/
240             _uiSetHighlighted:function (highlight, ev) {
241                 var self = this;
242                 SubMenu.superclass._uiSetHighlighted.call(self, highlight, ev);
243                 if (!highlight) {
244                     self.clearTimers();
245                     self.dismissTimer_ = S.later(self.hideMenu,
246                         self.get("menuDelay"),
247                         false, self);
248                 }
249             },
250 
251             containsElement:function (element) {
252                 var menu = getMenu(this);
253                 return menu && menu.containsElement(element);
254             },
255 
256             // 默认 addChild，这里里面的元素需要放到 menu 属性中
257             decorateChildrenInternal:function (ui, el, cls) {
258                 // 不能用 diaplay:none
259                 el.css("visibility", "hidden");
260                 var docBody = S.one(el[0].ownerDocument.body);
261                 docBody.prepend(el);
262                 var menu = new ui({
263                     srcNode:el,
264                     prefixCls:cls
265                 });
266                 this.__set("menu", menu);
267             },
268 
269             destructor:function () {
270                 var self = this,
271                     parentMenu = self.get("parent"),
272                     menu = getMenu(self);
273 
274                 self.clearTimers();
275 
276                 if (menu && menu.__bindDocClickToHide) {
277                     menu.__bindDocClickToHide = 0;
278                     Event.remove(doc, "click", _onDocClick, self);
279                 }
280 
281                 //当改菜单项所属的菜单隐藏后，该菜单项关联的子菜单也要隐藏
282                 if (parentMenu) {
283                     parentMenu.detach("hide", self._onParentHide, self);
284                 }
285 
286                 if (menu && !self.get("externalSubMenu")) {
287                     menu.destroy();
288                 }
289             }
290         },
291         {
292             ATTRS:{
293                 /**
294                  * The delay before opening the sub menu in milliseconds.  (This number is
295                  * arbitrary, it would be good to get some user studies or a designer to play
296                  * with some numbers).
297                  * @type {number}
298                  */
299                 menuDelay:{
300                     value:MENU_DELAY
301                 },
302                 /**
303                  * whether destroy submenu when destroy itself ,reverse result
304                  * @type {boolean}
305                  */
306                 externalSubMenu:{
307                     value:false
308                 },
309                 menuAlign:{},
310                 menu:{
311                 },
312                 decorateChildCls:{
313                     value:"popupmenu"
314                 }
315             },
316 
317             DefaultRender:SubMenuRender
318         }
319     );
320 
321 
322     Component.UIStore.setUIByClass("submenu", {
323         priority:Component.UIStore.PRIORITY.LEVEL2,
324         ui:SubMenu
325     });
326 
327     return SubMenu;
328 }, {
329     requires:['event', 'uibase', 'component', './menuitem', './submenurender']
330 });
