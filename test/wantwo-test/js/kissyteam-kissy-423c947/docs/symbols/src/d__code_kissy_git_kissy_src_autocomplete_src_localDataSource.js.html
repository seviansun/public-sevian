1 /**
  2  * local dataSource
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("autocomplete/localDataSource", function (S) {
  6 
  7     /**
  8      * Local dataSource for autoComplete
  9      * @memberOf AutoComplete
 10      * @class
 11      * @param {Object} cfg config
 12      * @param {Array} cfg.data array of static data for autoComplete
 13      * @param {Object} cfg.dataSourceCfg dataSource config
 14      * @param {Function} cfg.dataSourceCfg.parse parse data
 15      * @param {Boolean} cfg.dataSourceCfg.allowEmpty whether return all data when input is empty.default:true
 16      */
 17     function LocalDataSource(cfg) {
 18         LocalDataSource.superclass.constructor.apply(this, arguments);
 19     }
 20 
 21     function parser(inputVal, data) {
 22         var ret = [],
 23             count = 0;
 24 
 25         S.each(data, function (d) {
 26             if (d.indexOf(inputVal) != -1) {
 27                 ret.push(d);
 28             }
 29             count++;
 30         });
 31 
 32         return ret;
 33     }
 34 
 35     LocalDataSource.ATTRS = {
 36         data:{
 37             value:[]
 38         },
 39         dataSourceCfg:{
 40             value:{}
 41         }
 42     };
 43 
 44     S.extend(LocalDataSource, S.Base, {
 45         /**
 46          * Datasource interface. How to get data for autoComplete
 47          * @function
 48          * @name AutoComplete.LocalDataSource#fetchData
 49          * @param {String} inputVal current active input's value
 50          * @param {Function} callback callback to notify autoComplete when data is ready
 51          * @param {Object} context callback's execution context
 52          */
 53         fetchData:function (inputVal, callback, context) {
 54             var dataSourceCfg = this.get("dataSourceCfg"),
 55                 parse = dataSourceCfg.parse || parser,
 56                 data = this.get("data"),
 57                 allowEmpty = dataSourceCfg.allowEmpty;
 58             if (allowEmpty === false && !inputVal) {
 59             } else if (!inputVal && allowEmpty !== false) {
 60             } else {
 61                 data = parse(inputVal, data);
 62             }
 63             callback.call(context, data);
 64         }
 65     });
 66 
 67     return LocalDataSource;
 68 });
