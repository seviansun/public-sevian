1 /**
  2  * @fileOverview render aria from menu according to current menuitem
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/menurender", function(S, UA, UIBase, Component) {
  6 
  7     return UIBase.create(Component.Render, [
  8         UIBase.ContentBox.Render
  9     ], {
 10 
 11         renderUI:function() {
 12             var el = this.get("el");
 13             el .attr("role", "menu")
 14                 .attr("aria-haspopup", true);
 15             if (!el.attr("id")) {
 16                 el.attr("id", S.guid("ks-menu"));
 17             }
 18         },
 19 
 20         _uiSetActiveItem:function(v) {
 21             var el = this.get("el");
 22             if (v) {
 23                 var menuItemEl = v.get("el"),
 24                     id = menuItemEl.attr("id");
 25                 el.attr("aria-activedescendant", id);
 26                 // 会打印重复 ，每个子菜单都会打印，然后冒泡至父菜单，再打印，和该 menuitem 所处层次有关系
 27                 //S.log("menurender :" + el.attr("id") + " _uiSetActiveItem : " + v.get("content"));
 28             } else {
 29                 el.attr("aria-activedescendant", "");
 30                 //S.log("menurender :" + el.attr("id") + " _uiSetActiveItem : " + "");
 31             }
 32         },
 33 
 34         containsElement:function(element) {
 35             var el = this.get("el");
 36             return el[0] === element || el.contains(element);
 37         }
 38     }, {
 39         ATTRS:{
 40             activeItem:{}
 41         }
 42     });
 43 }, {
 44     requires:['ua','uibase','component']
 45 });
