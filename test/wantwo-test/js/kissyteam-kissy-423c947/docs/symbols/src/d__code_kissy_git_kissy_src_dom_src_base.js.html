1 /**
  2  * @fileOverview   dom
  3  * @author  yiminghe@gmail.com,lifesinger@gmail.com
  4  */
  5 KISSY.add('dom/base', function (S, UA, undefined) {
  6 
  7     var WINDOW=S.Env.host;
  8 
  9     function nodeTypeIs(node, val) {
 10         return node && node.nodeType === val;
 11     }
 12 
 13 
 14     var NODE_TYPE =
 15     /**
 16      * @lends DOM
 17      */
 18     {
 19         ELEMENT_NODE:1,
 20         ATTRIBUTE_NODE:2,
 21         TEXT_NODE:3,
 22         CDATA_SECTION_NODE:4,
 23         ENTITY_REFERENCE_NODE:5,
 24         ENTITY_NODE:6,
 25         PROCESSING_INSTRUCTION_NODE:7,
 26         COMMENT_NODE:8,
 27         DOCUMENT_NODE:9,
 28         DOCUMENT_TYPE_NODE:10,
 29         DOCUMENT_FRAGMENT_NODE:11,
 30         NOTATION_NODE:12
 31     };
 32     var DOM = {
 33 
 34         _isCustomDomain:function (win) {
 35             win = win || WINDOW;
 36             var domain = win.document.domain,
 37                 hostname = win.location.hostname;
 38             return domain != hostname &&
 39                 domain != ( '[' + hostname + ']' );	// IPv6 IP support
 40         },
 41 
 42         _genEmptyIframeSrc:function (win) {
 43             win = win || WINDOW;
 44             if (UA['ie'] && DOM._isCustomDomain(win)) {
 45                 return  'javascript:void(function(){' + encodeURIComponent("" +
 46                     "document.open();" +
 47                     "document.domain='" +
 48                     win.document.domain
 49                     + "';" +
 50                     "document.close();") + "}())";
 51             }
 52         },
 53 
 54         _NODE_TYPE:NODE_TYPE,
 55 
 56 
 57         /**
 58          * 是不是 element node
 59          */
 60         _isElementNode:function (elem) {
 61             return nodeTypeIs(elem, DOM.ELEMENT_NODE);
 62         },
 63 
 64         /**
 65          * elem 为 window 时，直接返回
 66          * elem 为 document 时，返回关联的 window
 67          * elem 为 undefined 时，返回当前 window
 68          * 其它值，返回 false
 69          */
 70         _getWin:function (elem) {
 71             return (elem && ('scrollTo' in elem) && elem['document']) ?
 72                 elem :
 73                 nodeTypeIs(elem, DOM.DOCUMENT_NODE) ?
 74                     elem.defaultView || elem.parentWindow :
 75                     (elem === undefined || elem === null) ?
 76                         WINDOW : false;
 77         },
 78 
 79         _nodeTypeIs:nodeTypeIs,
 80 
 81         // Ref: http://lifesinger.github.com/lab/2010/nodelist.html
 82         _isNodeList:function (o) {
 83             // 注1：ie 下，有 window.item, typeof node.item 在 ie 不同版本下，返回值不同
 84             // 注2：select 等元素也有 item, 要用 !node.nodeType 排除掉
 85             // 注3：通过 namedItem 来判断不可靠
 86             // 注4：getElementsByTagName 和 querySelectorAll 返回的集合不同
 87             // 注5: 考虑 iframe.contentWindow
 88             return o && !o.nodeType && o.item && !o.setTimeout;
 89         },
 90 
 91         _nodeName:function (e, name) {
 92             return e && e.nodeName.toLowerCase() === name.toLowerCase();
 93         }
 94     };
 95 
 96     S.mix(DOM, NODE_TYPE);
 97 
 98     return DOM;
 99 
100 }, {
101     requires:['ua']
102 });
