1 /**
  2  * @fileOverview default sync for model
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("mvc/sync", function (S, io, JSON) {
  6     var methodMap = {
  7         'create':'POST',
  8         'update':'POST', //'PUT'
  9         'delete':'POST', //'DELETE'
 10         'read':'GET'
 11     };
 12 
 13     /**
 14      * @memberOf MVC
 15      * @param self
 16      * @param method
 17      * @param options
 18      */
 19     function sync(self, method, options) {
 20         var type = methodMap[method],
 21             ioParam = S.merge({
 22                 type:type,
 23                 dataType:'json'
 24             }, options);
 25 
 26         var data = ioParam.data = ioParam.data || {};
 27         data['_method'] = method;
 28 
 29         if (!ioParam.url) {
 30             ioParam.url = S.isString(self.get("url")) ?
 31                 self.get("url") :
 32                 self.get("url").call(self);
 33         }
 34 
 35         if (method == 'create' || method == 'update') {
 36             data.model = JSON.stringify(self.toJSON());
 37         }
 38 
 39         return io(ioParam);
 40     }
 41 
 42     return sync;
 43 }, {
 44     requires:['ajax', 'json']
 45 });
