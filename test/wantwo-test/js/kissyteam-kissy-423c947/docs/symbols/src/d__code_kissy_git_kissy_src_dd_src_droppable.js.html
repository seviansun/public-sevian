1 /**
  2  * @fileOverview droppable for kissy
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("dd/droppable", function (S, Node, Base, DDM) {
  6 
  7     /**
  8      * make a node droppable
  9      * @memberOf DD
 10      * @class
 11      */
 12     function Droppable() {
 13         var self = this;
 14         Droppable.superclass.constructor.apply(self, arguments);
 15         self.addTarget(DDM);
 16         S.each([
 17         /**
 18          * @name DD.DDM#dropexit
 19          * @description fired after a draggable leaves a droppable
 20          * @event
 21          * @param e
 22          * @param e.drag current draggable object
 23          * @param e.drop current droppable object
 24          */
 25 
 26         /**
 27          * @name DD.Droppable#dropexit
 28          * @description fired after a draggable leaves a droppable
 29          * @event
 30          * @param e
 31          * @param e.drag current draggable object
 32          * @param e.drop current droppable object
 33          */
 34             "dropexit",
 35 
 36         /**
 37          * @name DD.DDM#dropenter
 38          * @description fired after a draggable object mouseenter a droppable object
 39          * @event
 40          * @param e
 41          * @param e.drag current draggable object
 42          * @param e.drop current droppable object
 43          */
 44 
 45         /**
 46          * @name DD.Droppable#dropenter
 47          * @description fired after a draggable object mouseenter a droppable object
 48          * @event
 49          * @param e
 50          * @param e.drag current draggable object
 51          * @param e.drop current droppable object
 52          */
 53 
 54             "dropenter",
 55 
 56         /**
 57          * @name DD.DDM#dropover
 58          * @description fired after a draggable object mouseover a droppable object
 59          * @event
 60          * @param e
 61          * @param e.drag current draggable object
 62          * @param e.drop current droppable object
 63          */
 64 
 65         /**
 66          * @name DD.Droppable#dropover
 67          * @description fired after a draggable object mouseover a droppable object
 68          * @event
 69          * @param e
 70          * @param e.drag current draggable object
 71          * @param e.drop current droppable object
 72          */
 73             "dropover",
 74 
 75         /**
 76          * @name DD.DDM#drophit
 77          * @description fired after drop a draggable onto a droppable object
 78          * @event
 79          * @param e
 80          * @param e.drag current draggable object
 81          * @param e.drop current droppable object
 82          */
 83 
 84         /**
 85          * @name DD.Droppable#drophit
 86          * @description fired after drop a draggable onto a droppable object
 87          * @event
 88          * @param e
 89          * @param e.drag current draggable object
 90          * @param e.drop current droppable object
 91          */
 92             "drophit"
 93         ], function (e) {
 94             self.publish(e, {
 95                 bubbles:1
 96             });
 97         });
 98         self._init();
 99     }
100 
101     Droppable.ATTRS =
102     /**
103      * @lends DD.Droppable#
104      */
105     {
106         /**
107          * droppable element
108          * @type String|HTMLElement
109          */
110         node:{
111             setter:function (v) {
112                 if (v) {
113                     return Node.one(v);
114                 }
115             }
116         },
117 
118         /**
119          * groups this droppable object belongs to. Default:true
120          * @type Object|boolean true to match any group
121          */
122         groups:{
123             value:true
124         }
125 
126     };
127 
128     function validDrop(dropGroups, dragGroups) {
129         if (dropGroups === true) {
130             return 1;
131         }
132         for (var d in dropGroups) {
133             if (dragGroups[d]) {
134                 return 1;
135             }
136         }
137         return 0;
138     }
139 
140     S.extend(Droppable, Base,
141         /**
142          * @lends DD.Droppable#
143          */
144         {
145             /**
146              * override by droppable-delegate override
147              * @private
148              */
149             getNodeFromTarget:function (ev, dragNode, proxyNode) {
150                 var node = this.get("node"),
151                     domNode = node[0];
152                 // 排除当前拖放和代理节点
153                 return domNode == dragNode ||
154                     domNode == proxyNode
155                     ? null : node;
156             },
157 
158             _init:function () {
159                 DDM._regDrop(this);
160             },
161 
162             _active:function () {
163                 var self = this,
164                     drag = DDM.get("activeDrag"),
165                     node = self.get("node"),
166                     dropGroups = self.get("groups"),
167                     dragGroups = drag.get("groups"),
168                     prefixCls = DDM.get("prefixCls");
169                 if (validDrop(dropGroups, dragGroups)) {
170                     DDM._addValidDrop(self);
171                     // 委托时取不到节点
172                     if (node) {
173                         node.addClass(prefixCls + "drop-active-valid");
174                     }
175                 } else if (node) {
176                     node.addClass(prefixCls + "drop-active-invalid");
177                 }
178             },
179 
180             _deActive:function () {
181                 var node = this.get("node"),
182                     prefixCls = DDM.get("prefixCls");
183                 if (node) {
184                     node.removeClass(prefixCls + "drop-active-valid")
185                         .removeClass(prefixCls + "drop-active-invalid");
186                 }
187             },
188 
189             __getCustomEvt:function (ev) {
190                 return S.mix({
191                     drag:DDM.get("activeDrag"),
192                     drop:this
193                 }, ev);
194             },
195 
196             _handleOut:function () {
197                 var self = this,
198                     ret = self.__getCustomEvt();
199                 self.get("node").removeClass(DDM.get("prefixCls") + "drop-over");
200                 /**
201                  * html5 => dragleave
202                  */
203                 self.fire("dropexit", ret);
204             },
205 
206             _handleEnter:function (ev) {
207                 var self = this,
208                     e = self.__getCustomEvt(ev);
209                 e.drag._handleEnter(e);
210                 self.get("node").addClass(DDM.get("prefixCls") + "drop-over");
211                 self.fire("dropenter", e);
212             },
213 
214 
215             _handleOver:function (ev) {
216                 var self = this,
217                     e = self.__getCustomEvt(ev);
218                 e.drag._handleOver(e);
219                 self.fire("dropover", e);
220             },
221 
222             _end:function () {
223                 var self = this,
224                     ret = self.__getCustomEvt();
225                 self.get("node").removeClass(DDM.get("prefixCls") + "drop-over");
226                 self.fire('drophit', ret);
227             },
228 
229             /**
230              * make this droppable' element undroppable
231              */
232             destroy:function () {
233                 DDM._unRegDrop(this);
234             }
235         });
236 
237     return Droppable;
238 
239 }, { requires:["node", "base", "./ddm"] });
