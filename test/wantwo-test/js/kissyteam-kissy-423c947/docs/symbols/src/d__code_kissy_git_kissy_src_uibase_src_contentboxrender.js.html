1 /**
  2  * @fileOverview 里层包裹层定义，适合mask以及shim
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("uibase/contentboxrender", function (S, Node, BoxRender) {
  6 
  7 
  8     function ContentBoxRender() {
  9     }
 10 
 11     ContentBoxRender.ATTRS = {
 12         contentEl:{},
 13         contentElAttrs:{},
 14         contentElCls:{},
 15         contentElStyle:{},
 16         contentTagName:{
 17             value:"div"
 18         },
 19         content:{
 20             sync:false
 21         }
 22     };
 23 
 24     /*
 25      ! contentEl 只能由组件动态生成
 26      */
 27     ContentBoxRender.HTML_PARSER = {
 28         content:function (el) {
 29             return el[0].innerHTML;
 30         }
 31     };
 32 
 33     var constructEl = BoxRender.construct;
 34 
 35     ContentBoxRender.prototype = {
 36 
 37         // no need ,shift create work to __createDom
 38         __renderUI:function () {
 39         },
 40 
 41         __createDom:function () {
 42             var self = this,
 43                 contentEl,
 44                 c = self.get("content"),
 45                 el = self.get("el"),
 46                 html = "",
 47                 elChildren = S.makeArray(el[0].childNodes);
 48 
 49             if (elChildren.length) {
 50                 html = el[0].innerHTML
 51             }
 52 
 53             // el html 和 c 相同，直接 append el的子节点
 54             if (c == html) {
 55                 c = "";
 56             }
 57 
 58             contentEl = new Node(constructEl(
 59                 self.get("prefixCls") + "contentbox "
 60                     + (self.get("contentElCls") || ""),
 61                 self.get("contentElStyle"),
 62                 undefined,
 63                 undefined,
 64                 self.get("contentTagName"),
 65                 self.get("contentElAttrs"),
 66                 c));
 67 
 68             contentEl.appendTo(el);
 69 
 70             self.__set("contentEl", contentEl);
 71             // on content,then read from box el
 72             if (!c) {
 73                 for (var i = 0, l = elChildren.length; i < l; i++) {
 74                     contentEl.append(elChildren[i]);
 75                 }
 76             } else if (typeof c !== 'string') {
 77                 contentEl.append(c);
 78             }
 79         },
 80 
 81         _uiSetContentElCls:function (cls) {
 82             this.get("contentEl").addClass(cls);
 83         },
 84 
 85         _uiSetContentElAttrs:function (attrs) {
 86             this.get("contentEl").attr(attrs);
 87         },
 88 
 89         _uiSetContentElStyle:function (v) {
 90             this.get("contentEl").css(v);
 91         },
 92 
 93         _uiSetContent:function (c) {
 94             if (typeof c == "string") {
 95                 this.get("contentEl").html(c);
 96             } else {
 97                 this.get("contentEl").empty().append(c);
 98             }
 99         }
100     };
101 
102     return ContentBoxRender;
103 }, {
104     requires:["node", "./boxrender"]
105 });
