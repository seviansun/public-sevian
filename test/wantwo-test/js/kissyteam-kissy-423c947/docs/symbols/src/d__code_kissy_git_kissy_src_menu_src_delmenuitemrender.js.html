1 /**
  2  * @fileOverview deletable menuitemrender
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/delmenuitemrender", function(S, Node, UIBase, Component, MenuItemRender) {
  6     var CLS = "menuitem-deletable",
  7         DEL_CLS = "menuitem-delete";
  8     var DEL_TMPL = '<span class="{prefixCls}' + DEL_CLS + '" title="{tooltip}">X<' + '/span>';
  9 
 10     function addDel(self) {
 11         self.get("contentEl").append(S.substitute(DEL_TMPL, {
 12             prefixCls:self.get("prefixCls"),
 13             tooltip:self.get("delTooltip")
 14         }));
 15     }
 16 
 17     return UIBase.create(MenuItemRender, {
 18         createDom:function() {
 19             addDel(this);
 20         },
 21         _uiSetContent:function(v) {
 22             var self = this;
 23             MenuItemRender.prototype._uiSetContent.call(self, v);
 24             addDel(self);
 25         },
 26 
 27         _uiSetDelTooltip:function() {
 28             this._uiSetContent(this.get("content"));
 29         }
 30     }, {
 31         ATTRS:{
 32             delTooltip:{}
 33         },
 34         HTML_PARSER:{
 35             delEl:function(el) {
 36                 return el.one(this.getCls(DEL_CLS));
 37             }
 38         },
 39         CLS:CLS,
 40         DEL_CLS:DEL_CLS
 41     });
 42 }, {
 43     requires:['node','uibase','component','./menuitemrender']
 44 });
