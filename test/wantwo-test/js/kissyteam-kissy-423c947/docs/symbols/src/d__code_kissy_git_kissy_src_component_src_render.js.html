1 /**
  2  * @fileOverview render base class for kissy
  3  * @author yiminghe@gmail.com
  4  * @see http://martinfowler.com/eaaDev/uiArchs.html
  5  */
  6 KISSY.add("component/render", function (S, UIBase, UIStore) {
  7 
  8     function tagFunc(self, classes, tag) {
  9         return self.getCls(classes.split(/\s+/).join(tag + " ") + tag);
 10     }
 11 
 12     return UIBase.create([UIBase.Box.Render], {
 13 
 14         __CLASS:"Component.Render",
 15 
 16         _completeClasses:function (classes, tag) {
 17             return tagFunc(this, classes, tag);
 18         },
 19 
 20         /**
 21          * @protected
 22          */
 23         _renderCls:function (componentCls) {
 24             var self = this;
 25             self.get("el").addClass(self.getCls(componentCls));
 26         },
 27 
 28         getCls:UIStore.getCls,
 29 
 30         getKeyEventTarget:function () {
 31             return this.get("el");
 32         },
 33 
 34         getContentElement:function () {
 35             return this.get("contentEl") || this.get("el");
 36         },
 37 
 38         /**
 39          * @protected
 40          */
 41         _uiSetFocusable:function (v) {
 42             var el = this.getKeyEventTarget(),
 43                 tabindex = el.attr("tabindex");
 44             if (tabindex >= 0 && !v) {
 45                 el.attr("tabindex", -1);
 46             } else if (!(tabindex >= 0) && v) {
 47                 el.attr("tabindex", 0);
 48             }
 49         },
 50 
 51         /**
 52          * @protected
 53          */
 54         _setHighlighted:function (v, componentCls) {
 55             var self = this, el = self.get("el");
 56             el[v ? 'addClass' : 'removeClass'](tagFunc(self, componentCls, "-hover"));
 57         },
 58 
 59         /**
 60          * @protected
 61          */
 62         _setDisabled:function (v, componentCls) {
 63             var self = this, el = self.get("el");
 64             el[v ? 'addClass' : 'removeClass'](tagFunc(self, componentCls, "-disabled"))
 65                 //不能被 tab focus 到
 66                 //support aria
 67                 .attr({
 68                     tabindex:v ? -1 : 0,
 69                     aria-disabled:v
 70                 });
 71 
 72         },
 73         /**
 74          * @protected
 75          */
 76         _setActive:function (v, componentCls) {
 77             var self = this;
 78             self.get("el")[v ? 'addClass' : 'removeClass'](tagFunc(self, componentCls, "-active"))
 79                 .attr("aria-pressed", !!v);
 80         },
 81         /**
 82          * @protected
 83          */
 84         _setFocused:function (v, componentCls) {
 85             var self = this, el = self.get("el");
 86             el[v ? 'addClass' : 'removeClass'](tagFunc(self, componentCls, "-focused"));
 87         }
 88 
 89     }, {
 90         ATTRS:{
 91             /**
 92              *  screen state
 93              */
 94             prefixCls:{},
 95             focusable:{}
 96         }
 97     });
 98 }, {
 99     requires:['uibase', './uistore']
100 });
