1 /**
  2  * @fileOverview Model and Control for button
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("button/base", function (S, Event, UIBase, Component, CustomRender) {
  6 
  7     var KeyCodes = Event.KeyCodes;
  8     /**
  9      * @name Button
 10      * @constructor
 11      * @extends Component.Controller
 12      */
 13     var Button = UIBase.create(Component.Controller, [UIBase.ContentBox],
 14         /**@lends Button.prototype */
 15         {
 16 
 17             bindUI:function () {
 18                 this.get("el").on("keyup", this._handleKeyEventInternal, this);
 19             },
 20 
 21             _handleKeyEventInternal:function (e) {
 22                 if (e.keyCode == KeyCodes.ENTER &&
 23                     e.type == "keydown" ||
 24                     e.keyCode == KeyCodes.SPACE &&
 25                         e.type == "keyup") {
 26                     return this._performInternal(e);
 27                 }
 28                 // Return true for space keypress (even though the event is handled on keyup)
 29                 // as preventDefault needs to be called up keypress to take effect in IE and
 30                 // WebKit.
 31                 return e.keyCode == KeyCodes.SPACE;
 32             },
 33 
 34             /* button 的默认行为就是触发 click*/
 35             _performInternal:function () {
 36                 var self = this;
 37                 self.fire("click");
 38             }
 39         }, {
 40             ATTRS:{
 41                 /**
 42                  * @inheritedDoc
 43                  * disabled:{}
 44                  */
 45                 value:{},
 46                 describedby:{
 47                     view:true
 48                 },
 49                 tooltip:{
 50                     view:true
 51                 },
 52                 collapseSide:{
 53                     view:true
 54                 }
 55             }
 56         });
 57 
 58     Button.DefaultRender = CustomRender;
 59 
 60 
 61     Component.UIStore.setUIByClass("button", {
 62         priority:Component.UIStore.PRIORITY.LEVEL1,
 63         ui:Button
 64     });
 65 
 66     return Button;
 67 
 68 }, {
 69     requires:['event', 'uibase', 'component', './customrender']
 70 });
