1 /**
  2  * @fileOverview KISSY.Dialog
  3  * @author  yiminghe@gmail.com, 乔花<qiaohua@taobao.com>
  4  */
  5 KISSY.add('overlay/dialog', function (S, Component, Overlay, UIBase, DialogRender, Aria) {
  6 
  7     function require(s) {
  8         return S.require("uibase/" + s);
  9     }
 10 
 11     /**
 12      * KISSY Dialog Component
 13      * @class
 14      * @name Dialog
 15      * @memberOf Overlay
 16      * @extends Overlay
 17      * @extends UIBase.StdMod
 18      * @extends UIBase.Drag
 19      * @extends UIBase.Constrain
 20      */
 21     var Dialog = UIBase.create(Overlay, [
 22         require("stdmod"),
 23         require("drag"),
 24         require("constrain"),
 25         Aria
 26     ],
 27         /**
 28          * @lends Overlay.Dialog#
 29          */
 30         {
 31             /**
 32              * see {@link UIBase.Box#show}
 33              * @name Overlay.Dialog#show
 34              * @function
 35              */
 36         },
 37 
 38         {
 39             ATTRS:/**
 40              * @lends Overlay.Dialog#
 41              */
 42             {
 43 
 44                 /**
 45                  * whether this component can be closed. Default:true
 46                  * @type Boolean
 47                  */
 48                 closable:{
 49                     value:true
 50                 },
 51 
 52                 /**
 53                  * Default: Dialog's header element
 54                  * see {@link DD.Draggable#handlers}
 55                  */
 56                 handlers:{
 57                     valueFn:function () {
 58                         var self = this;
 59                         return [
 60                             // 运行时取得拖放头
 61                             function () {
 62                                 return self.get("view").get("header");
 63                             }
 64                         ];
 65                     }
 66                 }
 67             }
 68         });
 69 
 70     Dialog.DefaultRender = DialogRender;
 71 
 72     Component.UIStore.setUIByClass("dialog", {
 73         priority:Component.UIStore.PRIORITY.LEVEL2,
 74         ui:Dialog
 75     });
 76 
 77     return Dialog;
 78 
 79 }, {
 80     requires:[ "component", "overlay/base", "uibase", 'overlay/dialogrender', './aria']
 81 });
