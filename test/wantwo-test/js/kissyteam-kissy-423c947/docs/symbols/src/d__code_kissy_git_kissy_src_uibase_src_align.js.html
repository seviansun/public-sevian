 13      * 得到影响元素显示的父亲元素
 14      */
 15     function getOffsetParent(element) {
 16         // ie 这个也不是完全可行
 17         /**
 18          <div style="width: 50px;height: 100px;overflow: hidden">
 19          <div style="width: 50px;height: 100px;position: relative;" id="d6">
 20          元素 6 高 100px 宽 50px<br/>
 21          </div>
 22          </div>
 23          **/
 24 //            if (UA['ie']) {
 25 //                return element.offsetParent;
 26 //            }
 27         var body = element.ownerDocument.body,
 28             positionStyle = DOM.css(element, 'position'),
 29             skipStatic = positionStyle == 'fixed' || positionStyle == 'absolute';
 30 
 31         for (var parent = element.parentNode;
 32              parent && parent != body;
 33              parent = parent.parentNode) {
 34 
 35             positionStyle = DOM.css(parent, 'position');
 36 
 37             skipStatic = skipStatic && positionStyle == 'static';
 38 
 39             var parentOverflow = DOM.css(parent, "overflow");
 40 
 41             // 必须有 overflow 属性，可能会隐藏掉子孙元素
 42             if (parentOverflow != 'visible' && (
 43                 // 元素初始为 fixed absolute ，遇到 父亲不是 定位元素忽略
 44                 // 否则就可以
 45                 !skipStatic ||
 46                     positionStyle == 'fixed' ||
 47                     positionStyle == 'absolute' ||
 48                     positionStyle == 'relative'
 49                 )) {
 50                 return parent;
 51             }
 52         }
 53         return null;
 54     }

 56     /**
 57      * 获得元素的显示部分的区域
 58      */
 59     function getVisibleRectForElement(element) {
 60         var visibleRect = {
 61             left:0,
 62             right:Infinity,
 63             top:0,
 64             bottom:Infinity
 65         };
 66 
 67         for (var el = element; el = getOffsetParent(el);) {
 68 
 69             var clientWidth = el.clientWidth;
 70 
 71             if (
 72             // clientWidth is zero for inline block elements in IE.
 73                 (!UA['ie'] || clientWidth !== 0)
 74             // on WEBKIT, body element can have clientHeight = 0 and scrollHeight > 0
 75             // && (!UA['webkit'] || clientHeight != 0 || el != body)
 76             // overflow 不为 visible 则可以限定其内元素
 77             // && (scrollWidth != clientWidth || scrollHeight != clientHeight)
 78             // offsetParent 已经判断过了
 79             // && DOM.css(el, 'overflow') != 'visible'
 80                 ) {
 81                 var clientLeft = el.clientLeft,
 82                     clientTop = el.clientTop,
 83                     pos = DOM.offset(el),
 84                     client = {
 85                         left:clientLeft,
 86                         top:clientTop
 87                     };
 88                 pos.left += client.left;
 89                 pos.top += client.top;
 90 
 91                 visibleRect.top = Math.max(visibleRect['top'], pos.top);
 92                 visibleRect.right = Math.min(visibleRect.right,
 93                     pos.left + el.clientWidth);
 94                 visibleRect.bottom = Math.min(visibleRect['bottom'],
 95                     pos.top + el.clientHeight);
 96                 visibleRect.left = Math.max(visibleRect.left, pos.left);
 97             }
 98         }
 99 
100         var scrollX = DOM.scrollLeft(),
101             scrollY = DOM.scrollTop();
102 
103         visibleRect.left = Math.max(visibleRect.left, scrollX);
104         visibleRect.top = Math.max(visibleRect['top'], scrollY);
105         visibleRect.right = Math.min(visibleRect.right, scrollX + DOM.viewportWidth());
106         visibleRect.bottom = Math.min(visibleRect['bottom'], scrollY + DOM.viewportHeight());
107 
108         return visibleRect.top >= 0 && visibleRect.left >= 0 &&
109             visibleRect.bottom > visibleRect.top &&
110             visibleRect.right > visibleRect.left ?
111             visibleRect : null;
112     }

114     function isFailed(status) {
115         for (var s in status) {
116             if (s.indexOf("fail") === 0) {
117                 return true;
118             }
119         }
120         return false;
121     }

123     function positionAtAnchor(alignCfg) {
124         var offset = alignCfg.offset,
125             node = alignCfg.node,
126             points = alignCfg.points,
127             self = this,
128             xy,
129             diff,
130             p1,
131             //如果没有view，就是不区分mvc
132             el = self.get('el'),
133             p2;
134 
135         offset = offset || [0, 0];
136         xy = el.offset();
137 
138         // p1 是 node 上 points[0] 的 offset
139         // p2 是 overlay 上 points[1] 的 offset
140         p1 = getAlignOffset(node, points[0]);
141         p2 = getAlignOffset(el, points[1]);
142 
143         diff = [p2.left - p1.left, p2.top - p1.top];
144         xy = {
145             left:xy.left - diff[0] + (+offset[0]),
146             top:xy.top - diff[1] + (+offset[1])
147         };
148 
149         return positionAtCoordinate.call(self, xy, alignCfg);
150     }

153     function positionAtCoordinate(absolutePos, alignCfg) {
154         var self = this, el = self.get('el');
155         var status = {};
156         var elSize = {width:el.outerWidth(), height:el.outerHeight()},
157             size = S.clone(elSize);
158         if (!S.isEmptyObject(alignCfg.overflow)) {
159             var viewport = getVisibleRectForElement(el[0]);
160             status = adjustForViewport(absolutePos, size, viewport, alignCfg.overflow || {});
161             if (isFailed(status)) {
162                 return status;
163             }
164         }
165 
166         self.set("x", absolutePos.left);
167         self.set("y", absolutePos.top);
168 
169         if (size.width != elSize.width || size.height != elSize.height) {
170             el.width(size.width);
171             el.height(size.height);
172         }
173 
174         return status;
175 
176     }

179     function adjustForViewport(pos, size, viewport, overflow) {
180         var status = {};
181         if (pos.left < viewport.left && overflow.adjustX) {
182             pos.left = viewport.left;
183             status.adjustX = 1;
184         }
185         // Left edge inside and right edge outside viewport, try to resize it.
186         if (pos.left < viewport.left &&
187             pos.left + size.width > viewport.right &&
188             overflow.resizeWidth) {
189             size.width -= (pos.left + size.width) - viewport.right;
190             status.resizeWidth = 1;
191         }
192 
193         // Right edge outside viewport, try to move it.
194         if (pos.left + size.width > viewport.right &&
195             overflow.adjustX) {
196             pos.left = Math.max(viewport.right - size.width, viewport.left);
197             status.adjustX = 1;
198         }
199 
200         // Left or right edge still outside viewport, fail if the FAIL_X option was
201         // specified, ignore it otherwise.
202         if (overflow.failX) {
203             status.failX = pos.left < viewport.left ||
204                 pos.left + size.width > viewport.right;
205         }
206 
207         // Top edge outside viewport, try to move it.
208         if (pos.top < viewport.top && overflow.adjustY) {
209             pos.top = viewport.top;
210             status.adjustY = 1;
211         }
212 
213         // Top edge inside and bottom edge outside viewport, try to resize it.
214         if (pos.top >= viewport.top &&
215             pos.top + size.height > viewport.bottom &&
216             overflow.resizeHeight) {
217             size.height -= (pos.top + size.height) - viewport.bottom;
218             status.resizeHeight = 1;
219         }
220 
221         // Bottom edge outside viewport, try to move it.
222         if (pos.top + size.height > viewport.bottom &&
223             overflow.adjustY) {
224             pos.top = Math.max(viewport.bottom - size.height, viewport.top);
225             status.adjustY = 1;
226         }
227 
228         // Top or bottom edge still outside viewport, fail if the FAIL_Y option was
229         // specified, ignore it otherwise.
230         if (overflow.failY) {
231             status.failY = pos.top < viewport.top ||
232                 pos.top + size.height > viewport.bottom;
233         }
234 
235         return status;
236     }

239     function flip(points, reg, map) {
240         var ret = [];
241         S.each(points, function (p) {
242             ret.push(p.replace(reg, function (m) {
243                 return map[m];
244             }));
245         });
246         return ret;
247     }

249     function flipOffset(offset, index) {
250         offset[index] = -offset[index];
251         return offset;
252     }

292     /**
293      * 获取 node 上的 align 对齐点 相对于页面的坐标
294      * @param node
295      * @param align
296      */
297     function getAlignOffset(node, align) {
298         var V = align.charAt(0),
299             H = align.charAt(1),
300             offset, w, h, x, y;
301 
302         if (node && !S.isWindow(node)) {
303             offset = node.offset();
304             w = node.outerWidth();
305             h = node.outerHeight();
306         } else {
307             offset = { left:DOM.scrollLeft(), top:DOM.scrollTop() };
308             w = DOM.viewportWidth();
309             h = DOM.viewportHeight();
310         }
311 
312         x = offset.left;
313         y = offset.top;
314 
315         if (V === 'c') {
316             y += h / 2;
317         } else if (V === 'b') {
318             y += h;
319         }
320 
321         if (H === 'c') {
322             x += w / 2;
323         } else if (H === 'r') {
324             x += w;
325         }
326 
327         return { left:x, top:y };
328     }

330     Align.prototype =
331     /**
332      * @lends UIBase.Align.prototype
333      */
334     {
335         _uiSetAlign:function (v) {
336             if (v) {
337                 this.align(v.node, v.points, v.offset, v.overflow);
338             }
339         },
340 
341         /*
342          对齐 Overlay 到 node 的 points 点, 偏移 offset 处
343          @function
344          @ignore
345          @param {Element} node 参照元素, 可取配置选项中的设置, 也可是一元素
346          @param {String[]} points 对齐方式
347          @param {Number[]} [offset] 偏移
348          */
349         align:function (node, points, offset, overflow) {
350             var self = this,
351                 flag = {};
352             if (node) {
353                 node = Node.one(node);
354             }
355             // 后面会改的，先保存下
356             overflow = S.clone(overflow || {});
357             offset = offset && [].concat(offset) || [0, 0];
358             if (overflow.failX) {
359                 flag.failX = 1;
360             }
361             if (overflow.failY) {
362                 flag.failY = 1;
363             }
364             var status = positionAtAnchor.call(self, {
365                 node:node,
366                 points:points,
367                 offset:offset,
368                 overflow:flag
369             });
370             // 如果错误调整重试
371             if (isFailed(status)) {
372                 if (status.failX) {
373                     points = flip(points, /[lr]/ig, {
374                         l:"r",
375                         r:"l"
376                     });
377                     offset = flipOffset(offset, 0);
378                 }
379 
380                 if (status.failY) {
381                     points = flip(points, /[tb]/ig, {
382                         t:"b",
383                         b:"t"
384                     });
385                     offset = flipOffset(offset, 1);
386                 }
387             }
388 
389             status = positionAtAnchor.call(self, {
390                 node:node,
391                 points:points,
392                 offset:offset,
393                 overflow:flag
394             });
395 
396             if (isFailed(status)) {
397                 delete overflow.failX;
398                 delete overflow.failY;
399                 positionAtAnchor.call(self, {
400                     node:node,
401                     points:points,
402                     offset:offset,
403                     overflow:overflow
404                 });
405             }
406         },
407 
408         /**
409          * 居中显示到可视区域, 一次性居中
410          * @param {undefined|String|HTMLElement|Node} node 对其元素，falsy 表示窗口可视区域
411          */
412         center:function (node) {
413             this.set('align', {
414                 node:node,
415                 points:["cc", "cc"],
416                 offset:[0, 0]
417             });
418         }
419     };
