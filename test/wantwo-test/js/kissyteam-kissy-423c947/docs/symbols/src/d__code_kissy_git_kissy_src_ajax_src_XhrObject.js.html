1 /**
  2  * @fileOverview encapsulation of io object . as transaction object in yui3
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("ajax/XhrObject", function (S, undefined) {
  6 
  7     var OK_CODE = 200,
  8         Promise = S.Promise,
  9         MULTIPLE_CHOICES = 300,
 10         NOT_MODIFIED = 304,
 11         // get individual response header from responseheader str
 12         rheaders = /^(.*?):[ \t]*([^\r\n]*)\r?$/mg;
 13 
 14     function handleResponseData(xhrObject) {
 15 
 16         // text xml 是否原生转化支持
 17         var text = xhrObject.responseText,
 18             xml = xhrObject.responseXML,
 19             c = xhrObject.config,
 20             cConverts = c.converters,
 21             xConverts = xhrObject.converters || {},
 22             type,
 23             responseData,
 24             contents = c.contents,
 25             dataType = c.dataType;
 26 
 27         // 例如 script 直接是js引擎执行，没有返回值，不需要自己处理初始返回值
 28         // jsonp 时还需要把 script 转换成 json，后面还得自己来
 29         if (text || xml) {
 30 
 31             var contentType = xhrObject.mimeType || xhrObject.getResponseHeader("Content-Type");
 32 
 33             // 去除无用的通用格式
 34             while (dataType[0] == "*") {
 35                 dataType.shift();
 36             }
 37 
 38             if (!dataType.length) {
 39                 // 获取源数据格式，放在第一个
 40                 for (type in contents) {
 41                     if (contents[type].test(contentType)) {
 42                         if (dataType[0] != type) {
 43                             dataType.unshift(type);
 44                         }
 45                         break;
 46                     }
 47                 }
 48             }
 49             // 服务器端没有告知（并且客户端没有mimetype）默认 text 类型
 50             dataType[0] = dataType[0] || "text";
 51 
 52             //获得合适的初始数据
 53             if (dataType[0] == "text" && text !== undefined) {
 54                 responseData = text;
 55             }
 56             // 有 xml 值才直接取，否则可能还要从 xml 转
 57             else if (dataType[0] == "xml" && xml !== undefined) {
 58                 responseData = xml;
 59             } else {
 60                 // 看能否从 text xml 转换到合适数据，并设置起始类型为 text/xml
 61                 S.each(["text", "xml"], function (prevType) {
 62                     var type = dataType[0],
 63                         converter = xConverts[prevType] && xConverts[prevType][type] ||
 64                             cConverts[prevType] && cConverts[prevType][type];
 65                     if (converter) {
 66                         dataType.unshift(prevType);
 67                         responseData = prevType == "text" ? text : xml;
 68                         return false;
 69                     }
 70                 });
 71             }
 72         }
 73         var prevType = dataType[0];
 74 
 75         // 按照转化链把初始数据转换成我们想要的数据类型
 76         for (var i = 1; i < dataType.length; i++) {
 77             type = dataType[i];
 78 
 79             var converter = xConverts[prevType] && xConverts[prevType][type] ||
 80                 cConverts[prevType] && cConverts[prevType][type];
 81 
 82             if (!converter) {
 83                 throw "no covert for " + prevType + " => " + type;
 84             }
 85             responseData = converter(responseData);
 86 
 87             prevType = type;
 88         }
 89 
 90         xhrObject.responseData = responseData;
 91     }
 92 
 93     /**
 94      * @class A class for constructing io request instances. !Do Not New By Yourself!
 95      * @extends KISSY.Promise
 96      * @memberOf IO
 97      */
 98     function XhrObject(c) {
 99         Promise.call(this);
100         S.mix(this, {
101             // 结构化数据，如 json
102             responseData:null,
103             config:c || {},
104             timeoutTimer:null,
105 
106             /**
107              * @field
108              * @memberOf IO.XhrObject#
109              * @description String typed data returned from server
110              */
111             responseText:null,
112             /**
113              * @field
114              * @memberOf IO.XhrObject#
115              * @description xml typed data returned from server
116              */
117             responseXML:null,
118             responseHeadersString:"",
119             responseHeaders:null,
120             requestHeaders:{},
121             /**
122              * @field
123              * @memberOf IO.XhrObject#
124              * @description <br>
125              * readyState of current request<br>
126              * 0: initialized<br>
127              * 1: send <br>
128              * 4: completed<br>
129              */
130             readyState:0,
131             state:0,
132             /**
133              * @field
134              * @memberOf IO.XhrObject#
135              * @description HTTP statusText of current request
136              */
137             statusText:null,
138             /**
139              * @field
140              * @memberOf IO.XhrObject#
141              * @description <br> HTTP Status Code of current request <br>
142              * eg:<br>
143              * 200 : ok<br>
144              * 404 : Not Found<br>
145              * 500 : Server Error<br>
146              */
147             status:0,
148             transport:null,
149             _defer:new S.Defer(this)
150         });
151     }
152 
153     S.extend(XhrObject, Promise,
154         /**
155          * @lends IO.XhrObject.prototype
156          */
157         {
158             // Caches the header
159             setRequestHeader:function (name, value) {
160                 var self = this;
161                 self.requestHeaders[ name ] = value;
162                 return self;
163             },
164 
165             /**
166              * get all response headers as string after request is completed
167              * @returns {String}
168              */
169             getAllResponseHeaders:function () {
170                 var self = this;
171                 return self.state === 2 ? self.responseHeadersString : null;
172             },
173 
174             /**
175              * get header value in response to specified header name
176              * @param {String} name header name
177              * @return {String} header value
178              */
179             getResponseHeader:function (name) {
180                 var match, self = this;
181                 if (self.state === 2) {
182                     if (!self.responseHeaders) {
183                         self.responseHeaders = {};
184                         while (( match = rheaders.exec(self.responseHeadersString) )) {
185                             self.responseHeaders[ match[1] ] = match[ 2 ];
186                         }
187                     }
188                     match = self.responseHeaders[ name ];
189                 }
190                 return match === undefined ? null : match;
191             },
192 
193             // Overrides response content-type header
194             overrideMimeType:function (type) {
195                 var self = this;
196                 if (!self.state) {
197                     self.mimeType = type;
198                 }
199                 return self;
200             },
201 
202             /**
203              * cancel this request
204              * @param {String} [statusText=abort] error reason as current request object's statusText
205              */
206             abort:function (statusText) {
207                 var self = this;
208                 statusText = statusText || "abort";
209                 if (self.transport) {
210                     self.transport.abort(statusText);
211                 }
212                 self._xhrReady(0, statusText);
213                 return self;
214             },
215 
216             /**
217              * get native XMLHttpRequest
218              * @since 1.3
219              */
220             getNativeXhr:function () {
221                 var transport;
222                 if (transport = this.transport) {
223                     return transport.nativeXhr;
224                 }
225             },
226 
227             _xhrReady:function (status, statusText) {
228                 var self = this;
229                 // 只能执行一次，防止重复执行
230                 // 例如完成后，调用 abort
231 
232                 // 到这要么成功，调用success
233                 // 要么失败，调用 error
234                 // 最终都会调用 complete
235                 if (self.state == 2) {
236                     return;
237                 }
238                 self.state = 2;
239                 self.readyState = 4;
240                 var isSuccess;
241                 if (status >= OK_CODE && status < MULTIPLE_CHOICES || status == NOT_MODIFIED) {
242 
243                     if (status == NOT_MODIFIED) {
244                         statusText = "notmodified";
245                         isSuccess = true;
246                     } else {
247                         try {
248                             handleResponseData(self);
249                             statusText = "success";
250                             isSuccess = true;
251                         } catch (e) {
252                             statusText = "parsererror : " + e;
253                         }
254                     }
255 
256                 } else {
257                     if (status < 0) {
258                         status = 0;
259                     }
260                 }
261 
262                 self.status = status;
263                 self.statusText = statusText;
264 
265                 var defer = self._defer;
266                 defer[isSuccess ? "resolve" : "reject"]([self.responseData, self.statusText, self]);
267             }
268         }
269     );
270 
271     return XhrObject;
272 });
