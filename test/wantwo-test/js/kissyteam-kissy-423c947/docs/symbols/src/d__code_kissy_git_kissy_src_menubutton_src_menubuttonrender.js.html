1 /**
  2  * @fileOverview render aria and drop arrow for menubutton
  3  * @author  yiminghe@gmail.com
  4  */
  5 KISSY.add("menubutton/menubuttonrender", function(S, UIBase, Button) {
  6 
  7     var MENU_BUTTON_TMPL = '<div class="{prefixCls}inline-block ' +
  8         '{prefixCls}menu-button-caption">{content}<' + '/div>' +
  9         '<div class="{prefixCls}inline-block ' +
 10         '{prefixCls}menu-button-dropdown"> <' + '/div>',
 11         CAPTION_CLS = "menu-button-caption",
 12         COLLAPSE_CLS = "menu-button-open";
 13 
 14     return UIBase.create(Button.Render, {
 15 
 16         createDom:function() {
 17             var innerEl = this.get("innerEl"),
 18                 html = S.substitute(MENU_BUTTON_TMPL, {
 19                     content:this.get("content") || "",
 20                     prefixCls:this.get("prefixCls")
 21                 });
 22             innerEl
 23                 .html(html)
 24                 //带有 menu
 25                 .attr("aria-haspopup", true);
 26         },
 27 
 28         _uiSetContent:function(v) {
 29             var caption = this.get("el").one("." + this.getCls(CAPTION_CLS));
 30             caption.html("");
 31             v && caption.append(v);
 32         },
 33 
 34         _uiSetCollapsed:function(v) {
 35             var self = this,
 36                 el = self.get("el"),
 37                 cls = self.getCls(COLLAPSE_CLS);
 38             el[v ? 'removeClass' : 'addClass'](cls).attr("aria-expanded", !v);
 39         },
 40 
 41         _uiSetActiveItem:function(v) {
 42             this.get("el").attr("aria-activedescendant",
 43                 (v && v.get("el").attr("id")) || "");
 44         }
 45     }, {
 46         ATTRS:{
 47             activeItem:{
 48             },
 49             collapsed:{
 50             }
 51         }
 52     });
 53 }, {
 54     requires:['uibase','button']
 55 });
