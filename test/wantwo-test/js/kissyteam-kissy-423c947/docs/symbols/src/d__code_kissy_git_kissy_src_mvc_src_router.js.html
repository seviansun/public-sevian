1 /**
  2  * @fileOverview simple router to get path parameter and query parameter from hash(old ie) or url(html5)
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add('mvc/router', function (S, Event, Base) {
  6     var queryReg = /\?(.*)/,
  7         each = S.each,
  8         // take a breath to avoid duplicate hashchange
  9         BREATH_INTERVAL = 100,
 10         grammar = /(:([\w\d]+))|(\\\*([\w\d]+))/g,
 11         // all registered route instance
 12         allRoutes = [],
 13         win = S.Env.host,
 14         location = win.location,
 15         history = win.history ,
 16         supportNativeHistory = !!(history && history['pushState']),
 17         __routerMap = "__routerMap";
 18 
 19     function findFirstCaptureGroupIndex(regStr) {
 20         var r, i;
 21         for (i = 0;
 22              i < regStr.length;
 23              i++) {
 24             r = regStr.charAt(i);
 25             // skip escaped reg meta char
 26             if (r == "\\") {
 27                 i++;
 28             } else if (r == "(") {
 29                 return i;
 30             }
 31         }
 32         throw new Error("impossible to not to get capture group in kissy mvc route");
 33     }
 34 
 35     function getHash() {
 36         // 不能 location.hash
 37         // http://xx.com/#yy?z=1
 38         // ie6 => location.hash = #yy
 39         // 其他浏览器 => location.hash = #yy?z=1
 40         return location.href.replace(/^[^#]*#?!?(.*)$/, '$1');
 41     }
 42 
 43     /**
 44      * get url fragment and dispatch
 45      */
 46     function getFragment() {
 47         if (Router.nativeHistory && supportNativeHistory) {
 48             return location.pathname.substr(Router.urlRoot.length) + location.search;
 49         } else {
 50             return getHash();
 51         }
 52     }
 53 
 54     /**
 55      * slash ------------- start
 56      */
 57 
 58     /**
 59      * whether string end with slash
 60      * @param str
 61      */
 62     function endWithSlash(str) {
 63         return S.endsWith(str, "/");
 64     }
 65 
 66     function startWithSlash(str) {
 67         return S.startsWith(str, "/");
 68     }
 69 
 70     function removeEndSlash(str) {
 71         if (endWithSlash(str)) {
 72             str = str.substring(0, str.length - 1);
 73         }
 74         return str;
 75     }
 76 
 77     function removeStartSlash(str) {
 78         if (startWithSlash(str)) {
 79             str = str.substring(1);
 80         }
 81         return str;
 82     }
 83 
 84     function addEndSlash(str) {
 85         return removeEndSlash(str) + "/";
 86     }
 87 
 88     function addStartSlash(str) {
 89         return "/" + removeStartSlash(str);
 90     }
 91 
 92     function equalsIgnoreSlash(str1, str2) {
 93         str1 = removeEndSlash(str1);
 94         str2 = removeEndSlash(str2);
 95         return str1 == str2;
 96     }
 97 
 98     /**
 99      * slash ------------------  end
100      */
101 
102     /**
103      * get full path from fragment for html history
104      * @param fragment
105      */
106     function getFullPath(fragment) {
107         return location.protocol + "//" + location.host +
108             removeEndSlash(Router.urlRoot) + addStartSlash(fragment)
109     }
110 
111     /**
112      * get query object from query string
113      * @param path
114      */
115     function getQuery(path) {
116         var m,
117             ret = {};
118         if (m = path.match(queryReg)) {
119             return S.unparam(m[1]);
120         }
121         return ret;
122     }
123 
124     /**
125      * match url with route intelligently (always get optimal result)
126      */
127     function dispatch() {
128         var path = getFragment(),
129             fullPath = path,
130             query,
131             arg,
132             finalRoute = 0,
133             finalMatchLength = -1,
134             finalRegStr = "",
135             finalFirstCaptureGroupIndex = -1,
136             finalCallback = 0,
137             finalRouteName = "",
138             finalParam = 0;
139 
140         path = fullPath.replace(queryReg, "");
141         // user input : /xx/yy/zz
142         each(allRoutes, function (route) {
143             var routeRegs = route[__routerMap],
144                 // match exactly
145                 exactlyMatch = 0;
146             each(routeRegs, function (desc) {
147                     var reg = desc.reg,
148                         regStr = desc.regStr,
149                         paramNames = desc.paramNames,
150                         firstCaptureGroupIndex = -1,
151                         m,
152                         name = desc.name,
153                         callback = desc.callback;
154                     if (m = path.match(reg)) {
155                         // match all result item shift out
156                         m.shift();
157 
158                         function genParam() {
159                             if (paramNames) {
160                                 var params = {};
161                                 each(m, function (sm, i) {
162                                     params[paramNames[i]] = sm;
163                                 });
164                                 return params;
165                             } else {
166                                 // if user gave directly reg
167                                 // then call callback with match result array
168                                 return [].concat(m);
169                             }
170                         }
171 
172                         function upToFinal() {
173                             finalRegStr = regStr;
174                             finalFirstCaptureGroupIndex = firstCaptureGroupIndex;
175                             finalCallback = callback;
176                             finalParam = genParam();
177                             finalRoute = route;
178                             finalRouteName = name;
179                             finalMatchLength = m.length;
180                         }
181 
182                         // route: /xx/yy/zz
183                         if (!m.length) {
184                             upToFinal();
185                             exactlyMatch = 1;
186                             return false;
187                         } else if (regStr) {
188 
189                             firstCaptureGroupIndex = findFirstCaptureGroupIndex(regStr);
190 
191                             // final route : /*
192                             // now route : /xx/*
193                             if (firstCaptureGroupIndex > finalFirstCaptureGroupIndex) {
194                                 upToFinal();
195                             }
196 
197                             // final route : /xx/:id/:id
198                             // now route :  /xx/:id/zz
199                             else if (
200                                 firstCaptureGroupIndex == finalFirstCaptureGroupIndex &&
201                                     finalMatchLength >= m.length
202                                 ) {
203                                 if (m.length < finalMatchLength) {
204                                     upToFinal()
205                                 } else if (regStr.length > finalRegStr.length) {
206                                     upToFinal();
207                                 }
208                             }
209 
210                             // first route has priority
211                             else if (!finalRoute) {
212                                 upToFinal();
213                             }
214                         }
215                         // if exists user-given reg router rule then update value directly
216                         else {
217                             upToFinal();
218                         }
219                     }
220                 }
221             );
222 
223             if (exactlyMatch) {
224                 return false;
225             }
226         });
227 
228 
229         if (finalParam) {
230             query = getQuery(fullPath);
231             finalCallback.apply(finalRoute, [finalParam, query]);
232             arg = {
233                 name:name,
234                 paths:finalParam,
235                 query:query
236             };
237             finalRoute.fire('route:' + name, arg);
238             finalRoute.fire('route', arg);
239         }
240     }

242     /**
243      * transform route declaration to router reg
244      * @param str
245      *         /search/:q
246      *         /user/*path
247      */
248     function transformRouterReg(self, str, callback) {
249         var name = str,
250             paramNames = [];
251 
252         if (S.isFunction(callback)) {
253             // escape keyword from regexp
254             str = S.escapeRegExp(str);
255 
256             str = str.replace(grammar, function (m, g1, g2, g3, g4) {
257                 paramNames.push(g2 || g4);
258                 // :name
259                 if (g2) {
260                     return "([^/]+)";
261                 }
262                 // *name
263                 else if (g4) {
264                     return "(.*)";
265                 }
266             });
267 
268             return {
269                 name:name,
270                 paramNames:paramNames,
271                 reg:new RegExp("^" + str + "$"),
272                 regStr:str,
273                 callback:callback
274             };
275         } else {
276             return {
277                 name:name,
278                 reg:callback.reg,
279                 callback:normFn(self, callback.callback)
280             };
281         }
282     }
283 
284     /**
285      * normalize function by self
286      * @param self
287      * @param callback
288      */
289     function normFn(self, callback) {
290         if (S.isFunction(callback)) {
291             return callback;
292         } else if (S.isString(callback)) {
293             return self[callback];
294         }
295         return callback;
296     }

298     function _afterRoutesChange(e) {
299         var self = this;
300         self[__routerMap] = {};
301         self.addRoutes(e.newVal);
302     }
303 
304     /**
305      * @class
306      * @memberOf MVC
307      */
308     function Router() {
309         var self = this;
310         Router.superclass.constructor.apply(self, arguments);
311         self.on("afterRoutesChange", _afterRoutesChange, self);
312         _afterRoutesChange.call(self, {newVal:self.get("routes")});
313         allRoutes.push(self);
314     }

324          }

328 
329     S.extend(Router, Base,
330         /**
331          * @lends MVC.Router#
332          */
333         {
334             /**
335              *
336              * @param routes
337              * @example
338              * <code>
339              *   {
340              *     "/search/:param":"callback"
341              *     // or
342              *     "search":{
343              *       reg:/xx/,
344              *       callback:fn
345              *     }
346              *   }
347              * </code>
348              */
349             addRoutes:function (routes) {
350                 var self = this;
351                 each(routes, function (callback, name) {
352                     self[__routerMap][name] = transformRouterReg(self, name, normFn(self, callback));
353                 });
354             }
355         },
356         /**
357          * @lends MVC.Router
358          */
359         {
360 
361             navigate:function (path, opts) {
362                 if (getFragment() !== path) {
363                     if (Router.nativeHistory && supportNativeHistory) {
364                         history['pushState']({}, "", getFullPath(path));
365                         // pushState does not fire popstate event (unlike hashchange)
366                         // so popstate is not statechange
367                         // fire manually
368                         dispatch();
369                     } else {
370                         location.hash = "!" + path;
371                     }
372                 } else if (opts && opts.triggerRoute) {
373                     dispatch();
374                 }
375             },
376             /**
377              *
378              * @param {object} opts
379              * @param {Function} opts.success
380              * @param {Function} opts.error
381              * @param {Function} opts.complete
382              */
383             start:function (opts) {
384                 opts = opts || {};
385 
386                 opts.urlRoot = opts.urlRoot || "";
387 
388                 var urlRoot,
389                     nativeHistory = opts.nativeHistory,
390                     locPath = location.pathname,
391                     hash = getFragment(),
392                     hashIsValid = location.hash.match(/#!.+/);
393 
394                 urlRoot = Router.urlRoot = opts.urlRoot;
395                 Router.nativeHistory = nativeHistory;
396 
397                 if (nativeHistory) {
398 
399                     if (supportNativeHistory) {
400                         // http://x.com/#!/x/y
401                         // =>
402                         // http://x.com/x/y
403                         // =>
404                         // process without refresh page and add history entry
405                         if (hashIsValid) {
406                             if (equalsIgnoreSlash(locPath, urlRoot)) {
407                                 // put hash to path
408                                 history['replaceState']({}, "", getFullPath(hash));
409                                 opts.triggerRoute = 1;
410                             } else {
411                                 S.error("location path must be same with urlRoot!");
412                             }
413                         }
414                     }
415                     // http://x.com/x/y
416                     // =>
417                     // http://x.com/#!/x/y
418                     // =>
419                     // refresh page without add history entry
420                     else if (!equalsIgnoreSlash(locPath, urlRoot)) {
421                         location.replace(addEndSlash(urlRoot) + "#!" + hash);
422                         return;
423                     }
424 
425                 }
426 
427                 // prevent hashChange trigger on start
428                 setTimeout(function () {
429                     if (nativeHistory && supportNativeHistory) {
430                         Event.on(win, 'popstate', dispatch);
431                     } else {
432                         Event.on(win, "hashchange", dispatch);
433                         opts.triggerRoute = 1;
434                     }
435 
436                     // check initial hash on start
437                     // in case server does not render initial state correctly
438                     // when monitor hashchange ,client must be responsible for dispatching and rendering.
439                     if (opts.triggerRoute) {
440                         dispatch();
441                     }
442                     opts.success && opts.success();
443 
444                 }, BREATH_INTERVAL);
445             }
446         });
447 
448     return Router;
