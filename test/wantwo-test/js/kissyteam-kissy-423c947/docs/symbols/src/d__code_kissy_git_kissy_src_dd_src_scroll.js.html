1 /**
  2  * @fileOverview auto scroll for drag object's container
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("dd/scroll", function (S, DDM, Base, Node, DOM) {
  6 
  7     var TAG_DRAG = "__dd-scroll-id-",
  8         win = S.Env.host,
  9         stamp = S.stamp,
 10         RATE = [10, 10],
 11         ADJUST_DELAY = 100,
 12         DIFF = [20, 20],
 13         DESTRUCTORS = "__dd_scrolls";
 14 
 15     /**
 16      * make parent node scroll while dragging
 17      * @memberOf DD
 18      * @class
 19      */
 20     function Scroll() {
 21         var self = this;
 22         Scroll.superclass.constructor.apply(self, arguments);
 23         self[DESTRUCTORS] = {};
 24     }
 25 
 26     Scroll.ATTRS =
 27     /**
 28      * @lends DD.Scroll#
 29      */
 30     {
 31         /**
 32          * node to be scrolled while dragging
 33          * @type {window|String|HTMLElement}
 34          */
 35         node:{
 36             // value:window：不行，默认值一定是简单对象
 37             valueFn:function () {
 38                 return Node.one(win);
 39             },
 40             setter:function (v) {
 41                 return Node.one(v);
 42             }
 43         },
 44         /**
 45          * adjust velocity. default:[10,10]. larger faster
 46          * @type Number[]
 47          */
 48         rate:{
 49             value:RATE
 50         },
 51         /**
 52          * the margin to make node scroll. default: [20,20].
 53          * easier to scroll for node if larger.
 54          * @type number[]
 55          */
 56         diff:{
 57             value:DIFF
 58         }
 59     };
 60 
 61 
 62     var isWin = S.isWindow;
 63 
 64     S.extend(Scroll, Base,
 65         /**
 66          * @lends DD.Scroll#
 67          */
 68         {
 69             /**
 70              * @private
 71              * @param node
 72              */
 73             getRegion:function (node) {
 74                 if (isWin(node[0])) {
 75                     return {
 76                         width:DOM.viewportWidth(),
 77                         height:DOM.viewportHeight()
 78                     };
 79                 } else {
 80                     return {
 81                         width:node.outerWidth(),
 82                         height:node.outerHeight()
 83                     };
 84                 }
 85             },
 86 
 87             /**
 88              * @private
 89              * @param node
 90              */
 91             getOffset:function (node) {
 92                 if (isWin(node[0])) {
 93                     return {
 94                         left:DOM.scrollLeft(),
 95                         top:DOM.scrollTop()
 96                     };
 97                 } else {
 98                     return node.offset();
 99                 }
100             },
101 
102             /**
103              * @private
104              * @param node
105              */
106             getScroll:function (node) {
107                 return {
108                     left:node.scrollLeft(),
109                     top:node.scrollTop()
110                 };
111             },
112 
113             /**
114              * @private
115              * @param node
116              */
117             setScroll:function (node, r) {
118                 node.scrollLeft(r.left);
119                 node.scrollTop(r.top);
120             },
121 
122             /**
123              * make node not to scroll while this drag object is dragging
124              * @param {DD.Draggable} drag
125              */
126             detachDrag:function (drag) {
127                 var tag,
128                     destructors = this[DESTRUCTORS];
129                 if (!(tag = stamp(drag, 1, TAG_DRAG)) ||
130                     !destructors[tag]
131                     ) {
132                     return;
133                 }
134                 destructors[tag].fn();
135                 delete destructors[tag];
136             },
137 
138             /**
139              * make node not to scroll at all
140              */
141             destroy:function () {
142                 var self = this,
143                     destructors = self[DESTRUCTORS];
144                 for (var d in destructors) {
145                     self.unAttach(destructors[d].drag);
146                 }
147             },
148 
149             /**
150              * make node to scroll while this drag object is dragging
151              * @param {DD.Draggable} drag
152              */
153             attachDrag:function (drag) {
154                 var self = this,
155                     node = self.get("node"),
156                     tag = stamp(drag, 0, TAG_DRAG),
157                     destructors = self[DESTRUCTORS];
158 
159                 if (destructors[tag]) {
160                     return;
161                 }
162 
163                 var rate = self.get("rate"),
164                     diff = self.get('diff'),
165                     event,
166                     /*
167                      目前相对 container 的偏移，container 为 window 时，相对于 viewport
168                      */
169                     dxy,
170                     timer = null;
171 
172                 // fix https://github.com/kissyteam/kissy/issues/115
173                 // dragDelegate 时 可能一个 dragDelegate对应多个 scroll
174                 // check container
175                 function checkContainer() {
176                     if (isWin(node[0])) {
177                         return 0;
178                     }
179                     // 判断 proxyNode，不对 dragNode 做大的改变
180                     var mousePos = drag.mousePos,
181                         r = DDM.region(node);
182 
183                     if (!DDM.inRegion(r, mousePos)) {
184                         clearTimeout(timer);
185                         timer = 0;
186                         return 1;
187                     }
188                     return 0;
189                 }
190 
191                 function dragging(ev) {
192                     // 给调用者的事件，框架不需要处理
193                     // fake 也表示该事件不是因为 mouseover 产生的
194                     if (ev.fake) {
195                         return;
196                     }
197 
198                     if (checkContainer()) {
199                         return;
200                     }
201 
202                     // 更新当前鼠标相对于拖节点的相对位置
203                     event = ev;
204                     dxy = S.clone(drag.mousePos);
205                     var offset = self.getOffset(node);
206                     dxy.left -= offset.left;
207                     dxy.top -= offset.top;
208                     if (!timer) {
209                         checkAndScroll();
210                     }
211                 }
212 
213                 function dragEnd() {
214                     clearTimeout(timer);
215                     timer = null;
216                 }
217 
218                 drag.on("drag", dragging);
219 
220                 drag.on("dragend", dragEnd);
221 
222                 destructors[tag] = {
223                     drag:drag,
224                     fn:function () {
225                         drag.detach("drag", dragging);
226                         drag.detach("dragend", dragEnd);
227                     }
228                 };
229 
230                 function checkAndScroll() {
231                     if (checkContainer()) {
232                         return;
233                     }
234 
235                     var r = self.getRegion(node),
236                         nw = r.width,
237                         nh = r.height,
238                         scroll = self.getScroll(node),
239                         origin = S.clone(scroll),
240                         diffY = dxy.top - nh,
241                         adjust = false;
242 
243                     if (diffY >= -diff[1]) {
244                         scroll.top += rate[1];
245                         adjust = true;
246                     }
247 
248                     var diffY2 = dxy.top;
249 
250                     if (diffY2 <= diff[1]) {
251                         scroll.top -= rate[1];
252                         adjust = true;
253                     }
254 
255                     var diffX = dxy.left - nw;
256 
257                     if (diffX >= -diff[0]) {
258                         scroll.left += rate[0];
259                         adjust = true;
260                     }
261 
262                     var diffX2 = dxy.left;
263 
264                     if (diffX2 <= diff[0]) {
265                         scroll.left -= rate[0];
266                         adjust = true;
267                     }
268 
269                     if (adjust) {
270                         self.setScroll(node, scroll);
271                         timer = setTimeout(checkAndScroll, ADJUST_DELAY);
272                         // 不希望更新相对值，特别对于相对 window 时，相对值如果不真正拖放触发的 drag，是不变的，
273                         // 不会因为程序 scroll 而改变相对值
274 
275                         // 调整事件，不需要 scroll 监控，达到预期结果：元素随容器的持续不断滚动而自动调整位置.
276                         event.fake = true;
277                         if (isWin(node[0])) {
278                             // 当使 window 自动滚动时，也要使得拖放物体相对文档位置随 scroll 改变
279                             // 而相对 node 容器时，只需 node 容器滚动，拖动物体相对文档位置不需要改变
280                             scroll = self.getScroll(node);
281                             event.left += scroll.left - origin.left;
282                             event.top += scroll.top - origin.top;
283                         }
284                         // 容器滚动了，元素也要重新设置 left,top
285                         if (drag.get("move")) {
286                             drag.get("node").offset(event);
287                         }
288                         drag.fire("drag", event);
289                     } else {
290                         timer = null;
291                     }
292                 }
293 
294             }
295         });
296 
297     // for compatibility
298     var ScrollPrototype = Scroll.prototype;
299     ScrollPrototype.attach = ScrollPrototype.attachDrag;
300     ScrollPrototype.unAttach = ScrollPrototype.detachDrag;
301     return Scroll;
302 }, {
303     requires:['./ddm', 'base', 'node', 'dom']
304 });
