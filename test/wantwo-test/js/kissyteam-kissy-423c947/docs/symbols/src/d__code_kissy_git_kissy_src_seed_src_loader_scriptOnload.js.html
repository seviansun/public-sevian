  5 (function (S) {
  6     if (typeof require !== 'undefined') {
  7         return;

 10     var CSS_POLL_INTERVAL = 30,
 11         win = S.Env.host,
 12         utils = S.Loader.Utils,
 13         /**
 14          * central poll for link node
 15          */
 16             timer = 0,
 17 
 18         monitors = {
 19             /**
 20              * node.id:{callback:callback,node:node}
 21              */
 22         };
 23 
 24     function startCssTimer() {
 25         if (!timer) {
 26             // S.log("start css polling");
 27             cssPoll();
 28         }
 29     }
 30 
 31     // single thread is ok
 32     function cssPoll() {
 33         for (var url in monitors) {
 34             var callbackObj = monitors[url],
 35                 node = callbackObj.node,
 36                 loaded = 0;
 37             if (utils.isWebKit) {
 38                 if (node['sheet']) {
 39                     S.log("webkit loaded : " + url);
 40                     loaded = 1;
 41                 }
 42             } else if (node['sheet']) {
 43                 try {
 44                     var cssRules;
 45                     if (cssRules = node['sheet'].cssRules) {
 46                         S.log('firefox  ' + cssRules + ' loaded : ' + url);
 47                         loaded = 1;
 48                     }
 49                 } catch (ex) {
 50                     // S.log('firefox  ' + ex.name + ' ' + ex.code + ' ' + url);
 51                     // if (ex.name === 'NS_ERROR_DOM_SECURITY_ERR') {
 52                     if (ex.code === 1000) {
 53                         S.log('firefox  ' + ex.name + ' loaded : ' + url);
 54                         loaded = 1;
 55                     }
 56                 }
 57             }
 58 
 59             if (loaded) {
 60                 if (callbackObj.callback) {
 61                     callbackObj.callback.call(node);
 62                 }
 63                 delete monitors[url];
 64             }
 65         }
 66         if (S.isEmptyObject(monitors)) {
 67             timer = 0;
 68             // S.log("end css polling");
 69         } else {
 70             timer = setTimeout(cssPoll, CSS_POLL_INTERVAL);
 71         }
 72     }
 73 
 74     S.mix(utils, {
 75         scriptOnLoad:win.addEventListener ?
 76             function (node, callback) {
 77                 node.addEventListener('load', callback, false);
 78             } :
 79             function (node, callback) {
 80                 node.onreadystatechange = function () {
 81                     var self = this, rs = self.readyState;
 82                     if (/loaded|complete/i.test(rs)) {
 83                         self.onreadystatechange = null;
 84                         callback.call(self);
 85                     }
 86                 };
 87             },
 88 
 89         /**
 90          * monitor css onload across browsers
 91          * 暂时不考虑如何判断失败，如 404 等
 92          * @see <pre>
 93          *  - firefox 不可行（结论4错误）：
 94          *    - http://yearofmoo.com/2011/03/cross-browser-stylesheet-preloading/
 95          *  - 全浏览器兼容
 96          *    - http://lifesinger.org/lab/2011/load-js-css/css-preload.html
 97          *  - 其他
 98          *    - http://www.zachleat.com/web/load-css-dynamically/
 99          *  </pre>
100          */
101         styleOnLoad:win.attachEvent ?
102             // ie/opera
103             function (node, callback) {
104                 // whether to detach using function wrapper?
105                 function t() {
106                     node.detachEvent('onload', t);
107                     S.log('ie/opera loaded : ' + node.href);
108                     callback.call(node);
109                 }
110 
111                 node.attachEvent('onload', t);
112             } :
113             // refer : http://lifesinger.org/lab/2011/load-js-css/css-preload.html
114             // 暂时不考虑如何判断失败，如 404 等
115             function (node, callback) {
116                 var href = node.href,
117                     arr;
118                 arr = monitors[href] = {};
119                 arr.node = node;
120                 arr.callback = callback;
121                 startCssTimer();
122             }
