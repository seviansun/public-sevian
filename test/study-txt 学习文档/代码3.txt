        $(
        function() {
            $("div").click(
            function() {
                $("<div>").appendTo("body").css(
                (function(e) {
                    var styleObj = {};
                    for (var key in e.style) {
                        if (e.style[key] != "") {
                            styleObj[key] = e.style[key];
                        }
                    }
                    return styleObj;
                } (this))
                );
            }
            );
        }
        );
